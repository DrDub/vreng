76 {

79  center.v[0] += lasting * move.lspeed.v[0];

91 void Sheet::updateTime(time_t s, time_t us, float *lasting)
92 {
93  if (move.aspeed.v[0] == aspeed) // turning right
94  move.ttl = MIN(ABSF(deltaAngle(pos.az, aright)/move.aspeed.v[0]), move.ttl);
95  else if (move.aspeed.v[0] == -aspeed) // turning left
96  move.ttl = MIN(ABSF(deltaAngle(pos.az, aleft)/move.aspeed.v[0]), move.ttl);
97 
98  if (! updateLasting(s, us, lasting)) {
99  if (book->fwd > 0) book->forward(NULL, 0, 0);
100  book->fwd--;
101  if (book->bwd > 0) book->backward(NULL, 0, 0);
102  book->bwd--;
103  }
104 }

157 void Sheet::pullNext()
158 {
159  // if (state == LEFT) return;
160  getDim(size);
161  float ttl = ABSF(deltaAngle(pos.az, aleft) / aspeed);
162  float dx = -size.v[1] * (cos(aleft) + cos(aright));
163  float dy = size.v[1] * (sin(aleft) + sin(aright));
164  clearV3(move.lspeed);
165  clearV3(move.aspeed);
166  move.lspeed.v[0] = dx / ttl;
167  move.lspeed.v[1] = dy / ttl;
168  move.aspeed.v[0] = -aspeed;
169  initImposedMovement(ttl);
170  //state = LEFT;
171 }
