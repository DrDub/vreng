149 float _3ds::getRadius()
150 {
151  float max_radius = 0.;
152 
153  if (loaded) {
154  for (int i=0; i < _3dsModel.numOfObjects; i++) {
155  if (_3dsModel.pObject.size() <= 0) break;
156  tObject *pObject = &_3dsModel.pObject[i];
157  for (int j=0; j < pObject->numOfFaces; j++) {
158  for (int whichVertex=0; whichVertex < 3; whichVertex++) {
159  int vi = pObject->pFaces[j].vertIndex[whichVertex];
160  float r= pObject->pVerts[vi].x * pObject->pVerts[vi].x +
161  pObject->pVerts[vi].y * pObject->pVerts[vi].y +
162  pObject->pVerts[vi].z * pObject->pVerts[vi].z ;
163  if (r > max_radius) max_radius = r;
164  }
165  }
166  }
167  }
168  return sqrt(max_radius);
169 }

250 void _3ds::draw()
251 {
252  if (! loaded) return;
253 
254  if (currentScale != desiredScale) setScale(desiredScale);
255 
256  for (int i=0; i < _3dsModel.numOfObjects; i++) {
257  if (_3dsModel.pObject.size() <= 0) break;
258  tObject *pObject = &_3dsModel.pObject[i]; // Get the current object
259 
260  if (pObject->bHasTexture) {
261  glColor3ub(255, 255, 255); // Reset the color to normal again
262 
263  // Bind the texture map to the object by it's materialID
264  bindTexture2D(_3dsModel.pMaterials[pObject->materialID].texureId);
265  }
266  else {
267  glDisable(GL_TEXTURE_2D);
268  glEnable(GL_COLOR);
269  glEnable(GL_COLOR_MATERIAL);
270  glColor3ub(255, 200, 200); // Reset the color to normal again
271  }
272 
273  glBegin(GL_TRIANGLES); // Begin drawing
274 
275  for (int j=0; j < pObject->numOfFaces; j++) { // all faces
276  for (int whichVertex=0; whichVertex < 3; whichVertex++) {
277  int vi = pObject->pFaces[j].vertIndex[whichVertex];
278  if (vi < 0) continue; //DAX BUG: segfault
279  glNormal3f(pObject->pNormals[vi].x, pObject->pNormals[vi].y, pObject->pNormals[vi].z);
280  if (pObject->bHasTexture) {
281  if (pObject->pTexVerts)
282  glTexCoord2f(pObject->pTexVerts[vi].x, pObject->pTexVerts[vi].y);
283  }
284  else {
285  // Make sure there is a valid material/color assigned to this object
286  if (_3dsModel.pMaterials.size() && pObject->materialID >= 0) {
287 
288  // Get and set the color that the object is, since it must not have a texture
289  uint8_t *pColor = _3dsModel.pMaterials[pObject->materialID].color;
290  glColor3ub(pColor[0], pColor[1], pColor[2]);
291  }
292  }
293  glVertex3f(pObject->pVerts[vi].x, pObject->pVerts[vi].y, pObject->pVerts[vi].z);
294  }
295  }
296  glEnd(); // End the drawing
297  glColor3ub(255,255,255);
298  glDisable(GL_TEXTURE_2D);
299  glDisable(GL_COLOR_MATERIAL);
300  }
301 }

304 bool _3ds::importModel(t3dsModel *pModel)
305 {
306  tChunk currentChunk = {0};
307 
308  if (!pModel || !fp) return false;
309 
310  readChunk(&currentChunk);
311  if (currentChunk.ID != PRIMARY) {
312  error("Unable to load PRIMARY chuck from file");
313  return false;
314  }
315 
316  // Now we actually start reading in the data. processNextChunk() is recursive
317  processNextChunk(pModel, &currentChunk);
318 
319  // After we have read the whole 3DS file, we want to calculate our own vertex normals.
320  computeNormals(pModel);
321 
322  File::closeFile(fp);
323  return true;
324 }

327 void _3ds::processNextChunk(t3dsModel *pModel, tChunk *pPreviousChunk)
328 {
329  tObject newObject = {0}; // This is used to add to our object list
330  t3dsMaterialInfo newTexture; // This is used to add to our material list
331  unsigned int version = 0; // This will hold the file version
332  int buffer[50000] = {0}; // This is used to read past unwanted data
333  tChunk currentChunk = {0}; // The current chunk to load
334  tChunk tempChunk = {0}; // A temp chunk for holding data
335 
336  // Continue to read the sub chunks until we have reached the length.
337  // After we read ANYTHING we add the bytes read to the chunk and then check
338  while (pPreviousChunk->bytesRead < pPreviousChunk->length) {
339  readChunk(&currentChunk);
340 
341  switch (currentChunk.ID) {
342  case VERSION_3DS: // This holds the version of the file
343  // This chunk has an unsigned short that holds the file version.
344  currentChunk.bytesRead += fread(&version, 1, currentChunk.length - currentChunk.bytesRead,fp);
345  File::localEndian(&version,sizeof(unsigned int));
346 
347  if (version > 0x03) error("This 3DS file is over version 3 so it may load incorrectly");
348  break;
349  case OBJECTINFO: // This holds the version of the mesh
350  // This chunk holds the version of the mesh. It is also the head of the MATERIAL
351  // and OBJECT chunks. From here on we start reading in the material and object info.
352  readChunk(&tempChunk);
353 
354  // Get the version of the mesh
355  tempChunk.bytesRead += fread(&version, 1, tempChunk.length - tempChunk.bytesRead, fp);
356  File::localEndian(&version,sizeof(unsigned int));
357 
358  currentChunk.bytesRead += tempChunk.bytesRead;
359 
360  // Go to the next chunk, which is the object has a texture, it should be MATERIAL, then OBJECT
361  processNextChunk(pModel, &currentChunk);
362  break;
363  case MATERIAL: // This holds the material information
364  // This chunk is the header for the material info chunks
365  pModel->numOfMaterials++;
366 
367  // Add a empty texture structure to our texture list.
368  pModel->pMaterials.push_back(newTexture);
369 
370  // Proceed to the material loading function
371  processNextMaterialChunk(pModel, &currentChunk);
372  break;
373  case OBJECT: // This holds the name of the object being read
374  // This chunk is the header for the object info chunks. It also
375  // holds the name of the object.
376  pModel->numOfObjects++; // Increase the object count
377 
378  // Add a new tObject node to our list of objects (like a link list)
379  pModel->pObject.push_back(newObject);
380 
381  // Initialize the object and all it's data members
382  memset(&(pModel->pObject[pModel->numOfObjects - 1]), 0, sizeof(tObject));
383 
384  // Get the name of the object and store it, then add the read bytes
385  currentChunk.bytesRead += getString(pModel->pObject[pModel->numOfObjects - 1].strName);
386 
387  // Now proceed to read in the rest of the object information
388  processNextObjectChunk(pModel, &(pModel->pObject[pModel->numOfObjects - 1]), &currentChunk);
389  break;
390  case EDITKEYFRAME:
391  //processNextKeyFrameChunk(pModel, currentChunk);
392  // Read past this chunk and add the bytes read
393  currentChunk.bytesRead += fread(buffer, 1, currentChunk.length - currentChunk.bytesRead, fp);
394  break;
395  default:
396  // If we didn't care about a chunk, then we get here. We still need
397  // to read past the unknown or ignored chunk
398  currentChunk.bytesRead += fread(buffer, 1, currentChunk.length - currentChunk.bytesRead, fp);
399  break;
400  }
401 
402  pPreviousChunk->bytesRead += currentChunk.bytesRead;
403  }
404  // Free the current chunk
405  //delete currentChunk;
406  //currentChunk = pPreviousChunk;
407 }

410 void _3ds::processNextObjectChunk(t3dsModel *pModel, tObject *pObject, tChunk *pPreviousChunk)
411 {
412  int buffer[50000] = {0}; // This is used to read past unwanted data
413 
414  // Allocate a new chunk to work with
415  tChunk currentChunk = {0};
416 
417  // Continue to read these chunks until we read the end of this sub chunk
418  while (pPreviousChunk->bytesRead < pPreviousChunk->length) {
419  readChunk(&currentChunk);
420 
421  switch (currentChunk.ID) { // Check which chunk we just read
422 
423  case OBJECT_MESH: // This lets us know that we are reading a new object
424  // We found a new object, so let's read in it's info using recursion
425  processNextObjectChunk(pModel, pObject, &currentChunk);
426  break;
427  case OBJECT_VERTICES: // This is the objects vertices
428  readVertices(pObject, &currentChunk);
429  break;
430  case OBJECT_FACES: // This is the objects face information
431  readVertexIndices(pObject, &currentChunk);
432  break;
433  case OBJECT_MATERIAL: // This holds the material name that the object has
434  // This chunk holds the name of the material that the object has assigned to it.
435  // We now will read the name of the material assigned to this object
436  readObjectMaterial(pModel, pObject, &currentChunk);
437  break;
438  case OBJECT_UV: // This holds the UV texture coordinates for the object
439  // This chunk holds all of the UV coordinates for our object. Let's read them in.
440  readUVCoordinates(pObject, &currentChunk);
441  break;
442  default:
443  currentChunk.bytesRead += fread(buffer, 1, currentChunk.length - currentChunk.bytesRead, fp);
444  break;
445  }
446 
447  pPreviousChunk->bytesRead += currentChunk.bytesRead;
448  }
449  // Free the current chunk
450  //delete currentChunk;
451  //currentChunk = pPreviousChunk;
452 }

455 void _3ds::processNextMaterialChunk(t3dsModel *pModel, tChunk *pPreviousChunk)
456 {
457  int buffer[50000] = {0}; // This is used to read past unwanted data
458  tChunk currentChunk = {0};
459 
460  while (pPreviousChunk->bytesRead < pPreviousChunk->length) {
461  readChunk(&currentChunk);
462 
463  switch (currentChunk.ID) {
464  case MATNAME: // This chunk holds the name of the material
465  // Here we read in the material name
466  currentChunk.bytesRead += fread(pModel->pMaterials[pModel->numOfMaterials - 1].strName, 1, currentChunk.length - currentChunk.bytesRead, fp);
467  break;
468  case MATDIFFUSE: // This holds the R G B color of our object
469  readColorChunk(&(pModel->pMaterials[pModel->numOfMaterials - 1]), &currentChunk);
470  break;
471  case MATMAP: // This is the header for the texture info
472  // Proceed to read in the material information
473  processNextMaterialChunk(pModel, &currentChunk);
474  break;
475  case MATMAPFILE: // This stores the file name of the material
476  // Here we read in the material's file name
477  currentChunk.bytesRead += fread(pModel->pMaterials[pModel->numOfMaterials - 1].strFile, 1, currentChunk.length - currentChunk.bytesRead, fp);
478  break;
479  default:
480  currentChunk.bytesRead += fread(buffer, 1, currentChunk.length - currentChunk.bytesRead, fp);
481  break;
482  }
483 
484  pPreviousChunk->bytesRead += currentChunk.bytesRead;
485  }
486  // Free the current chunk
487  //delete currentChunk;
488  //currentChunk = pPreviousChunk;

529 void _3ds::readVertexIndices(tObject *pObject, tChunk *pPreviousChunk)
530 {
531  uint16_t index = 0; // This is used to read in the current face index
532 
533  // Read in the number of faces that are in this object (int)
534  pPreviousChunk->bytesRead += fread(&pObject->numOfFaces, 1, 2, fp);
535  File::localEndian(&pObject->numOfFaces,sizeof(int));
536 
537  // Alloc enough memory for the faces and initialize the structure
538  pObject->pFaces = new t3dsFace [pObject->numOfFaces];
539  memset(pObject->pFaces, 0, sizeof(t3dsFace) * pObject->numOfFaces);
540 
541  for (int i=0; i < pObject->numOfFaces; i++) { // all faces
542  // Next, we read in the A then B then C index for the face, but ignore the 4th value.
543  // The fourth value is a visibility flag for 3D Studio Max, we don't care about this.
544  for (int j=0; j < 4; j++) {
545  // Read the first vertice index for the current face
546  pPreviousChunk->bytesRead += fread(&index, 1, sizeof(index), fp);
547  File::localEndian(&index,sizeof(unsigned short));
548  if (j < 3) // Store the index in our face structure.
549  pObject->pFaces[i].vertIndex[j] = index;
550  }
551  }
552 }

573 void _3ds::readVertices(tObject *pObject, tChunk *pPreviousChunk)
574 {
575  // Read in the number of vertices (int)
576  pPreviousChunk->bytesRead += fread(&(pObject->numOfVerts), 1, 2, fp);
577  File::localEndian(&pObject->numOfVerts,sizeof(int));
578 
579  //error("readVertices: numOfVerts=%d", pObject->numOfVerts);
580 
581  // Allocate the memory for the verts and initialize the structure
582  pObject->pVerts = new Vec3 [pObject->numOfVerts];
583  memset(pObject->pVerts, 0, sizeof(Vec3) * pObject->numOfVerts);
584 
585  // Read in the array of vertices (an array of 3 floats)
586  pPreviousChunk->bytesRead += fread(pObject->pVerts, 1, pPreviousChunk->length - pPreviousChunk->bytesRead, fp);
587 
588  // Now we should have all of the vertices read in. Because 3D Studio Max
589  // Models with the Z-Axis pointing up (strange and ugly I know!), we need
590  // to flip the y values with the z values in our vertices. That way it
591  // will be normal, with Y pointing up. If you prefer to work with Z pointing
592  // up, then just delete this next loop. Also, because we swap the Y and Z
593  // we need to negate the Z to make it come out correctly.
594 
595  // Go through all of the vertices that we just read and swap the Y and Z values
596  for (int i=0; i < pObject->numOfVerts; i++) {
597  File::localEndian(&pObject->pVerts[i].x,sizeof(float));
598  File::localEndian(&pObject->pVerts[i].y,sizeof(float));
599  File::localEndian(&pObject->pVerts[i].z,sizeof(float));
600 
601  float fTempY = pObject->pVerts[i].y;
602 
603  // Set the Y value to the Z value
604  pObject->pVerts[i].y = pObject->pVerts[i].z;
605 
606  // Set the Z value to the Y value,
607  // but negative Z because 3D Studio max does the opposite.
608  pObject->pVerts[i].z = -fTempY;
609  }
610 }

613 void _3ds::readObjectMaterial(t3dsModel *pModel, tObject *pObject, tChunk *pPreviousChunk)
614 {
615  char strMaterial[255] = {0}; // This is used to hold the objects material name
616  int buffer[50000] = {0}; // This is used to read past unwanted data
617 
618  pPreviousChunk->bytesRead += getString(strMaterial);
619 
620  // Now that we have a material name, we need to go through all of the materials
621  // and check the name against each material. When we find a material in our material
622  // list that matches this name we just read in, then we assign the materialID
623  // of the object to that material index. You will notice that we passed in the
624  // model to this function. This is because we need the number of textures.
625  // Yes though, we could have just passed in the model and not the object too.
626 
627  // Go through all of the textures
628  for (int i=0; i < pModel->numOfMaterials; i++) {
629  // If the material we just read in matches the current texture name
630  if (strcmp(strMaterial, pModel->pMaterials[i].strName) == 0) {
631  // Set the material ID to the current index 'i' and stop checking
632  pObject->materialID = i;
633 
634  // Now that we found the material, check if it's a texture map.
635  // If the strFile has a string length of 1 and over it's a texture
636  if (strlen(pModel->pMaterials[i].strFile) > 0)
637  // Set the object's flag to say it has a texture map to bind.
638  pObject->bHasTexture = true;
639  break;
640  }
641  else
642  // Set the ID to -1 to show there is no material for this object
643  pObject->materialID = -1;
644  }
645  // Read past the rest of the chunk since we don't care about shared vertices
646  pPreviousChunk->bytesRead += fread(buffer, 1, pPreviousChunk->length - pPreviousChunk->bytesRead, fp);
647 }

650 void _3ds::computeNormals(t3dsModel *pModel)
651 {
652  Vec3 v1, v2, vNormal, vPoly[3];
653  int i;
654 
655  // If there are no objects, we can skip this part
656  if (pModel->numOfObjects <= 0) return;
657 
658  // What are vertex normals? And how are they different from other normals?
659  // Well, if you find the normal to a triangle, you are finding a "Face Normal"
660  // If you give OpenGL a face normal for lighting, it will make your object look
661  // really flat and not very round. If we find the normal for each vertex, it makes
662  // the smooth lighting look. This also covers up blocky looking objects and they appear
663  // to have more polygons than they do. Basically, what you do is first
664  // calculate the face normals, then you take the average of all the normals around each
665  // vertex. It's just averaging. That way you get a better approximation for that vertex.
666 
667  // Go through each of the objects to calculate their normals
668  for (int index=0; index < pModel->numOfObjects; index++) {
669  tObject *pObject = &(pModel->pObject[index]); // Get the current object
670 
671  Vec3 *pNormals = new Vec3 [pObject->numOfFaces];
672  Vec3 *pTempNormals = new Vec3 [pObject->numOfFaces];
673  pObject->pNormals = new Vec3 [pObject->numOfVerts];
674  //error("Debug: numOfFaces=%d",pObject->numOfFaces);
675 
676  for (i=0; i < pObject->numOfFaces; i++) { // all of the faces
677  // To cut down LARGE code, we extract the 3 points of this face
678  vPoly[0] = pObject->pVerts[pObject->pFaces[i].vertIndex[0]];
679  vPoly[1] = pObject->pVerts[pObject->pFaces[i].vertIndex[1]];
680  vPoly[2] = pObject->pVerts[pObject->pFaces[i].vertIndex[2]];
681 
682  // Now calculate the face normals (Get 2 vectors and find the cross product of those 2)
683  v1 = Vector(vPoly[0], vPoly[2]); // Get the vector of the polygon (we just need 2 sides for the normal)
684  v2 = Vector(vPoly[2], vPoly[1]); // Get a second r of the polygon
685  vNormal = Cross(v1, v2); // Return the cross product of the 2 vectors (normalize vector, but not a unit vector)
686  pTempNormals[i] = vNormal; // Save the un-normalized normal for the vertex normals
687  vNormal = Normalize(vNormal); // Normalize the cross product to give us the polygons normal
688  pNormals[i] = vNormal; // Assign the normal to the list of normals
689  }
690 
692  Vec3 vSum = {0.0, 0.0, 0.0};
693  Vec3 vZero = vSum;
694  int shared = 0;
695 
696  for (i=0; i < pObject->numOfVerts; i++) { // all the vertices
697  for (int j=0; j < pObject->numOfFaces; j++) { // all the triangles
698  // Check if the vertex is shared by another face
699  if (pObject->pFaces[j].vertIndex[0] == i ||
700  pObject->pFaces[j].vertIndex[1] == i ||
701  pObject->pFaces[j].vertIndex[2] == i) {
702  vSum = AddVector(vSum, pTempNormals[j]);// Add the un-normalized normal of the shared face
703  shared++; // Increase the number of shared triangles
704  }
705  }
706 
707  // Get the normal by dividing the sum by the shared. We negate the shared so it has the normals pointing out.
708  pObject->pNormals[i] = DivideVectorByScaler(vSum, float(-shared));
709  pObject->pNormals[i] = Normalize(pObject->pNormals[i]);
710  vSum = vZero; // Reset the sum
711  shared = 0; // Reset the shared
712  }
713 
714  // Free our memory and start over on the next object
715  if (pTempNormals) delete[] pTempNormals;
716  if (pNormals) delete[] pNormals;
717  }
718 }
