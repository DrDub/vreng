76 void River::inits()
77 {
78  if (waves > DEF_WAVES) waves = DEF_WAVES;
79  mesh = new float[2*waves*sizeof(float)+1];
80  phase = new float[waves*sizeof(float)];
81  speed = new float[waves*sizeof(float)];
82  ampl = new float[waves*sizeof(float)];
83  srand(time(0));
84 
85  for (int i=0; i<waves ; i++) {
86  mesh[2*i] = (float) (rand()%(int)ceil(width));
87  mesh[2*i+1] = (float) (rand()%(int)ceil(depth));
88  phase[i] = (float) (rand()%10);
89  speed[i] = (float) (rand()%100)/300;
90  ampl[i] = 0.005* (float) (rand()%100)/100;
91  }
92 }

101 void River::wave(float a, float b)
102 {
103  float x=0, y=0, z=0;
104  float vx,vy;
105 
106  for (int i=0; i<waves ; i++) {
107  vx = mesh[2*i];
108  vy = mesh[2*i+1];
109  x += vx * width * sin(a*vx+b*vy + phase[i]) * ampl[i];
110  y += vy * depth * sin(a*vx+b*vy + phase[i]) * ampl[i];
111  }
112  glNormal3f(x, y, 1);
113 
114  for (int i=0; i<waves ; i++) {
115  vx = mesh[2*i];
116  vy = mesh[2*i+1];
117  z += cos(a*vx+b*vy + phase[i])*ampl[i];
118  }
119  glVertex3d(a, b, z);
120 }

140  for (int b=0; b < 2*ceil(depth) ; b++) {
141  wave((float) a, (float) b);
142  wave((float) a+1, (float) b);
143  wave((float) a+1, (float) b+1);
144  wave((float) a, (float) b+1);
145  }
146  }
147  glEnd();
148  for (int i=0; i<waves ; i++)
149  phase[i] += speed[i];
150 
151  glDisable(GL_LIGHTING);
152  glPopMatrix();
153  //DAX glPopAttrib();
154 }
