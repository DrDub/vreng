86 void Thunder::draw()
87 {
88  for (int i=0; i < 4; i++) color[i] = 1;
89 
90  for (int n=0; n < number; n++) {
91  s_point pt;
92  struct timeval tv;
93 
94  gettimeofday(&tv, NULL);
95  srand((time_t) tv.tv_usec);
96 
97  int sign = rand()%2 - 1;
98  sign = (sign < 0) ? -1 : 1; // [-1,1]
99 
100  int div = rand()%12 + 12; // [12..24]
101  // starting point
102  pt.x = pos.x + ((rand()%20) - 20/2); // xorig +- 10
103  pt.y = pos.y + ((rand()%40) - 40/2); // yorig +- 20
104  pt.z = pos.z + ((rand()%10) - 10/2); // zorig +- 5
105  // increment
106  inc.x = pt.x / div;
107  inc.y = pt.y / div;
108  inc.z = (pt.z + 10) / div;
109 
110  dlist = glGenLists(number);
111  glNewList(dlist + n, GL_COMPILE);
112  glBegin(GL_LINE_STRIP);
113  glVertex3f(pt.x, pt.y, pt.z);
114  for (int i=0; i < div; i++) {
115  glVertex3f(pt.x, pt.y, pt.z);
116  // next point
117  pt.x += (sign * (inc.x + ((rand()%6) - 6/2)));
118  pt.y += (sign * (inc.y + ((rand()%6) - 6/2)));
119  //pt.z -= inc.z + ((rand()%2) - 1);
120  pt.z -= inc.z; // fall down
121  }
122  glEnd();
123  glEndList();
124  }
125 }
