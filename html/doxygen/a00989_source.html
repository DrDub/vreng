99 void Render::config(bool _quality)
100 {
101  static bool configured = false;
102  if (! configured) {
103  glFrontFace(GL_CCW); // trigo
104  glEnable(GL_DEPTH_TEST);
105  glEnable(GL_ALPHA_TEST);
106  glEnable(GL_CULL_FACE);
107  glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
108  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
109 
110  // anti aliasing
111  if (_quality) {
112  glEnable(GL_LINE_SMOOTH);
113  glEnable(GL_POLYGON_SMOOTH);
114  }
115  else {
116  glDisable(GL_LINE_SMOOTH);
117  glDisable(GL_POLYGON_SMOOTH);
118  }
119 
120  GLenum hint = _quality ? GL_NICEST : GL_FASTEST;
121  glHint(GL_PERSPECTIVE_CORRECTION_HINT, hint);
122  glHint(GL_POINT_SMOOTH_HINT, hint);
123  glHint(GL_LINE_SMOOTH_HINT, hint);
124  glHint(GL_POLYGON_SMOOTH_HINT, hint);
125  glHint(GL_FOG_HINT, hint);
126  if (::g.pref.infogl == true)
127  trace(DBG_FORCE, "version=%s vendor=%s renderer=%s, depth=%d texturess=%d stencil=%d clips=%d", glGetString(GL_VERSION), glGetString(GL_VENDOR), glGetString(GL_RENDERER), haveDepth(), haveTextures(), haveStencil(), haveClips());
128  configured = true;
129  }
130  camera();
131 }

234 /*
235  * Rendering specific
236  */
237 void Render::specific(uint32_t num, uint8_t pri)
238 {
239  switch (pri) {
240  case WObject::RENDER_LOW: // low render : background
241  specificStill(num, pri); // still objects : walls
242  specificMobile(num, pri); // mobile objects : particle
243  specificInvisible(num, pri);// invisible objects : sun
244  break;
245  case WObject::RENDER_NORMAL: // normal render
246  specificStill(num, pri); // still objects
247  specificMobile(num, pri); // mobile objects
248  specificInvisible(num, pri);// invisible objects : transform
249  break;
250  case WObject::RENDER_HIGH: // high render : models
251  specificStill(num, pri); // still objects
252  specificMobile(num, pri); // mobile objects
253  specificInvisible(num, pri);
254  break;
255  }
256 }

258 /*
259  * Render all solids.
260  * - renders solids in displaylists
261  * - makes specific rendering for special solids
262  */
263 void Render::solids(bool zsel = false)
264 {
265  uint32_t objectsnumber = WObject::getObjectsNumber();
266 
267  // Find the local user solid
268  list<Solid*>::iterator psu;
269  for (list<Solid*>::iterator ps = solidList.begin(); ps != solidList.end() ; ps++) {
270  if ((*ps)->object()->type == USER_TYPE) {
271  if (! strcmp((*ps)->object()->getInstance(), localuser->getInstance()))
272  psu = ps; // local user
273  }
274  }
275 
277  //
278  trace2(DBG_VGL, "\n--- LOW");
279  trace2(DBG_VGL, "\nspeci: ");
280  for (uint32_t i=1; i < objectsnumber; i++) { // do specific rendering by objects themself
281  specific(i, WObject::RENDER_LOW); // particules
282  }
283  trace2(DBG_VGL, "\nopaque: ");
284  solidsOpaque(zsel, psu, WObject::RENDER_LOW);
285  trace2(DBG_VGL, "\ntranslucid: ");
286  solidsTranslucid(zsel, psu, WObject::RENDER_LOW);
287 
289  //
290  trace2(DBG_VGL, "\n--- NORMAL");
291  trace2(DBG_VGL, "\nspeci: ");
292  for (uint32_t i=1; i < objectsnumber; i++) { // do specific rendering by objects themself
293  specific(i, WObject::RENDER_NORMAL);
294  }
295  trace2(DBG_VGL, "\nopaque: ");
296  solidsOpaque(zsel, psu, WObject::RENDER_NORMAL);
297  trace2(DBG_VGL, "\ntranslucid: ");
298  solidsTranslucid(zsel, psu, WObject::RENDER_NORMAL);
299 
301  //
302  trace2(DBG_VGL, "\n--- HIGH");
303  trace2(DBG_VGL, "\nspeci: ");
304  for (uint32_t i=1; i < objectsnumber; i++) { // do specific rendering by objects themself
305  specific(i, WObject::RENDER_HIGH);
306  }
307  trace2(DBG_VGL, "\nspeci: ");
308  solidsOpaque(zsel, psu, WObject::RENDER_HIGH);
309  trace2(DBG_VGL, "\ntranslucid: ");
310  solidsTranslucid(zsel, psu, WObject::RENDER_HIGH);
311  trace2(DBG_VGL, "\n");
312 
313  // Renders flashy edges and ray impact
314  for (list<Solid*>::iterator ps = solidList.begin(); ps != solidList.end() ; ps++) {
315  if ((*ps) && ((*ps)->isflashy || (*ps)->isflary)) {
316  //trace(DBG_FORCE, "render flashy %s", (*ps)->object()->getInstance());
317  (*ps)->display3D(Solid::PLAIN, Solid::FLASHRAY);
318  }
319  }
320 
321  // if buffer selection mode is on, don't do anything else: end
322  if (zsel) return;
323 
324  // Renders local user last
325  if (localuser)
326  (*psu)->display3D(Solid::PLAIN, Solid::LOCALUSER);
327 }

445 /*
446  * 3D Selection with picking method.
447  */
448 uint16_t Render::getBufferSelection(GLint x, GLint y, GLint z)
449 {
450  //static GLuint selectbuf[4 * 256];
451 
452  // set selection mode
453  memset(selectbuf, 0, sizeof(selectbuf));
454  glSelectBuffer(sizeof(selectbuf), selectbuf);
455  glRenderMode(GL_SELECT);
456  glInitNames();
457  glPushName(0);
458 
459  GLint vp[4];
460  glGetIntegerv(GL_VIEWPORT, vp);
461  GLint width = vp[2];
462  GLint height = vp[3];
463  GLfloat top = cam_user.near * tan(cam_user.fovy * M_PI_180);
464  GLfloat bottom = -top;
465  GLfloat ratio = (GLfloat) width / (GLfloat) height;
466  GLfloat right = top * ratio;
467  GLfloat left = -right;
468 
469  glMatrixMode(GL_PROJECTION);
470  glPushMatrix();
471  glLoadIdentity();
472  glFrustum(left + (x-0.1) * (right-left) / (width-1),
473  left + (x+0.1) * (right-left) / (width-1),
474  top - (y+0.1) * (top-bottom) / (height-1),
475  top - (y-0.1) * (top-bottom) / (height-1),
476  cam_user.near, cam_user.far);
477  glTranslatef(0, 0, -cam_user.near);
478 
479  // eye position
480  cameraPosition();
481 
482  // redraw the solids into the selection buffer
483  solids(true);
484 
485  // we put the normal mode back
486  glMatrixMode(GL_PROJECTION);
487  glPopMatrix();
488 
489  // scan the selection buffer, find the nearest hit
490  GLint hits = glRenderMode(GL_RENDER);
491  GLuint** hitlist = new GLuint*[hits];
492  GLuint *psel = selectbuf;
493  for (int i=0; i < hits; i++) {
494  if (::g.pref.dbgtrace) error("hit=%d/%d numobject=%d", i, hits, psel[3]);
495  hitlist[i] = psel;
496  psel += 3 + psel[0]; // next hit
497  }
498 
499  uint16_t objectselected = 0;
500 
501  if (hits > 0) {
502  qsort((void *)hitlist, hits, sizeof(GLuint *), compareHit);
503  int n = z % hits;
504  objectselected = hitlist[n][3];
505  }
506  if (hitlist) delete[] hitlist;
507  return objectselected;
508 }

510 /* Qsort function for elements of the selection buffer. */
511 int Render::compareHit(const void *t1, const void *t2)
512 {
513  GLuint s1 = (*((GLuint **)t1))[1]; // please don't touch that
514  GLuint s2 = (*((GLuint **)t2))[1];
515 
516  if (s1 > s2) return 1;
517  if (s1 < s2) return -1;
518  return 0;

574 /*
575  * Converts (wx,wy) window coordinates into a vector (dir) representing
576  * the direction of the line from the user's eyes to the point he clicked on.
577  * Vector dir is returned.
578  * Static method.
579  */
580 void Render::clickDirection(GLint wx, GLint wy, V3 *dir)
581 {
582  GLint vp[4];
583  //dax glGetIntegerv(GL_VIEWPORT, vp);
584  ::g.gui.getScene()->getCoords(vp[0], vp[1], vp[2], vp[3]); // get viewport
585 
586 #if 0 //dax not used
587  //glPopMatrix();
588  //glPushMatrix();
589  GLdouble x1, y1, z1, x2, y2, z2, mview[16], mproj[16];
590  glGetDoublev(GL_MODELVIEW_MATRIX, mview);
591  glGetDoublev(GL_PROJECTION_MATRIX, mproj);
592  gluUnProject((GLdouble)wx, (GLdouble)(vp[3]-wy), (GLdouble)0., mview, mproj, vp, &x1, &y1, &z1);
593  gluUnProject((GLdouble)wx, (GLdouble)(vp[3]-wy), (GLdouble)1., mview, mproj, vp, &x2, &y2, &z2);
594  //error("wx=%d wy=%d o1=%.2f %.2f %.2f o2=%.2f %.2f %.2f", wx,wy,x1,y1,z1,x2,y2,z2);
595  dir->v[0] = (GLfloat) (x1 - x2);
596  dir->v[1] = (GLfloat) (y1 - y2);
597  dir->v[2] = (GLfloat) (z1 - z2);
598 #else
599  GLfloat ex = localuser->pos.x + User::NEAR;
600  GLfloat ey = localuser->pos.y;
601  GLfloat ez = localuser->pos.z + localuser->height/2 - 0.10;
602  GLfloat tx = (GLfloat) User::FAR;
603  GLfloat ty = vp[2]/2 - (GLfloat) wx;

662 /* Get the list of drawed object on the user's screen. */
663 WObject** Render::getDrawedObjects(int *nbhit)
664 {
665  // set selection mode
666  glSelectBuffer(sizeof(selectbuf), selectbuf);
667  glRenderMode(GL_SELECT);
668  glInitNames();
669  glPushName(0);
670 
671  GLint vp[4];
672  glGetIntegerv(GL_VIEWPORT, vp);
673 
674  GLint width = vp[2];
675  GLint height = vp[3];
676  GLfloat top = cam_user.near * tan(cam_user.fovy * M_PI_180);
677  GLfloat bottom = -top;
678  GLfloat ratio = (GLfloat) width / (GLfloat) height;
679  GLfloat right = top * ratio;
680  GLfloat left = -right;
681 
682  glMatrixMode(GL_PROJECTION);
683  glPushMatrix();
684  glLoadIdentity();
685  glFrustum(left + ((width/2.0)-(width/2.0)) * (right-left) / (width-1),
686  left + ((width/2.0)+(width/2.0)) * (right-left) / (width-1),
687  top - ((height/2.0)+(height/2.0)) * (top-bottom) / (height-1),
688  top - ((height/2.0)-(height/2.0)) * (top-bottom) / (height-1),
689  cam_user.near, cam_user.far);
690  glTranslatef(0, 0, -cam_user.near);
691 
692  // eye position
693  cameraPosition();
694 
695  // redraw the solids into the selection buffer
696  solids(true);
697 
698  // we put the normal mode back
699  glMatrixMode(GL_PROJECTION);
700  glPopMatrix();
701 
702  // scan the selection buffer
703  GLint hits = glRenderMode(GL_RENDER);
704  if (hits < 0) return NULL;
705 
706  GLuint** hitlist = new GLuint*[hits];
707  GLuint *psel = selectbuf;
708  for (int i=0; i < hits; i++) {
709  hitlist[i] = psel;
710  psel += 3 + psel[0]; // next hit
711  }
712  qsort((void *) hitlist, hits, sizeof(GLuint *), compareHit);
713 
714  int count = 0;
715  if (! strcasecmp((WObject::byNum((uint16_t) hitlist[0][3]))->typeName(), "User")) {
716  hitlist[0] = NULL;
717  count++;
718  }
719 
720  WObject** selectlist = new WObject*[hits-count];
721 
722  for (int i=0; i < hits; i++) {
723  if (hitlist[i]) selectlist[i] = WObject::byNum(hitlist[i][3]);
724  }
725  if (hitlist) delete[] hitlist;
726  *nbhit = hits - count;
727  return selectlist;
728 }
