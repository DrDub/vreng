72 static uint8_t *convert_format(uint8_t *data, uint8_t srccomp, uint8_t dstcomp, uint16_t x, uint16_t y)
73 {
74  if (dstcomp == srccomp) return data;
75 
76  uint8_t *good = new uint8_t[dstcomp * x * y];
77  if (good == NULL) {
78  delete[] data;
79  error("PSD out of memory");
80  return NULL;
81  }
82 
83  //error("srccomp=%d dstcomp=%d x=%d y=%d", srccomp,dstcomp,x,y);
84  for (int j=0; j < y; j++) {
85  uint8_t *src = data + j * x * srccomp;
86  uint8_t *dst = good + j * x * dstcomp;
87  #define COMBO(a,b) ((a)*8+(b))
88  #define CASE(a,b) case COMBO(a,b): for (int i=x-1; i >= 0; --i, src += a, dst += b)
89  // convert source image with srccomp channel to one with dstcomp channel;
90  // avoid switch per pixel, so use switch per scanline and massive macros
91  switch (COMBO(srccomp, dstcomp)) {
92  CASE(1,2) dst[0]=src[0], dst[1]=255;
93  break;
94  CASE(1,3) dst[0]=dst[1]=dst[2]=src[0];
95  break;
96  CASE(1,4) dst[0]=dst[1]=dst[2]=src[0], dst[3]=255;
97  break;
98  CASE(2,1) dst[0]=src[0];
99  break;
100  CASE(2,3) dst[0]=dst[1]=dst[2]=src[0];
101  break;
102  CASE(2,4) dst[0]=dst[1]=dst[2]=src[0], dst[3]=src[1];
103  break;
104  CASE(3,4) dst[0]=src[0],dst[1]=src[1],dst[2]=src[2],dst[3]=255;
105  break;
106  CASE(3,1) dst[0]=compute_y(src[0],src[1],src[2]);
107  break;
108  CASE(3,2) dst[0]=compute_y(src[0],src[1],src[2]), dst[1]=255;
109  break;
110  CASE(4,1) dst[0]=compute_y(src[0],src[1],src[2]);
111  break;
112  CASE(4,2) dst[0]=compute_y(src[0],src[1],src[2]), dst[1]=src[3];
113  break;
114  CASE(4,3) dst[0]=src[0],dst[1]=src[1],dst[2]=src[2];
115  break;
116  }
117  #undef CASE
118  }
119  delete[] data;
120  return good;
121 }

123 static uint8_t *psd_load(stbi *s, uint16_t *x, uint16_t *y, uint8_t *srccomp, uint8_t dstcomp)
124 {
125  uint16_t nbpixels;
126  uint8_t nbchannels;
127  uint8_t compression, channel;
128  uint16_t count, len;
129  uint16_t w, h;
130  uint8_t *data;
131 
132  // Check identifier
133  if (get32(s) != 0x38425053) { // "8BPS"
134  error("corrupt PSD image"); return NULL;
135  }
136  // Check file type version.
137  if (get16(s) != 1) { error("unsupported version of PSD image"); return NULL; }
138  // Skip 6 reserved bytes.
139  skip(s, 6);
140  // Read the number of channels (R, G, B, A, etc).
141  nbchannels = get16(s);
142  if (nbchannels > 16) { error("unsupported number of channels in PSD image"); return NULL; }
143  // Read the rows and columns of the image.
144  h = get32(s);
145  w = get32(s);
146  // Make sure the depth is 8 bits.
147  if (get16(s) != 8) { error("PSD bit depth is not 8 bit"); return NULL; }
148  // Make sure the color mode is RGB.
149  // 0: Bitmap
150  // 1: Grayscale
151  // 2: Indexed color
152  // 3: RGB color
153  // 4: CMYK color
154  // 7: Multichannel
155  // 8: Duotone
156  // 9: Lab color
157  if (get16(s) != 3) { error("PSD is not in RGB color format"); return NULL; }
158  skip(s, get32(s)); // Skip the Mode Data. (It's the palette for indexed color)
159  skip(s, get32(s)); // Skip the image resources. (resolution, pen tool paths, etc)
160  skip(s, get32(s)); // Skip the reserved data.
161  // Find out if the data is compressed.
162  // Known values: 0: no compression 1: RLE compressed
163  compression = get16(s);
164  if (compression > 1) { error("PSD has an unknown compression format"); return NULL; }
165 
166  // Create the destination image.
167  data = new uint8_t[4 * w*h];
168  if (!data) { error("PSD out of memory"); return NULL; }
169  nbpixels = w * h;
170 
171  // Finally, the image data.
172  if (compression) {
173  // RLE as used by .PSD and .TIFF
174  // Loop until you get the number of unpacked bytes you are expecting:
175  // Read the next source byte into n.
176  // If n is between 0 and 127 inclusive, copy the next n+1 bytes literally.
177  // Else if n is between -127 and -1 inclusive, copy the next byte -n+1 times.
178  // Else if n is 128, noop.
179  // Endloop
180 
181  // The RLE-compressed data is preceeded by a 2-byte data count for each row in the data,
182  // which we're going to just skip.
183  skip(s, h * nbchannels * 2);
184 
185  // Read the RLE data by channel.
186  for (channel=0; channel < 4; channel++) {
187  uint8_t *p = data + channel;
188  if (channel >= nbchannels) {
189  // Fill this channel with default data.
190  for (int i=0; i < nbpixels; i++) *p = (channel == 3 ? 255 : 0), p += 4;
191  } else {
192  // Read the RLE data.
193  count = 0;
194  while (count < nbpixels) {
195  len = get8(s);
196  if (len == 128) {
197  ; // No-op.
198  } else if (len < 128) {
199  // Copy next len+1 bytes literally.
200  len++;
201  count += len;
202  while (len) {
203  *p = get8(s);
204  p += 4;
205  len--;
206  }
207  } else if (len > 128) {
208  uint32_t val;
209  // Next -len+1 bytes in the dest are replicated from next source byte.
210  // (Interpret len as a negative 8-bit int.)
211  len ^= 0x0FF;
212  len += 2;
213  val = get8(s);
214  count += len;
215  while (len) {
216  *p = val;
217  p += 4;
218  len--;
219  }
220  }
221  }
222  }
223  }
224  } else {
225  // We're at the raw image data. It's each channel in order (Red, Green, Blue, Alpha, ...)
226  // where each channel consists of an 8-bit value for each pixel in the image.
227 
228  // Read the data by channel.
229  for (channel=0; channel < 4; channel++) {
230  uint8_t *p = data + channel;
231  if (channel > nbchannels) {
232  // Fill this channel with default data.
233  for (int i=0; i < nbpixels; i++) *p = (channel == 3) ? 255 : 0, p += 4;
234  } else {
235  // Read the data.
236  count = 0;
237  for (int i=0; i < nbpixels; i++) {
238  *p = get8(s);
239  p += 4;
240  }
241  }
242  }
243  }
244 
245  if (dstcomp && dstcomp != 4) {
246  data = convert_format(data, 4, dstcomp, w, h);
247  if (data == NULL) return NULL; // convert_format frees input on failure
248  }
249  if (srccomp) *srccomp = nbchannels;
250  *y = h;
251  *x = w;
252 
253  return data;
254 }
