227 {
228  FILE* menufp = null;
229  UMenu* dynMenu = null;
230  char attr[100], val[100];

233  fscanf(menufp, "%s %s", attr, val);
234  //cerr << "dynmenu: " << attr << " " <<val << endl;

237  dynMenu = new UMenu();

246  fscanf(menufp, "%s %s", attr, val);

589 void Widgets::processKey(long keysym, int keychar, bool press)
590 {
591  // converts the X keysym into a Vreng change key (vrkey) and returns
592  // a keymask which is an hexa value for marking released keys in the KRmask
593  int vrkey;
594  long keymask = convertKey(keysym, keychar, vrkey);
595  if (keymask == 0) return; // return if null (undefined or not a vrkey)
596 
597  if (postponedKRcount < 0) {
598  fprintf(stderr, "!negative KRcount => reset\n"); // should never happen!
599  postponedKRmask = 0;
600  postponedKRcount = 0;
601  }
602 
603  if (press) {
604  if (postponedKRmask & keymask) {
605  // le Press annule le Release correspondant dans le KRmask
606  postponedKRmask &= ~keymask; // remove keymask from KRmask
607  postponedKRcount--; // une touche en moins dans KRlist
608  }
609  else { // traitement normal d'un Press
610  //fprintf(stderr, "KPress change or activate Key( %d ) \n", vrkey);
611  if (vrkey >= MAXKEYS || vrkey < 0)
612  return;
613 
614  struct timeval time;
615  gettimeofday(&time, NULL);
616  changeKey(vrkey, press, time.tv_sec, time.tv_usec);
617  }
618  }
619  else { // release
620  // too many keys stored ==> flush the KRlist
621  if (postponedKRcount > KRKey::KRBUF_MAX) {
622  fprintf(stderr, "too many keys stored => flush: KRmask=%lx\n", postponedKRmask);
623  flushPostponedKRs();
624  }
625  else { // normal case: postpone until we come back to the mainLoop
626  // so that this Key Release event can possibly be annulate
627  // if there is a subsequent (and corresponding) Key Press event

648 static long convertKey(long keycode, int keychar, int& vrkey)
649 {
650  long keymask = 0;
651  vrkey = 0;
652 
653  if (keycode == UKey::Up) { keymask = 1<<0; vrkey = KEY_AV; } // move forward
654  else if (keycode == UKey::Down) { keymask = 1<<1; vrkey = KEY_AR; } // move backward
655  else if (keycode == UKey::Left) { keymask = 1<<2; vrkey = KEY_GA; } // turn left
656  else if (keycode == UKey::Right) { keymask = 1<<3; vrkey = KEY_DR; } // turn right
657  else if (keycode == UKey::PageUp) { keymask = 1<<11; vrkey = KEY_JU; } // move up
658  else if (keycode == UKey::PageDown) { keymask = 1<<12; vrkey = KEY_JD; } // move down
659  else if (keycode == UKey::Insert) { keymask = 1<<6; vrkey = KEY_MT; } // roll up
660  else if (keycode == UKey::Delete) { keymask = 1<<7; vrkey = KEY_DE; } // roll down
661  else if (keycode == UKey::Home) { keymask = 1<<8; vrkey = KEY_HZ; } // stand up
662  else if (keycode == UKey::End) { keymask = 1<<13; vrkey = KEY_VI; } // accelerator
663  else if (keycode == UKey::BackSpace) {
664  Widgets::callAction(UserAction::UA_ASPEEDLESS); return 0; // decrease aspeed
665  }
666  else if (keycode == UKey::Tab) {
667  Widgets::callAction(UserAction::UA_ASPEEDMORE); return 0; // increase aspeed
668  }
669  else {
670  switch (keychar) {
671  case '<': keymask = 1<<4; vrkey = KEY_SG; break; // left translation
672  case '>': keymask = 1<<5; vrkey = KEY_SD; break; // right translation
673  case 'l': keymask = 1<<9; vrkey = KEY_TL; break; // tilt left
674  case 'r': keymask = 1<<10; vrkey = KEY_TR; break; // tilt right
675  case 'u': keymask = 1<<11 | 1<<13; vrkey = KEY_JU; break; // up translation
676  case ' ': keymask = 1<<13; vrkey = KEY_VI; break; // accelerator
677  case '=': Widgets::callAction(UserAction::UA_FOVYDEF); return 0; // original fovy
678  case '-': Widgets::callAction(UserAction::UA_FOVYLESS); return 0; // decrease fovy
679  case '+': Widgets::callAction(UserAction::UA_FOVYMORE); return 0; // increase fovy
680  case '.': Widgets::callAction(UserAction::UA_LSPEEDDEF); return 0; // original lspeed
681  case 's': Widgets::callAction(UserAction::UA_LSPEEDLESS); return 0; // decrease lspeed
682  case 'f': Widgets::callAction(UserAction::UA_LSPEEDMORE); return 0; // increase lspeed
683  case ',': Widgets::callAction(UserAction::UA_ASPEEDDEF); return 0; // original aspeed
684  case 'b': Widgets::callAction(UserAction::UA_BULLET); return 0; // launche bullet
685  case 'd': Widgets::callAction(UserAction::UA_DART); return 0; // launche dart
686  case 'v': Widgets::callAction(UserAction::UA_SWITCHVIEW); return 0; // switch view
687  case 'x': Widgets::callAction(UserAction::UA_TPVIEWROTL); return 0;
688  case 'c': Widgets::callAction(UserAction::UA_TPVIEWROTR); return 0;
689  case 'q': Widgets::callAction(UserAction::UA_TPVIEWROTU); return 0;
690  case 'w': Widgets::callAction(UserAction::UA_TPVIEWROTD); return 0;
691  case 'p': Widgets::callAction(UserAction::UA_TPVIEWROTN); return 0;
692  case 'm': Widgets::callAction(UserAction::UA_TPVIEWROTF); return 0;
693  case 'D': Widgets::callAction(UserAction::UA_PITCHMORE); return 0; // increase pitch
694  case 'U': Widgets::callAction(UserAction::UA_PITCHLESS); return 0; // decrease pitch
695  case 'R': Widgets::callAction(UserAction::UA_ROLLMORE); return 0; // increase roll
696  case 'L': Widgets::callAction(UserAction::UA_ROLLLESS); return 0; // decrease roll
697  case '^': Widgets::callAction(UserAction::UA_FLYAWAY); return 0; // flyaway
698  case '$': Widgets::callAction(UserAction::UA_TOLAND); return 0; // toland
699  default: return 0; break; // == 0 => undefined key
700  }
701  }
702  return keymask;
703 }
