82 void Txf::httpReader(void *_txf, Http *http)
83 {
84  Txf *txf = (Txf *) _txf;
85  if (! txf) return;
86 
87  FILE *f = Cache::openCache(txf->getUrl(), http);
88  if (! f) { error("can't open %s", txf->getUrl()); return; }
89 
90  txf->texfont = new struct TexFont[1];
91  txf->texfont->tgi = NULL;
92  txf->texfont->tgvi = NULL;
93  txf->texfont->lut = NULL;
94  txf->texfont->teximage = NULL;
95 
96  int format, stride, width, height, max_glyph, min_glyph, endianness, swap;
97  GLfloat w, h, xstep, ystep;
98  char tmp;
99 
100  char fileid[4];
101  int got = fread((char *) fileid, 1, 4, f);
102  if (got != 4 || strncmp(fileid, "\377txf", 4)) {
103  lastError = "not a texture font file."; goto error;
104  }
105 
106  got = fread((char *) &endianness, sizeof(int), 1, f);
107  if (got == 1 && endianness == 0x12345678) swap = 0;
108  else if (got == 1 && endianness == 0x78563412) swap = 1;
109  else { lastError = "not a texture font file."; goto error; }
110 
111 #define EXPECT(v) if (got != (int)v) {lastError="premature end of file"; goto error;}
112  got = fread((char *) &format, sizeof(int), 1, f); EXPECT(1);
113  got = fread((char *) &txf->texfont->tex_width, sizeof(int), 1, f); EXPECT(1);
114  got = fread((char *) &txf->texfont->tex_height, sizeof(int), 1, f); EXPECT(1);
115  got = fread((char *) &txf->texfont->max_ascent, sizeof(int), 1, f); EXPECT(1);
116  got = fread((char *) &txf->texfont->max_descent, sizeof(int), 1, f); EXPECT(1);
117  got = fread((char *) &txf->texfont->num_glyphs, sizeof(int), 1, f); EXPECT(1);
118 
119  if (swap) {
120  SWAPL(&format, tmp);
121  SWAPL(&txf->texfont->tex_width, tmp);
122  SWAPL(&txf->texfont->tex_height, tmp);
123  SWAPL(&txf->texfont->max_ascent, tmp);
124  SWAPL(&txf->texfont->max_descent, tmp);
125  SWAPL(&txf->texfont->num_glyphs, tmp);
126  }
127  trace(DBG_IMG, "txfReader: fmt=%d w=%d h=%d ascent=%d descent=%d glyphs=%d", format, txf->texfont->tex_width, txf->texfont->tex_height, txf->texfont->max_ascent, txf->texfont->max_descent, txf->texfont->num_glyphs);
128 
129  txf->texfont->tgi = new TexGlyphInfo[txf->texfont->num_glyphs];
130  if (txf->texfont->tgi == NULL) { lastError = "out of memory."; goto error; }
131  got = fread((char *) txf->texfont->tgi, sizeof(TexGlyphInfo), txf->texfont->num_glyphs, f);
132  EXPECT(txf->texfont->num_glyphs);
133 
134  if (swap) {
135  for (int i=0; i < txf->texfont->num_glyphs; i++) {
136  SWAPS(&txf->texfont->tgi[i].c, tmp);
137  SWAPS(&txf->texfont->tgi[i].x, tmp);
138  SWAPS(&txf->texfont->tgi[i].y, tmp);
139  }
140  }
141  txf->texfont->tgvi = new TexGlyphVertexInfo[txf->texfont->num_glyphs];
142  if (txf->texfont->tgvi == NULL) { lastError = "out of memory."; goto error; }
143 
144  w = txf->texfont->tex_width;
145  h = txf->texfont->tex_height;
146  xstep = 0.5 / w;
147  ystep = 0.5 / h;
148  for (int i=0; i < txf->texfont->num_glyphs; i++) {
149  TexGlyphInfo *tgi = &txf->texfont->tgi[i];
150  txf->texfont->tgvi[i].t0[0] = tgi->x / w + xstep;
151  txf->texfont->tgvi[i].t0[1] = tgi->y / h + ystep;
152  txf->texfont->tgvi[i].v0[0] = tgi->xoffset;
153  txf->texfont->tgvi[i].v0[1] = tgi->yoffset;
154  txf->texfont->tgvi[i].t1[0] = (tgi->x + tgi->width) / w + xstep;
155  txf->texfont->tgvi[i].t1[1] = tgi->y / h + ystep;
156  txf->texfont->tgvi[i].v1[0] = tgi->xoffset + tgi->width;
157  txf->texfont->tgvi[i].v1[1] = tgi->yoffset;
158  txf->texfont->tgvi[i].t2[0] = (tgi->x + tgi->width) / w + xstep;
159  txf->texfont->tgvi[i].t2[1] = (tgi->y + tgi->height) / h + ystep;
160  txf->texfont->tgvi[i].v2[0] = tgi->xoffset + tgi->width;
161  txf->texfont->tgvi[i].v2[1] = tgi->yoffset + tgi->height;
162  txf->texfont->tgvi[i].t3[0] = tgi->x / w + xstep;
163  txf->texfont->tgvi[i].t3[1] = (tgi->y + tgi->height) / h + ystep;
164  txf->texfont->tgvi[i].v3[0] = tgi->xoffset;
165  txf->texfont->tgvi[i].v3[1] = tgi->yoffset + tgi->height;
166  txf->texfont->tgvi[i].advance = tgi->advance;
167  }
168 
169  min_glyph = txf->texfont->tgi[0].c;
170  max_glyph = txf->texfont->tgi[0].c;
171  for (int i=1; i < txf->texfont->num_glyphs; i++) {
172  if (txf->texfont->tgi[i].c < min_glyph) min_glyph = txf->texfont->tgi[i].c;
173  if (txf->texfont->tgi[i].c > max_glyph) max_glyph = txf->texfont->tgi[i].c;
174  }
175  txf->texfont->min_glyph = min_glyph;
176  txf->texfont->range = max_glyph - min_glyph + 1;
177 
178  txf->texfont->lut = new TexGlyphVertexInfo*[txf->texfont->range];
179  if (txf->texfont->lut == NULL) { lastError = "out of memory."; goto error; }
180  for (int i=0; i < txf->texfont->num_glyphs; i++)
181  txf->texfont->lut[txf->texfont->tgi[i].c - txf->texfont->min_glyph] = &txf->texfont->tgvi[i];
182 
183  uint8_t *texbitmap;
184 
185  switch (format) {
186  case TXF_FORMAT_BYTE:
187  txf->texfont->teximage = new uint8_t[txf->texfont->tex_width * txf->texfont->tex_height];
188  if (txf->texfont->teximage == NULL) { lastError = "out of memory."; goto error; }
189  got = fread((char *) txf->texfont->teximage, 1, txf->texfont->tex_width*txf->texfont->tex_height, f);
190  EXPECT(txf->texfont->tex_width * txf->texfont->tex_height);
191  break;
192  case TXF_FORMAT_BITMAP:
193  width = txf->texfont->tex_width;
194  height = txf->texfont->tex_height;
195  stride = (width + 7) >> 3;
196  if ((texbitmap = new uint8_t[stride * height]) == NULL) {
197  lastError = "out of memory."; goto error;
198  }
199  got = fread((char *) texbitmap, 1, stride * height, f);
200  EXPECT(stride * height);
201  if ((txf->texfont->teximage = new uint8_t[width * height]) == NULL) {
202  lastError = "out of memory."; goto error;
203  }
204  for (int i=0; i < height; i++) {
205  for (int j=0; j < width; j++) {
206  if (texbitmap[i * stride + (j >> 3)] & (1 << (j & 7)))
207  txf->texfont->teximage[i * width + j] = 255;
208  }
209  }
210  if (texbitmap) delete[] texbitmap;
211  texbitmap = NULL;
212  break;
213  }
214  File::closeFile(f);
215  return;
216 
217 error:
218  error("txf err: %s", lastError);
219  if (txf->texfont) {
220  delete[] txf->texfont->tgi;
221  delete[] txf->texfont->tgvi;
222  delete[] txf->texfont->lut;
223  delete[] txf->texfont->teximage;
224  delete[] txf->texfont; txf->texfont = NULL;
225  }
226  File::closeFile(f);
227  return;
228 }

230 TexGlyphVertexInfo * Txf::getTCVI(int c)
231 {
232  TexGlyphVertexInfo *tgvi;
233 
234  /* Automatically substitute uppercase letters with lowercase if not
235  uppercase available (and vice versa). */
236 lastchance:
237  if ((c >= texfont->min_glyph) && (c < texfont->min_glyph + texfont->range)) {
238  if ((tgvi = texfont->lut[c - texfont->min_glyph]) != 0) return tgvi;
239  if (islower(c)) {
240  c = toupper(c);
241  if ((c >= texfont->min_glyph) && (c < texfont->min_glyph + texfont->range))
242  return texfont->lut[c - texfont->min_glyph];
243  }
244  if (isupper(c)) {
245  c = tolower(c);
246  if ((c >= texfont->min_glyph) && (c < texfont->min_glyph + texfont->range))
247  return texfont->lut[c - texfont->min_glyph];
248  }
249  }
250  error("texfont: unavailable font character \"%c\" (%d)", isprint(c) ? c : ' ', c);
251  c = ' ';
252  goto lastchance;
253  /* NOTREACHED */
254 }

256 GLuint Txf::establishTexture(GLuint texobj, GLboolean setupMipmaps)
257 {
258  if (texfont) {
259  if (texfont->texobj == 0) {
260  if (texobj == 0) glGenTextures(1, &texfont->texobj);
261  else texfont->texobj = texobj;
262  }
263 
264  glBindTexture(GL_TEXTURE_2D, texfont->texobj);
265 
266  /* Use GL_INTENSITY4 as internal texture format since we want to use as
267  little texture memory as possible. */
268  if (setupMipmaps) {
269  gluBuild2DMipmaps(GL_TEXTURE_2D, GL_INTENSITY4,
270  texfont->tex_width, texfont->tex_height,
271  GL_LUMINANCE, GL_UNSIGNED_BYTE, texfont->teximage);
272  }
273  else {
274  glTexImage2D(GL_TEXTURE_2D, 0, GL_INTENSITY4,
275  texfont->tex_width, texfont->tex_height, 0,
276  GL_LUMINANCE, GL_UNSIGNED_BYTE, texfont->teximage);
277  }
278  return texfont->texobj;
279  }
280  else return (GLuint) 0;
281 }
