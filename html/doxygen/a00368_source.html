46 void Clip::parser(char *l)
47 {
48  l = tokenize(l);
49  begin_while_parse(l) {
50  l = parse()->parseAttributes(l, this);
51  if (!l) break;

68 
69  purl = strrchr(names.url, '.');
70  purl++;
71  if (!stringcmp(purl, "mpg") || !stringcmp(purl, "mpeg")) fmt = FMT_MPEG;
72  else if (!stringcmp(purl, "video")) {
73  fmt = FMT_RTPAV;

102 void Clip::startRtp()
103 {
104  if (state == RTP_ACTIVE) return; // busy
105 
106  char *pfile, *end;
107  pfile = strrchr(names.url, '/');
108  pfile++;
109  end = strchr(pfile, '.');
110  *end = '\0';
111 
112  char group[GROUP_LEN];
113  Channel::getGroup(World::current()->getChan(), group);
114 
115  char cmd[URL_LEN];
116  sprintf(cmd, "%s%s?%s&%s&%d&%d", RTP_URL, PLAY_CGI,
117  pfile,
118  group,
119  Channel::getPort(World::current()->getChan()) + 2,
120  Channel::currentTtl());
121  postCgi(cmd);
122  state = RTP_ACTIVE;
123 }
