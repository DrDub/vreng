47 void Button::parser(char *l)
48 {
49  char str_action0[ACTION_LEN];
50  char str_action1[ACTION_LEN];
51  memset(str_action0, 0, sizeof(str_action0));
52  memset(str_action1, 0, sizeof(str_action1));
53 
54  defaults();
55  l = tokenize(l);
56  begin_while_parse(l) {
57  l = parse()->parseAttributes(l, this);
58  if (!l) break;
59  if (!stringcmp(l, "use")) {
60  l = parse()->parseQuotedString(l, use_names, "use"); // refers object to use
61  }
62  else if (!stringcmp(l, "state")) l = parse()->parseUInt8(l, &pos.state, "state");
63  else if (!stringcmp(l, "method0")) l = parse()->parseString(l, str_action0, "method0");
64  else if (!stringcmp(l, "method1")) l = parse()->parseString(l, str_action1, "method1");
65  }
66  end_while_parse(l);
67 
68  char *p, *pname = new char[OBJNAME_LEN * USENAME_MAX];
69  strcpy(pname, use_names);
70  p = strchr(pname, ' ');
71  if (p) *p = '\0';
72 
73  WObject *puse = getObjectByName(pname);
74  num_action0 = indexAction(puse->type, str_action0);
75  if (*str_action1) num_action1 = indexAction(puse->type, str_action1);
76  if (num_action1 == 0 || num_action1 == 255) num_action1 = num_action0;
77  delete[] pname;
78 }

104 void Button::commut(Button *button, void *d, time_t s, time_t us)
105 {
106  int idxaction = (button->pos.state == 0) ? button->num_action0 : button->num_action1;
107  button->pos.state = 1 - button->pos.state;
108 
109  WObject *psel = NULL;
110 
111  char *puse = strtok(button->use_names, " \t");
112  while (puse) {
113  if ((psel = button->getObjectByName(puse)) != NULL) {
114  if ((idxaction <= OBJECTSNUMBER) && isAction(psel->type, idxaction))
115  doAction(psel->type, idxaction, psel, d, s, us);
116  else
117  warning("commut: object named %s has an unknown action name %d", puse, idxaction);
118  }
119  else {
120  warning("button: no object named %s", puse);
121  }
122  puse = strtok(NULL, " \t"); // next name
123  }
124 }
