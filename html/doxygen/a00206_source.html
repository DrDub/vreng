106 void Gui::addChannelSources(int channel, int table[], int table_count)
107 {
108  if (channel >= int(channel_sources.size())) {
109  channel_sources.resize(channel+1, null);
110  }
111  channel_sources[channel] = new ChannelSources(table_count);
112  for (int k=0; k < table_count; k++) {
113  USource* s = new USource(table[k]);
114  s->onAction(ucall(table[k], ::NetIncoming));
115  channel_sources[channel]->push_back(s);
116  }
117 }

119 void Gui::removeChannelSources(int channel)
120 {
121  if (channel >= int(channel_sources.size())) return;
122 
123  ChannelSources* css = channel_sources[channel];
124  if (css == null) return;
125 
126  for (unsigned int k=0; k < css->size(); k++) delete (*css)[k];
127  delete css;
128  channel_sources[channel] = null;
129 }

152 void Gui::removeCart(WObject *cart, int action)
153 {
154  if (!cart) { error("Gui::removeCart: NULL item!"); return; }
155 
156  if (cart->isGui()) {
157  widgets->setInfobar(null);
158  widgets->carts.remove(*cart->getGui()); // remove item's box
159  cart->resetGui();
160  }
161  if (! localuser) return;
162 
163  switch (action) {
164  case Cart::LEAVE:
165  notice("%s left in %s world", NN(cart->getInstance()),NN(World::current()->getName()));
166  localuser->cart->leaveIntoWorld(cart);
167  break;
168  case Cart::REMOVE:
169  notice("%s removed from cart", NN(cart->getInstance()));
170  localuser->cart->removeFromCart(cart);
171  break;
172  }
173 }

226 void Gui::gotoWorld(const UStr& url_or_name)
227 {
228  if (url_or_name.empty()) return;
229 
230  const char* urlorname = url_or_name.c_str();
231  char urlvre[URL_LEN], chanstr[CHAN_LEN];
232  Vac *vac = Vac::current();
233 
234  strcpy(chanstr, DEF_VRE_CHANNEL);
235  if (strchr(urlorname, '/')) { // url or path
236  strcpy(urlvre, urlorname);
237  if (! Cache::check(urlvre)) return; // bad url
238  if (! vac->resolveWorldUrl(urlvre, chanstr)) {
239  if (strcmp(chanstr, DEF_VRE_CHANNEL)) return; // url not found
240  }
241  }
242  else { // worldname
243  if (! vac->getUrlAndChannel(urlorname, urlvre, chanstr)) return; // world not found
244  }
245  notice("goto %s at %s", urlvre, chanstr);
246 
247  World::current()->quit();
248  delete Channel::current(); // delete old Channel
249  World::enter(urlvre, chanstr, true);
250  World::current()->setChanAndJoin(chanstr); // join new channel
251 
252  if (audioactive) Audio::start(chanstr);
253 }
