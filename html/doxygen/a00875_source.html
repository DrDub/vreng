32 /* When something is available on file-descriptor "fd" */
33 void NetIncoming(int fd)
34 {
35  while (1) {
36  fd_set set;
37  struct timeval timeout;
38 
39  FD_ZERO(&set);
40  FD_SET(fd, &set);
41  timeout.tv_sec = timeout.tv_usec = 0;
42  if (select(FD_SETSIZE, &set, NULL, NULL, &timeout) == 0)
43  break;
44 
45  /*
46  * read the packet
47  */
48  struct sockaddr_in from; // sender unicast address
49  int r;
50 
51  Payload *pp = new Payload(); // create the payload to get the data
52 
53  if ((r = pp->recvPayload(fd, &from)) <= 0) {
54  //error("NetIncoming: from %lx", ntohl(from.sin_addr.s_addr));
55  delete pp;
56  return; // read error <0 or ignore == 0
57  }
58 
59  /*
60  * check weither this packet belongs to VREP (VREng Protocol)
61  */
62  uint8_t cmd;
63  if (pp->len > 0 && pp->getPayload("c", &cmd) >= 0) {
64 #if 0 //debug
65  trace(DBG_FORCE, "Incoming from %lx/%x len=%d cmd=%02x",
66  ntohl(from.sin_addr.s_addr), ntohs(from.sin_port),
67  pp->len, cmd);
68 #endif
69  //
70  // parse VREP
71  //
72  switch (cmd) {
73  case VREP_CREATE: case VREP_CREATE_V1:
74  pp->incomingCreate(&from);
75  delete pp;
76  return;
77  case VREP_DELTA: case VREP_DELTA_V1:
78  pp->incomingDelta(&from);
79  delete pp;
80  return;
81  case VREP_QUERY: case VREP_QUERY_V1:
82  pp->incomingQuery(&from);
83  delete pp;
84  return;
85  case VREP_DELETE: case VREP_DELETE_V1:
86  pp->incomingDelete(&from);
87  delete pp;
88  return;
89  default:
90  error("Incoming unknown: X'%02x' fd=%d from %lx/%x (mine is %lx/%x)",
91  cmd, fd, ntohl(from.sin_addr.s_addr), ntohs(from.sin_port),
92  NetObject::getMyHostId(), NetObject::getMyPortId());
93  delete pp;
94  return;
95  }
96  }
97  else {
98  // empty or invalid payload
99  pp->incomingOther(&from, r);
100  error("NetIncoming other= %lx (%x)", ntohl(from.sin_addr.s_addr), r);
101  delete pp;
102  return;
103  }
104  } //end infinite loop
105 }
