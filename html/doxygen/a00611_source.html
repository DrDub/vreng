63 bool VRSql::connect()
64 {
65  if (::g.pref.fast == true)
66  return false;
67 
68 #ifdef HAVE_MYSQL_REAL_CONNECT
69  if ((mysql = mysql_init(mysql)) != NULL) {
70  if (! mysql_real_connect(mysql, DEF_MYSQL_SERVER, USER, PASSWD, DB, 0, NULL, 0)) {
71  warning("VRSql: %s can't connect %s", USER, DEF_MYSQL_SERVER);
72 #if HAVE_MYSQL_ERROR
73  if (mysql_errno(mysql))
74  error("mysql_error: %s", mysql_error(mysql));
75 #endif
76  return false;
77  }
78  }
79  else {
80  warning("VRSql: %s can't init %s", USER, DEF_MYSQL_SERVER);
81  return false;
82  }
83  return true;
84 #else
85  if ((mysql = mysql_connect(NULL, DEF_MYSQL_SERVER, USER, PASSWD)) == NULL) {
86  warning("VRSql: %s can't connect %s", USER, DEF_MYSQL_SERVER);
87  return false;
88  }
89  if (mysql && mysql_select_db(mysql, DB) != 0) {
90  error("VRSql: can't find database %s", DB);
91  return false;
92  }
93  return true;
94 #endif
95 }

162 int VRSql::getInt(const char *table, const char *col, const char *object, const char *world, uint16_t irow)
163 {
164  sprintf(sqlcmd, "select SQL_CACHE %s from %s where %s='%s%s%s'",
165  col, table, COL_NAME, object, (*world) ? "@" : "", world);
166  if (! query(sqlcmd)) return ERR_MYSQL;
167  if ((ressql = result()) == NULL) return ERR_MYSQL;
168  mysql_data_seek(ressql, irow);
169  if ((row = mysql_fetch_row(ressql)) == NULL) {
170  // then insert col into the table
171  insertCol(table, col, object, world);
172  mysql_free_result(ressql);
173  return ERR_MYSQL; // no row
174  }
175  int val = (row[0] == NULL ? ERR_MYSQL : atoi(row[0]));
176  mysql_free_result(ressql);
177  return val;
178 }

242 int VRSql::getSubstring(const char *table, const char *pattern, uint16_t irow, char *retstring)
243 {
244  sprintf(sqlcmd, "select SQL_CACHE %s from %s where %s regexp '%s'",
245  COL_NAME, table, COL_NAME, pattern);
246  if (! query(sqlcmd)) return ERR_MYSQL;
247  if ((ressql = result()) == NULL) return ERR_MYSQL;
248  mysql_data_seek(ressql, irow);
249  if ((row = mysql_fetch_row(ressql)) == NULL) {
250  mysql_free_result(ressql);
251  return ERR_MYSQL; // no row
252  }
253  if (row[0] == NULL) {
254  mysql_free_result(ressql);
255  return ERR_MYSQL;
256  }
257  if (! strstr(row[0], pattern)) {
258  mysql_free_result(ressql);
259  return ERR_MYSQL; // no match
260  }
261  if (retstring) strcpy(retstring, row[0]);
262  mysql_free_result(ressql);
263  return irow;
264 }
