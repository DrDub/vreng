61 bool Cauldron::whenIntersect(WObject *pcur, WObject *pold)
62 {
63  switch (pcur->type) {
64  case USER_TYPE:
65  pcur->pos.z += DELTAZ;
66  break;
67  case BALL_TYPE:
68  srand((uint32_t) time(NULL));
69  if (pcur->pos.z > pos.z + DELTAZ) {
70  pcur->pos.x += ((float) drand48() * 2 -1) * Ball::RADIUS / 2;
71  pcur->pos.y += ((float) drand48() * 2 -1) * Ball::RADIUS / 2;
72  }
73  else pcur->pos.z += DELTAZ; // jump upon cauldron
74  break;
75  case STEP_TYPE:
76  return false;
77  default:
78  pold->copyPositionAndBB(pcur);
79  return true;
80  }
81  pcur->updatePositionAndGrid(pold);
82  return true;
83 }
