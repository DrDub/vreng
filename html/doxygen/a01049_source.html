53 Img * Img::loadPCX(void *tex, ImageReader read_func)
54 {
55  PcxInfo pcxInfo;
56  struct { uint8_t r, g, b; } palette[256];
57 
58  Texture *_tex = (Texture *) tex;
59  FILE *f;
60  if ((f = Cache::openCache(_tex->url, _tex->http)) == NULL) return NULL;
61 
62  /* loads the Header */
63  fread((char *) &pcxInfo, 1, sizeof(pcxInfo), f);
64 
65  /* test if this is really a PCX file */
66  if (pcxInfo.Manufacturer != 0x0A) {
67  error("loadPCX: not a PCX format: %x", pcxInfo.Manufacturer);
68  return NULL;
69  }
70 
71  /* only decode one type of PCX */
72  if ((pcxInfo.Version != 0x05) || (pcxInfo.BitsPerPixel != 8) ||
73  (pcxInfo.NPlanes != 0x01))
74  return NULL;
75 
76  /* gets the image dimensions */
77  uint16_t width = ((pcxInfo.Xmax_l | (pcxInfo.Xmax_h << 8)) - (pcxInfo.Xmin_l | (pcxInfo.Xmin_h << 8))) + 1;
78  uint16_t height = ((pcxInfo.Ymax_l | (pcxInfo.Ymax_h << 8)) - (pcxInfo.Ymin_l | (pcxInfo.Ymin_h << 8))) + 1;
79  uint16_t bpl = (pcxInfo.BytesPerLine_l | (pcxInfo.BytesPerLine_h << 8));
80 
81  /* temporary indexed buffer */
82  uint8_t *pix = new uint8_t[width * height];
83 
84  /* start decoding the file */
85  uint8_t *tmp = pix;
86  for (int y=0; y < height; y++) {
87  int count = 0;
88  uint8_t data = 0x00;
89 
90  /* decodes one line */
91  int x = 0;
92  while (x < bpl) {
93  if (count > 0) {
94  if (x < width)
95  *(tmp++) = data;
96  x++;
97  count--;
98  } else {
99  data = getc(f);
100  if ((data & 0xC0) == 0xC0) {
101  count = data & 0x3F;
102  data = getc(f);
103  } else {
104  count = 1;
105  }
106  }
107  }
108  }
109 
110  /* decodes the palette */
111  if (getc(f) != 0x0C) {
112  delete[] pix;
113  return NULL;
114  }
115  for (int i=0; i < 256; i++) {
116  palette[i].r = getc(f);
117  palette[i].g = getc(f);
118  palette[i].b = getc(f);
119  }
120 
121  /* allocates image */
122  Img *img = new Img(width, height, Img::RGB);
123 
124  /* converts the indexed buffer to RGB */
125  tmp = pix;
126  uint8_t *pixmap = img->pixmap;
127  for (int x=0; x < width * height; x++) {
128  *(pixmap++) = palette[*tmp].r;
129  *(pixmap++) = palette[*tmp].g;
130  *(pixmap++) = palette[*tmp].b;
131  tmp++;
132  }
133  delete[] pix;
134  File::closeFile(f);
135 
136  return img;
137 }
