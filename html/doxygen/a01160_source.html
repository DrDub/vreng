143 bool Dxf::draw(DXF_file *dxffile)
144 {
145  if (! loaded) return false;
146  if (! dxffile) return false;
147 
148  if (! dxffile->objects || (dxffile->objects && dxffile->objects->num_objects <= 0)) {
149  error("dxf: nothing to draw !");
150  return false;
151  }
152 
153  if (currentScale != desiredScale) setScale(desiredScale);
154  if (flgcolor) {
155  glMaterialfv(GL_FRONT, GL_DIFFUSE, mat_diffuse);
156  glMaterialfv(GL_FRONT, GL_AMBIENT, mat_ambient);
157  glMaterialfv(GL_FRONT, GL_SPECULAR, mat_specular);
158  }
159 
160  Point3D *v1 = newPoint3D(1,0,0);
161  Point3D *v2 = newPoint3D(0,1,0);
162  glEnable(GL_NORMALIZE);
163 
164  for (int i=0 ; i < dxffile->objects->num_objects ; i++) {
165  PolyMeshObj *pmesh = dxffile->objects->objects[i];
166  if (pmesh) {
167  for (int j=0; j < pmesh->num_polygons; j++) {
168  Polygon *pol = pmesh->polygons[j];
169  if (pol && pol->num_points > 0) {
170 
171  glBegin(GL_POLYGON);
172  for (int k=0; k < pol->num_points; k++) {
173  if (k==0 || (k>0 && pol->points.refs[k] != pol->points.refs[k-1])) {
174  glVertex3d(pol->points.refs[k]->x,
175  pol->points.refs[k]->y,
176  pol->points.refs[k]->z);
177  //error("i=%d j=%d k=%d %.2f %.2f %.2f", i,j,k,pol->points.refs[k]->x,pol->points.refs[k]->y,pol->points.refs[k]->z);
178  }
179  glTexCoord3d(pol->points.refs[k]->x,
180  pol->points.refs[k]->y,
181  pol->points.refs[k]->z);
182  }
183  if (pol->num_points >= 3) {
184  copyPoint3D(v1, pol->points.refs[2]);
185  copyPoint3D(v2, pol->points.refs[1]);
186  subPoint3D(v1, pol->points.refs[0]);
187  subPoint3D(v2, pol->points.refs[0]);
188  glNormal3d(v1->y *(v2->z)-v1->z*(v2->y),
189  -v1->x*(v2->z)+v1->z*(v2->x),
190  v1->x*(v2->y)-v1->y*(v2->x));
191  }
192  glEnd();
193  }
194  }
195  }
196  }
197  deletePoint3D(v1);
198  deletePoint3D(v2);
199  return true;
200 }
