157 void Flare::render(float *from, float delta)
158 {
159  float at[3], light[3], view_dir[3], tmp[3], light_dir[3], pos[3], dx[3], dy[3], center[3], axis[3], sx[3], sy[3], dot;
160  float near = 1.;
161 
162 #if 0 //TEST
163  at[0] = localuser->pos.x;
164  at[1] = localuser->pos.y;
165  at[2] = localuser->pos.z;
166 #endif
167 #if 1
168  from[0] = localuser->pos.x;
169  from[1] = localuser->pos.y;
170  from[2] = localuser->pos.z;
171 #endif
172  light[0] = sin(delta * 0.73) + 2.;
173  light[1] = sin(delta * 0.678) * 0.5 + 2.;
174  light[2] = sin(delta * 0.895) * 0.5 + 2.;
175  //error("flare: %.2f %.2f %.2f", light[0],light[1],light[2]);
176 
177  vdiff(view_dir, at, from);
178  vnorm(view_dir); // view_dir = normalize(at-from)
179  vscale(tmp, view_dir, near);
180  vadd(center, from, tmp); // center = from + near * view_dir
181  vdiff(light_dir, light, from);
182  vnorm(light_dir); // light_dir = normalize(light-from)
183  dot = vdot(light_dir, view_dir);
184  vscale(tmp, light_dir, near/dot);
185  vadd(light, from, light_dir); // light = from + dot(light,view_dir)*near*light_dir
186  vdiff(axis, light, center);
187  vcopy(dx, axis); // axis = light - center
188  vnorm(dx); // dx = normalize(axis)
189  vcross(dy, dx, view_dir); // dy = cross(dx,view_dir)
190 
191  glPushAttrib(GL_ALL_ATTRIB_BITS);
192  glPushMatrix();
193  glDisable(GL_DEPTH_TEST);
194  glDisable(GL_LIGHTING);
195  glDisable(GL_DITHER);
196  glEnable(GL_TEXTURE_2D);
197  glEnable(GL_BLEND);
198  glBlendFunc(GL_ONE, GL_ONE);
199  //dax glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
200 
201  for (int i=0; i < num_flares; i++) {
202  vscale(sx, dx, flares[i].scale * scale);
203  vscale(sy, dy, flares[i].scale * scale);
204 
205  glColor3fv(flares[i].color);
206 
207  // bind textures
208  if (flares[i].type < 0) { // shines
209  glBindTexture(GL_TEXTURE_2D, shineTex[num_shine]);
210  num_shine = (num_shine + 1) % MAX_SHINE; // [0..9]
211  }
212  else { // flares
213  glBindTexture(GL_TEXTURE_2D, flareTex[flares[i].type]);
214  }
215 
216  vscale(tmp, axis, flares[i].loc);
217  vadd(pos, center, tmp); // pos = center + flares[i].loc * axis
218 
219  glBegin(GL_QUADS);
220  glTexCoord2f(0, 0); vadd(tmp, pos, sx); vadd(tmp, tmp, sy); glVertex3fv(tmp); // pos+sx+sy
221  glTexCoord2f(1, 0); vdiff(tmp, pos, sx); vadd(tmp, tmp, sy); glVertex3fv(tmp); // pos-sx+sy
222  glTexCoord2f(1, 1); vdiff(tmp, pos, sx); vdiff(tmp, tmp, sy); glVertex3fv(tmp); // pos-sx-sy
223  glTexCoord2f(0, 1); vadd(tmp, pos, sx); vdiff(tmp, tmp, sy); glVertex3fv(tmp); // pos+sx-sy
224  glEnd();
225  }
226  glDisable(GL_BLEND);
227  glDisable(GL_TEXTURE_2D);
228  glEnable(GL_DITHER);
229  glEnable(GL_LIGHTING);
230  glEnable(GL_DEPTH_TEST);
231  glPopMatrix();
232  glPopAttrib();
233 }

235 void Flare::setTexture(GLuint texid)
236 {
237  glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
238  glBindTexture(GL_TEXTURE_2D, texid);
239  glTexEnvf(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, GL_MODULATE);
240  glTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
241  glTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
242 
243  Texture *tex = Texture::getEntryById(texid);
244  if (tex && tex->img) {
245  uint16_t w = tex->img->width;
246  uint16_t h = tex->img->height;
247  uint8_t *pix = tex->img->pixmap;
248  glTexImage2D(GL_TEXTURE_2D, 0, 1, w, h, 0, GL_LUMINANCE, GL_UNSIGNED_BYTE, pix);
249  }
