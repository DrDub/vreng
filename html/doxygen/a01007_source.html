116 Vicinity::Distance Vicinity::computeDistance(WObject *obj1, WObject *obj2)
117 {
118  float coordobj1[6];
119  float coordobj2[6];
120  float minDistance[3];
121  int typeInclusion[3];
122 
123  coordobj1[0] = obj1->pos.x;
124  coordobj1[1] = obj1->pos.y;
125  coordobj1[2] = obj1->pos.z;
126  coordobj1[3] = obj1->pos.bbsize.v[0];
127  coordobj1[4] = obj1->pos.bbsize.v[1];
128  coordobj1[5] = obj1->pos.bbsize.v[2];
129  coordobj2[0] = obj2->pos.x;
130  coordobj2[1] = obj2->pos.y;
131  coordobj2[2] = obj2->pos.z;
132  coordobj2[3] = obj2->pos.bbsize.v[0];
133  coordobj2[4] = obj2->pos.bbsize.v[1];
134  coordobj2[5] = obj2->pos.bbsize.v[2];
135 
136  float obj1A[3], obj1B[3], obj2A[3], obj2B[3];
137 
138  for (int i=0; i<3; i++) {
139  obj1A[i] = (coordobj1[i]-coordobj1[i+3]);
140  obj1B[i] = (coordobj1[i]+coordobj1[i+3]);
141  obj2A[i] = (coordobj2[i]-coordobj2[i+3]);
142  obj2B[i] = (coordobj2[i]+coordobj2[i+3]);
143  }
144 
145  //on calcul la distance suivant les 3 axes
146  for (int i=0; i<3; i++)
147  minDistance[i] = MIN(ABSF(obj2B[i]-obj1A[i]), ABSF(obj2A[i]-obj1B[i]));
148 
149  //on calcul les inclusions d'objets
150  for (int i=0; i<3; i++) {
151  //pour diminuer les tests on inverse en cas de besoin les 2 objets
152  if (obj2A[i] < obj1A[i]) {
153  for (int i=0; i<3; i++) {
154  float objtmp;
155  objtmp = obj2A[i];
156  obj2A[i] = obj1A[i];
157  obj1A[i] = objtmp;
158  objtmp = obj2B[i];
159  obj2B[i] = obj1B[i];
160  obj1B[i] = objtmp;
161  }
162  }
163 
164  if ((obj1A[i] < obj2A[i]) && (obj2A[i] < obj1B[i])) {
165  if (obj2B[i] < obj1B[i]) typeInclusion[i] = DIST_INCL;
166  else typeInclusion[i] = DIST_INTER;
167  } else {
168  if (minDistance[i] < 0.5) typeInclusion[i] = DIST_STUCK;
169  else if (minDistance[i] > 15) typeInclusion[i] = DIST_FAR;
170  else typeInclusion[i] = DIST_NEAR;
171  }
172  }
173 
174  if ((typeInclusion[0] == DIST_INCL) &&
175  (typeInclusion[1] == DIST_INCL) &&
176  (typeInclusion[2] == DIST_INCL))
177  return DIST_INCL;
178  if ((typeInclusion[0] == DIST_FAR) ||
179  (typeInclusion[1] == DIST_FAR) ||
180  (typeInclusion[2] == DIST_FAR))
181  return DIST_FAR;
182  if ((typeInclusion[0] == DIST_NEAR) ||
183  (typeInclusion[1] == DIST_NEAR) ||
184  (typeInclusion[2] == DIST_NEAR))
185  return DIST_NEAR;
186  if ((typeInclusion[0] == DIST_STUCK) ||
187  (typeInclusion[1] == DIST_STUCK) ||
188  (typeInclusion[2] == DIST_STUCK))
189  return DIST_STUCK;
190  if ((typeInclusion[0] == DIST_INTER) ||
191  (typeInclusion[1] == DIST_INTER) ||
192  (typeInclusion[2] == DIST_INTER))
193  return DIST_INTER;
194  warning("computeDistance:: default distance value");
195  return DIST_INTER;
196 }

198 Vicinity::Size Vicinity::computeSize(WObject *obj)
199 {
200  Size objSize = SIZE_NORMAL;
201 
202  if ((obj->pos.bbsize.v[0] > (10*localuser->pos.bbsize.v[0]))

208  && (obj->pos.bbsize.v[2] > (2*localuser->pos.bbsize.v[2]))) {
209  objSize = SIZE_LARGE;
210  } else if ((obj->pos.bbsize.v[0] < (1*localuser->pos.bbsize.v[0]))
211  && (obj->pos.bbsize.v[1] < (0.5*localuser->pos.bbsize.v[1]))
212  && (obj->pos.bbsize.v[2] < (0.5*localuser->pos.bbsize.v[2]))) {
213  objSize = SIZE_SMALL;
214  } else if ((obj->pos.bbsize.v[0] < (0.5*localuser->pos.bbsize.v[0]))
215  && (obj->pos.bbsize.v[1] < (0.2*localuser->pos.bbsize.v[1]))
216  && (obj->pos.bbsize.v[2] < (0.2*localuser->pos.bbsize.v[2]))) {

223 int Vicinity::compareByDistance(const void *t1, const void *t2)
224 {
225  Vicin s1 = *((Vicin*) t1);
226  Vicin s2 = *((Vicin*) t2);
227 
228  //increasing order
229  if (s1.dist > s2.dist) return 1;
230  if (s1.dist < s2.dist) return -1;
231  if (s1.size > s2.size) return -1;
232  if (s1.size < s2.size) return 1;
233  return 0;
234 }

241 int Vicinity::compareByInterest(const void *t1, const void *t2)
242 {
243  Vicin s1 = *((Vicin*) t1);
244  Vicin s2 = *((Vicin*) t2);
245  int size1 = ((int) s1.size);
246  int size2 = ((int) s2.size);
247  int dist1 = (int) s1.dist;
248  int dist2 = (int) s2.dist;
249 
250  //qd un objet est loin on en tient pas compte
251  if ((dist1 == 4) && (dist2 < 4)) return 1;
252  if ((dist2 == 4) && (dist1 < 4)) return -1;
253 
254  //si la taille de l'objet est plus grand ou si la distance est plus proche
255  if ((size1 > size2) && (dist1 < dist2)) return -1;
256  if ((size1 < size2) && (dist1 > dist2)) return 1;
257 
258  //un objet grand et pres est plus important qu'un objet petit et loin
259  if ((size1 > size2) && (dist1 > dist2)) {
260  int distu1 = ((int) computeDistance(localuser,s1.object));
261  int distu2 = ((int) computeDistance(localuser,s2.object));
262 
263  //si un objet est grand loin size1 dist1 et un autre petit pres size2 dist2
264  //on regarde la distance de leur centre a l'utilisateur.
265  if (distu1 > distu2) return 1;
266  else return -1;
267  }
268  if ((size1 < size2) && (dist1 < dist2)) {
269  int distu1 = ((int) computeDistance(localuser, s1.object));
270  int distu2 = ((int) computeDistance(localuser, s2.object));
271  if (distu1 > distu2) return 1;
272  else return -1;
273  }
274  //si un objet est de taille proche de l'utilisateur il est plus important
275  //TODO: trouver une formule magique pour rendre encore plus pertinent le test
276  return 0;
277 }
