82 // the mouse is pressed on the canvas
83 void Navig::mousePressCB(UMouseEvent& e)
84 {
85  int x = (int) e.getX(), y = (int) e.getY();
86  int button = 0;
87  int bid = e.getButton();
88  if (bid == UMouseEvent::LeftButton) button = 1;
89  else if (bid == UMouseEvent::MidButton) button = 2;
90  else if (bid == UMouseEvent::RightButton) button = 3;
91 
92  // GL graphics can be performed until the current function returns
93  GLSection gls(&gw.scene);
94 
95  if (gw.gui.carrier && gw.gui.carrier->isTaking()) // events are sent to Carrier
96  gw.gui.carrier->mouseEvent(x, y, button);
97 #if 0 //dax
98  else if (gw.gui.board && gw.gui.board->isDrawing()) // events are sent to Board
99  gw.gui.board->mouseEvent(x, y, button);
100 #endif
101  else if (gw.gui.vnc) // events are sent to Vnc
102  gw.gui.vnc->mouseEvent(x, y, button);
103  else if (button == 2) // button 2: fine grain selection
104  mousePressB2(e, x, y, button);
105  else // buttons 1 or 3: navigator or info menu
106  mousePressB1orB3(e, x, y, button);
107 }

188 // Press Buttons 1 or 3: navigator or info menu
189 void Navig::mousePressB1orB3(UMouseEvent& e, int x, int y, int button)
190 {
191  //Sound::playSound(CLICKSND);
192 
193  // desactivate previous object
194  WObject* prev_object = gw.gui.getSelectedObject();
195  if (prev_object && prev_object->isValid()) {
196  prev_object->resetFlashy();
197  prev_object->resetRay();
198  }
199 
200  // current object
201  depthsel = 0;
202  WObject* object = gw.getPointedObject(x, y, objinfo, depthsel);
203  if (object && object->isValid()) {
204  gw.gui.selected_object = object;
205  trace(DBG_GUI, "clic [%d %d] on %s", x, y, object->getInstance());
206 
207  if (object->names.url[0]) selected_object_url = object->names.url;
208  else selected_object_url.clear();
209  gw.messages.performRequest(object);
210  // Vrelet: calculate the clic vector and do the clic method on the object
211  if (gw.gui.vrelet && button == 1) object->click(x, y);
212  if (gw.gui.board && button == 1) object->click(x, y);
213 
214  selectObject(objinfo, button);
215 
216  if (button == 3) { // show object menu
217  object_menu.open(e);
218  opened_menu = object_menu;
219  }
220  else if (button == 1) { // navigator
221  navig_menu.open(e); // show(e, 0, 0);
222  opened_menu = navig_menu;
223  if (object->solid) {
224  object->setFlashy(); // flashes the edges of the solid
225  object->setRay(x, y); // launches ray
226  }
227  }
228  }
229  else
230  gw.setRayDirection(x, y); // launches stipple ray
231 }
