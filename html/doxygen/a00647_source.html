55 void Plane::makeSolid()
56 {
57  V3 dim, rel;
58  uint8_t nbsec;
59  char s[BUFSIZ*2], materials[BUFSIZ];
60  GLfloat dif[4], amb[4], spe[4], emi[4], a;
61  GLint shi[1];
62 
63  getDim(dim);
64  // decomposer le solid d'origine en plusieurs petits solides
65  // pour faire de multiples bounding boxes gerables aux collisions
66  if (dim.v[2] < 0.2) return; // seuil de 20 cm en hauteur
67 
68  getMaterials(dif, amb, spe, emi, shi, &a);
69  //TODO: getTextures()
70  sprintf(materials, "dif=\"%.2f %.2f %.2f\" amb=\"%.2f %.2f %.2f\" spe=\"%.2f %.2f %.2f\" emi=\"%.2f %.2f %.2f\" shi=\"%d\" a=\"%.2f\" %s ", dif[0], dif[1], dif[2], amb[0], amb[1], amb[2], spe[0], spe[1], spe[2], emi[0], emi[1], emi[2], shi[0], a, (tex != NULL) ? tex : "");
71  nbsec = (uint8_t) ceil(dim.v[2] / 0.2);
72  float xsec = dim.v[0] / nbsec;
73  float ysec = dim.v[1] / nbsec;
74  float zsec = dim.v[2] / nbsec;
75  //error("dim=%.2f %.2f %.2f", dim.v[0],dim.v[1],dim.v[2]);
76  //error("nbsec=%d zsec=%.2f %.2f %.2f", nbsec, xsec, ysec, zsec);
77  deleteSolids();
78 
79  // le premier solide est au centre
80  sprintf(s, "solid dim=\"%.2f %.2f %.2f\" %s />", 2*dim.v[0], 2*dim.v[1], 2*zsec, materials);
81  //error("s: %s", s);
82  parse()->parseSolid(s, SEP, this);
83  for (int i=1; i < nbsec; i++) {
84  for (int j=0; j<2; j++) {
85  // les autres solides s'ecartent progressivement du centre en negatif et positif
86  int k = 2*j-1; // k=-1, k=1
87  rel.v[0] = k * (2*i - 1) * (xsec/2) * cos(pos.az);
88  rel.v[1] = k * (2*i - 1) * (ysec/2) * sin(pos.az);
89  rel.v[2] = k * (2*i - 1) * zsec / 2;
90  if (fabsf(dim.v[0]) < fabsf(dim.v[1])) {
91  sprintf(s, "solid rel=\"%.2f %.2f %.2f 0 0\" dim=\"%.2f %.2f %.2f\" %s />", rel.v[0], rel.v[1], rel.v[2], 2*dim.v[0], 2*dim.v[1], 2*zsec, materials);
92  }
93  else {
94  sprintf(s, "solid rel=\"%.2f %.2f %.2f 0 0\" dim=\"%.2f %.2f %.2f\" %s />", rel.v[0], rel.v[1], rel.v[2], 2*dim.v[0], 2*dim.v[1], 2*zsec, materials);
95  }
96  //error("s%d: rel=\"%.2f %.2f %.2f 0 0\" dim=\"%.2f %.2f %.2f\"", i, rel.v[0], rel.v[1], rel.v[2], 2*dim.v[0], 2*dim.v[1], 2*zsec);
97  parse()->parseSolid(s, SEP, this);
98  }
99  }
100 }

119 bool Plane::whenIntersect(WObject *pcur, WObject *pold)
120 {
121  switch (pcur->type) {
122 
123  case USER_TYPE:
124  case THING_TYPE:
125  if (pos.ax > 0.01) { // inclinated plane
126  pcur->pos.z += Step::JUMP; // * Sin(pos.ax); // up
127  pcur->updatePositionAndGrid(pold);
128  }
129  projectPosition(pcur, pold);
130  break;
131 
132  case BALL_TYPE:
133  {
134 #if 1 //cpax FIXME
135  V3 normal;
136  computeNormal(pold, &normal);
137  bounceTrajectory(pold, &normal);
138  pold->setLasting(0);
139  pold->disablePermanentMovement();
140  pold->copyPosAndBB(pcur->pos);
141 #else
142  projectPosition(pcur, pold);
143 #endif
144  }
145  break;
146 
147  case STEP_TYPE:
148  case GUIDE_TYPE:
149  return false;
150 
151  default:
152  pold->copyPosAndBB(pcur->pos);
153  }
154  return true;
