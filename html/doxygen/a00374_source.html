56 void Clock::changePermanent(float lasting)
57 {
58  switch (needle % 3) {
59  case SECOND: // secondes
60  pos.ax = ((float) sec * M_PI / 30) + M_PI_2;
61  break;
62  case MINUTE: // minutes
63  if (min != min_last) {
64  pos.ax = ((float) min * M_PI / 30) + M_PI_2;
65  min_last = min;
66  }
67  break;
68  case HOUR: // hours
69  if (hour != hour_last) {
70  pos.ax = ((float) hour * M_PI / 6) +
71  ((float) min * M_PI / 360) + M_PI_2;
72  hour_last = hour;
73  }
74  break;
75  }
76  sec_last = sec;
77 
78  // Can be used with changing Bgcolor
79  daytime = 12 - 4 * Cos(360 * (float) yday/365.);
80  risetime = 13 - daytime/2;
81  falltime = 13 + daytime/2;
