38 /*
39  * Transform a point (column vector) by a 4x4 matrix. I.e. out = m * in
40  * Input: m - the 4x4 matrix
41  * in - the 4x1 vector
42  * Output: out - the resulting 4x1 vector.
43  */
44 static void
45 transform_point(GLdouble out[4], const GLdouble m[16], const GLdouble in[4])
46 {
47 #define M(row,col) m[col*4+row]
48  out[0] = M(0,0) * in[0] + M(0,1) * in[1] + M(0,2) * in[2] + M(0,3) * in[3];
49  out[1] = M(1,0) * in[0] + M(1,1) * in[1] + M(1,2) * in[2] + M(1,3) * in[3];
50  out[2] = M(2,0) * in[0] + M(2,1) * in[1] + M(2,2) * in[2] + M(2,3) * in[3];
51  out[3] = M(3,0) * in[0] + M(3,1) * in[1] + M(3,2) * in[2] + M(3,3) * in[3];
52 #undef M
53 }

55 /*
56  * Perform a 4x4 matrix multiplication (product = a x b).
57  * Input: a, b - matrices to multiply
58  * Output: product - product of a and b
59  */
60 static void
61 matmul(GLdouble * product, const GLdouble * a, const GLdouble * b)
62 {
63  /* This matmul was contributed by Thomas Malik */
64  GLdouble temp[16];
65 
66 #define A(row,col) a[(col<<2)+row]
67 #define B(row,col) b[(col<<2)+row]
68 #define T(row,col) temp[(col<<2)+row]
69  /* i-te Zeile */
70  for (GLint i=0; i<4; i++) {
71  T(i,0) = A(i,0)*B(0,0) + A(i,1)*B(1,0) + A(i,2)*B(2,0) + A(i,3)*B(3,0);
72  T(i,1) = A(i,0)*B(0,1) + A(i,1)*B(1,1) + A(i,2)*B(2,1) + A(i,3)*B(3,1);
73  T(i,2) = A(i,0)*B(0,2) + A(i,1)*B(1,2) + A(i,2)*B(2,2) + A(i,3)*B(3,2);
74  T(i,3) = A(i,0)*B(0,3) + A(i,1)*B(1,3) + A(i,2)*B(2,3) + A(i,3)*B(3,3);
75  }
76 #undef A
77 #undef B
78 #undef T
79  memcpy((void *) product, (void *) temp, (size_t) 16 * sizeof(GLdouble));
80 }
81 /*
82  * Compute inverse of 4x4 transformation matrix.
83  * Code contributed by Jacques Leroy jle@star.be
84  * Return GL_TRUE for success, GL_FALSE for failure (singular matrix)
85  */
86 static GLboolean
87 invert_matrix(const GLdouble * m, GLdouble * out)
88 {
89 /* NB. OpenGL Matrices are COLUMN major. */
90 #define SWAP_ROWS(a, b) { GLdouble *_tmp = a; (a)=(b); (b)=_tmp; }
91 #define MAT(m,r,c) (m)[(c)*4+(r)]
92 
93  GLdouble wtmp[4][8];
94  GLdouble m0, m1, m2, m3, s;
95  GLdouble *r0, *r1, *r2, *r3;
96 
97  r0 = wtmp[0], r1 = wtmp[1], r2 = wtmp[2], r3 = wtmp[3];
98 
99  r0[0] = MAT(m, 0, 0), r0[1] = MAT(m, 0, 1),
100  r0[2] = MAT(m, 0, 2), r0[3] = MAT(m, 0, 3),
101  r0[4] = 1.0, r0[5] = r0[6] = r0[7] = 0.0,
102  r1[0] = MAT(m, 1, 0), r1[1] = MAT(m, 1, 1),
103  r1[2] = MAT(m, 1, 2), r1[3] = MAT(m, 1, 3),
104  r1[5] = 1.0, r1[4] = r1[6] = r1[7] = 0.0,
105  r2[0] = MAT(m, 2, 0), r2[1] = MAT(m, 2, 1),
106  r2[2] = MAT(m, 2, 2), r2[3] = MAT(m, 2, 3),
107  r2[6] = 1.0, r2[4] = r2[5] = r2[7] = 0.0,
108  r3[0] = MAT(m, 3, 0), r3[1] = MAT(m, 3, 1),
109  r3[2] = MAT(m, 3, 2), r3[3] = MAT(m, 3, 3),
110  r3[7] = 1.0, r3[4] = r3[5] = r3[6] = 0.0;
111 
112  /* choose pivot - or die */
113  if (fabs(r3[0]) > fabs(r2[0]))
114  SWAP_ROWS(r3, r2);
115  if (fabs(r2[0]) > fabs(r1[0]))
116  SWAP_ROWS(r2, r1);
117  if (fabs(r1[0]) > fabs(r0[0]))
118  SWAP_ROWS(r1, r0);
119  if (0.0 == r0[0])
120  return GL_FALSE;
121 
122  /* eliminate first variable */
123  m1 = r1[0] / r0[0];
124  m2 = r2[0] / r0[0];
125  m3 = r3[0] / r0[0];
126  s = r0[1];
127  r1[1] -= m1 * s;
128  r2[1] -= m2 * s;
129  r3[1] -= m3 * s;
130  s = r0[2];
131  r1[2] -= m1 * s;
132  r2[2] -= m2 * s;
133  r3[2] -= m3 * s;
134  s = r0[3];
135  r1[3] -= m1 * s;
136  r2[3] -= m2 * s;
137  r3[3] -= m3 * s;
138  s = r0[4];
139  if (s != 0.0) {
140  r1[4] -= m1 * s;
141  r2[4] -= m2 * s;
142  r3[4] -= m3 * s;
143  }
144  s = r0[5];
145  if (s != 0.0) {
146  r1[5] -= m1 * s;
147  r2[5] -= m2 * s;
148  r3[5] -= m3 * s;
149  }
150  s = r0[6];
151  if (s != 0.0) {
152  r1[6] -= m1 * s;
153  r2[6] -= m2 * s;
154  r3[6] -= m3 * s;
155  }
156  s = r0[7];
157  if (s != 0.0) {
158  r1[7] -= m1 * s;
159  r2[7] -= m2 * s;
160  r3[7] -= m3 * s;
161  }
162 
163  /* choose pivot - or die */
164  if (fabs(r3[1]) > fabs(r2[1]))
165  SWAP_ROWS(r3, r2);
166  if (fabs(r2[1]) > fabs(r1[1]))
167  SWAP_ROWS(r2, r1);
168  if (0.0 == r1[1])
169  return GL_FALSE;
170 
171  /* eliminate second variable */
172  m2 = r2[1] / r1[1];
173  m3 = r3[1] / r1[1];
174  r2[2] -= m2 * r1[2];
175  r3[2] -= m3 * r1[2];
176  r2[3] -= m2 * r1[3];
177  r3[3] -= m3 * r1[3];
178  s = r1[4];
179  if (0.0 != s) {
180  r2[4] -= m2 * s;
181  r3[4] -= m3 * s;
182  }
183  s = r1[5];
184  if (0.0 != s) {
185  r2[5] -= m2 * s;
186  r3[5] -= m3 * s;
187  }
188  s = r1[6];
189  if (0.0 != s) {
190  r2[6] -= m2 * s;
191  r3[6] -= m3 * s;
192  }
193  s = r1[7];
194  if (0.0 != s) {
195  r2[7] -= m2 * s;
196  r3[7] -= m3 * s;
197  }
198 
199  /* choose pivot - or die */
200  if (fabs(r3[2]) > fabs(r2[2]))
201  SWAP_ROWS(r3, r2);
202  if (0.0 == r2[2])
203  return GL_FALSE;
204 
205  /* eliminate third variable */
206  m3 = r3[2] / r2[2];
207  r3[3] -= m3 * r2[3], r3[4] -= m3 * r2[4],
208  r3[5] -= m3 * r2[5], r3[6] -= m3 * r2[6], r3[7] -= m3 * r2[7];
209 
210  /* last check */
211  if (0.0 == r3[3])
212  return GL_FALSE;
213 
214  s = 1.0 / r3[3]; /* now back substitute row 3 */
215  r3[4] *= s;
216  r3[5] *= s;
217  r3[6] *= s;
218  r3[7] *= s;
219 
220  m2 = r2[3]; /* now back substitute row 2 */
221  s = 1.0 / r2[2];
222  r2[4] = s * (r2[4] - r3[4] * m2), r2[5] = s * (r2[5] - r3[5] * m2),
223  r2[6] = s * (r2[6] - r3[6] * m2), r2[7] = s * (r2[7] - r3[7] * m2);
224  m1 = r1[3];
225  r1[4] -= r3[4] * m1, r1[5] -= r3[5] * m1,
226  r1[6] -= r3[6] * m1, r1[7] -= r3[7] * m1;
227  m0 = r0[3];
228  r0[4] -= r3[4] * m0, r0[5] -= r3[5] * m0,
229  r0[6] -= r3[6] * m0, r0[7] -= r3[7] * m0;
230 
231  m1 = r1[2]; /* now back substitute row 1 */
232  s = 1.0 / r1[1];
233  r1[4] = s * (r1[4] - r2[4] * m1), r1[5] = s * (r1[5] - r2[5] * m1),
234  r1[6] = s * (r1[6] - r2[6] * m1), r1[7] = s * (r1[7] - r2[7] * m1);
235  m0 = r0[2];
236  r0[4] -= r2[4] * m0, r0[5] -= r2[5] * m0,
237  r0[6] -= r2[6] * m0, r0[7] -= r2[7] * m0;
238 
239  m0 = r0[1]; /* now back substitute row 0 */
240  s = 1.0 / r0[0];
241  r0[4] = s * (r0[4] - r1[4] * m0), r0[5] = s * (r0[5] - r1[5] * m0),
242  r0[6] = s * (r0[6] - r1[6] * m0), r0[7] = s * (r0[7] - r1[7] * m0);
243 
244  MAT(out, 0, 0) = r0[4];
245  MAT(out, 0, 1) = r0[5], MAT(out, 0, 2) = r0[6];
246  MAT(out, 0, 3) = r0[7], MAT(out, 1, 0) = r1[4];
247  MAT(out, 1, 1) = r1[5], MAT(out, 1, 2) = r1[6];
248  MAT(out, 1, 3) = r1[7], MAT(out, 2, 0) = r2[4];
249  MAT(out, 2, 1) = r2[5], MAT(out, 2, 2) = r2[6];
250  MAT(out, 2, 3) = r2[7], MAT(out, 3, 0) = r3[4];
251  MAT(out, 3, 1) = r3[5], MAT(out, 3, 2) = r3[6];
252  MAT(out, 3, 3) = r3[7];
253 
254  return GL_TRUE;
255 
256 #undef MAT
257 #undef SWAP_ROWS
258 }

260 /* projection du point (objx,objy,obz) sur l'ecran (winx,winy,winz) */
261 GLint
262 gluProject(GLdouble objx, GLdouble objy, GLdouble objz,
263  const GLdouble model[16], const GLdouble proj[16],
264  const GLint viewport[4],
265  GLdouble * winx, GLdouble * winy, GLdouble * winz)
266 {
267  /* matrice de transformation */
268  GLdouble in[4], out[4];
269 
270  /* initilise la matrice et le vecteur a transformer */
271  in[0] = objx;
272  in[1] = objy;
273  in[2] = objz;
274  in[3] = 1.0;
275  transform_point(out, model, in);
276  transform_point(in, proj, out);
277 
278  /* d'ou le resultat normalise entre -1 et 1 */
279  if (in[3] == 0.0)
280  return GL_FALSE;
281 
282  in[0] /= in[3];
283  in[1] /= in[3];
284  in[2] /= in[3];
285 
286  /* en coordonnees ecran */
287  *winx = viewport[0] + (1 + in[0]) * viewport[2] / 2;
288  *winy = viewport[1] + (1 + in[1]) * viewport[3] / 2;
289  /* entre 0 et 1 suivant z */
290  *winz = (1 + in[2]) / 2;
291  return GL_TRUE;
292 }

294 /* transformation du point ecran (winx,winy,winz) en point objet */
295 GLint
296 gluUnProject(GLdouble winx, GLdouble winy, GLdouble winz,
297  const GLdouble model[16], const GLdouble proj[16],
298  const GLint viewport[4],
299  GLdouble * objx, GLdouble * objy, GLdouble * objz)
300 {
301  /* matrice de transformation */
302  GLdouble m[16], A[16];
303  GLdouble in[4], out[4];
304 
305  /* transformation coordonnees normalisees entre -1 et 1 */
306  in[0] = (winx - viewport[0]) * 2 / viewport[2] - 1.0;
307  in[1] = (winy - viewport[1]) * 2 / viewport[3] - 1.0;
308  in[2] = 2 * winz - 1.0;
309  in[3] = 1.0;
310 
311  /* calcul transformation inverse */
312  matmul(A, proj, model);
313  invert_matrix(A, m);
314 
315  /* d'ou les coordonnees objets */
316  transform_point(out, m, in);
317  if (out[3] == 0.0)
318  return GL_FALSE;
319  *objx = out[0] / out[3];
320  *objy = out[1] / out[3];
321  *objz = out[2] / out[3];
322  return GL_TRUE;
323 }
