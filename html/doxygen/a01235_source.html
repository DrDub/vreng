57 void Audio::start(const char *chan)
58 {

60 
61  char *p, newchan[CHAN_LEN];

63 
64  getnewchan(chan, newchan);

66  p = strrchr(newchan, '/');

68 
69 #if defined(WIN32) && !defined(CYGWIN32) // _spawn
70  if (toolid == RAT_TOOL) {
71  pid = _spawnlp(_P_NOWAIT, "rat", "rat", "-t", ttl,
72  "-f", "pcm", "-C", "VREng", newchan, NULL);
73  }
74  if (toolid == VAT_TOOL) {
75  pid = _spawnlp(_P_NOWAIT, "vat", "vat", "-aAsM", "-t", ttl,
76  "-C", "VREng", newchan, NULL);
77  }
78  if (toolid == FPHONE_TOOL) {
79  pid = _spawnlp(_P_NOWAIT, "fphone", "fphone", "-t", ttl,
80  newchan, NULL);
81  }
82 #else
83  switch (pid = fork()) {
84  case -1:
85  error("%s audio", e_fork);
86  break;
87  case 0:
88  if (toolid == RAT_TOOL) {
89  execlp("rat", "rat", "-f", "pcm", "-C", "VREng", "-t", ttl, newchan, (void*)NULL);
90  error("%s rat", e_exec);
91  }
92  if (toolid == VAT_TOOL) {
93  execlp("vat", "vat", "-aAsM", "-C", "VREng", "-t", ttl, newchan, (void*)NULL);
94  error("%s vat", e_exec);
95  }
96  if (toolid == FPHONE_TOOL) {
97  execlp("fphone", "fphone", "/T", ttl, newchan, (void*)NULL);
98  error("%s fphone", e_exec);
99  }
100  signal(SIGCHLD, SIG_IGN);
101  exit(1); // child death
102  default:
103  break;
104  }
105 #endif
106 }
