58 void Gate::parser(char *l)
59 {
60  defaults();
61  l = tokenize(l);
62  begin_while_parse(l) {
63  l = parse()->parseAttributes(l, this);
64  if (!l) break;
65  if (! stringcmp(l, "url")) l = parse()->parseUrl(l, names.url);
66  else if (! stringcmp(l, "world")) l = parse()->parseString(l, names.url, "world");
67  else if (! stringcmp(l, "channel")) l = parse()->parseString(l, chan, "channel");
68  else if (! stringcmp(l, "mode")) {
69  char modestr[6];
70  l = parse()->parseString(l, modestr, "mode");
71  if (! stringcmp(modestr, "link")) link = true;
72  else if (! stringcmp(modestr, "auto")) automatic = true;
73  }
74  else if (! stringcmp(l, "entry")) {
75  l = parse()->parseVector3f(l, entry, "entry");
76  flagentry = true;
77  trace(DBG_WO, "gate: entry=%.2f,%.2f,%.2f", entry[0], entry[1], entry[2]);
78  }
79  }
80  end_while_parse(l);
81 }

118 void Gate::enter()
119 {
120  /* save url because World::quit frees gate */
121  char *new_url = strdup(names.url);
122  if (!new_url) { error("enter: can't strdup url=%s", new_url); return; }
123 
124  if (link) { // without channel
125  World::current()->quit();
126  World::enter(new_url, NULL, World::NEW);
127  World::current()->linked(); // linked world
128  }
129  else { // with channel
130  if (strcmp(names.url, Universe::current()->url) == 0) {
131  sprintf(chan, "%s/%u/%d",
132  Universe::current()->group, Universe::current()->port, Channel::currentTtl());
133  trace(DBG_IPMC, "initial channel = %s", chan);
134  }
135 
136 #if 0 //dax
137  // call here the VACS (VREng Address Cache Server) to get the channel string
138  Vac *vac = Vac::current();
139  if (! vac->getChannel(names.url, chan)) {
140  // this url is not in the cache, we need to ask to the vacs to resolve it
141  if (vac->resolveWorldUrl(names.url, chan)) {
142  trace(DBG_FORCE, "enter: resolveWorldUrl url=%s channel=%s", names.url, chan);
143  }
144  else {
145  warning("enter: warning resolveWorldUrl failed from Vac: url=%s", names.url);
146  if (! *chan)
147  strcpy(chan, DEF_VRE_CHANNEL); // no given channel, forced to the default
148  }
149  }
150 #endif //dax
151  trace(DBG_IPMC, "enter: getChannel=%s url=%s", chan, new_url);
152  char *new_chan = strdup(chan);
153 
154  World::current()->quit(); // quit the current world
155  delete Channel::current(); // delete Channel
156  Sound::playSound(GATESND);
157 
158  World::enter(new_url, new_chan, World::NEW);
159  Channel::join(new_chan);
160  trace(DBG_IPMC, "enter: join channel=%s url=%s", new_chan, new_url);
161 
162  //TODO declareJoinWorldToManager(new_url, new_chan, worlds->plocaluser->getInstance());
163 
164  if (audioactive) Audio::start(new_chan);
165  free(new_chan);
166  }
167  free(new_url);
168 }

170 /* when an intersection occurs */
171 bool Gate::whenIntersect(WObject *pcur, WObject *pold)
172 {
173  switch (pcur->type) {
174  case USER_TYPE:
175  if (automatic) {
176  if (pcur != localuser) {
177  notice("Another user was just pushed out of this world");
178  return true;
179  }
180  else {
181  collidecnt = 0;
182  if (flagentry) {
183  flagentry = false;
184  pcur->updatePositionAndGrid(pcur->pos);
185  }
186  enter();
187  }
188  }
189  else {
190  if (collidecnt < 20) {
191  pold->copyPositionAndBB(pcur);
192  notice("Warning! You are near the gate %s in=%d", getInstance(), collidecnt);
193  collidecnt++;
194  }
195  else if (collidecnt < 40) {
196  pold->copyPositionAndBB(pcur);
197  notice("Warning! If you insist you'll enter in %s in=%d", getInstance(), collidecnt);
198  collidecnt++;
199  }
200  else {
201  enter(); // enter by force
202  collidecnt = 0;
203  }
204  }
205  break;
206  case BULLET_TYPE:
207  case DART_TYPE:
208  notice("%s:%s hits %s:%s", pcur->names.type, pcur->getInstance(), names.type, getInstance());
209  pold->copyPositionAndBB(pcur);
210  pcur->toDelete();
211  break;
212  case BALL_TYPE:
213  projectPosition(pcur, pold);
214  break;
215  case GUIDE_TYPE:
216  return false;
217  default: // other objects
218  pold->copyPositionAndBB(pcur);
219  }
220  return true;
221 }
