64 void Particle::parser(char *l)
65 {
66  defaults();
67  l = tokenize(l);
68  begin_while_parse(l) {
69  l = parse()->parseAttributes(l, this);
70  if (!l) break;
71  if (!stringcmp(l, "system")) {
72  char systemstr[16];
73  l = parse()->parseString(l, systemstr, "system");
74  if (! stringcmp(systemstr, "fountain")) system = FOUNTAIN;
75  else if (! stringcmp(systemstr, "firework")) system = FIREWORK;
76  else if (! stringcmp(systemstr, "rain")) system = RAIN;
77  else if (! stringcmp(systemstr, "snow")) system = SNOW;
78  else system = WATERFALL;
79  }
80  else if (!stringcmp(l, "number")) l = parse()->parseUInt16(l, &number, "number");
81  else if (!stringcmp(l, "flow")) l = parse()->parseFloat(l, &flow, "flow");
82  else if (!stringcmp(l, "speed")) l = parse()->parseFloat(l, &speed, "speed");
83  else if (!stringcmp(l, "ground")) l = parse()->parseFloat(l, &ground, "ground");
84  else if (!stringcmp(l, "size")) l = parse()->parseUInt8(l, &pt_size, "size");
85  else if (!stringcmp(l, "color")) {l = parse()->parseVector3f(l, color, "color"); mycolor = 1; }
86  }
87  end_while_parse(l);
88 }

156 void Particle::generate(tParticle *p, float dt)
157 {
158  p->pos[0] = pos.x;
159  p->pos[1] = pos.y;
160  p->pos[2] = pos.z;
161  p->prev[0] = p->pos[0];
162  p->prev[1] = p->pos[1];
163  p->prev[2] = p->pos[2];
164 
165  switch (system) {
166  case WATERFALL:
167  points = true;
168  p->vel[0] = 2*((float) drand48()-.5);
169  p->vel[1] = 2*((float) drand48()-.5);
170  p->vel[2] = 0;
171  p->damp = .45*(float) drand48();
172  break;
173  case FOUNTAIN:
174  points = true;
175  p->vel[0] = 2*((float) drand48()-.5);
176  p->vel[1] = 2*((float) drand48()-.5);
177  p->vel[2] = .75*speed;
178  p->damp = .35*(float) drand48();
179  break;
180  case FIREWORK:
181  points = true;
182  p->vel[0] = 5*((float) drand48()-.5);
183  p->vel[1] = 5*((float) drand48()-.5);
184  p->vel[2] = 5*((float) drand48()-.5);
185  p->damp = .35*(float) drand48();
186  break;
187  case SNOW:
188  points = true;
189  p->vel[0] = 3*((float) drand48()-.5);
190  p->vel[1] = 3*((float) drand48()-.5);
191  p->vel[2] = .2*speed;
192  p->damp = .25*(float) drand48();
193  break;
194  case RAIN:
195  points = false;
196  p->vel[0] = 4*((float) drand48()-.5);
197  p->vel[1] = 4*((float) drand48()-.5);
198  p->vel[2] = 2*speed;
199  p->damp = .15*(float) drand48();
200  break;
201  }
202  p->alive = true;
203  timestep(p, dt * (float) drand48());
204 }

224 void Particle::bounce(tParticle *p, float dt)
225 {
226  if (! p->alive) return; // dead
227 
228  // since we know it is the ground plane, we only need to calculate s for a single dimension.
229  float s = -p->prev[2] / p->vel[2];
230 
231  p->pos[0] = (p->prev[0] + p->vel[0] * s + p->vel[0] * (dt-s) * p->damp);
232  p->pos[1] = (p->prev[1] + p->vel[1] * s + p->vel[1] * (dt-s) * p->damp);
233  p->pos[2] = (p->prev[2] + p->vel[2] * s + p->vel[2] * (dt-s) * p->damp);
234 
235  // damp the reflected velocity (since the particle hit something, it lost some energy)
236  p->vel[0] *= p->damp;
237  p->vel[1] *= p->damp;
238  p->vel[2] *= -p->damp; // reflect
239 }

246 void Particle::timestep(tParticle *p, float dt)
247 {
248  if (! p->alive) return; // dead
249 
250  p->vel[0] += 0;
251  p->vel[1] += 0;
252  p->vel[2] -= 2*dt; // slow down
253  p->prev[0] = p->pos[0];
254  p->prev[1] = p->pos[1];
255  p->prev[2] = p->pos[2];
256  if ((p->pos[2] + p->vel[2] * dt) > (ground + 0.005)) {
257  p->pos[0] += p->vel[0] * dt * speed/2;
258  p->pos[1] += p->vel[1] * dt * speed/2;
259  p->pos[2] += p->vel[2] * dt * speed/2;
260  }
261 }
