134 float Ase::getRadius()
135 {
136  double max_radius = 0.0;
137 
138  for (int i=0; i < ASEModel.numOfObjects; i++) {
139  if (ASEModel.pObject.size() <= 0) break;
140  tASEObject *pObject = &ASEModel.pObject[i];
141  for (int j=0; j < pObject->numOfFaces; j++)
142  for (int whichVertex = 0; whichVertex < 3; whichVertex++) {
143  int vertIndex = pObject->pFaces[j].vertIndex[whichVertex];
144  double r= pObject->pVerts[vertIndex].x * pObject->pVerts[vertIndex].x +
145  pObject->pVerts[vertIndex].y * pObject->pVerts[vertIndex].y +
146  pObject->pVerts[vertIndex].z * pObject->pVerts[vertIndex].z ;
147  if (r > max_radius) max_radius = r;
148  }
149  }
150  return sqrt(max_radius);
151 }

206 void Ase::draw()
207 {
208  if (! loaded) return;
209 
210  if (currentScale != desiredScale) setScale(desiredScale);
211 
212  for (int i=0; i < ASEModel.numOfObjects; i++) {
213  if (ASEModel.pObject.size() <= 0) break;
214 
215  tASEObject *pObject = &ASEModel.pObject[i]; // get the current object
216 
217  if (pObject->bHasTexture) {
218  glColor3ub(255, 200, 200); // the color to normal again
219 
220  bindTexture2D(ASEModel.pMaterials[pObject->materialID].texureId); // Bind the texture
221  }
222  else {
223  glDisable(GL_TEXTURE_2D);
224  glEnable(GL_COLOR);
225  glEnable(GL_COLOR_MATERIAL);
226  glColor3ub(255, 200, 200);
227  }
228 
229  glBegin(GL_TRIANGLES); // Begin drawing with our selected mode
230 
231  for (int j=0; j < pObject->numOfFaces; j++) {
232  for (int whichVertex=0; whichVertex < 3; whichVertex++) {
233  int vi = pObject->pFaces[j].vertIndex[whichVertex];
234 
235 #if 1 //DAX
236  if (vi < 0) continue; //BUG: segfault
237 #endif
238  glNormal3f(pObject->pNormals[ vi ].x, pObject->pNormals[ vi ].y, pObject->pNormals[ vi ].z);
239 
240  if (pObject->bHasTexture) {
241  if (pObject->pTexVerts) {
242  int coordIndex = pObject->pFaces[j].coordIndex[whichVertex];
243 
244  glTexCoord2f(pObject->pTexVerts[ coordIndex ].x, pObject->pTexVerts[ coordIndex ].y);
245  }
246  }
247  else {
248  if (ASEModel.pMaterials.size() && pObject->materialID >= 0) {
249  float *pColor = ASEModel.pMaterials[pObject->materialID].fColor;
250 
251  glColor3f(pColor[0], pColor[1], pColor[2]);
252  }
253  }
254  glVertex3f(pObject->pVerts[ vi ].x, pObject->pVerts[ vi ].y, pObject->pVerts[ vi ].z);
255  }
256  }
257  glEnd(); // End the drawing
258  glColor3f(0,0,0);
259  glDisable(GL_TEXTURE_2D);
260  glDisable(GL_COLOR_MATERIAL);
261  }
262 }

277 void Ase::readFile(tASEModel *pModel)
278 {
279  tASEMaterialInfo newMaterial; // This will be used to push on a new material
280  tASEObject newObject = {0}; // This will be used to push on a new object
281 
282  // This function is the head for reading in the .ase data and information.
283  // We count the objects then we count the materials.
284  // Then for each material and each object we read the data in.
285 
286  // This will return the number of objects stored in the .ase file
287  pModel->numOfObjects = getObjectCount();
288 
289  // This will return the number of materials stored in the .ase file
290  pModel->numOfMaterials = getMaterialCount();
291 
292  // Go through all the materials and fill in their data and info
293  for (int i=0; i < pModel->numOfMaterials; i++) {
294  // Add a new material to our list of materials using the STL "vector" class
295  pModel->pMaterials.push_back(newMaterial);
296 
297  // Get the material info for the current material. We add 1 because we
298  // want to start at 1 and i starts at 0.
299  getTextureInfo(&(pModel->pMaterials[i]), i + 1);
300  }
301 
302  // Go through all the objects and fill in their data and info
303  for (int i=0; i < pModel->numOfObjects; i++) {
304  // Add a new object to our list of objects using the STL "vector" class
305  pModel->pObject.push_back(newObject);
306 
307  // Set the material ID to -1 to initialize it. This will be changed
308  // if there is a texture/material assigned to this object.
309  pModel->pObject[i].materialID = -1;
310 
311  // Move the file pointer to the desired object. We add one because our
312  // object count starts at 1 and i starts at 0
313  moveToObject(i + 1);
314 
315  // Find out the number of vertices, faces and texture coordinates for this object,
316  // then allocate the memory needed to store that amount that needs to be read in.
317  readObjectInfo(&(pModel->pObject[i]), i + 1);
318 
319  // Read the vertices, face indices and texture coordinates for this object
320  readObjectData(pModel, &(pModel->pObject[i]), i + 1);
321  }
322 }

360 void Ase::getTextureInfo(tASEMaterialInfo *pTexture, int desiredMaterial)
361 {
362  char line[255] = {0};
363  int materialCount= 0;
364 
365  // In this function we check for an MATERIAL tag, then see if it's the
366  // material we want by the desiredMaterial number. If so we start reading it in.
367  rewind(fp);
368  while (!feof(fp)) {
369  fscanf(fp, "%s", line);
370  if (!strcmp(line, MATERIAL)) {
371  materialCount++;
372  // Check if it's the one we want to stop at
373  if (materialCount == desiredMaterial) break;
374  }
375  else fgets(line, sizeof(line), fp);
376  }
377 
378  // Go through the rest of the file until we hit the end
379  while (!feof(fp)) {
380  fscanf(fp, "%s", line);
381 
382  // If we found a MATERIAL tag stop because we went to far
383  if (!strcmp (line, MATERIAL)) {
384  // We hit a new material, so we need to stop
385  return;
386  }
387  // If we hit a MATERIAL_COLOR tag, we need to get the material's color
388  else if (!strcmp(line, MATERIAL_COLOR)) {
389  // Get the material RGB color of the object
390  fscanf(fp, " %f %f %f", &(pTexture->fColor[0]),
391  &(pTexture->fColor[1]),
392  &(pTexture->fColor[2]));
393  }
394  // If we hit a TEXTURE tag, we need to get the texture's name
395  else if (!strcmp(line, TEXTURE)) {
396  // Get the file name of the texture
397  getTextureName(pTexture);
398  }
399  // If we hit a MATERIAL_NAME tag, we need to get the material's name
400  else if (!strcmp(line, MATERIAL_NAME)) {
401  // Get the material name of the object
402  getMaterialName(pTexture);
403  }
404  // If we hit a UTILE tag, we need to get the U tile ratio
405  else if (!strcmp(line, UTILE)) {
406  // Read the U tiling for the U coordinates of the texture
407  pTexture->uTile = readFloat();
408  }
409  // If we hit a VTILE tag, we need to get the V tile ratio
410  else if (!strcmp(line, VTILE)) {
411  // Read the V tiling for the V coordinates of the texture
412  pTexture->vTile = readFloat();
413  }
414  // Otherwise ignore the data and read past it
415  else fgets(line, sizeof(line), fp);
416  }
417 }

420 void Ase::moveToObject(int desiredObject)
421 {
422  char line[255] = {0};
423  int cnt = 0;
424 
425  // takes the number of the desired object to move to.
426  // We read through the file from the beginning and increase a counter.
427  // When that counter gets to the desiredObject we then stop. Now the file
428  // pointer can read the object data from that specific object
429 
430  rewind(fp);
431  while(!feof(fp)) {
432  fscanf(fp, "%s", line);
433  if (!strcmp(line, OBJECT)) {
434  cnt++;
435  // Check if it's the one we want to stop at, if so stop reading
436  if (cnt == desiredObject)
437  return;
438  }
439  else fgets(line, sizeof(line), fp);
440  }
441 }

453 void Ase::readObjectInfo(tASEObject *pObject, int desiredObject)
454 {
455  char line[255] = {0};
456 
457  // In this function we position the file pointer to the object we want
458  // to read in, then we read until we find the associated tags, then read the info.
459 
460  // Go to the desired object to read from in the file
461  moveToObject(desiredObject);
462 
463  while (!feof(fp)) {
464  fscanf(fp, "%s", line);
465  if (!strcmp(line, NUM_VERTEX)) {
466  fscanf(fp, "%d", &pObject->numOfVerts);
467  pObject->pVerts = new Vec3[pObject->numOfVerts];
468  }
469  else if (!strcmp(line, NUM_FACES)) {
470  // Read in the number of faces for this object
471  fscanf(fp, "%d", &pObject->numOfFaces);
472  pObject->pFaces = new tASEFace [pObject->numOfFaces];
473  }
474  else if (!strcmp(line, NUM_TVERTEX)) {
475  // Read in the number of texture coordinates for this object
476  fscanf(fp, "%d", &pObject->numTexVertex);
477  pObject->pTexVerts = new Vec2[pObject->numTexVertex];
478  }
479  // If we hit the object tag we want to stop cause we went to far
480  else if (!strcmp(line, OBJECT)) {
481  // Return if we get to the next object
482  return;
483  }
484  else {
485  //We didn't find anything we want to check for so we read to the next line
486  fgets(line, sizeof(line), fp);
487  }
488  }
489 }

506 void Ase::readObjectData(tASEModel *pModel, tASEObject *pObject, int desiredObject)
507 {
508  // This will go through the file every time for each data tag and read
509  // in the data for that tag. We pass in the desired object to read it from.
510 
511  // Load the material ID for this object
512  getData(pModel, pObject, (char*)MATERIAL_ID, desiredObject);
513 
514  // Load the vertices for this object
515  getData(pModel, pObject, (char*)VERTEX, desiredObject);
516 
517  // Load the texture coordinates for this object
518  getData(pModel, pObject, (char*)TVERTEX, desiredObject);
519 
520  // Load the vertex faces list for this object
521  getData(pModel, pObject, (char*)FACE, desiredObject);
522 
523  // Load the texture face list for this object
524  getData(pModel, pObject, (char*)TFACE, desiredObject);
525 
526  // Load the texture for this object
527  getData(pModel, pObject, (char*)TEXTURE, desiredObject);
528 
529  // Load the U tile for this object
530  getData(pModel, pObject, (char*)UTILE, desiredObject);
531 
532  // Load the V tile for this object
533  getData(pModel, pObject, (char*)VTILE, desiredObject);

537 void Ase::getData(tASEModel *pModel, tASEObject *pObject, char *strDesiredData, int desiredObject)
538 {
539  char line[255] = {0};
540 
541  // In here, we move the file pointer to the desired object we want to read in,
542  // then we query all the tags that are being read in. If we find a valid tag,
543  // then we need to check if that's the tag that we want to read in. If so,
544  // then we read it in.
545 
546  // Move the file pointer to the desired object
547  moveToObject(desiredObject);
548 
549  while (! feof(fp)) {
550  fscanf(fp, "%s", line);
551 
552  if (!strcmp(line, OBJECT)) { return; }
553  // If we hit a vertex tag
554  else if (!strcmp(line, VERTEX)) {
555  if (!strcmp(strDesiredData, VERTEX)) {
556  readVertex(pObject);
557  }
558  }
559  // If we hit a texture vertex
560  else if (!strcmp(line, TVERTEX)) {
561  if (!strcmp(strDesiredData, TVERTEX)) {
562  readTextureVertex(pObject, pModel->pMaterials[pObject->materialID]);
563  }
564  }
565  // If we hit a vertice index to a face
566  else if (!strcmp(line, FACE)) {
567  if (!strcmp(strDesiredData, FACE)) {
568  readFace(pObject);
569  }
570  }
571  // If we hit a texture index to a face
572  else if (!strcmp(line, TFACE)) {
573  if (!strcmp(strDesiredData, TFACE)) {
574  readTextureFace(pObject);
575  }
576  }
577  // If we hit the material ID to the object
578  else if (!strcmp(line, MATERIAL_ID)) {
579  if (!strcmp(strDesiredData, MATERIAL_ID)) {
580  pObject->materialID = (int)readFloat();
581  return;
582  }
583  }
584  else fgets(line, sizeof(line), fp);
585  }
586 }

589 void Ase::readVertex(tASEObject *pObject)
590 {
591  int index = 0;
592 
593  fscanf(fp, "%d", &index); // read past the vertex index
594 
595  // Here we read in the vertice's (X, Y, Z). Since 3D Studio Max has Z pointing
596  // up, we are going to swap Y and Z to make it normal. Otherwise we would need
597  // to rotate it around the X axis by -90deg: glRotatef(-90, 1.0f, 0.0f, 0.0f);
598  fscanf(fp, "%f %f %f", &pObject->pVerts[index].x,
599  &pObject->pVerts[index].z,
600  &pObject->pVerts[index].y);
601 
602  // Also in 3D studio max the Z goes out of the screen so we want to swap that.
603  // We do that by negating the Z value of each vertex.
604  pObject->pVerts[index].z = -pObject->pVerts[index].z;
605 }

608 void Ase::readTextureVertex(tASEObject *pObject, tASEMaterialInfo texture)
609 {
610  int index = 0;
611 
612  fscanf(fp, "%d", &index); // read the index of the texture coordinate
613 
614  // Next, we read in the (U, V) texture coordinates.
615  fscanf(fp, "%f %f", &(pObject->pTexVerts[index].x), &(pObject->pTexVerts[index].y));
616 
617  // What is being done here is we are multiplying a X and Y tile factor
618  // to the UV coordinate. Usually the uTile and vTile is 1, but if it depends on
619  // your UVW map. If you made the texture tile more along the object, these would change.
620  pObject->pTexVerts[index].x *= texture.uTile;
621  pObject->pTexVerts[index].y *= texture.vTile;
622 
623  // We know this object has a texture so let's set this true
624  pObject->bHasTexture = true;
625 }

642 void Ase::readTextureFace(tASEObject *pObject)
643 {
644  int index = 0;
645 
646  fscanf(fp, "%d:", &index);
647 
648  // Now we read in the UV coordinate index for the current face.
649  // This will be an index into pTexCoords[] for each point in the face.

656 void Ase::computeNormals(tASEModel *pModel)
657 {
658  Vec3 v1, v2, vNormal, vPoly[3];
659 
660  if (pModel->numOfObjects <= 0) return;
661 
662  // Go through each of the objects to calculate their normals
663  for (int index = 0; index < pModel->numOfObjects; index++) {
664  // Get the current object
665  tASEObject *pObject = &(pModel->pObject[index]);
666  int i;
667 
668  // Here we allocate all the memory we need to calculate the normals
669  Vec3 *pNormals = new Vec3 [pObject->numOfFaces];
670  Vec3 *pTempNormals = new Vec3 [pObject->numOfFaces];
671  pObject->pNormals = new Vec3 [pObject->numOfVerts];
672 
673  for (i=0; i < pObject->numOfFaces; i++) { // go though all of the faces
674  // to cut down LARGE code, we extract the 3 points of this face
675  vPoly[0] = pObject->pVerts[pObject->pFaces[i].vertIndex[0]];
676  vPoly[1] = pObject->pVerts[pObject->pFaces[i].vertIndex[1]];
677  vPoly[2] = pObject->pVerts[pObject->pFaces[i].vertIndex[2]];
678 
679  // calculate the face normals (Get 2 vectors and find the cross product of those 2)
680  v1 = Vector(vPoly[0], vPoly[2]); // Get the vector of the polygon (we just need 2 sides for the normal)
681  v2 = Vector(vPoly[2], vPoly[1]); // Get a second vector of the polygon
682  vNormal = Cross(v1, v2); // Return the cross product of the 2 vectors (normalize vector, but not a unit vector)
683  pTempNormals[i] = vNormal; // Save the un-normalized normal for the vertex normals
684  vNormal = Normalize(vNormal); // Normalize the cross product to give us the polygons normal
685  pNormals[i] = vNormal; // Assign the normal to the list of normals
686  }
687 
688  // Now Get The Vertex Normals //
689  Vec3 vSum = {0.0, 0.0, 0.0};
690  Vec3 vZero = vSum;
691  int shared=0;
692 
693  for (i=0; i < pObject->numOfVerts; i++) { // go through all of the vertices
694  for (int j=0; j < pObject->numOfFaces; j++) { // go through all of the triangles
695  // check if the vertex is shared by another face
696  if (pObject->pFaces[j].vertIndex[0] == i ||
697  pObject->pFaces[j].vertIndex[1] == i ||
698  pObject->pFaces[j].vertIndex[2] == i) {
699  vSum = AddVector(vSum, pTempNormals[j]);// Add the un-normalized normal of the shared face
700  shared++; // Increase the number of shared triangles
701  }
702  }
703  // get the normal by dividing the sum by the shared. We negate the shared so it has the normals pointing out.
704  pObject->pNormals[i] = DivideVectorByScaler(vSum, float(-shared));
705  pObject->pNormals[i] = Normalize(pObject->pNormals[i]);
706  vSum = vZero; // Reset the sum
707  shared = 0; // Reset the shared
708  }
709  if (pTempNormals) delete[] pTempNormals;
710  if (pNormals) delete[] pNormals;
711  }
712 }
