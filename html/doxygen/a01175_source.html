209 void Lwo::httpReader(void *alwo, Http *http)
210 {
211  Lwo *lwo = (Lwo *) alwo;
212  if (! lwo) return;
213 
214  FILE *f = Cache::openCache(lwourl, http);
215  if (! f) { error("lwo: can't open %s", lwourl); return; }
216 
217  /* check for headers */
218  if (File::read_long(f) != ID_FORM) { error("lwoReader: error ID_FORM"); return; }
219  int32_t form_bytes = File::read_long(f);
220  int32_t read_bytes = 4;
221  if (File::read_long(f) != ID_LWOB) { error("lwoReader: not a LWOB file"); return; }
222 
223  /* read chunks */
224  while (read_bytes < form_bytes) {
225  int32_t chunk = File::read_long(f);
226  int32_t nbytes = File::read_long(f);
227  read_bytes += 8 + nbytes + nbytes%2;
228  switch (chunk) {
229  case ID_PNTS: lwo->readPnts(f, nbytes); break;
230  case ID_POLS: lwo->readPols(f, nbytes); break;
231  case ID_SRFS: lwo->readSrfs(f, nbytes); break;
232  case ID_SURF: lwo->readSurf(f, nbytes); break;
233  default: File::skip_byte(f, nbytes + nbytes%2);
234  }
235  }
236  File::closeFile(f);
237  return;
238 }
