110 /* Created from file */
111 void Hat::parser(char *l)
112 {
113  defaults();
114  l = tokenize(l);
115  begin_while_parse(l) {
116  l = parse()->parseAttributes(l, this); // <solid ... />
117  if (!l) break;
118  if (!stringcmp(l, "model=")) {
119  l = parse()->parseString(l, modelname, "model");
120  if (!stringcmp(modelname, "tophat")) model = TOPHAT;
121  else if (!stringcmp(modelname, "bowler")) model = BOWLER;
122  else if (!stringcmp(modelname, "chinese")) model = CHINESE;
123  else if (!stringcmp(modelname, "cardinal")) model = CARDINAL;
124  else if (!stringcmp(modelname, "country")) model = COUNTRY;
125  else if (!stringcmp(modelname, "clown")) model = CLOWN;
126  }
127  }
128  end_while_parse(l);
129 }
