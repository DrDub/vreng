114 /* system of equations handling permanent motion */
115 void Sun::changePermanent(float lasting)
116 {
117  float ang = DEG2RAD(light_rot);
118  float sina = sin(ang);
119  float cosa = cos(ang);
120 
121  light_diff[0] = MAX(0.9*(1+sina), 0); //1
122  light_diff[1] = 0.7 + 0.7*sina; //MAX(sina, 0); //0.5 + 0.5*sina;
123  light_diff[2] = 0.2 + 0.2*sina;
124  light_spec[0] = 1;
125  light_spec[1] = MAX(sina, 0);
126  light_spec[2] = MAX(sina, 0);
127  light_rot += 0.005; // in reallife: 360/86400
128 
129  pos.x = ox * cosa;
130  pos.y = oy * cosa;
131  pos.z = ox * sina;
132  //trace(DBG_FORCE, "%.1f %.1f %.1f %.1f", pos.x,pos.y,pos.z,ang);
133  if (flare) flare->setColor(light_diff);
134 }
