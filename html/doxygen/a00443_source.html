37 static float hrnd(float n) { return (n*rand())/RAND_MAX; }

39 static float nrnd(float d) // normal distribution randomizer
40 {
41  float r,v1,v2;
42  do {
43  v1=2*hrnd(1.)-1.;
44  v2=2*hrnd(1.)-1.;
45  r=v1*v1+v2*v2;
46  } while (r >= 1. || r == 0.);
47  r = sqrt(-2.*log(r)/r);
48  return v1*d*r;
49 }

201 void Fire::changePermanent(float dt)
202 {
203  struct sParticle *p;
204 
205  time += dt;
206 
207  if (anim) {
208  Vector3 a(0,0,1);
209  float da = pow(0.2, dt);
210 
211  p = particles;
212  for (int n=0; n < np; n++, p++) {
213  if ((p->t -= dt) <= 0) // kill it
214  *p = particles[--np];
215  else { // animate it
216  p->v += a * dt;
217  p->p += p->v * dt;
218  p->s.x = .67*FIRESIZE + .40*FIRESIZE*sin(M_PI*(p->t)/FIRELIFE);
219  p->s.z = .05*FIRESIZE + .77*FIRESIZE*(1-p->t/FIRELIFE);
220  p->a *= da;
221  }
222  }
223  while (np < FIREMAX && (time - lasttime >= FIREDELTA)) {
224  Vector3 dir(0,0,.25);
225  float f;
226  lasttime += FIREDELTA;
227  p = particles + np++;
228  do { f = nrnd(.2); } while (ABSF(f) > .45);
229  p->p = src + cyl[rand()%5].d * f;
230  p->v = dir;
231  p->t = lasttime + FIRELIFE - time;
232  p->a = FIREALPHA;
233  p->s.y = FIRESIZE;
234  p->s.x = .67*FIRESIZE + .40*FIRESIZE*sin(M_PI*(p->t)/FIRELIFE);
235  p->s.z = .05*FIRESIZE + .77*FIRESIZE*(1-p->t/FIRELIFE);
236  }
237  }
238  else lasttime += dt;
239 
240  Vector3 v, o;
241  Matrix3 m;
242  m.apply(src, &o);
243  o = src - o;
244  p = particles;
245  for (int n=0; n < np; n++, p++) {
246  m.apply(p->p, &v);
247  v += o;
248  p->ex = width * projx(v); //0.25
249  p->ey = height * projy(v); //0.75
250  projz(p->ez, v);
251  p->dx = width * p->s.x/v.y;
252  p->dy = height * p->s.z/v.y;
253  }
254 
255 #if 0 //dax
256  for (int n=0; n<4; n++) {
257  Quad fground;
258  Vector3 vg = src;
259  vg.y -= 2*FIRECYL;
260  if ((n+1)&2) vg += ds1; else vg -= ds1;
261  if ( n &2) vg += ds2; else vg -= ds2;
262  m.apply(vg, &v);
263  v += o;
264  fground.v[n].ex = projx(v);
265  fground.v[n].ey = projy(v);
266  projz(fground.v[n].ez, v);
267  }
268 #endif
269 }
