68 void Icon::changePosition(float lasting)
69 {
70  if (! taken) {
71  float ratio = MAX(::g.times.getRate() / 40., 1);
72  pos.x += lasting * move.lspeed.v[0] * ratio;
73  pos.y += lasting * move.lspeed.v[1] * ratio;
74  pos.z += lasting * move.lspeed.v[2] * ratio;
75  pos.az += lasting * move.aspeed.v[0] * ratio;
76  pos.ax += lasting * move.aspeed.v[2] * ratio;
77  }
78  else {
79  float off = 0.4;
80  pos.x = localuser->pos.x + off * Cos(localuser->pos.az);
81  pos.y = localuser->pos.y + off * Sin(localuser->pos.az);
82  pos.z = localuser->pos.z + 0.6;
83  pos.az = localuser->pos.az + M_PI_2;
84  }
85 }

129 {

132  ptok++;

144 Icon::Icon(User *user, void *d)
145 {
146  char *infos = (char *) d;
147  ifile = NULL;
148  ofile = NULL;
149  char *action = NULL;
150  vref = NULL;
151  char icon[URL_LEN];
152  *icon = 0;
153 
154  /* parameters transmission */
155  for (char *pt = strtok(infos, "&"); pt ; pt = strtok(NULL, "&")) {
156  if (! stringcmp(pt, "<url=")) { pt = getParam(pt); strcpy(names.url, pt); taken = true; }
157  else if (! stringcmp(pt, "<file=")) { pt = getParam(pt); ifile = strdup(pt); }
158  else if (! stringcmp(pt, "<ofile=")) { pt = getParam(pt); strcpy(names.url, pt); }
159  else if (! stringcmp(pt, "<name=")) { pt = getParam(pt); strcpy(names.named, pt); }
160  else if (! stringcmp(pt, "<icon=")) { pt = getParam(pt); strcpy(icon, pt); }
161  else if (! stringcmp(pt, "<action=")) { pt = getParam(pt); action = strdup(pt); }
162  else if (! stringcmp(pt, "<vref=")) {
163  pt = strchr(pt, '=');
164  pt++;
165  char *p2;
166  if ((p2 = strchr(pt, '>'))) *p2 = 0;
167  vref = strdup(pt);
168  }
169  }
170 
171  if (vref) { // local saved document exists
172  parser(vref);
173 
174  // get the last loaded texture
175  int texid = Texture::getIdByUrl(names.url);
176  Texture *tclast = Texture::getEntryById(texid);
177  if (! tclast) {
178  tex = new char[sizeof(ICO_DEF) + 1];
179  strcpy(tex, ICO_DEF); // default texture
180  }
181  else {
182  tex = new char[strlen(tclast->url) + 1];
183  strcpy(tex, tclast->url);
184  }
185  taken = false;
186  }
187 
188  else { // new document named interactively by hand
189  /* position */
190  float off = 0.4;
191  pos.x = user->pos.x + off * Cos(user->pos.az);
192  pos.y = user->pos.y + off * Sin(user->pos.az);
193  pos.z = user->pos.z + 0.6; // visible by eyes
194  pos.az = user->pos.az + M_PI_2;
195 
196  /* texture */
197  if (*icon) {
198  tex = new char[strlen(icon) + 1];
199  strcpy(tex, icon);
200  }
201  else {
202  // default binding icon to document
203  char ext[8] = "";
204  memset(ext, 0, sizeof(ext));
205  if (Format::getExt(names.url, ext)) {
206  tex = new char[URL_LEN];
207  Format::getImgByExt(ext, tex);
208  }
209  else {
210  tex = new char[sizeof(ICO_DEF) + 1];
211  strcpy(tex, ICO_DEF);
212  }
213  }
214 
215  if (ifile) { // private local document
216  if (*names.url) {
217  // public url given by user
218  Cache::download(ifile, ofile, "inout");
219  }
220  else {
221  // build local ofile in ~/public_html/vreng/
222  ofile = new char[URL_LEN];
223  sprintf(ofile, "%s/public_html", getenv("HOME"));
224  if (access(ofile, R_OK|W_OK|X_OK) == 0) {
225  strcat(ofile, "/vreng/");
226  if (access(ofile, R_OK|W_OK|X_OK) == -1)
227  mkdir(ofile, 0755);
228  strcat(ofile, ifile);
229  FILE *fin, *fout;
230  if ((fin = File::openFile(ifile, "r")) && (fout = File::openFile(ofile, "w"))) {
231  char buf[2];
232  while (fread(buf, 1, 1, fin))
233  fwrite(buf, 1, 1, fout);
234  File::closeFile(fin);
235  File::closeFile(fout);
236  chmod(ofile, 0644);
237 
238  //FIXME: define local http_server
239  sprintf(names.url, "http://%s/~%s/vreng/%s", DEF_HTTP_SERVER, getenv("USER"), ifile);
240  }
241  else {
242  error("can't open %s or %s: %s (%d)", ifile, ofile, strerror(errno), errno);
243  free(ifile); ifile = NULL;
244  delete[] ofile;
245  }
246  }
247  else {
248  error("can't access %s", ofile);
249  free(ifile); ifile = NULL;
250  delete[] ofile;
251  }
252  }
253  }
254  makeSolid();
255  }
256 
257  // local creation
258  defaults();
259 
260  enableBehavior(REMOVABLE);
261  enableBehavior(NO_ELEMENTARY_MOVE);
262  setRenderPrior(RENDER_HIGH);
263 
264  initializeMobileObject(1);
265  ttl = (taken) ? MAXFLOAT : 0;
266  initImposedMovement(ttl);
267  disablePermanentMovement();
268 
269  // network creation
270  createVolatileNetObject(PROPS);
271 
272  // document's owner
273  setOwner();
274 
275  trace(DBG_WO, "Icon: url=%s icon=%s name=%s owner=%s", urlName(), tex, getInstance(), ownerName());
276 
277  if (action) {
278  if (! stringcmp(action, "pin")) pin(this, NULL, 0L, 0L);
279  else if (! stringcmp(action, "push")) push(this, NULL, 0L, 0L);
280  else if (! stringcmp(action, "carry")) carry(this, NULL, 0L, 0L);
281  }
282 }

310 bool Icon::whenIntersect(WObject *pcur, WObject *pold)
311 {
312  switch (pcur->type) {
313  case WALL_TYPE:
314  pos.x += pcur->pos.x - pold->pos.x;
315  pos.y += pcur->pos.y - pold->pos.y;
316  pos.z += pcur->pos.z - pold->pos.z;
317  pos.az += pcur->pos.az - pold->pos.az;
318  pos.ax += pcur->pos.ax - pold->pos.ax;
319  updatePositionAndGrid(pold);
320  break;
321  case USER_TYPE:
322  projectPosition(pcur, pold);
323  break;
324  case ICON_TYPE:
325  pcur->pos.x += 0.1;
326  pcur->pos.y += 0.1;
327  pcur->updatePositionAndGrid(pold);
328  break;
329  default:
330  pold->copyPositionAndBB(pcur);
331  }
332  return true;
333 }
