181 {
182  Source *pso, *psolast;
183  int i = 0;
184 
185  for (psolast = pso = source; pso && (i < nbsources); pso = pso->next, i++) {
186  if (pso->ssrc == _ssrc) {
187  if (psolast == NULL) // no source found
188  source = NULL;
189  else {
190  // MS : this is a NOOP if pso == source
191  // Bad things happen after that
192  // psolast->next = pso->next;
193  if (pso == source)
194  source = pso->next;

197  }
198  setLostPackets(pso->lost);
199  nbsources = Source::decrSourcesNumber();
200  trace(DBG_RTP, "nbsources--=%d", nbsources);
201  delete pso; // delete Source

203  break;

344 int Session::buildSDES(rtcp_common_t *prtcp_hdr, uint8_t* pkt)
345 {
346  uint8_t xitem = 0;
347  int len = 0;
348  uint8_t items[PKTSIZE-sizeof(rtcp_sr_t)-sizeof(rtcp_common_t)-sizeof(ssrc)];
349  SdesItem *pchunk;
350 
351  buildRTCPcommon(prtcp_hdr, RTCP_SDES);
352  memset(items, 0, sizeof(items));
353  prtcp_hdr->count = 1; // Only one SDES with multiple chunks
354 
355  /* fill chunks */
356  for (pchunk = mysdes; pchunk; pchunk = pchunk->si_next) {
357  items[xitem++] = pchunk->si_type;
358  items[xitem++] = pchunk->si_len;
359  memcpy(items+xitem, pchunk->si_str, pchunk->si_len);
360  xitem += pchunk->si_len;
361  }
362  items[xitem++] = RTCP_SDES_END; // end of chunks is indicated by a zero
363 
364  while (xitem % 4) // Pad to 4 bytes word boundary and store zeros there
365  items[xitem++] = 0;
366 
367  /* SDES header */
368  prtcp_hdr->length = htons(1 + (xitem >> 2));
369  memcpy(pkt, prtcp_hdr, sizeof(rtcp_common_t));
370  len += sizeof(rtcp_common_t);
371  memcpy(pkt+len, &ssrc, sizeof(ssrc));
372  len += sizeof(ssrc);
373  memcpy(pkt+len, items, xitem);
374  len += xitem;
375 
376  trace(DBG_RTP, "setSDES: ssrc=%x len=%d xitem=%d", ssrc, len, xitem);
377  return len;
378 }

401 int Session::sendRTCPPacket(const struct sockaddr_in *to, uint8_t pt)
402 {
403  int pkt_len = 0, r, sd;
404  rtcp_common_t rtcp_hdr;
405  struct sockaddr_in *sin_rtcp;
406  uint8_t pkt[PKTSIZE];
407 
408  switch (pt) {
409  case RTCP_SR: // send a SR
410  pkt_len = buildSR(&rtcp_hdr, pkt);
411  break;
412  case RTCP_SDES: // send a SDES
413  pkt_len = buildSDES(&rtcp_hdr, pkt);
414  break;
415  case RTCP_BYE: // send a BYE
416  pkt_len = buildBYE(&rtcp_hdr, pkt);
417  break;
418  }
419  if ((sin_rtcp = Channel::getSaRTCP(to)) == NULL) {
420  error("sendRTCPPacket: sin_rtcp NULL"); return -1;
421  }
422  if ((sd = Channel::getFdSendRTCP(sin_rtcp)) < 0) {
423  error("sendRTCPPacket: sd <0"); return -1;
424  }
425 
426  r = Rtp::sendPacket(sd, pkt, pkt_len, sin_rtcp);
427  statSendRTCP(pkt_len);
428  return r;
429 }
