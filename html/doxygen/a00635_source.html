191 /* parse begin of line */
192 int Parse::parseLine(char *_line, int *ptag_type)
193 {
194  char *pline = strdup(_line);
195  char *ptok = skipSpace(pline);
196 
197  if (*ptok == '<') {
198  if (commented) {
199  FREE(pline);
200  return TAG_COMMENT;
201  }
202  ptok++; // after '<'
203 
204  // check <!--
205  if ((!stringcmp(ptok, "!--"))) {
206  commented = true;
207  FREE(pline);
208  return TAG_COMMENT;
209  }
210  else if ((!stringcmp(ptok, "!DOCTYPE"))) {
211  FREE(pline);
212  return TAG_DOCTYPE;
213  }
214 
215  // check <vre ... > </vre>
216  else if (!stringcmp(ptok, "vre>") || (!stringcmp(ptok, "vre") && isspace(ptok[3]))) {
217  FREE(pline);
218  return TAG_BEGINFILE;
219  }
220  else if (!stringcmp(ptok, "/vre>")) {
221  FREE(pline);
222  return TAG_ENDFILE;
223  }
224 
225  // check <head> </head>
226  else if ((!stringcmp(ptok, "head>"))) {
227  FREE(pline);
228  return TAG_HEAD;
229  }
230  else if ((!stringcmp(ptok, "/head>"))) {
231  FREE(pline);
232  return TAG_HEAD;
233  }
234 
235  // check <meta ... />
236  else if ((!stringcmp(ptok, "meta"))) {
237  FREE(pline);
238  return TAG_META;
239  }
240 
241  // check <scene> </scene>
242  else if (!stringcmp(ptok, "scene>") || (!stringcmp(ptok, "scene") && isspace(ptok[3]))) {
243  FREE(pline);
244  return TAG_SCENE;
245  }
246  else if ((!stringcmp(ptok, "/scene>"))) {
247  FREE(pline);
248  return TAG_SCENE;
249  }
250 
251  // check <local> </local>
252  else if ((!stringcmp(ptok, "local>"))) {
253  FREE(pline);
254  return TAG_LOCAL;
255  }
256  else if ((!stringcmp(ptok, "/local>"))) {
257  FREE(pline);
258  return TAG_LOCAL;
259  }
260 
261  else { // object description follows
262  const OClass *oclass = NULL;
263  char *q, c = 0;
264 
265  if ((q = nextSpace(ptok))) { // either ' ' or '>'
266  c = *q;
267  *q = 0;
268  }
269  if ((oclass = OClass::getOClass(ptok))) {
270  *ptag_type = oclass->type_id;
271  if (! OClass::isValidType(*ptag_type))
272  error("At line %d, *ptag_type=%d ptok=%s", numline, *ptag_type, ptok);
273  memset(tagobj, 0, sizeof(tagobj));
274  strncpy(tagobj, ptok, TAG_LEN-1); // memorize object tag
275  if (q) *q = c; // restore sep
276  FREE(pline);
277  return TAG_OBJECT;
278  }
279  FREE(pline);
280  return TAG_OBJECT;
281  }
282  }
283 
284  // not a '<'
285  if ((!stringcmp(ptok, "-->"))) {
286  FREE(pline);
287  return TAG_COMMENT;
288  }
289  if (commented) {
290  FREE(pline);
291  return TAG_COMMENT;
292  }
293  FREE(pline);
294  return TAG_OBJECT;
295 }

297 /* parse vre data, called by World::httpReader */
298 int Parse::parseVreLines(char *buf, int bufsiz)
299 {
300  int maxlen = 0; // length
301  char *tmpline = NULL;
302  static char *line = NULL;
303 
304  if (! line) { // need a new line
305  tmpline = new char[bufsiz];
306  memset(tmpline, 0, bufsiz);
307  maxlen = 0;
308  }
309  else { // old line: copy previous line in tmpline
310  maxlen = strlen(line);
311  tmpline = new char[bufsiz + maxlen];
312  memset(tmpline, 0, bufsiz);
313  strcpy(tmpline, line);
314  }
315 
316  // copy buf at the end of tmpline
317  memcpy(tmpline + maxlen, buf, bufsiz);
318  maxlen += bufsiz;
319 
320  int iol = 0; // index in line;
321  int bol = 0; // index begin of line
322  numline = 1; // line number
323 
324  while (1) {
325  int eol = 0; // index end of line
326 
327  for (iol = bol; iol < maxlen; iol++) {
328  if (tmpline[iol] == '\n') {
329  numline++;
330  if (tmpline[iol-1] == '\\') { // end of physical line, line follows
331  tmpline[iol-1] = tmpline[iol] = ' ';
332  }
333  else { // end of logical line
334  tmpline[iol] = ' '; // replace '\n' by ' '
335  eol = iol;
336  iol = maxlen + 1; // HUGLY! (to do maxlen + 2)
337  }
338  }
339  }
340 
341  if (iol == maxlen + 2) { // end of logical line
342  // allocate a new line
343  //FIXME segfault if (line) delete[] line;
344  line = new char[eol - bol + 2];
345 
346  // build the target line from tmpline
347  memcpy(line, tmpline + bol, eol - bol);
348  line[eol - bol] = '\0';
349  //trace(DBG_FORCE, "line: %s %d %d", line, eol, bol);
350 #ifdef WIN32 // crlf
351  if (line[eol - bol - 1] == '\r') line[eol - bol - 1] = '\0';
352 #endif // WIN32
353 
354  // discard empty lines
355  if (*line == '\0') {
356  if (line)
357  delete[] line;
358  line = NULL;
359  bol = eol + 1; // begin of next line
360  continue;
361  }
362 
363  char *p = skipSpace(line);
364 
365  // discard comments begining with a '#'
366  if (*p == '#') {
367  delete[] line;
368  bol = eol + 1; // begin of next line
369  continue;
370  }
371  //trace(DBG_FORCE, "line=%s", line);
372 
373  WObject *wobject = NULL;
374 
375  switch (parseLine(line, &tag_type)) {
376  case TAG_BEGINFILE:
377  case TAG_HEAD:
378  case TAG_SCENE:
379  if (line)
380  delete[] line;
381  line = NULL;
382  break;
383 
384  case TAG_DOCTYPE:
385  if (line)
386  delete[] line;
387  line = NULL;
388  break; // end of parsing
389 
390  case TAG_ENDFILE:
391  if (line)
392  delete[] line;
393  line = NULL;
394  return 0; // end of parsing
395 
396  case TAG_META:
397  if ((p = strstr(line, "=\"refresh\"")))
398  World::current()->setPersistent(false);
399  if ((p = strstr(line, "/>"))) {
400  if (line)
401  delete[] line;
402  line = NULL;
403  }
404  break;
405 
406  case TAG_COMMENT:
407  if ((p = strstr(line, "-->"))) {
408  commented = false;
409  if (line)
410  delete[] line;
411  line = NULL;
412  }
413  break;
414 
415  case TAG_LOCAL: // <local>, </local>
416  {
417  const OClass *oclass;
418 
419  if ((p = strstr(line, "-->"))) {
420  commented = false;
421  break;
422  }
423  strcpy(tagobj, "transform");
424  if ((oclass = OClass::getOClass(tagobj)))
425  tag_type = oclass->type_id;
426  if (strstr(line, "<local>")) strcpy(line, "push ");
427  else strcpy(line, "pop ");
428  //trace(DBG_FORCE, "LOCAL: type=%d line=%s", tag_type, line);
429  if ((wobject = OClass::creatorInstance(tag_type, line)) == NULL)
430  return -1;
431  }
432  break;
433 
434  case TAG_OBJECT:
435  //trace(DBG_FORCE, "type=%d line=%s", tag_type, line);
436  if ((p = strstr(line, "-->"))) {
437  commented = false;
438  break;
439  }
440 
441  // check end of the object </...>
442  char closetag[TAG_LEN +4];
443  sprintf(closetag, "</%s>", tagobj);
444  if ((p = strstr(line, closetag)) == NULL) continue; // not end of object
445  //trace(DBG_FORCE, "</TAG> %s", line);
446 
447  /*
448  * here, we must have reached the end of the object
449  */
450  if (tag_type != UNKNOWN_TYPE) {
451  if (! OClass::isValidType(tag_type)) {
452  error("At line %d, type=%d line=%s", numline, tag_type, line);
453  return -1;
454  }
455 
456  // skip <type to get attributes
457  char *attr = line;
458  if (isspace(*line))
459  attr = skipSpace(line);
460  if (*attr == '<')
461  attr = nextSpace(attr);
462  if (*attr == '>')
463  ++attr;
464 
465  if (::g.pref.dbgtrace) trace(DBG_FORCE, "[%d] %s", tag_type, line);
466 
467  /*
468  * call the class creator with object attributes
469  */
470  progression('o');
471  ::g.times.object.start();
472  if ((wobject = OClass::creatorInstance(tag_type, attr)) == NULL) {
473  error("At line %d, type=%d line=%s", numline, tag_type, line);
474  return -1;
475  }
476 #if 0 //dax
477  if (new_world > 1) ::g.render.render();
478 #endif
479  ::g.times.object.stop();
480  }
481  break;
482  }
483 
484  if (line)
485  delete[] line;
486  line = NULL;
487  bol = eol + 1; // begin of next line = end of current line + \n
488  }
489  else break; // next buffer
490  }
491 
492  if (line)
493  delete[] line;
494  line = NULL;
495 
496  line = new char[maxlen - bol + 2];
497 
498  // copy the end of previous tmpline into line
499  memcpy(line, tmpline + bol, maxlen - bol);
500  line[maxlen - bol] = '\0';
501 
502  delete[] tmpline;
503  return 1;
504 }

531  strcpy(geometry, ++p);

548  }
549  }

552  return strtok(l, SEP);

613 char * Parse::parsePosition(char *ptok, Pos &pos)
614 {
615  if (!stringcmp(ptok, "pos=")) ptok = skipEqual(ptok);
616  ptok = skipQuotes(ptok); // get pos.x
617  if (ptok) if (*ptok == 0) ptok = nextToken();
618  if (isFloat(ptok)) pos.x = (float) atof(ptok);
619  else return nextToken();
620 
621  ptok = skipComma(ptok); // get pos.y
622  if (isFloat(ptok)) pos.y = (float) atof(ptok);
623  else return nextToken();
624 
625  ptok = skipComma(ptok); // get pos.z
626  if (isFloat(ptok)) {
627  pos.z = (float) atof(ptok);
628  if (ptok[strlen(ptok) - 1] == '"') { // "x,y,z"
629  return nextToken();
630  }
631  }
632  else return nextToken();
633 
634  ptok = skipComma(ptok); // get pos.az or next token
635  if (!ptok) return nextToken();
636  if (isFloat(ptok)) {
637  pos.az = (float) atof(ptok);
638  if (ptok[strlen(ptok) - 1] == '"') { // "x,y,z,az"
639  return nextToken();
640  }
641  }
642  else // pos.az not float
643  return ptok;
644 
645  ptok = skipComma(ptok); // get pos.ax
646  if (!ptok) return nextToken();
647  if (isFloat(ptok)) {
648  pos.ax = (float) atof(ptok);
649  if (ptok[strlen(ptok) - 1] == '"') { // "x,y,z,az,ax"
650  return nextToken(); // normal end
651  }
652  }
653  else // pos.ax not float
654  if (!ptok) return ptok;
655  return ptok; //dummy
