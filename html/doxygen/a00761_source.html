116 Thing::Thing(World *pw, void *d, time_t s, time_t u)
117 {
118  char *str = (char *) d; // name transmitted
119  if (!str) return;
120 
121  char *p = str;
122  while (*p && !isdigit(*p)) p++;
123  int n = atoi(p);
124  if (n) oid = n;
125  strcpy(names.named, str);
126  strcpy(names.type, typeName()); // need names.type for MySql
127 
128  /* local creation */
129  defaults();
130 
131 #if HAVE_MYSQL
132  // we don't know anything about the geometry except from MySql
133  geometry = new char[BUFSIZ];
134  if (! psql)
135  psql = VRSql::getVRSql();
136  if (psql && explicitName()) {
137  psql->getGeom(this);
138  psql->getOwner(this);
139  psql->getPos(this);
140  }
141  if (geometry && isprint(*geometry)) {
142  parse()->parseSolids(geometry, SEP, this);
143  }
144  else warning("Thing: %s has no geometry", names.named);
145 #endif
146 
147  enableBehavior(DYNAMIC);
148  behavior();
149 }

196 bool Thing::whenIntersect(WObject *pcur, WObject *pold)
197 {
198  switch (pcur->type) {
199  case USER_TYPE:
200  // pushes the thing
201  move.lspeed.v[0] = (lspeed / 2) * Cos(pcur->pos.az);
202  move.lspeed.v[1] = (lspeed / 2) * Sin(pcur->pos.az);
203  initImposedMovement(1); // 1 sec
204  break;
205  default:
206  pold->copyPositionAndBB(pcur);
207  break;
208  }
209  return true;
210 }
