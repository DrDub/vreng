54 void Render::cameraProjection(GLfloat fovy, GLfloat near, GLfloat far)
55 {
56  GLint vp[4];
57  glGetIntegerv(GL_VIEWPORT, vp);
58 
59  GLint width = vp[2];
60  GLint height = vp[3];
61  GLfloat ratio = (GLfloat) width / (GLfloat) height;
62  GLfloat top = near * tan(fovy * M_PI_180);
63  GLfloat bottom = -top;
64  GLfloat right = top * ratio;
65  GLfloat left = -right;
66  cam_user.fovy = fovy;
67  cam_user.near = near;
68  cam_user.far = far;
69 
70  glMatrixMode(GL_PROJECTION);
71  glLoadIdentity();
72  glFrustum(left, right, bottom, top, near, far);
73  glTranslatef(0, 0, -near); // orig -0.4
74  glMatrixMode(GL_MODELVIEW);
75 }

99 void Render::cameraPosition()
100 {
101  M4 vr_mat;
102 
103  glMatrixMode(GL_MODELVIEW);
104  glDisable(GL_SCISSOR_TEST);
105  glPolygonMode(GL_FRONT_AND_BACK, GL_FILL); //default
106 
107  switch (view) {
108 
109  case VIEW_FIRST_PERSON:
110  vr_mat = MulM4(TranslateM4(0, 0, 0.15),
111  MulM4(RotateM4(pitch, UX), camera_pos));
112  break;
113 
114  case VIEW_THIRD_PERSON:
115  vr_mat = MulM4(TranslateM4(0, localuser->height/6, thirdPerson_Near-.8),
116  MulM4(RotateM4(M_PI_2/6 + thirdPerson_xRot + pitch, UX),
117  MulM4(RotateM4(thirdPerson_yRot, UY),
118  camera_pos)
119  )
120  );
121  break;
122 
123  case VIEW_THIRD_PERSON_FAR:
124  vr_mat = MulM4(TranslateM4(0, localuser->height/6, thirdPerson_Near-2.4),
125  MulM4(RotateM4(M_PI_2/6 + thirdPerson_xRot + pitch, UX),
126  MulM4(RotateM4(thirdPerson_yRot, UY),
127  camera_pos)
128  )
129  );
130  break;
131 
132  case VIEW_TURN_AROUND:
133  turnAround += (M_PI/18) / MAX(::g.times.getRate() / 5, 1);
134  vr_mat = MulM4(TranslateM4(0, localuser->height/4, -2),
135  MulM4(RotateM4(M_PI_2/4, UX),
136  MulM4(RotateM4(thirdPerson_yRot + turnAround, UY),
137  camera_pos)
138  )
139  );
140  break;
141 
142  case VIEW_VERTICAL:
143  vr_mat = MulM4(RotateM4(M_PI_2, UX),
144  MulM4(TranslateM4(0, -0.5, 0), // 50cm top
145  camera_pos)
146  );
147  break;
148 
149  case VIEW_VERTICAL_FAR:
150  vr_mat = MulM4(RotateM4(M_PI_2, UX),
151  MulM4(TranslateM4(0, -3, 0), // 3m top
152  camera_pos)
153  );
154  break;
155 
156  case VIEW_GROUND_LEVEL:
157  vr_mat = MulM4(TranslateM4(0, localuser->height - 0.10, 0), camera_pos);
158  break;
159 
160  case VIEW_WIRED:
161  glPolygonMode(GL_FRONT_AND_BACK, GL_LINE);
162  vr_mat = MulM4(RotateM4(pitch, UX), camera_pos);
163  break;
164 
165  case VIEW_SCISSOR:
166  vr_mat = camera_pos;
167  break;
168  }
169 
170  // transpose Vreng to OpenGl coordinates
171  GLfloat gl_mat[16];
172  M4toV16(&vr_mat, gl_mat);
173  glLoadMatrixf(gl_mat);
174 }

262 V3 Render::getVisiblePosition(WObject *po)
263 {
264  bool seen = false;
265  GLint vp[4];
266  GLdouble mview[16], mproj[16];
267 
268  ::g.gui.getScene()->getCoords(vp[0], vp[1], vp[2], vp[3]);
269  GLint w = vp[2];
270  GLint h = vp[3];
271 
272  GLfloat xa = po->pos.x;
273  GLfloat ya = po->pos.y;
274  GLfloat za = po->pos.z;
275  GLfloat dxa = po->pos.bbsize.v[0];
276  GLfloat dya = po->pos.bbsize.v[1];
277  GLfloat dza = po->pos.bbsize.v[2];
278 
279  glGetDoublev(GL_MODELVIEW_MATRIX, mview);
280  glGetDoublev(GL_PROJECTION_MATRIX, mproj);
281 
282  GLdouble x[5], y[5], z[5];
283  gluProject(xa , ya , za , mview, mproj, vp, &x[0],&y[0],&z[0]);
284  gluProject(xa+dxa, ya+dya, za+dza, mview, mproj, vp, &x[1],&y[1],&z[1]);
285  gluProject(xa-dxa, ya+dya, za+dza, mview, mproj, vp, &x[2],&y[2],&z[2]);
286  gluProject(xa+dxa, ya-dya, za-dza, mview, mproj, vp, &x[3],&y[3],&z[3]);
287  gluProject(xa-dxa, ya-dya, za-dza, mview, mproj, vp, &x[4],&y[4],&z[4]);
288 
289  for (int i=0; i<5; i++)
290  y[i] = vp[3] - y[i];
291 
292  for (int i=0; i<5; i++) {
293  if ((x[i] > w) || (x[i] < 0) || (y[i] > h) || (y[i] < 0)) continue;
294  uint16_t n = getBufferSelection((GLint) x[i], (GLint) y[i], 0);
295  WObject *o = WObject::byNum(n);
296  if (o && !strcasecmp(po->getInstance(), o->getInstance())) {
297  seen = true;
298  break;
299  }
300  }
301  return newV3(x[0], y[0], seen);
302 }

304 void Render::calculateFov(GLfloat posx, GLfloat posy, GLfloat posz, GLfloat rotz, char* filename)
305 {
306  GLint vp[4];
307  GLdouble mview[16], mproj[16];
308 
309  glGetIntegerv(GL_VIEWPORT, vp);
310  glGetDoublev(GL_MODELVIEW_MATRIX, mview);
311  glGetDoublev(GL_PROJECTION_MATRIX, mproj);
312 
313  GLint w = vp[2]; if (w & 1) ++w; //even values
314  GLint h = vp[3]; if (h & 1) ++h;
315 
316  // crash on some machines, use GL_BACK to render (test: va vers @dax@10,6,-5,1.0@ !)
317  glDrawBuffer(GL_BACK);
318  glPushMatrix();
319  // transformations to have the good position
320  glRotatef(-90, 1,0,0);
321  glRotatef(90, 0,0,1);
322  glRotatef(-RAD2DEG(rotz), 0,0,1);
323  glTranslatef(-posx, -posy, -posz);
324 
325  // redraw the scene (solids only)
326  clearGLBuffer();
327  solids(false);
328  glPopMatrix();
329 
330  // we use the back, then we read the back
331  uint8_t *pixjpg = Ogl::copyPixels(::g.pref.width3D, ::g.pref.height3D, GL_BACK);
332 
333  // saving in a jpeg file
334  int jpegquality = 1;
335  Img::saveJPG(filename, ::g.pref.width3D, ::g.pref.height3D, jpegquality, pixjpg);
336  delete[] pixjpg;
337 
338  // rendering reinitializes buffers then we draw in the front and the back
339  render();
340 }
