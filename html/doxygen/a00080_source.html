145 void Scene::init()
146 {
147  GLSection gls(this);
148  //NB: render.config() must be called before world.init() which is called in startCB())
149  ::g.render.config(::g.pref.quality3D);
150 
151  // complete the initialization of VRENG
152  ::g.startCB();
153 
154  // timer qui lance le reaffichage de la scene (repaint) tous les delta t
155  render_timer.onAction(ucall(this, &UBox::repaint));
156 
157  // le 1er arg. est le delai entre deux frames successives (si besoin est, changer
158  // cette valeur au lieu de mettre un select() dans paintGL, ce qui est incorrect)
159  // le 2e arg., -1, signifie qu'on repete indefiniement
160  render_timer.start((time_t) (::g.pref.frame_delay/1000.), (int) (-1));
161 
162  // timer qui execute NetTimeout() tous les delta t
163  net_timer.onAction(ucall(this, &Scene::netTimeoutCB));
164  net_timer.start(net_delay, -1);
165  message.show(false);
166  is_initialized = true; // the scene is initialized and ready for rendering
167 }

176 /*
177  * Main loop of the engine
178  */
179 void Scene::paintScene()
180 {
181  if (! is_startCB_launched) {
182  is_startCB_launched = true;
183  UAppli::addTimeout(500, 1, ucall(this, &Scene::init));
184  }
185  if (! is_visible || ! is_initialized) return;
186 
187  // at least one postponed Key Release event
188  if (gw.pendingPostponedKRs()) gw.flushPostponedKRs();
189 
190  // Computes current world
191  ProfileTime& tsimul = ::g.times.simul;
192  tsimul.start();
193  if (World::current())
194  World::current()->compute(tsimul.start_time.tv_sec, tsimul.start_time.tv_usec);
195  tsimul.stop();
196  trace(DBG_WO, "world computed");
197 
198  // General rendering
199  ProfileTime& trender = ::g.times.render;
200  trender.start();
201  ::g.render.render();
202  trender.stop();
203 
204  // Displays misc infos
205  if (hudvisible) {
206  updateHud();
207  hudbox.show(true);
208  }
209  else hudbox.show(false);
210 
211  // check if video capture is running
212  if (gw.capture.isCapturingVideo()) gw.capture.writeVideoFrame();
213 
214  cycles++; // increments cycles
215 }
