54 void Lift::parser(char *l)
55 {
56  defaults();
57  l = tokenize(l);
58  begin_while_parse(l) {
59  l = parse()->parseAttributes(l, this);
60  if (!l) break;
61  if (!stringcmp(l, "height")) l = parse()->parseFloat(l, &height, "height");
62  else if (!stringcmp(l, "speed")) l = parse()->parseFloat(l, &speed, "speed");
63  else if (!stringcmp(l, "top")) l = parse()->parseInt(l, &top, "top");
64  else if (!stringcmp(l, "bot")) l = parse()->parseInt(l, &bot, "bot");
65  }
66  end_while_parse(l);
67 }

99 void Lift::changePosition(float lasting)
100 {
101  if (ABSF(dest - pos.z) <= 0.1) return; // same level
102 
103  int sign = (dest > pos.z) ? 1 : -1;
104  pos.z += sign * lasting * speed;
105  if (state == CARRYING) {
106  localuser->pos.z += sign * lasting * speed; // user follows the lift
107  //dax localuser->updatePositionAndGrid(localuser->pos);
108  localuser->updatePosition();
109  localuser->updateCamera(pos);
110  }
111 }

113 void Lift::up()
114 {
115  dest = MIN(pos.z + height, height * top);
116  if ((pos.z + 0.1) >= top) return;
117  initImposedMovement((dest - pos.z) / speed);
118 }

120 void Lift::down()
121 {
122  dest = MAX(pos.z - height, height * bot);
123  if ((pos.z - 0.1) <= bot) return;
124  initImposedMovement((dest - pos.z) / speed);
125 }

150 /* somebody intersects */
151 bool Lift::whenIntersect(WObject *pcur, WObject *pold)
152 {
153  switch (pcur->type) {
154  case USER_TYPE:
155  if (! state) { // user comes in
156  state = CARRYING;
157  if (pcur->pos.z >= pold->pos.z) { // same level
158  pcur->pos.z += DEF_STEP; // moving up
159  pcur->updatePositionAndGrid(pold);
160  }
161  else { // inside from up
162  pold->copyPositionAndBB(pcur);
163  }
164  }
165  else { // user already inside
166  pold->copyPositionAndBB(pcur);
167  }
168  break;
169  default:
170  pold->copyPositionAndBB(pcur);
171  break;
172  }
173  return true;
174 }
