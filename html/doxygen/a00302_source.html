218 void Ball::changePosition(float lasting)
219 {
220  if (! taken) {
221  pos.x += lasting * move.lspeed.v[0] * ratio();
222  pos.y += lasting * move.lspeed.v[1] * ratio();
223  pos.z += lasting * move.lspeed.v[2] * ratio();
224  pos.az += lasting * move.aspeed.v[0] * ratio();
225  pos.ax += lasting * move.aspeed.v[2] * ratio();
226 
227  // reduce speed
228  move.lspeed.v[0] /= 1.02;
229  move.lspeed.v[1] /= 1.02;
230  move.lspeed.v[2] /= 1.02;
231  move.aspeed.v[0] /= 1.02;
232  move.aspeed.v[2] /= 1.02;
233  }
234  else { // ball is taken by user
235  float off = 0.4; //(pos.x > localuser->pos.x) ? 0.4 : -0.4;
236  pos.x = localuser->pos.x + off * Cos(localuser->pos.az);
237  pos.y = localuser->pos.y + off * Sin(localuser->pos.az);
238  pos.z = localuser->pos.z + 0.5;
239  }
240 }

255 bool Ball::whenIntersect(WObject *pcur, WObject *pold)
256 {
257  switch (pcur->type) {
258  case BALL_TYPE:
259  pcur->pos.x += ((float) drand48() * 2 - 1) * RADIUS / 2;
260  pcur->pos.y += ((float) drand48() * 2 - 1) * RADIUS / 2;
261  break;
262  case USER_TYPE:
263  if (! taken) {
264  /* user pushes the ball on the ground */
265  move.lspeed.v[0] = (lspeed / 2) * Cos(pcur->pos.az);
266  move.lspeed.v[1] = (lspeed / 2) * Sin(pcur->pos.az);
267  ttl = 1; // 1 sec
268  initImposedMovement(ttl);
269  pcur->updatePositionAndGrid(pold);
270  }
271  pold->copyPositionAndBB(pcur);
272  break;
273  case STEP_TYPE:
274  return false;
275  default:
276  pold->copyPositionAndBB(pcur);
277  //pcur->pos.z += RADIUS;
278  break;
279  }
280  pcur->updatePositionAndGrid(pold);
281  return true;
282 }
