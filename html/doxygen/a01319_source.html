47 /*
48  * We use a fixed key to store passwords, since we assume that our local
49  * file system is secure but nonetheless don't want to store passwords
50  * as plaintext.
51  */
52 uint8_t fixedkey[8] = {23,82,107,6,35,78,88,7};

54 /*
55  * Encrypt a password and store it in a file. Returns 0 if successful,
56  * 1 if the file could not be written.
57  */
58 int vncEncryptAndStorePasswd(char *passwd, char *fname)
59 {
60  FILE *fp;
61  uint8_t encryptedPasswd[8];
62 
63  if ((fp = File::openFile(fname, "w")) == NULL) return 1;
64 
65  chmod(fname, S_IRUSR|S_IWUSR);
66 
67  /* pad password with nulls */
68  for (uint32_t i=0; i < 8; i++)
69  encryptedPasswd[i] = (i < strlen(passwd)) ? passwd[i] : '\0';
70 
71  /* Do encryption in-place - this way we overwrite our copy of the plaintext
72  password */
73  deskey(fixedkey, EN0);
74 
75  des(encryptedPasswd, encryptedPasswd);
76 
77  for (int i=0; i < 8; i++)
78  putc(encryptedPasswd[i], fp);
79  File::closeFile(fp);
80  return 0;
81 }

83 /*
84  * Decrypt a password from a file. Returns a pointer to a newly allocated
85  * string containing the password or a null pointer if the password could
86  * not be retrieved for some reason.
87  */
88 char * vncDecryptPasswdFromFile(char *fname)
89 {
90  FILE *fp;
91  uint8_t *passwd = new uint8_t[9];
92 
93  if ((fp = File::openFile(fname, "r")) == NULL) return NULL;
94 
95  for (int i=0; i < 8; i++) {
96  int ch = getc(fp);
97  if (ch == EOF) {
98  File::closeFile(fp);
99  return NULL;
100  }
101  passwd[i] = ch;
102  }
103  File::closeFile(fp);
104  deskey(fixedkey, DE1);
105  des(passwd, passwd);
106  passwd[8] = '\0';
107 
108  return (char *) passwd;
109 }

114 void vncEncryptBytes(uint8_t *bytes, char *passwd)
115 {
116  uint8_t key[8];

118  /* key is simply password padded with nulls */
