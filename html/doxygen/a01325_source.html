146 /*
147  * sendRFBEvent is an action which sends an RFB event. It can be used in two
148  * ways. Without any parameters it simply sends an RFB event corresponding to
149  * the X event which caused it to be called. With parameters, it generates a
150  * "fake" RFB event based on those parameters. The first parameter is the
151  * event type, either "ptr", "keydown", "keyup" or "key" (down&up). For a
152  * "key" event the second parameter is simply a keysym string as understood by
153  * XStringToKeysym(). For a "ptr" event, the following three parameters are
154  * just X, Y and the button mask (0 for all up, 1 for button1 down, 2 for
155  * button2 down, 3 for both, etc).
156  */
157 void VNCClient::sendRFBEvent(char **params, unsigned int *num_params)
158 {
159  KeySym ks;
160  int buttonMask, x, y;
161 
162  if (viewonly || !framebuffer) return;
163 
164  if (strncasecmp(params[0], "key", 3) == 0) {
165  if (*num_params != 2) {
166  error("invalid params: sendRFBEvent(key|keydown|keyup,<keysym>)");
167  return;
168  }
169  ks = XStringToKeysym(params[1]);
170  if (ks == NoSymbol) {
171  error("Invalid keysym '%s' passed to sendRFBEvent", params[1]);
172  return;
173  }
174  if (strcasecmp(params[0], "keydown") == 0)
175  rfbproto.sendKeyEvent(ks, 1);
176  else if (strcasecmp(params[0], "keyup") == 0)
177  rfbproto.sendKeyEvent(ks, 0);
178  else if (strcasecmp(params[0], "key") == 0) {
179  rfbproto.sendKeyEvent(ks, 1);
180  rfbproto.sendKeyEvent(ks, 0);
181  }
182  else {
183  error("invalid event '%s' passed to sendRFBEvent", params[0]);
184  return;
185  }
186  }
187  else if (strcasecmp(params[0], "ptr") == 0) {
188  if (*num_params == 4) {
189  x = atoi(params[1]);
190  y = atoi(params[2]);
191  buttonMask = atoi(params[3]);
192  rfbproto.sendPointerEvent(x, y, buttonMask);
193  }
194  else {
195  error("invalid params: sendRFBEvent(ptr,x,y,buttonMask)");
196  return;
197  }
198  }
199  else
200  error("invalid event '%s' passed to sendRFBEvent", params[0]);
201 }

355 bool VNCClient::handleHextile32(int rx, int ry, int rw, int rh)
356 {
357  uint32_t bg, fg;
358  int i;
359  uint8_t *ptr;
360  int x, y, w, h;
361  int sx, sy, sw, sh;
362  uint8_t subencoding;
363  uint8_t nSubrects;
364  VNCRGB pixel;
365 
366  trace(DBG_VNC, "handleHextile32: rx=%d ry=%d rw=%d rh=%d",rx,ry,rw,rh);
367 
368  for (y = ry; y < ry+rh; y += 16) {
369  for (x = rx; x < rx+rw; x += 16) {
370  w = h = 16;
371  if (rx+rw - x < 16)
372  w = rx+rw - x;
373  if (ry+rh - y < 16)
374  h = ry+rh - y;
375 
376  if (!rfbproto.VNC_sock.ReadFromRFBServer((char *)&subencoding, 1))
377  return false;
378  if (subencoding & rfbHextileRaw) {
379  if (!rfbproto.VNC_sock.ReadFromRFBServer(rfbbuffer, w * h * 4))
380  return false;
381 
382  handleRAW32(x, y, w, h);
383  continue;
384  }
385 
386  if (subencoding & rfbHextileBackgroundSpecified)
387  if (!rfbproto.VNC_sock.ReadFromRFBServer((char *)&bg, sizeof(bg)))
388  return false;
389 
390  pixel = VNCRGB(swap32IfLE(bg));
391  fillRect(x, y, w, h, pixel);
392 
393  if (subencoding & rfbHextileForegroundSpecified)
394  if (!rfbproto.VNC_sock.ReadFromRFBServer((char *)&fg, sizeof(fg)))
395  return false;
396  if (!(subencoding & rfbHextileAnySubrects))
397  continue;
398 
399  if (!rfbproto.VNC_sock.ReadFromRFBServer((char *)&nSubrects, 1))
400  return false;
401 
402  ptr = (uint8_t *) rfbbuffer;
403 
404  if (subencoding & rfbHextileSubrectsColoured) {
405  if (!rfbproto.VNC_sock.ReadFromRFBServer(rfbbuffer, nSubrects * 6))
406  return false;
407 
408  for (i=0; i < nSubrects; i++) {
409  GET_PIXEL32(fg, ptr);
410  sx = rfbHextileExtractX(*ptr);
411  sy = rfbHextileExtractY(*ptr);
412  ptr++;
413  sw = rfbHextileExtractW(*ptr);
414  sh = rfbHextileExtractH(*ptr);
415  ptr++;
416  pixel = VNCRGB(swap32IfLE(fg));
417  fillRect(x+sx, y+sy, sw, sh, pixel);
418  }
419  }
420  else {
421  if (!rfbproto.VNC_sock.ReadFromRFBServer(rfbbuffer, nSubrects * 2))
422  return false;
423 
424  for (i=0; i < nSubrects; i++) {
425  sx = rfbHextileExtractX(*ptr);
426  sy = rfbHextileExtractY(*ptr);
427  ptr++;
428  sw = rfbHextileExtractW(*ptr);
429  sh = rfbHextileExtractH(*ptr);
430  ptr++;
431  pixel = VNCRGB(swap32IfLE(fg));
432  fillRect(x+sx, y+sy, sw, sh, pixel);
433  }
434  }
435  }
436  }
437  return true;
438 }

440 bool VNCClient::handleRFBServerMessage()
441 {
442  rfbServerToClientMsg msg;
443 
444  if (!framebuffer)
445  return false;
446  if (!rfbproto.VNC_sock.ReadFromRFBServer((char *) &msg, 1))
447  return false;
448 
449  switch (msg.type) {
450 
451  case rfbSetColourMapEntries:
452  {
453  error("handleRFBServerMessage: rfbSetColourMapEntries not supported yet");
454  uint16_t rgb[3];
455 
456  if (!rfbproto.VNC_sock.ReadFromRFBServer(((char *)&msg) + 1,
457  sz_rfbSetColourMapEntriesMsg - 1))
458  return false;
459 
460  msg.scme.firstColour = swap16IfLE(msg.scme.firstColour);
461  msg.scme.nColours = swap16IfLE(msg.scme.nColours);
462 
463  for (int i=0; i < msg.scme.nColours; i++) {
464  if (!rfbproto.VNC_sock.ReadFromRFBServer((char *)rgb, 6))
465  return false;
466  }
467  break;
468  }
469 
470  case rfbFramebufferUpdate:
471  {
472  //trace(DBG_VNC, "handleRFBServerMessage: rfbFramebufferUpdate");
473 
474  rfbFramebufferUpdateRectHeader rect;
475  int linesToRead;
476  int bytesPerLine;
477  int i;
478 
479  if (!rfbproto.VNC_sock.ReadFromRFBServer(((char *)&msg.fu) + 1, sz_rfbFramebufferUpdateMsg - 1))
480  return false;
481 
482  msg.fu.nRects = swap16IfLE(msg.fu.nRects);
483 
484  for (i=0; i < msg.fu.nRects; i++) {
485  if (!rfbproto.VNC_sock.ReadFromRFBServer((char *)&rect, sz_rfbFramebufferUpdateRectHeader))
486  return false;
487 
488  rect.r.x = swap16IfLE(rect.r.x);
489  rect.r.y = swap16IfLE(rect.r.y);
490  rect.r.w = swap16IfLE(rect.r.w);
491  rect.r.h = swap16IfLE(rect.r.h);
492 
493  rect.encoding = swap32IfLE(rect.encoding);
494 
495  if ((rect.r.x + rect.r.w > rfbproto.si.framebufferWidth) ||
496  (rect.r.y + rect.r.h > rfbproto.si.framebufferHeight)) {
497  error("Rect too large: %dx%d at (%d, %d)",
498  rect.r.w, rect.r.h, rect.r.x, rect.r.y);
499  return false;
500  }
501  if ((rect.r.h * rect.r.w) == 0) {
502  error("Zero rfbproto.size rect - ignoring");
503  continue;
504  }
505 
506  switch (rect.encoding) {
507 
508  case rfbEncodingRaw:
509  bytesPerLine = rect.r.w * rfbproto.pixFormat.bitsPerPixel / 8;
510  linesToRead = sizeof(rfbbuffer) / bytesPerLine;
511 
512  //trace(DBG_VNC, "handleRFBServerMessage: rfbFramebufferUpdate rfbEncodingRaw bytesPerLine=%d linesToRead=%d", bytesPerLine, linesToRead);
513 
514  while (rect.r.h > 0) {
515  if (linesToRead > rect.r.h)
516  linesToRead = rect.r.h;
517  if (!rfbproto.VNC_sock.ReadFromRFBServer(rfbbuffer,bytesPerLine * linesToRead))
518  return false;
519 
520  if (!handleRAW32(rect.r.x, rect.r.y, rect.r.w, rect.r.h))
521  return false;
522  rect.r.h -= linesToRead;
523  rect.r.y += linesToRead;
524  }
525  break;
526 
527  case rfbEncodingCopyRect:
528  {
529  rfbCopyRect cr;
530  if (!rfbproto.VNC_sock.ReadFromRFBServer((char *) &cr, sz_rfbCopyRect))
531  return false;
532 
533  cr.srcX = swap16IfLE(cr.srcX);
534  cr.srcY = swap16IfLE(cr.srcY);
535 
536  if (!handleCR(cr.srcX, cr.srcY, rect.r.x, rect.r.y, rect.r.w, rect.r.h))
537  return false;
538  break;
539  }
540 
541  case rfbEncodingRRE:
542  if (!handleRRE32(rect.r.x,rect.r.y,rect.r.w,rect.r.h))
543  return false;
544  break;
545  case rfbEncodingCoRRE:
546  if (!handleCoRRE32(rect.r.x,rect.r.y,rect.r.w,rect.r.h))
547  return false;
548  break;
549  case rfbEncodingHextile:
550  if (!handleHextile32(rect.r.x,rect.r.y,rect.r.w,rect.r.h))
551  return false;
552  break;
553 
554  default:
555  error("Unknown rect encoding %d", (int) rect.encoding);
556  return false;
557  }
558  }
559  if (!sendIncrementalFramebufferUpdateRequest())
560  return false;
561  break;
562  }
563 
564  case rfbBell:
565  //XBell(dpy,100);
566  break;
567 
568  case rfbServerCutText:
569  if (!rfbproto.VNC_sock.ReadFromRFBServer(((char *) &msg) + 1, sz_rfbServerCutTextMsg - 1))
570  return false;
571 
572  msg.sct.length = swap32IfLE(msg.sct.length);
573 
574  if (serverCutText) delete[] serverCutText;
575  serverCutText = NULL;
576  serverCutText = new char[msg.sct.length+1];
577 
578  if (!rfbproto.VNC_sock.ReadFromRFBServer(serverCutText, msg.sct.length))
579  return false;
580 
581  serverCutText[msg.sct.length] = 0;
582  newServerCutText = true;
583  break;
584 
585  default:
586  error("Unknown message type %d from VNC server", msg.type);
587  return false;
588  }
589  return true;
590 }
