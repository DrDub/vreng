81 void Off::httpReader(void *_off, Http *http)
82 {
83  Off *off = (class Off *) _off;
84  if (! off) return;
85 
86  FILE *f = Cache::openCache(off->getUrl(), http);
87 
88  char line[80];
89  /* Get info header: vertices_number normals_number polygons_number */
90  fgets(line, sizeof(line), f);
91  getOff3i(line, &off->vn, &off->nn, &off->pn);
92  fgets(line, sizeof(line), f); // empty line
93 
94  /* Get vertices */
95  off->v = new float[3 * off->vn];
96  for (int i=0; i < off->vn; i++) {
97  fgets(line, sizeof(line), f);
98  getOff3f(line, &off->v[i*3+0], &off->v[i*3+1], &off->v[i*3+2], off->scale);
99  }
100  fgets(line, sizeof(line), f); // empty line
101 
102  /* Get normals */
103  off->n = new float[3 * off->nn];
104  for (int i=0; i < off->nn; i++) {
105  fgets(line, sizeof(line), f);
106  getOff3f(line, &off->n[i*3+0], &off->n[i*3+1], &off->n[i*3+2], off->scale);
107  }
108  fgets(line, sizeof(line), f); // empty line
109 
110  /* Get polygons */
111  off->p = new int[6 * off->pn];
112  for (int i=0; i < off->pn; i++) {
113  fgets(line, sizeof(line), f);
114  int k = strlen(line);
115  for (int j=0; j < k; j++) {
116  int a = 0;
117  if (line[j] == 'v') a = j+2;
118  if (line[j] == 'n') {
119  line[j-1] = 0;
120  getOff3i(&line[a+2], &off->p[i*6+0], &off->p[i*6+1], &off->p[i*6+2]);
121  getOff3i(&line[j+2], &off->p[i*6+3], &off->p[i*6+4], &off->p[i*6+5]);
122  break;
123  }
124  }
125  }
126  File::closeFile(f);
127 }
