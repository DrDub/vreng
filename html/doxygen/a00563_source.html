40 void Light::parser(char *l)
41 {
42  defaults();
43  l = tokenize(l);
44  begin_while_parse(l) {
45  l = parse()->parseAttributes(l, this);
46  if (!l) break;
47  if (! stringcmp(l, "fog=")) l = parse()->parseFloat(l, &fog, "fog");

61 Light::Light(char *l)
62 {
63  parser(l);
64 
65  initializeObject(INVISIBLE);
66  addToList(lightList);
67 
68  if (!pos.x && !pos.y && !pos.z && !pos.az && !pos.ax) islight = false;
69  else islight = true;
70 
71  if (islight) {
72  GLfloat light_diffuse[] = {1, 1, 1, 1};
73  GLfloat light_ambient[] = {0.6, 0.6, 0.6, 1};
74  GLfloat light_specular[] = {0.8, 0.8, 0.8, 1};
75  GLfloat light_spot_direction[3];
76  GLfloat light_spot_cutoff[1];
77 
78  light_ambient[0] = 0.1;
79  light_ambient[1] = 0.1;
80  light_ambient[2] = 0.1;
81  light_ambient[3] = 1;
82  light_position[0] = pos.x;
83  light_position[1] = pos.y;
84  light_position[2] = pos.z;
85  light_position[3] = pos.az; // w = 0 (dir) | 1 (pos)
86  if (light_position[3] == 0) { // directionnel
87  if (pos.ax > 0 && pos.ax <= 90) {
88  light_spot_cutoff[0] = pos.ax; // 0 < spot < 90
89  light_mode = DIR;
90  }
91  else light_mode = POS;
92  }
93  else light_mode = POS; // positionnel
94 
95  // reset orientation
96  pos.az = 0;
97  pos.ax = 0;
98 
99  glEnable(GL_LIGHT2);
100 
101  glLightfv(GL_LIGHT2, GL_DIFFUSE, light_diffuse);
102  glLightfv(GL_LIGHT2, GL_AMBIENT, light_ambient);
103  glLightfv(GL_LIGHT2, GL_SPECULAR, light_specular);
104 
105  if (light_mode == DIR) { // spot
106  light_spot_direction[0] = light_position[0];
107  light_spot_direction[1] = light_position[1];
108  light_spot_direction[2] = light_position[2];
109  glLightfv(GL_LIGHT2, GL_SPOT_DIRECTION, light_spot_direction);
110  glLightfv(GL_LIGHT2, GL_SPOT_CUTOFF, light_spot_cutoff);
111  //glLightf(GL_LIGHT2, GL_CONSTANT_ATTENUATION, 1);
112  }
113  else
114  glLightfv(GL_LIGHT2, GL_POSITION, light_position);
115  }
116  if (fog) {
117  glFogi(GL_FOG_MODE, GL_EXP);
118  glFogf(GL_FOG_DENSITY, fog);
119  glFogfv(GL_FOG_COLOR, color);
120  }
121 }
