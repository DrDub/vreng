117 inline void getDistanceFromAndOnBone(Vertex *vertex, BoneVertex *a, BoneVertex *b, float *time, float *dist)
118 {
119  Vect3D nullVector(0., 0., 0.);
120  Vect3D aPosition; aPosition = a->initialMatrix * nullVector;
121  Vect3D bPosition; bPosition = b->initialMatrix * nullVector;
122  Vect3D abVector; abVector = bPosition - aPosition;
123  Vect3D paVector; paVector = aPosition - vertex->initialPosition;
124 
125  *time = -Vect3D::dotProduct(paVector, abVector) / Vect3D::dotProduct(abVector, abVector);
126  *time = CROP(0.0001, *time, 0.9999);
127 
128  Vect3D phVector;
129  phVector = abVector;
130  phVector = phVector * *time;
131  phVector = phVector + paVector;
132  *dist = Vect3D::dotProduct(phVector, phVector);
133  *dist = (*dist > 0.00000001 ? *dist : 0.00000001);
134 }

136 inline float BoneAnimator::getWeight(Vertex *vertex, BoneVertex *node)
137 {
138  float result = 0;
139  float dist, time;
140 
141  if (node->father != NULL) {
142  getDistanceFromAndOnBone(vertex, node->father, node, &time, &dist);
143  result += (time) / (dist * dist);
144  }
145 
146  for (int i=0; i<node->children; i++) {
147  getDistanceFromAndOnBone(vertex, node->child[i], node, &time, &dist);
148  result += (time) / (dist * dist);
149  }
150 
151  result /= node->children + (node->father == NULL ? 0:1);
152  result *= node->influenceScaleFactor;
153  return result;
154 }

166 // -> Will call the method to generate links and then update the weight
167 // and optionally remove unsignificant links (that would speed down the cpu)
168 void BoneAnimator::generateLinkList()
169 {
170  // We need both meshToMove and skeleton !
171  if (meshToMove == NULL) return; // To avoid NULL pointer exception
172  if ( skeleton == NULL) return; // To avoid NULL pointer exception
173 
174  BoneVertex *tempNode;
175  tempNode = skeleton->findChild("root");
176  if (tempNode != NULL)
177  tempNode->influenceScaleFactor = 10.;
178  tempNode = skeleton->findChild("lipsRoot");
179  if (tempNode != NULL)
180  tempNode->influenceScaleFactor = 10.;
181  tempNode = skeleton->findChild("frontRoot");
182  if (tempNode != NULL)
183  tempNode->influenceScaleFactor = 10.;
184 
185  // We start link generation with an empty link list of course
186  emptyLinkList();
187 
188  // If the mesh has not compiled hos vertices list, we'll do it
189  if (! meshToMove->vertexListCompiled) meshToMove->compileVertexList();
190 
191  // First, we need to generate the initial matrices for
192  // all the nodes of the skeleton
193  glPushMatrix();
194  glLoadIdentity();
195  skeleton->generateInitialMatrix();
196  glPopMatrix();
197 
198  //--- We'll store the skeleton nodes into a list
199  BoneList < BoneVertex > nodeList;
200  BoneVertex **node;
201  int nodes;
202 
203  addNodeAndChildren(skeleton, &nodeList);
204  node = nodeList.getNiceTable(&nodes);
205 
206  //=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-=-
207  // Now we'll go throw each vertex of the mesh calculating
208  // the weight of this node
209  BoneList < BoneLink > temporaryLinkList;
210  BoneLink **temporaryLink;
211  BoneLink *temp;
212  int temporaryLinks;
213 #ifdef DEBUG_FACE
214  int progress = -1;
215  int progressold = -1;
216 #endif
217 
218  for (int i=0; i<meshToMove->vertices; i++) {
219 #ifdef DEBUG_FACE
220  // Prints info on progress...
221  progressold = progress;
222  progress = (i * 10) / (meshToMove->vertices - 1);
223  if (progress != progressold) {
224  fprintf(stderr, "%i%%... ", progress * 10);
225  fflush(stderr);
226  }
227 #endif
228 
229  temporaryLinkList.empty();
230  for (int j=0; j<nodes; j++) {
231  // And create a link between the vertex and the node with an influence
232  // proportional to the inverse of the distance ( so far vertices
233  // will be less influenced by the node than near vertices )
234  temp = new BoneLink(meshToMove->vertex[i], node[j], getWeight(meshToMove->vertex[i], node[j]));
235  // We now save this new link in our list
236  temporaryLinkList.addElement(temp);
237  }
238  temporaryLink = temporaryLinkList.getNiceTable(&temporaryLinks);
239 
240  // Now sorting the links per weight
241  for (int j=1; j<temporaryLinks; j++) {
242  temp = temporaryLink[j];
243  int k = j;
244  while ((k > 0) && (temporaryLink[k-1]->weight < temp->weight)) {
245  temporaryLink[k] = temporaryLink[k-1];
246  k--;
247  }
248  temporaryLink[k] = temp;
249  }
250 
251  // Now removing unsignificant links
252  float seuil = 0.3 * temporaryLink[0]->weight;
253  for (int j=0; j<temporaryLinks; j++) {
254  if (temporaryLink[j]->weight < seuil)
255  { delete temporaryLink[j]; temporaryLink[j] = NULL; }
256  }
257 
258  // Record the selected links in the list
259  normalize (temporaryLink, temporaryLinks);
260  for (int j=0; j<temporaryLinks; j++)
261  if (temporaryLink[j] != NULL) linkList.addElement(temporaryLink[j]);
262  }
263 
264  // Now that we have all the links, we may compile the link list in here
265  compileLinkList();
266 
267  for (int i=0; i<links; i++) link[i]->notifyTarget();
268  for (int i=0; i<meshToMove->vertices; i++) meshToMove->vertex[i]->compileLinkList();
269 
270 #ifdef DEBUG_FACE
271  fprintf(stderr, "\n");
272  fflush(stderr);
273 #endif
274  trace(DBG_MAN, "selected links: [%2.2f%%]", (links * 100.) / (meshToMove->vertices * nodes));
275 }

447 void BoneAnimator::animate()
448 {
449  if (! boneEditing) {
450  // First we reset to 0,0,0 all the vertices of the mesh
451  for (int i=0; i < meshToMove->vertices; i++) {
452  Vertex *currentVertex = meshToMove->vertex[i];
453  if (currentVertex->links != 0) {
454  currentVertex->currentPosition.reset();
455  currentVertex->currentNormal.reset();
456  }
457  else {
458  currentVertex->currentPosition = currentVertex->initialPosition;
459  currentVertex->currentNormal = currentVertex->initialNormal;
460  }
461  }
462 
463  // Then we generate the current matrix of each part of the skeleton
464  glPushMatrix();
465  glLoadIdentity();
466  skeleton->generateCurrentMatrix();
467  glPopMatrix();
468 
469  // And now we can perform the recursive calls to create our new 3d mesh
470  animateSkeletonNode(skeleton);
471  }
472  else
473  for (int i=0; i < meshToMove->vertices; i++) {
474  meshToMove->vertex[i]->currentPosition = meshToMove->vertex[i]->initialPosition;
475  meshToMove->vertex[i]->currentNormal = meshToMove->vertex[i]->initialNormal;
476  }

480 inline void BoneAnimator::animateSkeletonNode(BoneVertex *node)
481 {
482  // The first time, the link list won't be compiled for this node
483  // so we'll dot it
484  if (! node->linkListCompiled) node->compileLinkList();
485 
486  BoneLink *zeLink;
487  Vertex *zeVertex;
488  Vect3D tempPos, normal;
489 
490  // We now look at each link in this node and find the related vertex
491  for (int i=0; i < node->links; i++) {
492  zeLink = node->link[i];
493  zeVertex = zeLink->vertex;
494 
495  // Now that we have M1 ( Initial matrix for this node )
496  // M2 ( Current matrix for this node )
497  // and w ( Weight of this link for the vertex )
498  // We increase newPosition as follozs :
499  // newPosition += w . M2 . M1-1 . initialPosition
500  // Since the weight are normalized to 100% [0.0f .. 1.0f], we
501  // should have a normalized result (means no scaling here) for all
502  // the vertices
503 
504  tempPos = zeVertex->initialPosition;
505  tempPos *= node->initialMatrixInverted;
506  tempPos *= node->currentMatrix;
507  tempPos *= zeLink-> weight;
508  zeVertex->currentPosition += tempPos;
509 #if 0 //DEBUG
510  normal = zeVertex->initialNormal;
511  normal *= node->initialRotMatrixInverted;
512  normal *= node->currentRotMatrix;
513  normal *= zeLink->weight;
514  zeVertex->currentNormal += normal;
515 #endif
516  }
517 
518  // And now, we'll add the other links actions
519  for (int i=0; i < node->children; i++)
520  animateSkeletonNode(node->child[i]);
521 }
