59 void Guide::parser(char *l)
60 {
61  defaults();
62  l = tokenize(l);
63  begin_while_parse(l) {
64  l = parse()->parseAttributes(l, this);
65  if (!l) break;
66  if (! stringcmp(l, "path=")) l = parse()->parsePath(l, &path[1], &pts);
67  else if (! stringcmp(l, "color")) l = parse()->parseVector3f(l, color, "color");
68  else if (! stringcmp(l, "mode=")) {
69  char modestr[16];
70  l = parse()->parseString(l, modestr, "mode");
71  if (! stringcmp(modestr, "one-way")) oneway = true;
72  else if (! stringcmp(modestr, "perpetual")) perpetual = true;
73  }
74  }
75  end_while_parse(l);
76 
77  if (pts == 0) { warning("no path"); return; }
78 
79  // start point = guide initial
80  path[0][0] = pos.x;
81  path[0][1] = pos.y;
82  path[0][2] = pos.z;
83  path[0][3] = path[1][3];
84  path[0][4] = 1; // tempo 1 sec
85  pts++;
86  if (oneway) {
87  path[pts][3] = 0;
88  path[pts][4] = 0;
89  }
90  else {
91  // round-trip: end point = start point
92  for (int i=0; i<5 ; i++) path[pts+1][i] = path[0][i];
93  path[pts+1][3] = 0; // speed = 0 for end point
94  path[pts+1][4] = 0;
95  pts++;
96  }
97 }

144 {

147  if (! pause) {

150  pos.y += ((path[pt+1][1] - path[pt][1]) * path[pt][3] / 100) / nn;

156 void Guide::motion(float *dx, float *dy, float *dz)
157 {
158  float nn = norm();
159 
160  if (pause) {
161  *dx = *dy = *dz = 0;
162  }

165  *dy = ((path[pt+1][1] - path[pt][1]) * path[pt][3] / 50) / nn;
166  *dz = ((path[pt+1][2] - path[pt][2]) * path[pt][3] / 50) / nn;
167  updatePositionAndGrid(pos);
168  pos.x += *dx;
169  pos.y += *dy;
170  pos.z += *dz;
171  updatePosition();
172  }

197 bool Guide::whenIntersect(WObject *pcur, WObject *pold)
198 {
199  static bool first = true;
200 
201  if (perpetual) return true;
202 
203  if (pcur->type != USER_TYPE) {
204  pold->copyPositionAndBB(pcur);
205  return true;
206  }
207  // user only
208  inside = true;
209  if (restored) {
210  restored = false;
211  first = true;
212  }
213 
214  if (first) {
215  // save initial position of the user
216  uinitial[0] = pold->pos.x;
217  uinitial[1] = pold->pos.y;
218  uinitial[2] = pold->pos.z;
219  uinitial[3] = pold->pos.az;
220  localuser->pos.x = pos.x;
221  localuser->pos.y = pos.y;
222  localuser->pos.z += (pos.z + pos.bbsize.v[2]); // jump on the skate
223  localuser->pos.az = pos.az;
224  if (path[pt][4]) { // pause
225  signal(SIGALRM, sigguide);
226  alarm((uint32_t) path[pt][4]);
227  pause = true;
228  }
229  first = false;
230  }
231 
232  if (path[pt][3]) {
233  /* user follows the guide */
234  float dx, dy, dz;
235  motion(&dx, &dy, &dz);
236  localuser->updatePositionAndGrid(localuser->pos);
237  localuser->pos.x += dx;
238  localuser->pos.y += dy;
239  localuser->pos.z += dz + .05; // + 1cm
240  //error("follow: %.2f %.2f %.2f, %.3f %.3f %.3f", localuser->pos.x,localuser->pos.y,localuser->pos.z,dx,dy,dz);
241  if (localuser->pos.x == pold->pos.x && localuser->pos.y == pold->pos.y)
242  pold->copyPositionAndBB(localuser);
243  else localuser->pos.z += DELTAZ;
244  updatePositionAndGrid(pold->pos); //HACK! I don't know why!
245  localuser->updatePositionAndGrid(localuser->pos);
246 
247  if ((floor(pos.x) == path[pt+1][0]) &&
248  (floor(pos.y) == path[pt+1][1])) { // new segment
249  pt++; // next point
250 
251  if (path[pt][3] == 0) goto endtour; // null speed
252 
253  if (path[pt][4]) { // pause
254  signal(SIGALRM, sigguide);
255  alarm((uint32_t) path[pt][4]); // set delay
256  pause = true;
257  float azn = atan((path[pt+1][1] - path[pt][1]) /
258  (path[pt+1][0] - path[pt][0]));
259  if ((path[pt+1][0] - path[pt][0]) < 0) azn += M_PI;
260  float azo = atan((path[pt][1] - path[pt-1][1]) /
261  (path[pt][0] - path[pt-1][0]));
262  if ((path[pt][0] - path[pt-1][0]) < 0) azo += M_PI;
263  float da = deltaAngle(azn, 0);
264  move.aspeed.v[0] = da / path[pt][4];
265  }
266  else {
267  signal(SIGALRM, SIG_IGN);
268  motion(&dx, &dy, &dz);
269  localuser->updatePositionAndGrid(localuser->pos);
270  localuser->pos.x += dx;
271  localuser->pos.y += dy;
272  localuser->pos.z += dz;
273  localuser->updatePosition();
274  }
275 
276  // new orientation
277  float az = atan((path[pt+1][1] - path[pt][1]) /
278  (path[pt+1][0] - path[pt][0]));
279  if ((path[pt+1][0] - path[pt][0]) < 0) az += M_PI;
280  pos.az = az;
281  localuser->pos.az = pos.az; // user takes same orientation than guide
282  }
283  // update user
284  localuser->updatePositionAndGrid(localuser->pos);
285  }
286  else {
287 endtour:
288  first = true;
289  restore((User *)localuser);
290  }
291  return true;
292 }
