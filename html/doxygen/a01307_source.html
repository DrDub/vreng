30 int Wget::start(const char *_url, char *file, const char opts[])
31 {
32 #if HAVE_WGET
33  bool cached = false;
34  char url[URL_LEN];
35  Url::abs(_url, url);
36 
37  if (file && (! strcmp(opts, "-qO"))) { // use the cache
38  cached = true;
39  if (Cache::setCacheName(_url, file) == 0) return 0;
40 
41  struct stat bufstat;
42  if (stat(file, &bufstat) == 0) {
43  if (bufstat.st_size != 0) return 1; // file yet in the cache
44  unlink(file); // remove empty file
45  }
46  }
47 
48  pid_t pid;
49  int status;
50  bool check = false;
51  if (!stringcmp(opts, "check")) check = true;
52 
53  switch (pid = fork()) {
54  case -1:
55  error("%s wget", e_fork);
56  return 0;
57  case 0: // child starts transfert
58  if (file == NULL) {
59  if (check) execlp("wget", "wget", "-q", "--spider", url, (char*)NULL);
60  else execlp("wget", "wget", "-q", url, (char*)NULL);
61  }
62  if ( *file == '-') {
63  execlp("wget", "wget", "-nv", "-O", "-", url, (char*)NULL);
64  }
65  else if (! stringcmp(opts, "inout")) // local copy
66  execlp("wget", "wget", "-i", url, "-O", file, (char*)NULL);
67  else
68  execlp("wget", "wget", "-O", file, url, (char*)NULL);
69  error("%s wget", e_exec);
70  exit(1);
71  default:
72  if (! cached && ! check) // not in cache
73  return 1;
74  status = my_wait(pid); // parent waits end of transfert
75  if (status) {
76  if (! check) {
77  error("wget status=%d file=%s", status, file);
78  unlink(file); // because wget leaves an empty file
79  }
80  return 0;
81  }
82 #if HAVE_UTIME
83  utime(file, NULL); // set current time
84 #endif
85  return 1;
86  }
87 #else
88  return 0;
89 #endif
90 }
