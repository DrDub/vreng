47 void Fountain::parser(char *l)
48 {
49  defaults();
50  l = tokenize(l);
51  begin_while_parse(l) {
52  l = parse()->parseAttributes(l, this);
53  if (!l) break;
54  else if (! stringcmp(l, "number")) l = parse()->parseUInt16(l, &number, "number");
55  else if (! stringcmp(l, "flow")) l = parse()->parseFloat(l, &flow, "flow");
56  else if (! stringcmp(l, "speed")) l = parse()->parseFloat(l, &speed, "speed");
57  else if (! stringcmp(l, "ground")) l = parse()->parseFloat(l, &ground, "ground");
58  else if (! stringcmp(l, "color")) l = parse()->parseVector3f(l, color, "color");
59  else if (! stringcmp(l, "size")) {
60  l = parse()->parseUInt8(l, &pt_size, "size");
61  if (pt_size <= 0) pt_size = 1;
62  }
63  }
64  end_while_parse(l);
65 }
