204 /* Initialize a pair of sockets for communicating with the child app */
205 VjcSocket::VjcSocket(uint16_t _listenPort, const char *_destHost, uint16_t _destPort)
206 {
207  listenPort = _listenPort;
208  destPort = _destPort;
209  sadest = 0;
210  sdr = -1;
211  sdw = -1;
212 
213  struct sockaddr_in *sa;
214  if ((sa = new struct sockaddr_in[1]) == NULL) {
215  //error("VjcSocket: can't new sa");
216  return;
217  }
218  memset(sa, 0, sizeof(struct sockaddr_in));
219  sa->sin_family = AF_INET;
220  sa->sin_port = htons(destPort);
221 
222  struct hostent *hp;
223  if ((hp = my_gethostbyname(_destHost, AF_INET)) == NULL) {
224  error("VjcSocket: unknown server %s", _destHost);
225  if (sa) delete[] sa;
226  return;
227  }
228  memcpy(&sa->sin_addr, hp->h_addr_list[0], hp->h_length);
229  sadest = sa;
230 
231  statecon = SOCK_CLOSED;
232 }

234 /* Opens the receiver socket for this socket pair */
235 int VjcSocket::openRecv()
236 {
237  if ((sdr = Socket::openDatagram()) < 0) return 0;
238 
239  for (int tries = 0; tries<10; tries++) {
240  if (Socket::bindSocket(sdr, INADDR_ANY, listenPort + tries) != -1) {
241  listenPort += tries;
242  statecon = SOCK_RCVOPEN;
243  notice("Vjc receiver bound to port %d", listenPort);
244  return 1;
245  }
246  }
247  Socket::closeDatagram(sdr);
248  return 0;
249 }

261 /* Tries a connect to the client app */
262 int VjcSocket::connectSend()
263 {
264  struct timeval timeout;
265  timeout.tv_sec = 10;
266  timeout.tv_usec = 0;
267 
268  if (setsockopt(sdw, SOL_SOCKET, SO_RCVTIMEO, (char *)&timeout, sizeof(timeout)) < 0)
269  error("setsockopt failed\n");
270  if (connect(sdw, (const struct sockaddr *) sadest, sizeof(struct sockaddr_in)) < 0) {
271  if ((errno == EINPROGRESS) || (errno == EALREADY)) {
272  // Socket set to non-blocking to prevent an app freeze
273  return 1;
274  }
275  else {
276  //error("connectSend: %s", strerror(errno));
277  return 0;
278  }
279  }
280  else statecon = SOCK_OPEN;
281  return 1;
282 }

284 /* Check if the non-blocking connect call on the send socket finished or not */
285 bool VjcSocket::isConnected()
286 {
287  fd_set set;
288  struct timeval timeout;
289 
290  FD_ZERO(&set);
291  FD_SET(sdw, &set);
292  timeout.tv_sec = 0;
293  timeout.tv_usec = 0;
294  if (select(sdw+1, NULL, &set, NULL, &timeout) > 0) {
295  statecon = SOCK_OPEN; // The socket finished connecting
296  return true;
297  }
298  return false;
299 }

301 /* Opens a VjcSocket. This call is 'non-blocking' */
302 int VjcSocket::openSocket()
303 {
304  // check if it's not already open
305  if (statecon == SOCK_OPEN) return 1;
306  // Open the receiver
307  if (statecon == SOCK_CLOSED) if (openRecv() == 0) return -1;
308  // Open the sender
309  if (statecon == SOCK_RCVOPEN) if (openSend() == 0) return -1;
310  // Connect the sender
311  if (statecon == SOCK_CONNECT) if (connectSend() == 0) return -1;
312  if (statecon == SOCK_OPEN) return 1; // Everything is connected
313  // One of the steps didn't finish yet.
314  return 0;
315 }

520 /* Packs this message into an uint8_t array */
521 uint8_t *VjcMessage::toBytes(int *pktlen)
522 {
523  int datalen = cursor - VJC_HDR_LEN; // total length - header
524 
525  *pktlen = cursor; // total length is where our cursor currently is
526  cursor = VJC_HDR_DATALEN_IDX; // set the data length in the packet
527  put16(datalen);
528  cursor = VJC_HDR_SSRC_IDX; // set the ssrc (could have been updated)
529  put32(Vjc::getServer()->ssrc);
530  // Restore the cursor. This should actually never be needed, except
531  // if someone sends the same message twice, after updating it
532  cursor = *pktlen;
533  return data; // return a pointer to the data.
534 }

536 /* Send data over to the server */
537 int VjcMessage::sendData()
538 {
539  int pktlen = 0;
540  uint8_t *pkt;
541 
542  /* Check that the socket is ready */
543  Vjc *srv = Vjc::getServer();
544  if (srv && (srv->sock->statecon == VjcSocket::SOCK_CONNECT)) {
545  // The socket is still trying to connect. Check if it's done or not.
546  if (! srv->sock->isConnected()) return 0; // The connection is still in progress
547  }
548  if (srv && (srv->sock->statecon == VjcSocket::SOCK_OPEN)) {
549  /* Check if an SSRC change occured since we sent the REGISTER commands */
550  if (NetObject::getMySsrcId() != srv->ssrc) {
551  if (! ((getHeader().msg_type == VJC_MSGT_CTRL)
552  && (getHeader().msg_id == VJC_MSGV_REGISTER))) {
553 
554  VjcMessage *msg = new VjcMessage(srv, srv->ssrc, VJC_MSGT_CTRL, VJC_MSGV_UPDATE);
555  //error("Vjc: updating SSRC for Vjc (old:%d, new:%d)", srv->ssrc, NetObject::getMySsrcId());
556  msg->put32(NetObject::getMySsrcId());
557  pkt = msg->toBytes(&pktlen);
558  send(srv->sock->sdw, pkt, pktlen, 0);
559  if (msg) delete msg;
560  }
561  srv->ssrc = NetObject::getMySsrcId();
562  }
563 
564  /* send the message */
565  pkt = toBytes(&pktlen);
566  send(srv->sock->sdw, pkt, pktlen, 0);
567  //error("Vjc: sending %d bytes with '%02x%02x' %d %08x (%d %d)",
568  // sent, pkt[0], pkt[1], pkt[2], NetObject::getMySsrcId(),
569  // getHeader().msg_type,
570  // getHeader().msg_id);
571  return 1;
572  }
573  return 0;
574 }
575 
576 #define VJC_CHECK_OVERFLOW_2(a) \
577  if (cursor > maxlen-a) { \
578  error("Read past end of buffer (%d %d)", getHeader().msg_type, getHeader().msg_id); \
579  return NULL; \
580  }

582 /* Reads in data from the child process, if any is availabe */
583 VjcMessage * Vjc::getData(WObject *po)
584 {
585  /* check that the socket is really open */
586  Vjc *srv = getServer();
587  if ((! srv) || (srv->sock->statecon != VjcSocket::SOCK_OPEN)) return NULL;
588 
589  if (srv->lastMessage) goto haspack;
590 
591  uint8_t *pkt;
592 
593  /* set a 0 sec timeout:
594  * the select will return instantly if no data is available */
595  fd_set fds;
596  struct timeval to;
597 
598  FD_ZERO(&fds);
599  FD_SET(srv->sock->sdr, &fds);
600  to.tv_sec = 0;
601  to.tv_usec = 0;
602 
603  if (select(srv->sock->sdr+1, &fds, NULL, NULL, &to) > 0) {
604  int r;
605 
606  if ((pkt = new uint8_t[VjcMessage::MAX_PACKET]) == NULL) {
607  //error("Vjc::getData: can't new reception buffer");
608  return NULL;
609  }
610 
611  if ((r = recv(srv->sock->sdr, (void *) pkt, VjcMessage::MAX_PACKET, 0)) > 0) {
612  if (r < VJC_HDR_LEN) {
613  /* We didn't get a whole header */
614  error("Vjc::getData: dropped incomplete packet (%d)", r);
615  }
616  else {
617  /* Message size is big enough to contain a header */
618  srv->lastMessage = new VjcMessage(pkt, r);
619 
620  /* Check the header */
621  // TODO: check protocol id and version number
622  if (srv->lastMessage->getHeader().data_len > (VjcMessage::MAX_PACKET-VJC_HDR_LEN)) {
623  /* Header and real length don't agree */
624  error("Vjc::getData: illegal data length");
625  if (srv->lastMessage) delete srv->lastMessage;
626  srv->lastMessage = NULL;
627  }
628  else goto haspack;
629  }
630  }
631  // (r < 0) || (packet too short) || (invalid header)
632  if (pkt) delete[] pkt;
633  pkt = NULL;
634  return NULL;
635  }
636  return NULL; // select <= 0
637 
638 haspack:
639  // A message was read from the socket, or had been read previously
640  if (srv->lastMessage->isForObject(po)) {
641  VjcMessage *ret = srv->lastMessage;
642  srv->lastMessage = NULL;
643  return ret;
644  }
645  return NULL;
646 }
