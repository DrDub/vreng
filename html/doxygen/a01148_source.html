313 // Generating the initial matrix for this node
314 void BoneVertex::generateInitialMatrix()
315 {
316  // First, we'll need to know all the childs
317  if (! childListCompiled) compileChildList();
318 
319  glPushMatrix();
320 
321  // Now I let OpenGL calculate the matrix
322  glTranslatef(initialPosition.x, initialPosition.y, initialPosition.z);
323  glRotatef (initialAngle, initialAxis.x , initialAxis.y, initialAxis.z);
324  // I save it
325  glGetFloatv (GL_MODELVIEW_MATRIX, initialMatrix);
326 
327  // Here is a nice matrix inversion
328  // 1. tmp gets the transposed rotation part of the matrix
329  float tmp[16];
330  tmp[0] = initialMatrix[ 0]; tmp[4] = initialMatrix[ 1]; tmp[ 8] = initialMatrix[ 2]; tmp[12] = 0;
331  tmp[1] = initialMatrix[ 4]; tmp[5] = initialMatrix[ 5]; tmp[ 9] = initialMatrix[ 6]; tmp[13] = 0;
332  tmp[2] = initialMatrix[ 8]; tmp[6] = initialMatrix[ 9]; tmp[10] = initialMatrix[10]; tmp[14] = 0;
333  tmp[3] = 0; tmp[7] = 0; tmp[11] = 0; tmp[15] = 1;
334  // 2. tmpVect gets an inverted translation from the matrix
335  Vect3D tmpVect(-initialMatrix[12], -initialMatrix[13], -initialMatrix[14]);
336  // 3. tmpVect is to be done in the inverted coordinate system so multiply
337  tmpVect = tmp * tmpVect;
338  // 4. now we have the nice inverted matrix :)
339  tmp[12] = tmpVect.x;
340  tmp[13] = tmpVect.y;
341  tmp[14] = tmpVect.z;
342  // 5. Stores it !
343  for (int j=0; j<16 ; j++) initialMatrixInverted[j] = tmp[j];
344 
345  // Now, store the rotation matrices (for normals computation)
346  int off=0;
347  for (int i=0; i<4; i++) {
348  for (int j=0; j<4; j++, off++) {
349  if ((i<3) && (j<3)) {
350  initialRotMatrix [off] = initialMatrix [off];
351  initialRotMatrixInverted[off] = initialMatrixInverted[off];
352  }
353  else if (i==j) {
354  initialRotMatrix [off] = 1;
355  initialRotMatrixInverted[off] = 1;
356  }
357  else {
358  initialRotMatrix [off] = 0;
359  initialRotMatrixInverted[off] = 0;
360  }
361  }
362  }
363 
364  // Now, let's do the same for children :)
365  for (int i=0; i < children; i++)
366  child[i]->generateInitialMatrix();
367 
368  glPopMatrix();
369 }

371 // Now let's have a look to the current matrix, same code as above
372 void BoneVertex::generateCurrentMatrix()
373 {
374  if (! childListCompiled) compileChildList();
375 
376  glPushMatrix();
377 
378  glTranslatef(currentPosition.x, currentPosition.y, currentPosition.z);
379  glRotatef(currentAngle, currentAxis.x , currentAxis.y, currentAxis.z);
380  glGetFloatv(GL_MODELVIEW_MATRIX, currentMatrix);
381 
382  // Now we have the current transformation matrix,
383  // we'll extract the rotation part so we can
384  // move the normals of the mesh correctly to
385  // update the lighting !
386  currentRotMatrix[ 0] = currentMatrix[ 0];
387  currentRotMatrix[ 1] = currentMatrix[ 1];
388  currentRotMatrix[ 2] = currentMatrix[ 2];
389  currentRotMatrix[ 3] = 0;
390  currentRotMatrix[ 4] = currentMatrix[ 4];
391  currentRotMatrix[ 5] = currentMatrix[ 5];
392  currentRotMatrix[ 6] = currentMatrix[ 6];
393  currentRotMatrix[ 7] = 0;
394  currentRotMatrix[ 8] = currentMatrix[ 8];
395  currentRotMatrix[ 9] = currentMatrix[ 9];
396  currentRotMatrix[10] = currentMatrix[10];
397  currentRotMatrix[11] = 0;
398  currentRotMatrix[12] = 0;
399  currentRotMatrix[13] = 0;
400  currentRotMatrix[14] = 0;
401  currentRotMatrix[15] = 1;
402 
403  // And now, just generate the childrem matrices
404  for (int i=0; i<children; i++)
405  child[i]->generateCurrentMatrix();
406 
407  glPopMatrix();

422 void BoneVertex::readFromFile(FILE *fp, float scale)
423 {
424  char nameCurrent[512];
425  float posx, posy, posz;
426  float angle;
427  float axisx, axisy, axisz;
428 
429  readString(fp, nameCurrent);
430 
431  posx = readFloat(fp) * scale;
432  posy = readFloat(fp) * scale;
433  posz = readFloat(fp) * scale;
434 
435  angle = readFloat(fp);
436  axisx = readFloat(fp);
437  axisy = readFloat(fp);
438  axisz = readFloat(fp);
439 
440  setName(nameCurrent);
441  setInitialPosition(posx, posy, posz);
442  setInitialRotation(angle, axisx, axisy, axisz);
443 
444  int cpt = readInt(fp); // Number of children
445  for (int i=0; i<cpt; i++) {
446  BoneVertex *tmp = new BoneVertex();
447  addChild(tmp);
448  tmp->readFromFile(fp);
449  }
450 
451  compileChildList();
452 }
