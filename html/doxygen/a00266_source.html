50 void Accessory::parser(char *l)
51 {
52  defaults();
53  l = tokenize(l);
54  begin_while_parse(l) {
55  l = parse()->parseAttributes(l, this);
56  if (!l) break;
57  if (!stringcmp(l, "speed")) l = parse()->parseFloat(l, &speed, "speed");
58  else if (!stringcmp(l, "slowx")) l = parse()->parseFloat(l, &slowx, "slowx");
59  else if (!stringcmp(l, "slowy")) l = parse()->parseFloat(l, &slowy, "slowy");
60  else if (!stringcmp(l, "slowz")) l = parse()->parseFloat(l, &slowz, "slowz");
61  else if (!stringcmp(l, "shiftx")) l = parse()->parseFloat(l, &shiftx, "shiftx");
62  else if (!stringcmp(l, "shifty")) l = parse()->parseFloat(l, &shifty, "shifty");
63  else if (!stringcmp(l, "shiftz")) l = parse()->parseFloat(l, &shiftz, "shiftz");
64  }
65  end_while_parse(l);
66 }

88 void Accessory::changePermanent(float lasting)
89 {
90  if (! following) return;
91 
92  // distances
93  float distx = localuser->pos.x + shiftx - pos.x;
94  float disty = localuser->pos.y + shifty - pos.y;
95  float distz = localuser->pos.z + shiftz - pos.z;
96  float deltax = ABSF(distx / slowx);
97  float deltay = ABSF(disty / slowy);
98  float deltaz = ABSF(distz / slowz);
99  // progression
100  if (distx > 0) pos.x += deltax; else pos.x -= deltax;
101  if (disty > 0) pos.y += deltay; else pos.y -= deltay;
102  if (distz > 0) pos.z += deltaz; else pos.z -= deltaz;
103 
104  updatePosition();
105 }
