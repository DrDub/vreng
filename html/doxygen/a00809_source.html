53 /* Vrelet parser */
54 void Vrelet::parser(char *l)
55 {
56  bool goturl = false;
57 
58  defaults();
59  l = tokenize(l);
60  begin_while_parse(l) {
61  l = parse()->parseAttributes(l, this);
62  if (!l) break;
63  if (!stringcmp(l, "class=")) l = parse()->parseString(l, app, "class"); // java class
64  else if (!stringcmp(l, "incrx")) l = parse()->parseInt(l, &incrx, "incrx");
65  else if (!stringcmp(l, "incry")) l = parse()->parseInt(l, &incry, "incry");
66  else if (!stringcmp(l, "fx")) l = parse()->parseFloat(l, &fx, "fx");
67  else if (!stringcmp(l, "fy")) l = parse()->parseFloat(l, &fy, "fy");
68  else if (!stringcmp(l, "base")) {
69  // url in the .vre file, try to find out the codebase URL
70  l = parse()->parseString(l, url, "base");
71  goturl = true;
72  }
73  }
74  end_while_parse(l);
75 
76  if (! goturl) { // there's none in the .vre file.
77  char l2[URL_LEN];
78  char *last = NULL;
79  // if the world's url is like [...]/vre/v6/world.vre
80  // try to build an url like [...]/jar/vrengapp.jar
81  strcpy(l2, World::current()->getUrl());
82  if ((last = strrchr(l2, '/'))) *last = '\0';
83  if ((last = strrchr(l2, '/')) && (! stringcmp(last, "/vre"))) {
84  *last = '\0';
85  Url::abs(l2, url);
86  strcat(url, DEF_URL_JAR);
87  //error("Vrelet: computed codebase=%s", url);
88  }
89  else { // neither of the above methods worked, just put a default url
90  sprintf(url, "http://%s%s%s", Universe::current()->universe_name, Universe::current()->urlpfx, DEF_URL_JAR);
91  //error("Vrelet: default codebase=%s", url);
92  }
93  }
94 }

166 /* Deal with input from the client app */
167 void Vrelet::processClient()
168 {
169  bool processed = false;
170  int tag;
171  V3 color = newV3(0, 0, 0); //black;
172 
173  // check if the client has sent anything
174  VjcMessage *msg;
175  if ((msg = Vjc::getData(this)) == NULL) return;
176  tVjcHeader header = msg->getHeader();
177 
178  // Process the client's data
179  switch (header.msg_type) {
180 
181  case VJC_MSGT_ADD:
182  tag = msg->read16();
183  color = msg->readPoint3D();
184 
185  if ( (header.msg_id == VJC_MSGV_LINE)
186  || (header.msg_id == VJC_MSGV_LOOP)
187  || (header.msg_id == VJC_MSGV_FILL) ) {
188  // create a new line
189  trace(DBG_IFC, "vrelet get new line");
190  o2 = newObject2D(o2, header.msg_id, tag, color);
191  while (msg->hasData(2*sizeof(int32_t)))
192  addPoint(o2, msg->readPoint2D());
193  processed = true;
194  }
195  else if (header.msg_id == VJC_MSGV_CIRCLE) {
196  trace(DBG_IFC, "vrelet get new circle");
197  // Create a new circle
198  o2 = newObject2D(o2, header.msg_id, tag, color);
199  addPoint(o2, msg->readPoint2D());
200  addPoint(o2, msg->readPoint2D());
201  processed = true;
202  }
203  needRedraw = true;
204  break;
205 
206  case VJC_MSGT_DEL:
207  if (header.msg_id == VJC_MSGV_DELALL) {
208  freeObject2DList(o2);
209  o2 = NULL;
210  }
211  else {
212  tag = msg->read16();
213  o2 = removeObject2D(o2, header.msg_id, tag);
214  }
215  needRedraw = true;
216  break;
217 
218  case VJC_MSGT_POS: {
219  // get the objet ID
220  uint8_t type = msg->read8();
221  uint32_t src = msg->read32();
222  uint16_t port = msg->read16();
223  uint16_t id = msg->read16();
224 
225  // locate the object
226  WObject *who = ((type==0) ? this : OList::findInMobile(type, src, port, id));
227  if (! who) {
228  // we didn't find anything that matched
229  sendPosError(type, src, port, id);
230  }
231  else if (header.msg_id == VJC_MSGV_ASK) {
232  // message was a query - send the data back
233  sendPos(who);
234  processed = true;
235  }
236  else if ((header.msg_id == VJC_MSGV_SET)
237  || (header.msg_id == VJC_MSGV_UPD)) {
238  // message was a move, get the movement information
239  copyV3(&posDelta, msg->readDelta());
240  copyV3(&angDelta, msg->readDelta());
241 
242  if (who == this) // move myself (vrelet object)
243  wantDelta = header.msg_id;
244  else { // push someone else (vreng object)
245  WObject *pold = new WObject(); // needed for collision
246 
247  // make the changes
248  who->copyPositionAndBB(pold);
249  deltaPos(who);
250 
251  // update the object in the Vreng 3D thingies
252  who->updatePositionAndGrid(pold);
253 
254  // check if we need to move the camera (user)
255  if (who == localuser) who->updateCamera(pos);
256 
257  // propagate the changes
258 #if 0 //dax
259  OList *vicinityList = who->getVicinityList(pold);
260  who->generalIntersect(pold, vicinityList);
261  vicinityList->remove();
262 #endif
263  delete pold;
264  }
265  processed = true;
266  }
267  break;
268  }
269 
270  case VJC_MSGT_QUERY:
271  if (header.msg_id == VJC_MSGV_QTYPE) {
272  int type = msg->read32();
273  answerTypeQuery(type);
274  processed = true;
275  }
276  break;
277  }
278 
279  if (! processed) { //error("message ignore %d %d", header.msg_type, header.msg_id);
280  if (msg) delete msg;
281  msg = NULL;
282  }
283 }

285 /* React to a user click on our surface */
286 void Vrelet::click(V3 dir)
287 {
288  // calculate the intersection between the click vector and our primary surface
289 
290  // get two principal vectors and a normal one
291  V3 v, w, normal;
292  getSurfVecs(pos, &v, &w, &normal);
293 
294  // check whether the click comes from the right side of the surface
295  float sp = normal.v[0]*dir.v[0] + normal.v[1]*dir.v[1] + normal.v[2]*dir.v[2];
296  if (sp < 0) { warning("Vrelet::click: bad side!"); return; }
297 
298  // eye position
299  V3 e = newV3(localuser->pos.x, localuser->pos.y, localuser->pos.z + localuser->height/2 - 0.10);
300 
301  // object's center coordinates
302  V3 c = newV3(pos.x, pos.y, pos.z);
303 
304  // determine (x,y) coord. relativly to our surface
305  float det = detV3(dir, v, w);
306  float varx = calcV3(c, e, w, dir) / det;
307  float vary = calcV3(c, e, dir, v) / det;
308 
309  // scale these values
310  int x = (int) (varx * (float) incrx / fx);
311  int y = (int) (vary * (float) incry / fy);
312 
313  // notify the child
314  trace(DBG_IFC, "Vrelet::click: x=%d y=%d", x, y);
315  sendClick(x, y);
316 }

486 /* Draws the Vrelet's 2D object list */
487 void Vrelet::draw()
488 {
489  float rx = incrx / fx;
490  float ry = incry / fy;
491 
492  dlist = glGenLists(1);
493  glNewList(dlist, GL_COMPILE);
494  glTranslatef(0, 0, 0.02); // extra z so that the figures are drawn slightly
495  // above the surface
496  while (o2) {
497  trace(DBG_IFC, "draw2DList: o2->type=%d", o2->type);
498  switch (o2->type) {
499 
500  case VJC_MSGV_LINE:
501  case VJC_MSGV_LOOP:
502  case VJC_MSGV_FILL: {
503  tPointList *pts = o2->points;
504  if (! pts) return;
505 
506  glColor3f(o2->color.v[0], o2->color.v[1], o2->color.v[2]);
507 
508  if (o2->type == VJC_MSGV_LINE)
509  glBegin(GL_LINE_STRIP);
510  else if (o2->type == VJC_MSGV_LOOP)
511  glBegin(GL_LINE_LOOP);
512  else
513  glBegin(GL_POLYGON);
514 
515  for ( ; pts ; pts = pts->next)
516  glVertex2f(pts->point.v[0]/rx, pts->point.v[1]/ry);
517  glEnd();
518  }
519  break;
520 
521  case VJC_MSGV_CIRCLE: {
522  V3 diam = o2->points->point;
523  V3 center = o2->points->next->point;

532  glEnd();
533  }

535  }
