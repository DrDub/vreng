40 void Door::parser(char *l)
41 {
42  mecanism = NOMECANISM;
43  aopen = xopen = zopen = 0;
44  l = tokenize(l);
45  begin_while_parse(l) {
46  l = parse()->parseAttributes(l, this);
47  if (!l) break;
48  if (!stringcmp(l, "open")) {
49  l = parse()->parseFloat(l, &aopen, "open");
50  mecanism = ANGULAR;
51  }
52  else if (!stringcmp(l, "close")) {
53  l = parse()->parseFloat(l, &aclose, "close");
54  mecanism = ANGULAR;
55  }
56  else if (!stringcmp(l, "xopen")) {
57  l = parse()->parseFloat(l, &xopen, "xopen");
58  mecanism = SLIDING;
59  }
60  else if (!stringcmp(l, "xclose")) {
61  l = parse()->parseFloat(l, &xclose, "xclose");
62  mecanism = SLIDING;
63  }
64  else if (!stringcmp(l, "zopen")) {
65  l = parse()->parseFloat(l, &zopen, "zopen");
66  mecanism = VERTICAL;
67  }
68  else if (!stringcmp(l, "zclose")) {
69  l = parse()->parseFloat(l, &zclose, "zclose");
70  mecanism = VERTICAL;
71  }
72  else if (!stringcmp(l, "speed")) {
73  l = parse()->parseFloat(l, &speed, "speed");
74  }
75  }
76  end_while_parse(l);
77 }

80 Door::Door(char *l)
81 {
82  parser(l);
83 
84  switch (mecanism) {
85  case ANGULAR:
86  state = (aclose == pos.az) ? CLOSED : OPENED;
87  break;
88  case SLIDING:
89  state = (xclose == pos.x) ? CLOSED : OPENED;
90  break;
91  case VERTICAL:
92  state = (zclose == pos.z) ? CLOSED : OPENED;
93  break;
94  }
95 
96  /* calls persistency MySql server to know the door state */
97  getMySqlState();
98 
99  switch (mecanism) {
100  case ANGULAR:
101  pos.az = (state & CLOSED) ? aclose : aopen;
102  trace(DBG_WO, "Door: sql state: %d %.2f", state, pos.az);
103  break;
104  case SLIDING:
105  state = (xclose == pos.x) ? CLOSED : OPENED;
106  break;
107  case VERTICAL:
108  state = (zclose == pos.z) ? CLOSED : OPENED;
109  break;
110  }
111 
112  V3 bbs, bbc;
113  getRelativeBB(bbc, bbs);
114  size.v[0] = bbs.v[0];
115  center.v[0] = pos.x; // hinge
116  center.v[1] = pos.y;
117  center.v[2] = pos.z;
118  pos.x += size.v[0] * cos(pos.az);
119  pos.y += size.v[0] * sin(pos.az);
120  //dax trace(DBG_FORCE, "pos=(%.2f,%.2f,%.2f) center=(%.2f,%.2f,%.2f) size=%.2f", pos.x, pos.y, pos.z, center.v[0], center.v[1], center.v[2], size.v[0]);
121 
122  enableBehavior(PERSISTENT); // after init because mysql already done
123  setRenderPrior(RENDER_NORMAL);
124 
125  initializeMobileObject(1);
126  createPermanentNetObject(PROPS, ++oid);
127 }

129 void Door::updateTime(time_t s, time_t us, float *lasting)
130 {
131  switch (mecanism) {
132  case ANGULAR:
133  if (move.aspeed.v[0] == speed)
134  move.ttl = MIN(ABSF(deltaAngle(pos.az, aopen) / move.aspeed.v[0]), move.ttl);
135  else
136  move.ttl = MIN(ABSF(deltaAngle(pos.az, aclose) / move.aspeed.v[0]), move.ttl);
137  break;
138  case SLIDING:
139  if (move.lspeed.v[0] == speed)
140  move.ttl = MIN(ABSF((pos.x - xopen) / move.lspeed.v[0]), move.ttl);
141  else
142  move.ttl = MIN(ABSF((pos.x - xclose) / move.lspeed.v[0]), move.ttl);
143  break;
144  case VERTICAL:
145  if (move.lspeed.v[2] == speed)
146  move.ttl = MIN(ABSF((pos.z - zopen) / move.lspeed.v[2]), move.ttl);
147  else
148  move.ttl = MIN(ABSF((pos.z - zclose) / move.lspeed.v[2]), move.ttl);
149  break;
150  }
151 
152  if (! updateLasting(s, us, lasting) && state == CLOSED)
153  Sound::playSound(DOORCLOSESND);
154 }

207 void Door::open()
208 {
209  if (state == Door::OPENED || state == Door::LOCKED)
210  return;
211  clearV3(move.lspeed);
212  clearV3(move.aspeed);
213  switch (mecanism) {
214  case Door::ANGULAR:
215  move.aspeed.v[0] = speed;
216  initImposedMovement(ABSF(deltaAngle(pos.az, aopen) / move.aspeed.v[0]));
217  break;
218  case Door::SLIDING:
219  move.lspeed.v[0] = speed;
220  initImposedMovement(ABSF((pos.x - xopen) / move.lspeed.v[0]));
221  break;
222  case Door::VERTICAL:
223  move.lspeed.v[2] = speed;
224  initImposedMovement(ABSF((pos.z - zopen) / move.lspeed.v[2]));
225  break;
226  }
227  Sound::playSound(DOOROPENSND);
228  state = OPENED;
229  pos.moved = true; // has moved
230  updateMySqlState(state);
231 }

233 void Door::close()
234 {
235  if (state & Door::CLOSED)
236  return;
237  clearV3(move.lspeed);
238  clearV3(move.aspeed);
239  switch (mecanism) {
240  case Door::ANGULAR:
241  move.aspeed.v[0] = -speed;
242  initImposedMovement(ABSF(deltaAngle(pos.az, aclose) / move.aspeed.v[0]));
243  break;
244  case Door::SLIDING:
245  move.lspeed.v[0] = -speed;
246  initImposedMovement(ABSF((pos.x - xclose) / move.lspeed.v[0]));
247  break;
248  case Door::VERTICAL:
249  move.lspeed.v[2] = -speed;
250  initImposedMovement(ABSF((pos.z - zclose) / move.lspeed.v[2]));
251  break;
252  }
253  state = CLOSED;
254  pos.moved = true; // has moved
255  updateMySqlState(state);
256 }
