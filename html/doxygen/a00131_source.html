32 int Url::parser(char *url, char *host, char *scheme, char *path)
33 {
34  int urltype;
35  char *p;
36  static char prev_host[MAXHOSTNAMELEN] = ""; // first one is undefined
37 
38  /* parse scheme */
39  if (! url || ! strlen(url)) {
40  *scheme = '\0';
41  trace(DBG_HTTP, "BAD URL: %s", url);
42  return URLBAD;
43  }
44  if (! stringcmp(url, "http://")) {
45  urltype = URLHTTP;
46  strcpy(scheme, "http");
47  url += 7;
48  } else if (! stringcmp(url, "file:")) {
49  urltype = URLFILE;
50  strcpy(scheme, "file");
51  url += 5;
52  }
53  else { // incomplete URL
54  trace(DBG_HTTP, "relative URL: %s", url);
55  strcpy(scheme, "http");
56  strcpy(host, Universe::current()->universe_name);
57  if (*url != '/') {
58  // relative url
59  if (! isprint(*url)) {
60  error("Bad Url: %02x%02x%02x host=%s", url[0], url[1], url[2], host);
61  return URLBAD;
62  }
63  sprintf(path, "%s/%s", Universe::current()->urlpfx, url);
64  }
65  else { // /path
66  if (! stringcmp(url, "http://") || ! stringcmp(url, "/~")) {
67  // full qualified url
68  sprintf(path, "%s", url);
69  }
70  else {
71  // add url prefix before url
72  sprintf(path, "%s%s", Universe::current()->urlpfx, url);
73  }
74  }
75  return URLHTTP;
76  }
77 
78  /* then parse host's name */
79  if (urltype != URLFILE) {
80  p = prev_host;
81  while ((*url != ':') && (*url != '/') && (*url != '\0')) {
82  *host++ = *url;
83  *p++ = *url; // keep current host
84  url++;
85  }
86  *host = '\0';
87  *p = '\0';
88  }
89 
90  /* then parse pathname */
91  if (*url == '\0')
92  strcpy(path, "/");
93  else // file
94  strcpy(path, url);
95  return urltype;
96 }
