471 /* system of equations handling permanent motion */
472 void Mech::changePermanent(float lasting)
473 {
474  float angle;
475  static int step;
476 
477  if (!anim) return;
478 
479  if (step == 0 || step == 2) {
480  if (mframe >= 0 && mframe <= 21) {
481  if (mframe == 0)
482  mframe = 3;
483  angle = (180 / M_PI) * (acos(((cos((M_PI / 180) * mframe) * 2.043) + 1.1625) / 3.2059));
484  if (mframe > 0)
485  elevation = -(3.2055 - (cos((M_PI / 180) * angle) * 3.2055));
486  else
487  elevation = 0;
488 
489  if (step == 0) {
490  hip11 = (int) -(mframe * 1.7);
491  if (1.7 * mframe > 15)
492  heel1 = (int) (mframe * 1.7);
493  heel2 = 0;
494  ankle1 = (int) (mframe * 1.7);
495  if (mframe > 0)
496  hip21 = (int) angle;
497  else
498  hip21 = 0;
499  ankle2 = -hip21;
500  shoulder1 = (int) (mframe * 1.5);
501  shoulder2 = (int) -(mframe * 1.5);
502  elbow1 = (int) mframe;
503  elbow2 = (int) -mframe;
504  } else {
505  hip21 = (int) -(mframe * 1.7);
506  if (1.7 * mframe > 15)
507  heel2 = (int) (mframe * 1.7);
508  heel1 = 0;
509  ankle2 = (int) (mframe * 1.7);
510  if (mframe > 0)
511  hip11 = (int) angle;
512  else
513  hip11 = 0;
514  ankle1 = -hip11;
515  shoulder1 = (int) -(mframe * 1.5);
516  shoulder2 = (int) (mframe * 1.5);
517  elbow1 = (int) -mframe;
518  elbow2 = (int) mframe;
519  }
520  if (mframe == 21)
521  step++;
522  if (mframe < 21)
523  mframe = mframe + 3;
524  }
525  }
526  if (step == 1 || step == 3) {
527  /* for(x=21; x>=0; x=x-3){ */
528  if (mframe <= 21 && mframe >= 0) {
529  angle = (180 / M_PI) * (acos(((cos((M_PI / 180) * mframe) * 2.043) + 1.1625) / 3.2029));
530  if (mframe > 0)
531  elevation = -(3.2055 - (cos((M_PI / 180) * angle) * 3.2055));
532  else
533  elevation = 0;
534  if (step == 1) {
535  elbow2 = hip11 = (int) -mframe;
536  elbow1 = heel1 = (int) mframe;
537  heel2 = 15;
538  ankle1 = (int) mframe;
539  if (mframe > 0)
540  hip21 = (int) angle;
541  else
542  hip21 = 0;
543  ankle2 = -hip21;
544  shoulder1 = (int) (1.5 * mframe);
545  shoulder2 = (int) -(mframe * 1.5);
546  } else {
547  elbow1 = hip21 = (int) -mframe;
548  elbow2 = heel2 = (int) mframe;
549  heel1 = 15;
550  ankle2 = (int) mframe;
551  if (mframe > 0)
552  hip11 = (int) angle;
553  else
554  hip11 = 0;
555  ankle1 = -hip11;
556  shoulder1 = (int) -(mframe * 1.5);
557  shoulder2 = (int) (mframe * 1.5);
558  }
559  if (mframe == 0)
560  step++;
561  if (mframe > 0)
562  mframe = mframe - 3;
563  }
564  }
565  if (step == 4)
566  step = 0;
567 }
