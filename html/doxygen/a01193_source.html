145 float Obj::getRadius()
146 {
147  double max_radius = 0.0;
148 
149  for (int i=0; i < OBJModel.numOfObjects; i++) {
150  if (OBJModel.pObject.size() <= 0) break;
151  tOBJObject *pobject = &OBJModel.pObject[i];
152  for (int j=0; j < pobject->numOfFaces; j++)
153  for (int v = 0; v < 3; v++) {
154  int vi = pobject->pFaces[j].vertIndex[v];
155  double r= pobject->pVerts[vi].x * pobject->pVerts[vi].x +
156  pobject->pVerts[vi].y * pobject->pVerts[vi].y +
157  pobject->pVerts[vi].z * pobject->pVerts[vi].z ;
158  if (r > max_radius) max_radius = r;
159  }
160  }
161  return sqrt(max_radius);

240 void Obj::draw()
241 {
242  if (!loaded) return;
243 
244  if (currentScale != desiredScale) setScale(desiredScale);
245  if (flgcolor) {
246  glMaterialfv(GL_FRONT, GL_DIFFUSE, mat_diffuse);
247  glMaterialfv(GL_FRONT, GL_AMBIENT, mat_ambient);
248  glMaterialfv(GL_FRONT, GL_SPECULAR, mat_specular);
249  }
250 
251  for (int i=0; i < OBJModel.numOfObjects; i++) {
252  if (OBJModel.pObject.size() <= 0) break;
253 
254  tOBJObject *pobject = &OBJModel.pObject[i]; // get the current object
255 
256  glBegin(viewMode);
257  for (int j=0; j < pobject->numOfFaces; j++) {
258  for (int c=0; c < SIDES; c++) { // go through each corner of the triangle and draw it.
259  int vi = pobject->pFaces[j].vertIndex[c]; // get the vertex index for each point of the face
260  if (vi < 0 || vi > pobject->numOfVerts) continue; //DAX BUGFIX: to avoid segfault
261  glNormal3f(pobject->pNormals[vi].x, pobject->pNormals[vi].y, pobject->pNormals[vi].z);
262  if (pobject->pTexVerts) { // if the object has a texture associated with it
263  int ti = pobject->pFaces[j].coordIndex[c]; // get the texture coordinate index
264  glTexCoord2f(pobject->pTexVerts[ti].x, pobject->pTexVerts[ti].y);
265  }
266  glVertex3f(pobject->pVerts[vi].x, pobject->pVerts[vi].y, pobject->pVerts[vi].z);
267  }
268  }
269  glEnd();
270  }

285 void Obj::readFile(tOBJModel *pmodel)
286 {
287  char line[256];
288 
289  while (! feof(fp)) {
290  switch (fgetc(fp)) { // get the beginning character of the current line
291  case 'v': // check if we just read in a 'v' (vertice/normal/textureCoord)
292  if (bJustReadAFace) fillInObjectInfo(pmodel); // save the last object's info
293  readVertexInfo(); // see if it's a vertex "v", normal "vn", or UV coordinate "vt"
294  break;
295  case 'f': // check if we just read in a face header ('f')
296  readFaceInfo();
297  bJustReadAFace = true;
298  break;
299  default:
300  fgets(line, sizeof(line), fp);
301  break;
302  }
303  }
304  fillInObjectInfo(pmodel); // we are done reading in the file
305 }

308 void Obj::readVertexInfo()
309 {
310  Vec3 vNewVertex = {0};
311  Vec2 vNewTexCoord = {0};
312  char line[256];
313 
314  char ch = fgetc(fp); // next character
315 
316  if (ch == ' ') { // if we get a space it must have been a vertex ("v")
317  // here we read in a vertice. the format is "v x y z"
318  fscanf(fp, "%f %f %f", &vNewVertex.x, &vNewVertex.y, &vNewVertex.z);
319  fgets(line, sizeof(line), fp);
320  pVertices.push_back(vNewVertex); // add a new vertice to our list
321  }
322  else if (ch == 't') { // a texture coordinate ("vt")
323  // the format is "vt u v"
324  fscanf(fp, "%f %f", &vNewTexCoord.x, &vNewTexCoord.y);
325  fgets(line, sizeof(line), fp);
326  pTextureCoords.push_back(vNewTexCoord); // add a new texcoord to our list
327 
328  // set the flag that tells us this object has texture coordinates.
329  // we know that the face information will list the vertice AND UV index.
330  // for example, ("f 1 3 2" verses "f 1/1 2/2 3/3")
331  bObjectHasUV = true;
332  }
333  else { // otherwise it's probably a normal so we don't care ("vn")
334  if (ch == 'n') bObjectHasNormal = true;
335  // we calculate our own normals at the end so we read past them.
336  fgets(line, sizeof(line), fp);
337  }
338 }

341 void Obj::readFaceInfo()
342 {
343  tOBJFace newFace;
344  char line[255];
345 
346  if (bObjectHasUV) { // check if this object has texture coordinates
347  // here we read in the object's vertex and texture coordinate indices.
348  // the format: "f vertexIndex1/coordIndex1 vertexIndex2/coordIndex2 vertexIndex3/coordIndex3"
349  int vi[4]={0,0,0,0}, ti[4]={0,0,0,0}, ni[4]={0,0,0,0};
350 
351  for (int c=0; c < 4; c++) {
352  int r;
353  if (bObjectHasNormal) {
354  if (fscanf(fp, "%d/%d/%d", &vi[c], &ti[c], &ni[c]) == 3) {
355  newFace.vertIndex[c] = vi[c];
356  newFace.coordIndex[c] = ti[c];
357  }
358  else if (fscanf(fp, "%d//%d", &vi[c], &ni[c]) == 2) {
359  newFace.vertIndex[c] = vi[c];
360  }
361  else {
362  newFace.vertIndex[3] = vi[2];
363  newFace.coordIndex[3] = ti[2];
364  }
365  }
366  else {
367  if ((r = fscanf(fp, "%d/%d", &vi[c], &ti[c])) == 2) {
368  newFace.vertIndex[c] = vi[c];
369  newFace.coordIndex[c] = ti[c];
370  }
371  else {
372  newFace.vertIndex[3] = vi[2];
373  newFace.coordIndex[3] = ti[2];
374  }
375  }
376  }
377  }
378  else { // the object does NOT have texture coordinates
379  // here we read in just the object's vertex indices.
380  // the format: "f vertexIndex1 vertexIndex2 vertexIndex3"
381  fscanf(fp, "%d %d %d", &newFace.vertIndex[0], &newFace.vertIndex[1], &newFace.vertIndex[2]);
382  }
383  fgets(line, sizeof(line), fp); // purge current line
384  pFaces.push_back(newFace); // add the new face to our face list
385 }

388 void Obj::fillInObjectInfo(tOBJModel *pmodel)
389 {
390  tOBJObject newObject = {0};
391  int textureOffset = 0, vertexOffset = 0;
392  int i = 0;
393 
394  // we just finished reading in an object and need to increase the object count.
395  pmodel->numOfObjects++;
396 
397  // add a new object to the list of objects
398  pmodel->pObject.push_back(newObject);
399 
400  // get a pointer to the current object
401  tOBJObject *pobject = &(pmodel->pObject[pmodel->numOfObjects - 1]);
402 
403  // now that we have our list's full of information, we can get the size
404  pobject->numOfFaces = pFaces.size();
405  pobject->numOfVerts = pVertices.size();
406  pobject->numTexVertex = pTextureCoords.size();
407 
408  trace(DBG_MAN, "numOfObjects=%d numOfFaces=%d numOfVerts=%d numTexVertex=%d", pmodel->numOfObjects, pobject->numOfFaces, pobject->numOfVerts, pobject->numTexVertex);
409 
410  // if we read in any faces for this object (required)
411  if (pobject->numOfFaces)
412  pobject->pFaces = new tOBJFace [pobject->numOfFaces];
413 
414  // if we read in any vertices for this object (required)
415  if (pobject->numOfVerts)
416  pobject->pVerts = new Vec3 [pobject->numOfVerts];
417 
418  // if we read in any texture coordinates for this object (optional)
419  if (pobject->numTexVertex) {
420  pobject->pTexVerts = new Vec2 [pobject->numTexVertex];
421  pobject->bHasTexture = true;
422  }
423 
424  for (i=0; i < pobject->numOfFaces; i++) { // go through all of the faces
425  pobject->pFaces[i] = pFaces[i]; // copy the current face from the temporary list
426 
427  // if a new object is found in the file, the face and texture indices start
428  // from the last index that was used in the last object. That means that if
429  // the last one was 8, it would then go to 9 for the next object. We need to
430  // bring that back down to 1, so we just create an offset that we subtract
431  // from the vertex and UV indices.
432 
433  if (i == 0 && ! flgpart) { // check if this is the first face
434  // if the first index is not 1, then we must be past the first object
435  if (pobject->pFaces[0].vertIndex[0] != 1) {
436  // to create the offset we take the current starting point minus 1.
437  // lets say the last object ended at 8. we would then have 9 here.
438  // we want to then subtract 8 from the 9 to get back to 1.
439  vertexOffset = pobject->pFaces[0].vertIndex[0] - 1;
440 
441  // the same goes for texture coordinates, if we have them do the same
442  if (pobject->numTexVertex > 0)
443  textureOffset = pobject->pFaces[0].coordIndex[0] - 1; // take the current index
444  }
445  }
446  // because the face indices start at 1, we need to minus 1 from them due
447  // to arrays being zero based, this is VERY important!
448  for (int j=0; j<SIDES; j++) {
449  // for each index, minus 1 to conform with zero based arrays.
450  // we also need to add the vertex and texture offsets to subtract
451  // the total amount necessary for this to work. The first object
452  // will have a offset of 0 for both since it starts at 1.
453  pobject->pFaces[i].vertIndex[j] -= 1 + vertexOffset;
454  pobject->pFaces[i].coordIndex[j] -= 1 + textureOffset;
455  //int vi = pobject->pFaces[i].vertIndex[j];
456  //if (vi < 0) error("vi=%d vo=%d face=%d c=%d", vi, vertexOffset, i, j);
457  }
458  }
459  for (i=0; i < pobject->numOfVerts; i++) // go through all the vertices in the object
460  pobject->pVerts[i] = pVertices[i]; // copy the current vertice from the temporary list
461  for (i=0; i < pobject->numTexVertex; i++) // go through all of the texture coordinates
462  pobject->pTexVerts[i] = pTextureCoords[i]; // copy the current UV coordinate
463 
464  // .OBJ files don't have materials, we set the material ID to -1.
465  pobject->materialID = -1;
466 
467  pVertices.clear();
468  pFaces.clear();
469  pTextureCoords.clear();
470 
471  bObjectHasUV = false;
472  bObjectHasNormal = false;
473  bJustReadAFace = false;
474 }

477 void Obj::computeNormals(tOBJModel *pmodel)
478 {
479  Vec3 vVector1, vVector2, vNormal, vPoly[3];
480 
481  if (pmodel->numOfObjects <= 0) return; // if there are no objects, we can skip
482 
483  for (int index = 0; index < pmodel->numOfObjects; index++) { // for each of the objects
484  tOBJObject *pobject = &(pmodel->pObject[index]); // get the current object
485 
486  Vec3 *pNormals = new Vec3 [pobject->numOfFaces];
487  Vec3 *pTempNormals = new Vec3 [pobject->numOfFaces];
488  pobject->pNormals = new Vec3 [pobject->numOfVerts];
489 
490  for (int i=0; i < pobject->numOfFaces; i++) { // go though all of the faces
491  // we extract the 3 points of this face
492  int vx = pobject->pFaces[i].vertIndex[0];
493  int vy = pobject->pFaces[i].vertIndex[1];
494  int vz = pobject->pFaces[i].vertIndex[2];
495  int vc = pobject->numOfVerts;
496  if (vx > vc || vy > vc || vz > vc) {
497  error("Obj: vx=%d vy=%d vz=%d vc=%d", vx, vy, vz, vc);
498  continue; //dax BUG: segfault
499  }
500  vPoly[0] = pobject->pVerts[pobject->pFaces[i].vertIndex[0]];
501  vPoly[1] = pobject->pVerts[pobject->pFaces[i].vertIndex[1]];
502  vPoly[2] = pobject->pVerts[pobject->pFaces[i].vertIndex[2]];
503 
504  // let's calculate the face normals (get 2 vectors and find the cross product of those 2)
505  vVector1 = Vector(vPoly[0], vPoly[2]); // get the vector of the polygon (we just need 2 sides)
506  vVector2 = Vector(vPoly[2], vPoly[1]); // get a second vector of the polygon
507  vNormal = Cross(vVector1, vVector2); // return the cross product of the 2 vectors
508  pTempNormals[i] = vNormal; // save the un-normalized normal for the vertex normals
509  vNormal = Normalize(vNormal); // normalize the cross product to give us the polygons normal
510  pNormals[i] = vNormal; // assign the normal to the list of normals
511  }
512 
513  Vec3 vSum = {0.0, 0.0, 0.0};
514  Vec3 vZero = vSum;
515  int shared=0;
516 
517  // get the vertex normals
518  for (int i=0; i < pobject->numOfVerts; i++) { // go through all of the vertices
519  for (int j=0; j < pobject->numOfFaces; j++) {// go through all of the triangles
520  // check if the vertex is shared by another face
521  if (pobject->pFaces[j].vertIndex[0] == i ||
522  pobject->pFaces[j].vertIndex[1] == i ||
523  pobject->pFaces[j].vertIndex[2] == i) {
524  vSum = AddVector(vSum, pTempNormals[j]); // add the unnormalized normal of the shared face
525  shared++; // increase the number of shared triangles
526  }
527  }
528  // get the normal by dividing the sum by the shared. we negate so normals pointing out.
529  pobject->pNormals[i] = DivideVectorByScaler(vSum, float(-shared));
530 
531  // normalize the normal for the final vertex normal
532  pobject->pNormals[i] = Normalize(pobject->pNormals[i]);
533  vSum = vZero; // reset the sum
534  shared = 0; // reset the shared
535  }
536  if (pTempNormals) delete[] pTempNormals;
537  if (pNormals) delete[] pNormals;
538  }
