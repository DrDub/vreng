109 void Global::quitVreng(int signum)
110 {
111  static int inquit = 0;
112 
113  if (signum > 0) {
114  cerr << "Got signal " << signum <<
115  ", aborting VREng (pid=" << getpid() << " ppid=" << getppid() << ")" << endl;
116  signal(signum, SIG_DFL);
117  longjmp(sigctx, signum);
118  }
119  if (inquit > 0) {
120  trace(DBG_FORCE, "quit: inquit=%d sig=%d", inquit, signum);
121  if (inquit > 3) exit(signum); // force exit without quitting properly
122  else inquit++;
123  }
124  inquit++;
125 
126  printStats();
127 
128  // close modules properly
129  ::g.render.quit();
130  if (World::current()) World::current()->quit();
131  delete Channel::current();
132  delete Channel::getManager();
133 
134  // quit the application (and close the main window)
135  UAppli::quit(signum);
136 }
