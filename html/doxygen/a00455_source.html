65 void Flag::inits()
66 {
67  width = pos.bbsize.v[0];
68  height = pos.bbsize.v[2];
69 
70  texid = Texture::current();
71 
72  Wind *wind = World::current()->wind;
73  if (wind) { // get orientation and speed of the wind from meteo server
74  zrot = RAD2DEG(wind->getOrient()) + 180;
75  uint8_t speed = wind->getSpeed();
76  force = MAX(speed / 20, 1); // [1..5]
77  }
78  else {
79  zrot = RAD2DEG(pos.az);
80  force = 1;
81  }
82  //error("w=%.2f h=%.2f r=%.2f f=%d", width, height, zrot, force);

84  for (int x=0; x < DIM_FLAG + 1; x++) {
85  for (int y=0; y < DIM_FLAG + 1; y++) {
86  mesh[x][y][0] = (GLfloat) (x - DIM_FLAG/2) / 5.;
87  mesh[x][y][1] = (GLfloat) (y - DIM_FLAG/2) / 5.;
88  mesh[x][y][2] = (GLfloat) (sin(((x * (180./DIM_FLAG))/180.) * M_PI));
89  //error("mesh: %d,%d %.2f %.2f %.2f",x,y, mesh[x][y][0], mesh[x][y][1], mesh[x][y][2]);
