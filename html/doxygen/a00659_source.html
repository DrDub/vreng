53 void Rain::parser(char *l)
54 {
55  defaults();
56  l = tokenize(l);
57  begin_while_parse(l) {
58  l = parse()->parseAttributes(l, this);
59  if (!l) break;
60  else if (!stringcmp(l, "number")) l = parse()->parseUInt16(l, &number, "number");
61  else if (!stringcmp(l, "flow")) l = parse()->parseFloat(l, &flow, "flow");
62  else if (!stringcmp(l, "speed")) l = parse()->parseFloat(l, &speed, "speed");
63  else if (!stringcmp(l, "ground")) l = parse()->parseFloat(l, &ground, "ground");
64  else if (!stringcmp(l, "color")) l = parse()->parseVector3f(l, color, "color");
65  else if (!stringcmp(l, "size")) l = parse()->parseUInt8(l, &pt_size, "size");
66  }
67  end_while_parse(l);
68 }
