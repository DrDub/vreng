88 void Grid::draw()
89 {
90  Scene * s = ::g.gui.getScene();
91  if (!s || !s->isInitialized()) return;
92 
93  if (dlist != -1)
94  glDeleteLists(dlist, 1);
95  dlist = glGenLists(1);
96 
97  glNewList(dlist, GL_COMPILE);
98 
99  /*
100  * draw grid
101  */
102  int grid_3d_height = (grid_3d) ? inter_height : 0;
103 
104  glColor4f(red, green, blue, alpha);
105 
106  glBegin(GL_LINES);
107  for (int j = -grid_3d_height; j <= grid_3d_height; j++) {
108  float x, y, z;
109  z = (float) (j * grid_height) / inter_height;
110  for (int i = -inter_width; i < inter_width; i++) {
111  y = (float) (i * grid_width) / inter_width;
112  glVertex3f(-grid_depth, y, z);
113  glVertex3f( grid_depth, y, z);
114  }
115  for (int i = -inter_depth; i < inter_depth; i++) {
116  x = (float) (i * grid_depth) / inter_depth;
117  glVertex3f(x, -grid_width, z);
118  glVertex3f(x, grid_width, z);
119  }
120  }
121 
122  if (grid_3d) {
123  for (int j = -inter_width; j < inter_width; j++) {
124  float x, y;
125  y = (float) (j * grid_width) / inter_width;
126  for (int i = -inter_depth; i < inter_depth; i++) {
127  x = (float) (i * grid_depth) / inter_depth;
128  glVertex3f(x, y, -grid_height);
129  glVertex3f(x, y, grid_height);
130  }
131  }
132  }
133  glEnd();
134 
135  glEndList();
136 }

138 void Grid::render()
139 {
140  if (! visible)
141  return;
142 
143  glPushMatrix();
144  glDisable(GL_LIGHTING);
145  if (overlap)
146  glDisable(GL_DEPTH_TEST); // no overlap by other objects
147 
148  // get rid of translation movement:
149  GLfloat projection[16];
150  glGetFloatv(GL_MODELVIEW_MATRIX, projection);
151  projection[12] = projection[13] = projection[14] = 0;
152  glLoadMatrixf(projection);
153 
154  if (behavior == SFOLLOW) // load default view matrix
155  glLoadMatrixf(glmat);
156  else if (behavior == STICK) {
157  glPopMatrix();
158  glPushMatrix();
159  }
160 
161  glLineWidth(1);
162  glTranslatef(posx, posy, posz);
163  glRotatef(rotz, 0, 0, 1);
164  glRotatef(rotx, 0, 1, 0);
165  glRotatef(roty, 1, 0, 0);
166  glCallList(dlist);
167 
168  if (overlap)
169  glEnable(GL_DEPTH_TEST);
170  glEnable(GL_LIGHTING);
171  glPopMatrix();
172 }
