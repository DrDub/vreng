36 void V3d::readV3Dfile(BoneMesh *result, BoneVertex *skeletonRoot, char *filename, float scale)
37 {
38  FILE *fp = File::openFile(filename, "rb");
39  if (fp == NULL) return;
40 
41  trace(DBG_MAN, "readV3Dfile: reading mesh and skeleton from V3D file %s", filename);
42 
43  // Reading name
44  char name[512];
45  readString(fp, name);
46  result->setName(name);
47 
48  // Reading vertices
49  int vertices = readInt(fp);
50  for (int i=0; i<vertices; i++) {
51  float x = readFloat(fp) * scale;
52  float y = readFloat(fp) * scale;
53  float z = readFloat(fp) * scale;
54  result->addVertex(x,y,z);
55  }
56  trace(DBG_MAN, "Vertices added: %i", vertices);
57 
58  int facets = readInt(fp);
59  for (int i=0; i<facets; i++) {
60  // Reading indices
61  int index1 = readInt(fp);
62  int index2 = readInt(fp);
63  int index3 = readInt(fp);
64  result->addTriangle(index1, index2, index3);
65  Triangle *triangle = result->triangleList.getElemAt(i);
66  // Reading color
67  float r = readFloat(fp);
68  float g = readFloat(fp);
69  float b = readFloat(fp);
70  float a = readFloat(fp);
71  triangle->setColor(r,g,b,a);
72  // Reading texture coordinantes
73  triangle->u1 = readFloat(fp);
74  triangle->v1 = readFloat(fp);
75  triangle->u2 = readFloat(fp);
76  triangle->v2 = readFloat(fp);
77  triangle->u3 = readFloat(fp);
78  triangle->v3 = readFloat(fp);
79  }
80  trace(DBG_MAN, "Faces added : %i", facets);
81 
82  // Reading skeleton
83  skeletonRoot->readFromFile(fp, scale);
84  trace(DBG_MAN, "Skeleton added!");
85  File::closeFile(fp);
86  result->rebuildNormals();
87 }

92 {
93  FILE *fp = File::openFile(filename, "wb");
94  if (fp == NULL) return;
95 
96  if (! outMesh->vertexListCompiled) outMesh->compileVertexList();
97  if (! outMesh->triangleListCompiled) outMesh->compileTriangleList();
98 
99  int i, index1, index2, index3;
100 
101  trace(DBG_FORCE, "writeV3Dfile: writing mesh and skeleton to V3D file");
102 
103  // Writing name
104  writeString(fp, outMesh->getName());
105 
106  // Writing vertices coordinates
107  writeInt(fp, outMesh->vertices);
108  for (i=0; i<outMesh->vertices; i++) {
109  writeFloat(fp, outMesh->vertex[i]->initialPosition.x);
110  writeFloat(fp, outMesh->vertex[i]->initialPosition.y);
111  writeFloat(fp, outMesh->vertex[i]->initialPosition.z);
112  }
113  trace(DBG_FORCE, " Vertices added: %i", outMesh->vertices);
114 
115  // Writing triangles
116  writeInt(fp, outMesh->triangles);
117  for (i=0; i<outMesh->triangles; i++) {
118  index1 = 0;
119  index2 = 0;
120  index3 = 0;
121  while (outMesh->vertex[index1] != outMesh->triangle[i]->vertex1) index1++;
122  while (outMesh->vertex[index2] != outMesh->triangle[i]->vertex2) index2++;
123  while (outMesh->vertex[index3] != outMesh->triangle[i]->vertex3) index3++;
124  // Writing indices
125  writeInt(fp, index1);
126  writeInt(fp, index2);
127  writeInt(fp, index3);
128  // Writing triangle color
129  writeFloat(fp, outMesh->triangle[i]->colorRed);
130  writeFloat(fp, outMesh->triangle[i]->colorGreen);
131  writeFloat(fp, outMesh->triangle[i]->colorBlue);
132  writeFloat(fp, outMesh->triangle[i]->colorAlpha);
133  // Writing triangle texture coordinates
134  writeFloat(fp, outMesh->triangle[i]->u1);
135  writeFloat(fp, outMesh->triangle[i]->v1);
136  writeFloat(fp, outMesh->triangle[i]->u2);
137  writeFloat(fp, outMesh->triangle[i]->v2);
138  writeFloat(fp, outMesh->triangle[i]->u3);
139  writeFloat(fp, outMesh->triangle[i]->v3);
140  }
141  trace(DBG_FORCE, " Faces added : %i", outMesh->triangles);
142 
143  // Writing skeleton
144  skeletonRoot->writeToFile(fp);
145  trace(DBG_FORCE, " Skeleton added!");
146  File::closeFile(fp);
147 }
