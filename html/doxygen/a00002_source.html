64 FILE * Cache::openCache(const char *url, Http *http)
65 {
66  char filepath[PATH_LEN];
67  if (! setCacheName(url, filepath)) return NULL;
68  //error("url=%s http=%p", url, http);
69 
70  FILE *f;
71  if ((f = File::openFile(filepath, "r")) == NULL) {
72  if ((f = File::openFile(filepath, "w")) == NULL) {
73  error("httpReader: can't create %s", filepath);
74  return NULL;
75  }
76 
77  if (! http) return NULL;
78 
79  // writing the file into the cache
80  int c;
81  while ((c = http->getChar()) >= 0)
82  putc(c, f);
83  File::closeFile(f);
84  progression('c'); // c as cache
85  }
86  if (::g.pref.refresh) { // force reload
87  int c;
88  progression('h'); // h as http
89  while ((c = http->getChar()) >= 0)
90  putc(c, f);
91  File::closeFile(f);
92  progression('c');
93  }
94  // and open it for reading
95  f = File::openFile(filepath, "r");
96 
97  return f; // file is opened
98 }

146 {

156  curl_easy_setopt(hcurl, CURLOPT_USERAGENT, PACKAGE_NAME "/" PACKAGE_VERSION);
157  curl_easy_setopt(hcurl, CURLOPT_FOLLOWLOCATION, 1);
158  curl_easy_setopt(hcurl, CURLOPT_TIMEOUT, 60L);
159  curl_easy_setopt(hcurl, CURLOPT_VERBOSE, 1);
160  if (filename) {
161  if ((fp = File::openFile(filename, "wb")) == NULL) {
162  perror("open wb");
163  return 0;
164  }
165  curl_easy_setopt(hcurl, CURLOPT_VERBOSE, 0);
166  curl_easy_setopt(hcurl, CURLOPT_WRITEFUNCTION, NULL);
167  curl_easy_setopt(hcurl, CURLOPT_WRITEDATA, fp);
168  if (stringcmp(arg, "anon") == 0) {
169  curl_easy_setopt(hcurl, CURLOPT_USERPWD, "ftp:vreng@");
170  }
171  if (stringcmp(arg, "inout") == 0) {
172  curl_easy_setopt(hcurl, CURLOPT_READFUNCTION, NULL);
173  }
174  }
175  else if (stringcmp(arg, "check") == 0) {
176  curl_easy_setopt(hcurl, CURLOPT_VERBOSE, 0);
177  curl_easy_setopt(hcurl, CURLOPT_NOBODY, 1);
178  }
179  res = curl_easy_perform(hcurl);
180  if (res) {
181  error("curl: %s", curl_easy_strerror(res));
182  }
183  else {
184  curl_easy_getinfo(hcurl, CURLINFO_SIZE_DOWNLOAD, &size);
185  }
186  if (fp) File::closeFile(fp);
187  curl_easy_cleanup(hcurl);
188  return 1;
189  }
190  else {
191  error("curl: can't init");
192  return 0;
193  }
194 #else
195  return 0;
196 #endif
197 }

199 int Cache::download(const char *_url, char *filename, const char arg[])
200 {
201  char url[URL_LEN];
202 
203  Url::abs(_url, url);
204 
205  if (filename && (! strcmp(arg, "cache"))) { // use the cache
206  if (! Cache::setCacheName(_url, filename)) return 0;
207 
208  struct stat bufstat;
209  if (stat(filename, &bufstat) == 0) {
210  trace(DBG_TOOL, "download: %s found in cache", filename);
211  if (bufstat.st_size != 0)
212  return 1; // file yet in the cache
213  unlink(filename); // remove empty file
214  }
215  trace(DBG_TOOL, "download: download %s in %s", url, filename);
216  }
217 
218 #if HAVE_CURL
219  return curl(url, filename, arg);
220 #else
221  return Wget::start(url, filename, arg);
