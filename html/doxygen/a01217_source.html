106 void X3d::browseX3dTree(XMLNode* xmlnode, X3dShape* shape)
107 {
108  if (!xmlnode || !shape) return;
109 
110  XMLCSTR nodeName = xmlnode->getName();
111 
112  int attribLimit = xmlnode->nAttribute();
113  const char* attrib = NULL;
114 
115  if (isEqual(nodeName, "Shape")) { //new shape
116  //error("on ajoute un fils shape niveau %d", shape->level + 1);
117  X3dShape* son = new X3dShape(shape->level + 1);
118 
119  shape->childrenShapes.push_back(son);
120  shape = son; //we change of node
121 
122  for (int i=0; i < attribLimit; i++) {
123  attrib = xmlnode->getAttributeName(i);
124  if (isEqual(attrib, "DEF")) {
125  //error("we have an object DEF name: %s",xmlnode->getAttributeValue(i));
126  shape->name = xmlnode->getAttributeValue(i);
127  }
128  }
129  }
130  else if (isEqual(nodeName, "Transform")) {
131  if (xmlnode->nChildNode() != 0) {
132  //error("on ajoute un fils transform niveau %d", shape->level + 1);
133 
134  X3dShape* son = new X3dShape(shape->level + 1);
135 
136  shape->childrenShapes.push_back(son);
137  shape = son; //we change of node
138  }
139 
140  for (int i=0; i < attribLimit; i++) {
141  attrib = xmlnode->getAttributeName(i);
142  if (isEqual(attrib, "translation")) {
143  if (VectorTools::parseFloats(xmlnode->getAttributeValue(i),shape->translation,3))
144  shape->translationOn = true;
145  else error("Translation non correcte !");
146  }
147  else if (isEqual(attrib, "rotation")) {
148  if (VectorTools::parseFloats(xmlnode->getAttributeValue(i),shape->rotation,4))
149  shape->rotationOn = true;
150  else error("Rotation non correcte !");
151  }
152  else if (isEqual(attrib, "scale")) {
153  if (VectorTools::parseFloats(xmlnode->getAttributeValue(i),shape->scale,3))
154  shape->scaleOn = true;
155  else error("Scale non correcte !");
156  }
157  else if (isEqual(attrib, "DEF")) {
158  //error("we have an object DEF name: %s",xmlnode->getAttributeValue(i));
159  shape->name = xmlnode->getAttributeValue(i);
160  }
161  }
162  }
163  else if (isEqual(nodeName, "Material")) {
164  for (int i=0; i < attribLimit; i++) {
165  attrib = xmlnode->getAttributeName(i);
166  if (isEqual(attrib, "emissiveColor")) {
167  if (VectorTools::parseFloats(xmlnode->getAttributeValue(i),shape->emissive,3))
168  shape->emissiveOn = true;
169  }
170  else if (isEqual(attrib, "diffuseColor")) {
171  if (VectorTools::parseFloats(xmlnode->getAttributeValue(i),shape->diffuse,3))
172  shape->diffuseOn = true;
173  }
174  else if (isEqual(attrib, "specularColor")) {
175  if (VectorTools::parseFloats(xmlnode->getAttributeValue(i),shape->specular,3))
176  shape->specularOn = true;
177  }
178  else if (isEqual(attrib, "ambientIntensity")) {
179  if (VectorTools::parseFloats(xmlnode->getAttributeValue(i),&shape->ambientIntensity,1))
180  shape->ambientIntensityOn = true;
181  }
182  else if (isEqual(attrib, "shininess")) {
183  if (VectorTools::parseFloats(xmlnode->getAttributeValue(i),&shape->shininess,1))
184  shape->shininessOn = true;
185  }
186  else if (isEqual(attrib, "transparency")) {
187  if (VectorTools::parseFloats(xmlnode->getAttributeValue(i),&shape->transparency,1))
188  shape->transparencyOn = true;
189  }
190  else if (isEqual(attrib, "DEF")) {
191  //error("We have an object DEF name: %s", xmlnode->getAttributeValue(i));
192  shape->name = xmlnode->getAttributeValue(i);
193  }
194  }
195  }
196  else if (isEqual(nodeName, "ImageTexture")) {
197  //on modifie la texture
198  for (int i=0; i < attribLimit; i++) {
199  attrib = xmlnode->getAttributeName(i);
200  if (isEqual(attrib, "url"))
201  shape->texture = Texture::getFromCache(xmlnode->getAttributeValue(i));
202  }
203  }
204  else if (isEqual(nodeName, "Coordinate")) {
205  for (int i=0; i < attribLimit; i++) {
206  attrib = xmlnode->getAttributeName(i);
207  if (isEqual(attrib, "point"))
208  VectorTools::parseCertifiedVectors(xmlnode->getAttributeValue(i),&shape->meshInfos.Coordinate,3);
209  }
210  }
211  else if (isEqual(nodeName, "TextureCoordinate")) {
212  for (int i=0; i < attribLimit; i++) {
213  attrib = xmlnode->getAttributeName(i);
214  if (isEqual(attrib, "point"))
215  VectorTools::parseCertifiedVectors(xmlnode->getAttributeValue(i),&shape->meshInfos.TextureCoordinate,2);
216  }
217  }
218  else if (isEqual(nodeName, "Color")) {
219  for (int i=0; i < attribLimit; i++) {
220  attrib = xmlnode->getAttributeName(i);
221  if (isEqual(attrib, "color"))
222  VectorTools::parseCertifiedVectors(xmlnode->getAttributeValue(i),&shape->meshInfos.Color,3);
223  }
224  }
225  else if (isEqual(nodeName, "ColorInterpolator") ||
226  isEqual(nodeName, "PositionInterpolator") ||
227  isEqual(nodeName, "ScaleInterpolator") ||
228  isEqual(nodeName, "OrientationInterpolator") ||
229  isEqual(nodeName, "ScalarInterpolator")) {
230  Interpolator temp;
231 
232  if (temp.initArrays(xmlnode)) interpolators.push_back(temp);
233  else error("interpolator rejected !");
234  }
235  else if (isEqual(nodeName, "TimeSensor")) {
236  TimeSensor temp;
237 
238  if (temp.initSensor(xmlnode)) timeSensors.push_back(temp);
239  else error("timesensor rejected !");
240  }
241  else if (isEqual(nodeName, "ROUTE")) {
242  Route temp;
243 
244  if (temp.initRoute(xmlnode)) routes.push_back(temp);
245  else error("route rejected !");
246  }
247  else { //is it a PRIMITIVE ???
248  X3dShapes vrengPrimitiveIndex = X3DNONE;
249 
250  if (X3DNONE != (vrengPrimitiveIndex = isKnownPrimitive(xmlnode->getName()))) {
251  GLuint newPrimitive = drawPrimitive(vrengPrimitiveIndex);
252  shape->meshes.push_back(newPrimitive);
253  }
254  }
255 
256  //we take care of the children nodes
257  int nchild = xmlnode->nChildNode();
258  for (int i=0; i < nchild; i++) {
259  XMLNode mynode = xmlnode->getChildNode(i);
260  browseX3dTree(&mynode, shape);
261  }
262  if (isEqual(nodeName, "IndexedFaceSet")) {
263  shape->meshInfos.colorPerVertex = false;
264 
265  for (int i=0; i < attribLimit; i++) {
266  attrib = xmlnode->getAttributeName(i);
267  if (isEqual(attrib, "colorPerVertex")) {
268  if (isEqual(xmlnode->getAttributeValue(i), "TRUE"))
269  shape->meshInfos.colorPerVertex = true;
270  }
271  else if (isEqual(attrib, "coordIndex")) {
272  std::string temp(xmlnode->getAttributeValue(i));
273  uint32_t offset = temp.find("-1");
274  while (offset != (uint32_t) std::string::npos) {
275  temp.replace(offset, 2, ", ");
276  offset = temp.find("-1");
277  }
278  VectorTools::parseVectors(temp,&shape->meshInfos.coordIndex);
279  }
280  else if (isEqual(attrib,"texCoordIndex")) {
281  std::string temp(xmlnode->getAttributeValue(i));
282  uint32_t offset = temp.find("-1");
283  while (offset != (uint32_t) std::string::npos) {
284  temp.replace(offset, 2, ", ");
285  offset = temp.find("-1");
286  }
287  VectorTools::parseVectors(temp,&shape->meshInfos.texCoordIndex);
288  }
289  else if (isEqual(attrib,"colorIndex")) {
290  std::string temp(xmlnode->getAttributeValue(i));
291  uint32_t offset = temp.find("-1");
292  while (offset != (uint32_t) std::string::npos) {
293  temp.replace(offset, 2, ", ");
294  offset = temp.find("-1");
295  }
296  VectorTools::parseVectors(temp,&shape->meshInfos.colorIndex);
297  }
298  }
299  //We draw the complex mesh with the stored data
300  GLuint newPrimitive = drawMesh(& shape->meshInfos);
301  shape->meshes.push_back(newPrimitive);
302 
303  //we reset everything for the next indexedFaceSet
304  shape->meshInfos.coordIndex.clear();
305  shape->meshInfos.texCoordIndex.clear();
306  shape->meshInfos.colorIndex.clear();
307  shape->meshInfos.Coordinate.clear();
308  shape->meshInfos.TextureCoordinate.clear();
309  shape->meshInfos.Color.clear();
310  }

313 void X3d::setupInterpolators()
314 {
315  for (std::vector<Route>::iterator route = routes.begin(); route != routes.end(); route++) {
316  X3DOUTField TempOutfield = route->fromField;
317  X3DINField TempInfield = route->toField;
318 
319  //timer to interpolator route
320  if (TempOutfield == FRACTION_CHANGED && TempInfield == SET_FRACTION) {
321  TimeSensor* sensor = findItem<TimeSensor>(&timeSensors, route->fromNode);
322  Interpolator* interpolator = findItem<Interpolator>(&interpolators, route->toNode);
323 
324  if (sensor && interpolator) sensor->initTarget(interpolator);
325  else error("route sensor->interpolator buggee: de %p a %p", sensor, interpolator);
326  }
327 
328  else if (TempOutfield == VALUE_CHANGED && TempInfield >= TRANSLATION && TempInfield <= TRANSPARENCY) { //interpolator to shape field
329  Interpolator* interpolator = findItem<Interpolator>(&interpolators, route->fromNode);
330  X3dShape* targetShape = findShape(&rootShape, route->toNode);
331 
332  if (interpolator && targetShape) interpolator->initTarget(targetShape, TempInfield);
333  else error("route interpolator->x3dshape buggee: de %p a %p", interpolator, targetShape);
334  }
335  else
336  error("bugged route with improper fields from %s to %s", route->fromNode.c_str(), route->toNode.c_str());
337  } //end loop on routes
338 
339  //error("initialisation des liens d'animation done");
340  routes.clear();
341 }

513

515  uint32_t tempRealVertex = (uint32_t)meshInfos->coordIndex[polygonNum][vertexNum];

518  //if a vertex exists at that index

524  //one color per polygon

526 
527  //we really have a list of colors for that polygon
528  if (polygonNum < meshInfos->colorIndex.size()) {
529 
530  //if a color index exists for that vertex in the polygon
531  if (vertexNum < meshInfos->colorIndex[polygonNum].size()) {
532 
533  //this is the color index we retrieve from the index list
534  uint32_t tempRealColor = (uint32_t)meshInfos->colorIndex[polygonNum][vertexNum];

546  }
547  }
548  }
549  }
550  //texture coords to apply ?

565  meshInfos->Coordinate[realVertex][0],
566  meshInfos->Coordinate[realVertex][1],
567  meshInfos->Coordinate[realVertex][2]);
568  }
569  else error("coordonnees du sommet incorrectes");
570  }
571  else error("index de sommet hors-limites");
572  }
573  glEnd();
574  }
575  glEndList();
576  return X3dDisplayList;

579 void X3d::displayShape(X3dShape* myShape) //NOT RECURSIVE !!
580 {
581  //error("On display la shape: %d au niveau %d",(int) myShape, myShape->level);
582 
583  float reset[4] = {0,0,0,1};
584  glMaterialfv(GL_FRONT, GL_AMBIENT, reset);
585 
586  if (myShape->ambientIntensityOn) {
587  float ambient[4];
588  ambient[0] = myShape->ambientIntensity * myShape->diffuse[0];
589  ambient[1] = myShape->ambientIntensity * myShape->diffuse[1];
590  ambient[2] = myShape->ambientIntensity * myShape->diffuse[2];
591  ambient[3] = myShape->transparencyOn ? myShape->transparency : 1;
592 
593  glMaterialfv(GL_FRONT, GL_AMBIENT, ambient);
594  }
595  if (myShape->shininessOn)
596  glMaterialf(GL_FRONT, GL_SHININESS, 128 * myShape->shininess);
597  if (myShape->diffuseOn) {
598  myShape->diffuse[3] = myShape->transparencyOn ? myShape->transparency : 1;
599  glMaterialfv(GL_FRONT, GL_DIFFUSE, myShape->diffuse);
600  }
601  if (myShape->specularOn) {
602  myShape->specular[3] = myShape->transparencyOn ? myShape->transparency : 1;
603  glMaterialfv(GL_FRONT, GL_SPECULAR, myShape->specular);
604  }
605  if (myShape->emissiveOn) {
606  myShape->emissive[3] = myShape->transparencyOn ? myShape->transparency : 1;
607  glMaterialfv(GL_FRONT, GL_EMISSION, myShape->emissive);
608  }
609  if (myShape->translationOn)
