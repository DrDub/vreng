74 bool Wall::whenIntersect(WObject *pcur, WObject *pold)
75 {
76  switch (pcur->type) {
77  case USER_TYPE:
78  case THING_TYPE:
79  case BALL_TYPE:
80  projectPosition(pcur, pold);
81  break;
82  case ICON_TYPE:
83  // stick the icon on the wall
84  doAction(ICON_TYPE, Icon::STICK, this, pcur, 0, 0);
85  pold->copyPosAndBB(pcur->pos);
86  break;
87  case BULLET_TYPE:
88  case DART_TYPE:
89  pcur->toDelete();
90  break;
91  case STEP_TYPE: // escalator
92  case GUIDE_TYPE:
93  case WEB_TYPE:
94  return false;
95  default:
96  pold->copyPosAndBB(pcur->pos);
97  }
98  return true;
99 }
