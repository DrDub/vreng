132 void Web::pivot()
133 {
134  float nexta;
135 
136  acurr = pos.az;
137  atemp = 0;
138  switch (state) {
139  case 0: nexta = aback; break;
140  case 1: nexta = afront; break;
141  }
142  pos.moved = false;
143  clearV3(move.aspeed);
144  move.aspeed.v[0] = aspeed;
145  initImposedMovement(TTL);
146  if (text) { // get rid of previous text
147  text->delFromList();
148  }
149 
150  if (nexta == aback && legend) {
151  WObject *pleg = new WObject();
152  V3 dim;
153  float green[4] = {0,1,0,1}; // green
154  copyPositionAndBB(pleg);
155  getDim(dim); // get dim
156  pleg->pos.z += (dim.v[2] - 0.15); // 15cm under the top
157  pleg->pos.ax = pos.ax;
158  pleg->pos.az = afront;
159  pos.az = afront;
160  if (dim.v[0] > dim.v[1]) {
161  pleg->pos.x += (dim.v[0] + 0.001) * sin(afront); // 1mm near front face
162  pleg->pos.y -= (dim.v[1] - 0.05) * cos(afront); // 5cm from the left margin
163  }
164  else {
165  pleg->pos.y += (dim.v[0] + 0.001) * sin(afront); // 1mm near front face
166  pleg->pos.x -= (dim.v[1] - 0.05) * cos(afront); // 5cm from the left margin
167  }
168  text = new Text(legend, pleg->pos, 0.25, green);
169  text->setShifts(pleg->pos.x, pleg->pos.y, pleg->pos.z, pleg->pos.az + M_PI, pleg->pos.ax + M_PI_2);
170  delete pleg;
171  }
172  state ^= 1;
173  pos.moved = true; // has moved
174 }
