49 void Terrain::parser(char *l)
50 {
51  defaults();
52  l = tokenize(l);
53  begin_while_parse(l) {
54  l = parse()->parseAttributes(l, this);
55  if (!l) break;
56  if (!stringcmp(l, "height")) l = parse()->parseFloat(l, &height, "height");
57  else if (!stringcmp(l, "color")) l = parse()->parseVector3f(l, color, "color");
58  else if (!stringcmp(l, "level")) l = parse()->parseUInt8(l, &level, "level");
59  else if (!stringcmp(l, "div")) l = parse()->parseFloat(l, &div, "div");
60  else if (!stringcmp(l, "scale")) l = parse()->parseFloat(l, &scale, "scale");
61  }
62  end_while_parse(l);
63 }

94 void Terrain::heights(int l, int x, int y, float h, float f)
95 {
96  if (l == 0) return;
97 
98  int w = l * l;
99  if (y == 0) aux_heights(x, y, x+w, y, h);
100  if (x == 0) aux_heights(x, y, x, y+w, h);
101  aux_heights(x+w, y, x+w, y+w, h);
102  aux_heights(x, y+w, x+w, y+w, h);
103  mesh[(x+w/2)*size+y+w/2] = (mesh[x*size+y]+mesh[(x+w)*size+y]+mesh[x*size+y+w]+mesh[(x+w)*size+y+w])/4;
104  float m = f * mul;
105  heights(l-1, x, y, h/f, m);
106  heights(l-1, x+w/2, y, h/f, m);
107  heights(l-1, x, y+w/2, h/f, m);
108  heights(l-1, x+w/2, y+w/2, h/f, m);

192 void Terrain::setColor(float z)
