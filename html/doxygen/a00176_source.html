89 /*
90  * strcasecmp --
91  *
92  * Compares two strings, ignoring case differences.
93  *
94  * Results:
95  * Compares two null-terminated strings s1 and s2, returning -1, 0,
96  * or 1 if s1 is lexicographically less than, equal to, or greater
97  * than s2.
98  *
99  * Side effects:
100  * None.
101  */
102 int strcasecmp(const char *s1, const char *s2)
103 {
104  uint8_t u1, u2;
105 
106  for ( ; ; s1++, s2++) {
107  u1 = (uint8_t) *s1;
108  u2 = (uint8_t) *s2;
109  if ((u1 == '\0') || (charmap[u1] != charmap[u2])) {
110  break;
111  }
112  }
113  return charmap[u1] - charmap[u2];
114 }

116 /*
117  * strncasecmp --
118  *
119  * Compares two strings, ignoring case differences.
120  *
121  * Results:
122  * Compares up to length chars of s1 and s2, returning -1, 0, or 1
123  * if s1 is lexicographically less than, equal to, or greater
124  * than s2 over those characters.
125  *
126  * Side effects:
127  * None.
128  */
129 int strncasecmp(const char *s1, const char *s2, size_t length)
130 {
131  uint8_t u1, u2;
132 
133  for (; length != 0; length--, s1++, s2++) {
134  u1 = (uint8_t) *s1;
135  u2 = (uint8_t) *s2;
136  if (charmap[u1] != charmap[u2]) {
137  return charmap[u1] - charmap[u2];
138  }
139  if (u1 == '\0') {
140  return 0;
141  }
142  }
143  return 0;
144 }
