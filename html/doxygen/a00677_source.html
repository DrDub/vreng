46 void Slider::parser(char *l)
47 {
48  defaults();
49  l = tokenize(l);
50  begin_while_parse(l) {
51  l = parse()->parseAttributes(l, this);
52  if (!l) break;
53  if (!stringcmp(l, "incrx")) l = parse()->parseFloat(l, &incrx, "incrx");

55  else if (!stringcmp(l, "incrz")) l = parse()->parseFloat(l, &incrz, "incrz");
56  }
57  end_while_parse(l);
58 }

89 void Slider::changePosition(float lasting)
90 {
91  if (! capted) return;
92 
93  // check if user out
94  if (interAABB(capted, this) == 0) {
95  //error("user out %s", names.instance);
96  capted = NULL;
97  state = INACTIVE;
98  return;
99  }
100  //error("user in %s", names.instance);
101  // progression
102  capted->pos.x += incrx * cos(pos.az);
103  capted->pos.y += incry * sin(pos.az);
104  capted->pos.z += incrz;
105  capted->pos.az = pos.az;
106  capted->updatePosition();
107 }
