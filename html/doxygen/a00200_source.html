48  GLint vp[4];
49 
50  glGetIntegerv(GL_VIEWPORT, vp);
51  GLint w = vp[2] -2; // I don't know why -2, but the result is correct

92 void Capture::captureGl2ps(const char *ext)
93 {
94  FILE *fp;
95  GLint buffsize = 0, state = GL2PS_OVERFLOW;
96  GLuint form;
97  char name[World::WORLD_LEN];
98 
99  if (!strcmp(ext, "ps")) form = GL2PS_PS;
100  else if (!strcmp(ext, "eps")) form = GL2PS_EPS;
101  else if (!strcmp(ext, "pdf")) form = GL2PS_PDF;
102  else if (!strcmp(ext, "svg")) form = GL2PS_SVG;
103  else return;
104 
105  sprintf(name, "%s.%s", World::current()->getName(), ext);
106  if ((fp = File::openFile(name, "wb")) == 0) { perror("open"); return; }
107 
108  for (int i=1; state == GL2PS_OVERFLOW; i++) {
109  buffsize += 512*512;
110  gl2psBeginPage(name, PACKAGE_STRING, NULL, form,
111  GL2PS_SIMPLE_SORT, // GL2PS_BSP_SORT,
112  GL2PS_SILENT |
113  GL2PS_BEST_ROOT |
114  GL2PS_OCCLUSION_CULL |
115  GL2PS_NO_BLENDING |
116  GL2PS_DRAW_BACKGROUND |
117  GL2PS_USE_CURRENT_VIEWPORT |
118  GL2PS_SIMPLE_LINE_OFFSET,
119  GL_RGBA, 0, NULL, 0, 0, 0, buffsize, fp, name);
120  Render::sharedRender()->rendering();
121  state = gl2psEndPage();
122  }
123  File::closeFile(fp);
124  notice("capture done in file %s", name);
125 }

189 // stop capture video
190 void Capture::stopVideo()
191 {
192  char jpegbase[16];
193  char swfname[World::WORLD_LEN + 4]; // video flash (swf)
194  char cmd[BUFSIZ];
195 
196  int rate = int(ceil( float(MAX_CAPTURE_COUNT) / (time_t) ::g.times.capture.stop()));
197 
198  notice("sequence capture stops at %d images, rate=%d", capture_no + 1, rate);
199  sprintf(jpegbase, "/tmp/vreng-");
200  sprintf(swfname, "%s.swf", World::current()->getName());
201  sprintf(cmd, "IFS=' '; jpeg2swf -v -q 50 -r %d -o %s %s*.jpg; rm -f %s*.jpg", rate, swfname, jpegbase, jpegbase);
202  system(cmd);
203  notice("file %s created", swfname);
204  is_capturing = false;
205  capture_no =0;
206 }
