124 int Socket::handleBlocking(int sock, bool block)
125 {
126 #if HAVE_FCNTL
127  int flags;
128 
129  if ((flags = fcntl(sock, F_GETFL)) < 0) {
130  error("F_GETFL: %s (%d)", strerror(errno), errno);
131  return -1;
132  }
133  if (block)
134  flags &= ~O_NONBLOCK;
135  else
136  flags |= O_NONBLOCK;
137  if (fcntl(sock, F_SETFL, flags) < 0) {
138  error("F_SETFL: %s (%d)", strerror(errno), errno);
139  return -1;
140  }
141 #endif // HAVE_FCNTL
142  return sock;
143 }
