56 Source * Source::getSource(uint32_t _ssrc)
57 {
58  Channel *pchan = Channel::current(); // maybe indeterminated...
59  if (! pchan->session->source) return NULL;
60 
61  Source *pso, *psolast;
62  int i = 0;
63 
64  for (pso = psolast = pchan->session->source;
65  pso && (i < sourcesCounter);
66  pso = pso->next, i++) {
67  if (pso->ssrc == _ssrc) return pso; // registered source
68  psolast = pso;
69  }
70  /* not yet registered source */
71  if (! pso) {
72  /* no source found, we must create it now */
73  pchan->session->source = new Source(_ssrc);
74 
75  pchan->session->incrSources();
76  trace(DBG_RTP, "getSource: create source=%p", pchan->session->source);
77  return pchan->session->source;
78  }
79  else {
80  error("bad new source");
81  /* new incoming source: create this source */
82  psolast->next = new Source(_ssrc);
83 
84  pchan->session->incrSources();
85  return psolast->next;
86  }
87 }
