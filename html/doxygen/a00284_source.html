43 void AnimX3d::parser(char *l)
44 {
45  defaults();
46  l = tokenize(l);
47  begin_while_parse(l) {
48  l = parse()->parseAttributes(l, this);
49  if (!l) break;
50  if (!stringcmp(l, "url=")) l = parse()->parseUrl(l, names.url);
51  else if (!stringcmp(l, "dim=")) {
52  l = parse()->skipEqual(l);
53  l = parse()->skipQuotes(l); // to get pos.x
54  l = parse()->parseFloat(l, &dimx);
55  l = parse()->parseFloat(l, &dimy);
56  l = parse()->parseFloat(l, &dimz);
57  }
58  }
59  end_while_parse(l);
60 }
