339 {
340  type = type_id;
341  defaults();
342  getMemory(); // alloc geometries
343 
344  replicateVolatileNetObject(PROPS, _noid);
345  copyNoid(_noid);
346 
347  /* hack to retrieve the name and the mapping */
348  noh->getProperty(/* 0 */ PROPHNAME, pp);
349  noh->getProperty(/* 1 */ PROPMAPFRONT, pp);
350  noh->getProperty(/* 2 */ PROPMAPBACK, pp);
351 
352  int idxgeom, idxend = 0;
353  char s[1024];
354  memset(s, 0, sizeof(s));
355 
356  int idxvar = pp->tellPayload(); // note begin of var
357  trace(DBG_WO, "idxvar=%d[%02x]", idxvar, idxvar);
358  if (((idxgeom = pp->tellStrInPayload("shape=\"box\" size=")) > 0) ||
359  ((idxgeom = pp->tellStrInPayload("shape=\"man\" size=")) > 0) ||
360  ((idxgeom = pp->tellStrInPayload("shape=\"guy\" size=")) > 0)
361  ) {
362  /* get replicated user characteristics from the network */
363  trace(DBG_WO, "idxgeom=%d[%02x]", idxgeom, idxgeom);
364  noh->getProperty(/* 09 */ PROPMENSURATION, pp);
365  noh->getProperty(/* 10 */ PROPMAPLEFT, pp);
366  noh->getProperty(/* 11 */ PROPMAPRIGHT, pp);
367  noh->getProperty(/* 12 */ PROPVRE, pp);
368  noh->getProperty(/* 13 */ PROPWEB, pp);

370  noh->getProperty(/* 15 */ PROPRTCPNAME, pp);

372  noh->getProperty(/* 17 */ PROPMODEL, pp);
373  noh->getProperty(/* 18 */ PROPFACE, pp);
374  noh->getProperty(/* 19 */ PROPSEX, pp);
375  noh->getProperty(/* 20 */ PROPHEAD, pp);
376  noh->getProperty(/* 21 */ PROPSKIN, pp);
377  noh->getProperty(/* 22 */ PROPBUST, pp);
378  noh->getProperty(/* 23 */ PROPRAY, pp);
379  idxend = pp->tellPayload(); // note end of properties
380 
381  if (isalpha(*model))
382  sprintf(s, "%s />", mensuration);
383  else
384  sprintf(s, "%s yp=\"%s\" xp=\"%s\" />", mensuration, front, back);

390  }
391  trace(DBG_WO, "Replica: s=%s", s);
392  parse()->parseSolid(s, SEP, this);
393 
394  // get the variable properties
395  if (idxend > 0) {
396  trace(DBG_WO, "Replica: read var props, idxend=%d", idxend);
397  pp->seekPayload(idxvar); // begin prop var
398  for (int np = PROPBEGINVAR; np <= PROPENDVAR; np++) {
399  noh->getProperty(np, pp);
400  }
401  pp->seekPayload(idxend); // end properties
402  }
403  else { // never executed
404  error("never executed idxend=%d", idxend);
405  uint8_t _nbprop = noh->getPropertiesNumber();
406  trace(DBG_WO, "Replica: read all props");
407  for (int np = PROPBEGINVAR; np < _nbprop; np++) {
408  noh->getProperty(np, pp);
409  }
410  }
411 
412  notice("Avatar: rctpname=%s", rtcpname);
413  trace(DBG_WO, "Replica: web=%s vre=%s", web, vre);
414  trace(DBG_WO, "Replica: model=%s face=%s", model, face);
415 
416  initializeMobileObject(0);
417 
418  trace(DBG_WO, "replica: type=%s named=%s name=%s ssrc=%x rtcpname=%s email=%s", names.type, names.named, getInstance(), ssrc, rtcpname, email);

566 bool User::whenIntersect(WObject *pcur, WObject *pold)
567 {
568  // User has no control over ghost objects
569  if (pcur->isBehavior(COLLIDE_GHOST)) return true;
570 
571  switch (pcur->type) {
572  case BULLET_TYPE:
573  case DART_TYPE:
574  /* projectile intersects a stopped user: hit */
575  if (hit == 0) {
576  hit = 1;
577  notice("%s:%s hits %s", pcur->names.type, pcur->getInstance(), getInstance());
578  if (pcur->type == DART_TYPE) {
579  ((Dart *)pcur)->hit = 1;
580  noh->sendDelta(Dart::PROPHIT);
581  ((Dart *)pcur)->hit = 0;
582  Sound::playSound(OUILLESND);
583  }
584  else if (pcur->type == BULLET_TYPE) {
585  ((Bullet *)pcur)->hit = 1;
586  noh->sendDelta(Bullet::PROPHIT);
587  ((Bullet *)pcur)->hit = 0;
588  Sound::playSound(RUPSSND);
589  }
590  }
591  pold->copyPositionAndBB(pcur);
592  pcur->toDelete(); // delete projectile
593  break;
594  case BALL_TYPE:
595  pcur->pos.x += Ball::SHIFTX; // ball shifts
596  pcur->pos.y += Ball::SHIFTY;
597  pcur->updatePositionAndGrid(pold);
598  break;
599  case THING_TYPE:
600  case USER_TYPE:
601  /* user intersects an other user: slide */
602  pcur->pos.x += GOTHROUGH; // step to shift
603  pcur->pos.y += GOTHROUGH;
604  pcur->updatePositionAndGrid(pold);
605  break;
606  case STEP_TYPE:
607  return false; //TODO
608  default:
609  pcur->updatePositionAndGrid(pold);
610  //dax pold->copyPositionAndBB(pcur);
611  }
612  return true;
613 }
