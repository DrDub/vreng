69 void Motion::move(int mspeed)
70 {
71  struct timeval t;
72  gettimeofday(&t, NULL);
73 
74  if (mspeed == 0) { // stop
75  changeKey(minuskey, RELEASED, t.tv_sec, t.tv_usec);
76  changeKey(pluskey, RELEASED, t.tv_sec, t.tv_usec);
77  }
78  else {
79  if (! localuser) return;
80  float new_speed;
81 
82  if (accel > 0) {
83  new_speed = accel * mspeed;
84  }
85  else {
86  if (carrier && carrier->isTaking())
87  new_speed = Carrier::LSPEED;
88  else
89  new_speed = User::LSPEED; // !! incorect !!
90  }
91 
92  if (minuskey == 0) {
93  localuser->specialAction(fun, &new_speed, t.tv_sec, t.tv_usec);
94  changeKey(pluskey, PRESSED, t.tv_sec, t.tv_usec);
95  }
96  else { // minuskey > 0
97  if (new_speed > 0) {
98  localuser->specialAction(fun, &new_speed, t.tv_sec, t.tv_usec);
99  changeKey(minuskey, RELEASED, t.tv_sec, t.tv_usec);
100  changeKey(pluskey, PRESSED, t.tv_sec, t.tv_usec);
101  }
102  else {
103  new_speed = -new_speed;
104  localuser->specialAction(fun, &new_speed, t.tv_sec, t.tv_usec);
105  changeKey(minuskey, PRESSED, t.tv_sec, t.tv_usec);
106  changeKey(pluskey, RELEASED, t.tv_sec, t.tv_usec);
107  }
108  }
109  }
110 }
