65 void Head::parser(char *l)
66 {
67  defaults();
68  l = tokenize(l);
69  begin_while_parse(l) {
70  l = parse()->parseAttributes(l, this);
71  if (!l) break;
72  if (!stringcmp(l, "url=")) l = parse()->parseUrl(l, names.url);
73  else if (!stringcmp(l, "scale=")) l = parse()->parseFloat(l, &scale, "scale");
74  else if (!stringcmp(l, "color=")) l = parse()->parseVector3f(l, color, "color");
75  else if (!stringcmp(l, "model=")) {
76  char str[16];
77  l = parse()->parseString(l, str, "model");
78  if (!stringcmp(str, "male")) { model_e = MALE; }
79  else if (!stringcmp(str, "female")) { model_e = FEMALE; }
80  }
81  }
82  end_while_parse(l);
83 }

241 void Head::changePermanent(float lasting)
242 {
243  if (! taken) return;
244 
245  pos.x = localuser->pos.x + dx;
246  pos.y = localuser->pos.y + dy;
247  switch (model_t) {
248  case MODEL_LWO: pos.z = localuser->pos.z + (localuser->height / 2) - .17; break;
249  case MODEL_OFF: pos.z = localuser->pos.z - (localuser->height / 2) - .03; break;
250  case MODEL_3DS: pos.z = localuser->pos.z + (localuser->height / 2) - .15; break;
251  case MODEL_OBJ: pos.z = localuser->pos.z - (localuser->height / 2) - .10; break;
252  default: pos.z = localuser->pos.z - (localuser->height / 2) - .10; break;
253  }
254  switch (model_t) {
255  case MODEL_LWO: /* dax = M_PI_2; daz = -M_PI_2;*/ break;
256  case MODEL_OFF: daz = M_PI_2; break;
257  case MODEL_3DS: dax = M_PI_2; break;
258  case MODEL_OBJ: daz = M_PI; break;
259  }
260  pos.az = localuser->pos.az + daz;
261  pos.ax = localuser->pos.ax + dax;
262  pos.ay = localuser->pos.ay + day;
263  //error("head: %.2f %.2f %.2f %.2f %.2f %.2f", pos.x,pos.y,pos.z,pos.ax,pos.ay,pos.az);
264  updatePosition();
265 }
