45  char openurl[URL_LEN + 20];
46 
47  if (! newwindow)

50  sprintf(openurl, "openURL(%s)", url);

177  case MOZILLA_TOOL: launchmozilla("mozilla", url); break;
178  case NETSCAPE_TOOL: launchnetscape("netscape", url); break;
179  default: error("%s browser", e_tool);
