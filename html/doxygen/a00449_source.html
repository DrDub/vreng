50 void Firework::parser(char *l)
51 {
52  defaults();
53  l = tokenize(l);
54  begin_while_parse(l) {
55  l = parse()->parseAttributes(l, this);
56  if (!l) break;
57  else if (! stringcmp(l, "number")) l = parse()->parseUInt16(l, &number, "number");
58  else if (! stringcmp(l, "ttl")) l = parse()->parseFloat(l, &ttl, "ttl");
59  else if (! stringcmp(l, "speed")) l = parse()->parseFloat(l, &speed, "speed");
60  else if (! stringcmp(l, "color")) { l = parse()->parseVector3f(l, color, "color"); mycolor = 1; }
61  else if (! stringcmp(l, "size")) l = parse()->parseUInt8(l, &pt_size, "size");
62  }
63  end_while_parse(l);
64 }

111 void Firework::changePosition(float dt)
112 {
113  for (int i=0; i < number; i++) {
114  particles[i].pos[0] += particles[i].vel[0] * dt;
115  particles[i].pos[1] += particles[i].vel[1] * dt;
116  particles[i].pos[2] += particles[i].vel[2] * dt;
117  particles[i].vel[0] -= (rand()%2 ?1:-1) * dt;
118  particles[i].vel[1] -= (rand()%2 ?1:-1) * dt;
119  particles[i].vel[2] -= (rand()%2 ?1:-1) * dt;
120  if (fabs(particles[i].vel[0]) < 10e-3 &&
121  fabs(particles[i].vel[1]) < 10e-3 &&
122  fabs(particles[i].vel[2]) < 10e-3)
123  particles[i].alive = false;
124 
125  if (mycolor) {
126  particles[i].rgb[0] = color[0];
127  particles[i].rgb[1] = color[1];
128  particles[i].rgb[2] = color[2];
129  }
130  else {
131  particles[i].rgb[0] = 1;
132  particles[i].rgb[1] = (float) drand48();
133  particles[i].rgb[2] = (float) drand48();
134  }
135  }
136 }

138 void Firework::render()
139 {
140  if (state == INACTIVE) return;
141 
142  for (int i=0; i < number; i++) {
143  if (! particles[i].alive) continue; // dead
144  // check distance
145  if (sqr(particles[i].pos[0] - pos.x) +
146  sqr(particles[i].pos[1] - pos.y) +
147  sqr(particles[i].pos[2] - pos.z) < w*w + d*d + h*h) {
148  glPushAttrib(GL_ALL_ATTRIB_BITS);
149  glPushMatrix();
150  glEnable(GL_COLOR_MATERIAL);
151  glDisable(GL_LIGHTING);
152  glPointSize(pt_size);
153  glBegin(GL_POINTS);
154  glColor3fv(particles[i].rgb);
155  glVertex3fv(particles[i].pos);
156  glEnd();
157  glEnable(GL_LIGHTING);
158  glPopMatrix();
159  glPopAttrib();
160  }
161  else {
162  generate();
163  }
164  }
165 }
