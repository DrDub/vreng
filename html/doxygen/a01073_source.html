50 Img * Img::loadTGA(void *tex, ImageReader read_func)
51 {
52  uint16_t width, height;
53  uint8_t channel;
54  uint8_t rle, TgaInfo[18];
55  int size;
56 
57  Texture *_tex = (Texture *) tex;
58  FILE *f;
59  if ((f = Cache::openCache(_tex->url, _tex->http)) == NULL) return NULL;
60 
61  /* we read the header */
62  fread(&TgaInfo, 1, 18, f); // Read header
63  width = TgaInfo[12] + TgaInfo[13] * 256;
64  height = TgaInfo[14] + TgaInfo[15] * 256;
65  switch (TgaInfo[16]) { // Read only 32 && 24 bit per pixel
66  case 32:
67  channel = 4; // 32 bit per pixel (RGBA)
68  break;
69  case 24:
70  channel = 3; // 24 bit per pixel (RGB)
71  break;
72  default:
73  File::closeFile(f);
74  return NULL;
75  }
76  trace(DBG_IMG, "loadTGA: type=%d width=%d height=%d channel=%d desc=%x", TgaInfo[2], width, height, channel, TgaInfo[17]);
77  size = width * height * channel;
78 
79  Img *img = new Img(width, height, Img::RGB);
80 
81  // we read the data
82  uint8_t *data = new uint8_t[size]; // for RGB or RGBA image
83  if (!data) {
84  File::closeFile(f);
85  return NULL;
86  }
87 
88  fseek(f, TgaInfo[0], SEEK_CUR);
89  int i = 0, j;
90  uint8_t *pdata = data;
91 
92  switch (TgaInfo[2]) {
93  case 2: // Unmapped RGB image
94  fread(data, 1, size, f);
95  break;
96  case 10: // Run length encoded
97  while (i < size) {
98  fread(&rle, 1, 1, f);
99  if (rle & 0x80) {
100  rle ^= 0x80;
101  fread(pdata, 1, channel, f);
102  pdata += channel;
103  for (int j = 0; j < rle * channel; j++) {
104  *pdata = *(pdata - channel);
105  pdata++;
106  }
107  i += channel * (rle + 1);
108  }
109  else {
110  int k = channel * (rle + 1);
111  fread(pdata, 1, k, f);
112  pdata += k;
113  i += k;
114  }
115  }
116  break;
117  default:
118  File::closeFile(f);
119  delete[] data;
120  return NULL;
121  }
122  for (i=0, j=0; i < size; i += channel, j += 3) { // BGR -> RGB
123  img->pixmap[j] = data[i+2];
124  img->pixmap[j+1] = data[i+1];
125  img->pixmap[j+2] = data[i];
126  }
127  File::closeFile(f);
128  delete[] data;
129  return img;
130 }

132 void Img::saveTGA(const char *filename, GLint width, GLint height, const GLubyte *data)
133 {
134  FILE *f = File::openFile(filename, "w");
135  if (f) {
136  const GLubyte *pdata = data;
137 
138  fputc(0, f); /* ID Length, 0 => No ID */
139  fputc(0, f); /* Color Map Type, 0 => No color map included */
140  fputc(2, f); /* Image Type, 2 => Uncompressed, True-color Image */
141  fputc(0, f); /* Next five bytes are about the color map entries */
142  fputc(0, f); /* 2 bytes Index, 2 bytes length, 1 byte size */
143  fputc(0, f);
144  fputc(0, f);
145  fputc(0, f);
146  fputc(0, f); /* X-origin of Image */
147  fputc(0, f);
148  fputc(0, f); /* Y-origin of Image */
149  fputc(0, f);
150  fputc(width & 0xff, f); /* Image Width */
151  fputc((width>>8) & 0xff, f);
152  fputc(height & 0xff, f); /* Image Height */
153  fputc((height>>8) & 0xff, f);
154  fputc(24, f); /* Pixel Depth, 0x18 => 24 Bits */
155  fputc(0x20, f); /* Image Descriptor */
156  File::closeFile(f);
157 
158  f = File::openFile(filename, "ab"); /* reopen in binary append mode */
159  for (int y=height-1 ; y>=0; y--) {
160  for (int x=0; x<width; x++) {
161  int i = (y*width + x) * 4;
162  fputc(pdata[i+2], f); /* write blue */
163  fputc(pdata[i+1], f); /* write green */
164  fputc(pdata[i], f); /* write red */
165  }
166  }
167  File::closeFile(f);
168  }
169 }
