72 void Pitch::dragCB(UMouseEvent &e)
73 {
74  GLfloat dx = e.getX() - arrow_point.x;
75  GLfloat dy = arrow_point.y - e.getY();
76 
77  current_radius = sqrt(dx * dx + dy * dy);
78  delta = asin(dy / current_radius);
79 
80  if (dx <= 0) {
81  if (dy < 5 && dy > -5) delta = 0;
82  }
83  else {
84  if (dy >= 0) delta = M_PI/2; else delta = -M_PI/2;
85  }
86  angle = 180 * delta / M_PI;
87 
88  doAction();
89  repaint();
90 }
