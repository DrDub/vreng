92 void Cloud::makeSolid()
93 {
94  char s[256];
95  float minx, miny, minz, maxx, maxy, maxz;
96  minx = miny = minz = maxx = maxy = maxz = 0;
97 
98  for (int n=0; n < number; n++) {
99  minx = MIN(minx, obloids[n].x);
100  miny = MIN(miny, obloids[n].y);
101  minz = MIN(minz, obloids[n].z);
102  maxx = MAX(maxx, obloids[n].x);
103  maxy = MAX(maxy, obloids[n].y);
104  maxz = MAX(maxz, obloids[n].z);
105  }
106  sprintf(s, "solid shape=\"bbox\" dim=\"%f %f %f\" />", (maxx-minx)/2, (maxy-miny)/2, (maxz-minz) *.5/2);
107  parse()->parseSolid(s, SEP, this);
108 }

121 void Cloud::inits()

129  // wind values

160 void Cloud::changePermanent(float lasting)
161 {
162  if (state == INACTIVE) return;
163 
164  // progression
165  GLfloat dx = -speed * cos(pos.az);
166  GLfloat dy = -speed * sin(pos.az);
167 
168  for (int n=0; n < number; n++) {
169  obloids[n].x += dx; // next position
170  obloids[n].y += dy;
171  if ((fabs(obloids[n].x) > FAR) || (fabs(obloids[n].y) > FAR)) {
172  rebuild();
173  return;
174  }
175  // compute global position
176  pos.x += obloids[n].x;
177  pos.y += obloids[n].y;
178  pos.z += obloids[n].z;
179  }
180  // global position
181  pos.x /= (float) number;
182  pos.y /= (float) number;
183  pos.z /= (float) number;
184  //error("cloud: pos=%.2f %.2f %.2f", pos.x, pos.y, pos.z);
185 }
