53 static void readJpegData(Img *img, struct jpeg_decompress_struct& cinfo)
54 {
55  JSAMPARRAY buffer; // Output row buffer
56  int row_stride; // physical row width in output buffer
57 
58  // Step 4: set parameters for decompression
59 
60  // Step 5: Start decompressor
61  jpeg_start_decompress(&cinfo);
62 
63  row_stride = cinfo.output_width * cinfo.output_components;
64 
65  // Make a one-row-high sample array that will go away when done with image
66  buffer = (*cinfo.mem->alloc_sarray)
67  ((j_common_ptr) &cinfo, JPOOL_IMAGE, row_stride, 1);
68 
69  // Step 6: while (scan lines remain to be read)
70  trace(DBG_IMG, "readJpegData: width=%d height=%d channel=%d",
71  cinfo.output_width, cinfo.output_height, cinfo.output_components);
72  int k = 0;
73  int y = 0;
74  while (cinfo.output_scanline < cinfo.output_height) {
75  jpeg_read_scanlines(&cinfo, buffer, 1);
76  JSAMPROW prow = buffer[0];
77 
78  if (cinfo.output_components == 3) { // RGB
79  for (unsigned int x = 0; x < cinfo.output_width; x++) {
80  img->pixmap[k +0] = prow[0];
81  img->pixmap[k +1] = prow[1];
82  img->pixmap[k +2] = prow[2];
83  prow += 3;
84  k += 3;
85  }
86  }
87  else if (cinfo.output_components == 1) { // GREYSCALE
88  for (unsigned int x = 0; x < cinfo.output_width; x++) {
89  img->pixmap[k] = *prow;
90  prow++;
91  k++;
92  }
93  }
94  y++;
95  }
96 
97  // Step 7: Finish decompression
98  jpeg_finish_decompress(&cinfo);
99 
100  // Step 8: Release JPEG decompression object
101  jpeg_destroy_decompress(&cinfo);
102 }

118 Img * Img::loadJPG(void *tex, ImageReader read_func)
119 {
120 #if HAVE_JPEG
121  Texture *_tex = (Texture *) tex;
122  FILE *fp;
123  if ((fp = Cache::openCache(_tex->url, _tex->http)) == NULL) return NULL;
124 
125  struct jpeg_decompress_struct cinfo;
126 
127  /* error handling */
128  struct my_error_mgr jerr;
129  cinfo.err = jpeg_std_error(&jerr.pub);
130  jerr.pub.error_exit = my_error_exit;
131  if (setjmp(jerr.setjmp_buffer)) {
132  error("loadJPG: invalidData");
133  jpeg_destroy_decompress(&cinfo);
134  File::closeFile(fp);
135  return NULL;
136  }
137 
138  /* we read the header */
139  readJpegHeader(fp, cinfo);
140 
141  /* we read the data */
142  if (cinfo.image_width == 0 || cinfo.image_height == 0
143  // we can only decode RGB (num=3) or greayscale (num=1)
144  || (cinfo.num_components != 3 && cinfo.num_components != 1)) {
145  error("loadJPG: num_components=%d", cinfo.num_components);
146  jpeg_destroy_decompress(&cinfo);
147  File::closeFile(fp);
148  return NULL;
149  }
150 
151  Img *img = new Img(cinfo.image_width, cinfo.image_height, cinfo.num_components);
152  if (!img) {
153  error("loadJPG: can't new image");
154  jpeg_destroy_decompress(&cinfo);
155  File::closeFile(fp);
156  return NULL;
157  }
158 
159  readJpegData(img, cinfo);
160  File::closeFile(fp);
161 
162  return img;
163 #else
164  return NULL;
165 #endif
166 }

168 void Img::saveJPG(const char *filename, GLint width, GLint height, GLint quality, const GLubyte *buffer)
169 {
170 #if HAVE_JPEG
171  JSAMPLE *image_buffer = (JSAMPLE *) buffer;
172 
173  struct jpeg_compress_struct cinfo;
174  struct jpeg_error_mgr jerr;
175  FILE *outfile; /* target file */
176  JSAMPROW row_pointer[1]; /* pointer to JSAMPLE row[s] */
177  int row_stride; /* physical row width in image buffer */
178 
179  /* Step 1: allocate and initialize JPEG compression object */
180  cinfo.err = jpeg_std_error(&jerr);
181  jpeg_create_compress(&cinfo);
182 
183  /* Step 2: specify data destination (eg, a file) */
184  if ((outfile = File::openFile(filename, "wb")) == NULL) {
185  fprintf(stderr, "can't open %s\n", filename); return;
186  }
187  jpeg_stdio_dest(&cinfo, outfile);
188 
189  /* Step 3: set parameters for compression */
190  cinfo.image_width = width; /* image width and height, in pixels */
191  cinfo.image_height = height;
192  cinfo.input_components = 3; /* # of color components per pixel */
193  cinfo.in_color_space = JCS_RGB; /* colorspace of input image */
194  jpeg_set_defaults(&cinfo);
195  jpeg_set_quality(&cinfo, quality, TRUE /* limit to baseline-JPEG values */);
196 
197  /* Step 4: Start compressor */
198  jpeg_start_compress(&cinfo, TRUE);
199 
200  /* Step 5: while (scan lines remain to be written) */
201  row_stride = width * 3; /* JSAMPLEs per row in image_buffer */
202  while (cinfo.next_scanline < cinfo.image_height) {
203  row_pointer[0] = & image_buffer[cinfo.next_scanline * row_stride];
204  jpeg_write_scanlines(&cinfo, row_pointer, 1);
205  }
206 
207  /* Step 6: Finish compression */
208  jpeg_finish_compress(&cinfo);
209  File::closeFile(outfile);
210 
211  /* Step 7: release JPEG compression object */
212  jpeg_destroy_compress(&cinfo);
213 #endif
214 }
