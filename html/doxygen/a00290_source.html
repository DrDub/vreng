139 void Aoi::aoiQuit()
140 {
141  /* freeing only mobile object we're responsible for */
142  for (list<WObject*>::iterator it = mobileList.begin(); it != mobileList.end(); ++it) {
143  /* skipping local user object (as we do not want to remove it) */
144  if (*it == localuser) continue;
145  /* closes GUI */
146  if ((*it)->type == USER_TYPE) ::g.gui.removeUser((User *)(*it));
147  /* performs some clean up */
148  if ((*it)->type) delete *it;
149  }
150  //dax clearList(mobileList); //FIXME! don't comment else segfault
151 
152  /* explicit declaration to network so that there is minimal latency
153  on other VREng processes to see user leaving previous AoI */
154  localuser->noh->declareDeletion();
155  localuser->noh->deleteFromList();
156 
157  /* closes multicast channel now as we do not need it anymore */
158  delete Channel::current(); // delete Channel
159 }
