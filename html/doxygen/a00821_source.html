43 void Walls::httpReader(void *_walls, Http *http)
44 {
45  Walls *walls = (class Walls *) _walls;
46  if (! walls) return;
47 
48  Walls *pw = NULL, *prev = NULL;
49  float xs, xe, ys, ye, zs, ze, r, g, b;
50  char line[BUFSIZ];
51 
52  FILE *f = Cache::openCache(walls->getUrl(), http);
53  if (! f) { error("can't open %s", walls->getUrl()); return; }
54 
55  while (fgets(line, sizeof(line), f)) {
56  if (*line == '#') continue;
57  line[strlen(line) -1] = '\0';
58  if (! strlen(line)) continue;
59  if (sscanf(line, "%f %f %f %f %f %f %f %f %f",
60  &xs, &xe, &ys, &ye, &zs, &ze, &r, &g, &b) <= 0) {
61  break; // EOF
62  }
63 
64  pw = new Walls(xs, xe, ys, ye, zs, ze, r, g, b);
65 
66  if (! prev) wallsList = pw;
67  else prev->next = pw;
68  prev = pw;
69  }
70  if (pw) pw->next = NULL;
71  File::closeFile(f);
72 }

143 /* Intersection with an object */
144 int Walls::whenIntersect(const V3& center, const V3& size, V3& normal)
145 {
146  Walls *walls = wallsList;
147  V3 cur_normal;
148  bool first = true;
149 
150  float oxmin = center.v[0] - size.v[0];
151  float oxmax = center.v[0] + size.v[0];
152  float oymin = center.v[1] - size.v[1];
153  float oymax = center.v[1] + size.v[1];
154  float ozmin = center.v[2] - size.v[2];
155  float ozmax = center.v[2] + size.v[2];
156 
157  // we test the collision with all the walls
158  while (walls) {
159  float wxmin = (walls->xs < walls->xe) ? walls->xs : walls->xe;
160  float wxmax = (walls->xs > walls->xe) ? walls->xs : walls->xe;
161  float wymin = (walls->ys < walls->ye) ? walls->ys : walls->ye;
162  float wymax = (walls->ys > walls->ye) ? walls->ys : walls->ye;
163  float wzmin = (walls->zs < walls->ze) ? walls->zs : walls->ze;
164  float wzmax = (walls->zs > walls->ze) ? walls->zs : walls->ze;
165 
166  if ((((oxmin < wxmin) && (oxmax > wxmax)) ||
167  ((oxmin > wxmin) && (oxmax < wxmax)) ||
168  ((oxmax > wxmin) && (oxmax < wxmax)) ||
169  ((oxmin > wxmin) && (oxmin < wxmax)))
170  &&
171  (((oymin < wymin) && (oymax > wymax)) ||
172  ((oymin > wymin) && (oymax < wymax)) ||
173  ((oymax > wymin) && (oymax < wymax)) ||
174  ((oymin > wymin) && (oymin < wymax)))
175  &&
176  (((ozmin < wzmin) && (ozmax > wzmax)) ||
177  ((ozmin > wzmin) && (ozmax < wzmax)) ||
178  ((ozmax > wzmin) && (ozmax < wzmax)) ||
179  ((ozmin > wzmin) && (ozmin < wzmax)))) {
180  cur_normal.v[2] = 0.;
181  if (wxmin == wxmax) {
182  cur_normal.v[0] = 1.;
183  cur_normal.v[1] = 0.;
184  }
185  else {
186  cur_normal.v[1] = 1.;
187  cur_normal.v[0] = 0.;
188  }
189 
190  // determine which normal we take
191  if (dotprod(center.v[0] - wxmin, center.v[1] - wymin,
192  cur_normal.v[0], cur_normal.v[1]) < 0.) {
193  cur_normal.v[0] = -cur_normal.v[0];
194  cur_normal.v[1] = -cur_normal.v[1];
195  }
196  if (first) {
197  normal = cur_normal;
198  first = false;
199  }
200  else {
201  if (cmpnormal(&cur_normal, &normal) > 0.01) return 2;
202  }
203  }
204  walls = walls->next;
205  }
206  if (first) return 0; // no collision
207  else return 1;
208 }
