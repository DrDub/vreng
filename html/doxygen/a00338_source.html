89 void Bumper::changePosition(float lasting)
90 {
91  static uint8_t tilts = 5;
92 
93  if (lasting > move.ttl) {
94  setFrame((getFrame()+1) % getFrames()); // next frame
95  if (capted) {
96  float dx = (capted->pos.x - pos.x) * cos(localuser->pos.az);
97  float dy = (capted->pos.y - pos.y) * sin(localuser->pos.az);
98  if (dx < 0) localuser->pos.x += dx * forcex;
99  else localuser->pos.x += dx * forcex;
100  if (dy < 0) localuser->pos.y += dy * forcey;
101  else localuser->pos.y += dy * forcey;
102  localuser->pos.z = capted->pos.z;
103  capted = NULL;
104  }
105  tilts--;
106  if (tilts == 0) {
107  tilts = 5;
108  state = INACTIVE;
109  }
110  }
111 }

113 bool Bumper::whenIntersect(WObject *pcur, WObject *pold)
114 {
115  switch (pcur->type) {
116  case USER_TYPE:
117  case BALL_TYPE:
118  {
119  float dx = pcur->pos.x - pold->pos.x;
120  float dy = pcur->pos.y - pold->pos.y;
121  float dz = pcur->pos.z - pold->pos.z;
122  pcur->pos.x = pold->pos.x - (forcex * dx);
123  pcur->pos.y = pold->pos.y - (forcey * dy);
124  pcur->pos.z = pold->pos.z - (forcez * dz);
125  pcur->projectPosition(pold, pcur);
126  capted = pcur;
127  setRenderPrior(RENDER_HIGH);
128  enableImposedMovement();
129  state = ACTIVE;
130  break;
131  }
132  default:
133  pcur->projectPosition(pcur, pold);
134  break;
135  }
136  return true;
137 }
