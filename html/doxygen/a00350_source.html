65 void Carrier::take(WObject *po)

67  if (po->mode != MOBILE) { notice("%s is not mobile", po->getInstance()); return; }
68 
69  ::g.gui.showManipulator();

72  carriedObject = po;
73  carriedObject->move.lspeed.v[0] = 0;
