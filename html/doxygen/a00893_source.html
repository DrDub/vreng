75 /* Heuristic to avoid to send bunch of Query */
76 int Noid::filterQuery()
77 {
78  static Noid oldnoid;
79  static int countDelta = 0;
80 
81  if (equalNoid(oldnoid)) {
82  // last request was on this name
83  if (++countDelta <= 15)
84  // 15: 10 proprietes en moyenne, avec 15 on saute donc
85  // sans doute un "bloc" de deltas, mais sans doute pas deux
86  return 0; // cancel this request
87  }
88  // it's another one
89  oldnoid = *this;
90  countDelta = 0;
91  return 1;
92 }
