178 void Channel::decode(const char *chan_str, uint32_t *group, uint16_t *port, uint8_t *ttl)
179 {
180  char *ttlstr, *portstr, *groupstr;
181  char *chanstr = strdup(chan_str);
182 
183  groupstr = chanstr;
184  portstr = strchr(chanstr, '/');
185  if (portstr != NULL) *portstr = '\0';
186  *group = inet_addr(groupstr);
187  if (portstr != NULL) {
188  portstr++;
189  ttlstr = strchr(portstr, '/');
190  if (ttlstr != NULL) *ttlstr = '\0';
191  *port = atoi(portstr);
192  if (ttlstr != NULL) {
193  *ttlstr = '\0';
194  ttlstr++;
195  *ttl = atoi(ttlstr);
196  }
197  }
198  free(chanstr);
199 }

202 void Channel::naming()

236 int Channel::create(const char *chan_str, int **pfds)
237 {
238  cntfds = 0; // number of fd
239 
240  if (! chan_str) return 0;
241  decode(chan_str, &group, &port, &ttl);
242 
243  port &= ~1; // RTP compliant: even port
244 
245  /*
246  * RTP channel
247  */
248  memset(&sa_rtp, 0, sizeof(struct sockaddr_in));
249  memset(&sa_rtp_r, 0, sizeof(struct sockaddr_in));
250  sa_rtp.sin_family = AF_INET;
251  sa_rtp_r.sin_family = AF_INET;
252  if (::g.pref.reflector) {
253  sa_rtp_r.sin_port = 0;
254  sa_rtp_r.sin_addr.s_addr = htonl(INADDR_ANY);
255  sa_rtp.sin_port = htons(port - VRUM_PORT_OFFSET);
256  inet4_pton(vrum_addr, &sa_rtp.sin_addr.s_addr);
257  sa[SA_RTP_R] = &sa_rtp_r;
258  }
259  else { // native Multicast
260  sa_rtp.sin_port = htons(port);
261  sa_rtp.sin_addr.s_addr = group;
262  }
263  sa[SA_RTP] = &sa_rtp;
264 
265  if (::g.pref.reflector) {
266  if ((sd[SD_R_RTP] = createMcastRecvSocket(sa[SA_RTP_R])) < 0) {
267  warning("can't open receive socket RTP reflector");
268  return 0;
269  }
270  }
271  else { // native Multicast
272  if ((sd[SD_R_RTP] = createMcastRecvSocket(sa[SA_RTP])) < 0) {
273  warning("can't open receive socket RTP");
274  return 0;
275  }
276  }
277 
278  sd[SD_W_RTP] = Socket::createSendSocket(ttl);
279  cntfds += 2;
280  trace(DBG_RTP, "sa[SA_RTP]=%p", sa[SA_RTP]);
281  trace(DBG_RTP, "sd[SD_R_RTP]=%d, sd[SD_W_RTP]=%d",
282  sd[SD_R_RTP], sd[SD_W_RTP]);
283  trace(DBG_RTP, "SA_RTP: port=%d", sa[SA_RTP]->sin_port);
284 
285  /*
286  * RTCP channel
287  */
288  memset(&sa_rtcp, 0, sizeof(struct sockaddr_in));
289  memset(&sa_rtcp_r, 0, sizeof(struct sockaddr_in));
290  sa_rtcp.sin_family = AF_INET;
291  sa_rtcp_r.sin_family = AF_INET;
292  if (::g.pref.reflector) {
293  sa_rtcp_r.sin_port = 0;
294  sa_rtcp_r.sin_addr.s_addr = htonl(INADDR_ANY);
295  sa_rtcp.sin_port = htons(port - VRUM_PORT_OFFSET + 1);
296  inet4_pton(vrum_addr, &sa_rtcp.sin_addr.s_addr);
297  sa[SA_RTCP_R] = &sa_rtcp_r;
298  }
299  else { // native Multicast
300  sa_rtcp.sin_port = htons(port + 1); // not ++port !!!
301  sa_rtcp.sin_addr.s_addr = group;
302  }
303  sa[SA_RTCP] = &sa_rtcp;
304 
305  if (::g.pref.reflector) {
306  if ((sd[SD_R_RTCP] = createMcastRecvSocket(sa[SA_RTCP_R])) < 0) {
307  warning("can't open receive socket RTCP reflector");
308  return 0;
309  }
310  }
311  else { // native Multicast
312  if ((sd[SD_R_RTCP] = createMcastRecvSocket(sa[SA_RTCP])) < 0) {
313  warning("can't open receive socket RTCP");
314  return 0;
315  }
316  }
317 
318  sd[SD_W_RTCP] = Socket::createSendSocket(ttl);
319  cntfds += 2;
320  trace(DBG_RTP, "sa[SA_RTCP]=%p", sa[SA_RTCP]);
321  trace(DBG_RTP, "sd[SD_R_RTCP]=%d, sd[SD_W_RTCP]=%d",
322  sd[SD_R_RTCP], sd[SD_W_RTCP]);
323  trace(DBG_RTP, "SA_RTCP: port=%d", sa[SA_RTCP]->sin_port);
324 
325  /* UDP channel */
326  sa[SA_UDP] = NULL;
327 
328  /*
329  * RTP session initialization
330  */
331  session = new Session();
332  if (! session) { error("session: can't new"); return 0; }
333 
334  uint32_t oldssrc = 0;
335  World *world = NULL;
336  if ((world = World::worldByGroup(group)) != NULL)
337  oldssrc = world->getSsrc();
338  if ((ssrc = session->create(group, port, ttl, oldssrc)) == 0) {
339  error("create Channel: can't create session");
340  delete session;
341  session = NULL;
342  return 0;
343  }
344 
345  if (this == managerChannel) {
346  NetObject::setMyMgrSsrcId(ssrc);
347  session->mode = MANAGER_MODE;
348  }
349  else {
350  NetObject::setMySsrcId(ssrc);
351  session->mode = WORLD_MODE;
352 
353  if (world) {
354  world->setGroupAdr(group);
355  world->setSsrc(ssrc);
356  }
357  else if (World::current()) { // !! tester existence !!!
358  World::current()->setGroupAdr(group);
359  World::current()->setSsrc(ssrc);
360  }
361  }
362 
363  naming();
364 
365  *pfds = sd;
366  cntFd += cntfds;
367  return cntfds;
368 }
