44 void Mp3::launch(const char *tool, const char *url)
45 {

47  case -1:
48  error("%s %s", e_fork, tool);
49  break;
50  case 0:
51  switch (toolid) {
52  case MPG123_TOOL: execlp(tool, tool, "-qy", url, (char*)NULL); break;
53  case XAUDIO_TOOL: execlp(tool, tool, url, (char*)NULL); break;
54  case FREEAMP_TOOL: execlp(tool, tool, url, (char*)NULL); break;
55 #if MACOSX
56  case QUICKTIME_TOOL: execlp(tool, tool, "-a", QUICKTIME_PATH, url, (char*)NULL); break;
57  case VLC_TOOL: execlp(tool, tool, "--hide", "-a", VLC_PATH, url, "--args", "--play-and-exit", (char*)NULL); break;
58 #else
59  case VLC_TOOL: execlp(tool, tool, "--quiet", url, (char*)NULL); break;
60 #endif
61  }
62  error("%s %s", e_exec, tool);
63  signal(SIGCHLD, SIG_IGN);
64  exit(1);
65  default:
