26 uint8_t * Ogl::copyPixels(GLint width, GLint height, GLenum mode)
27 {
28 #if !defined(WITH_TINYGL)
29  // Alloc pixel bytes
30  int nbbytes = 3 * width * height;
31  uint8_t *pixogl = new uint8_t[nbbytes];
32  uint8_t *pixmap = new uint8_t[nbbytes];
33 
34  // Copy from OpenGL
35  glReadBuffer(mode);
36  glReadPixels(0, 0, width, height, GL_RGB, GL_UNSIGNED_BYTE, pixogl);
37 
38  /* Swapping lines in the buffer
39  * OpenGL writes from lower line going up
40  * but ijl works reverse from up to down
41  */
42  for (int j=0; j < height; j++) { // lines
43  for (int i=0; i < width; i++) { // columns
44  pixmap[((j*width)+i)*3+0] = pixogl[((height-1-j)*width+i)*3+0];
45  pixmap[((j*width)+i)*3+1] = pixogl[((height-1-j)*width+i)*3+1];
46  pixmap[((j*width)+i)*3+2] = pixogl[((height-1-j)*width+i)*3+2];
47  }
48  }
49 
50  delete[] pixogl;
51  return pixmap;
52 #else
53  return NULL;
54 #endif
55 }

58 bool Ogl::isGLextension(const char *ext)
59 {
60  const GLubyte *extentions = NULL;
61  const GLubyte *start;
62  GLubyte *where, *terminator;
63 
64  if (*ext == '\0')
65  return false;
66  if ((where = (GLubyte *) strchr(ext, ' ')))
67  return false;
68 
69 #if !defined(WITH_TINYGL)
70  extentions = glGetString(GL_EXTENSIONS);
71  start = extentions;
72  for (;;) {
73  where = (GLubyte *) strstr((const char *) start, ext);
74  if (! where)
75  break;
76  terminator = where + strlen(ext);
77  if (where == start || *(where - 1) == ' ')
78  if (*terminator == ' ' || *terminator == '\0')
79  return true;
80  start = terminator;
81  }
82 #endif
83  return false;
84 }
