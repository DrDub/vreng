39 {
40  switch (pid = fork()) {
41  case -1:
42  error("%s %s", e_fork, tool);
43  break;
44  case 0:
45  switch (toolid) {
46  case VLC_TOOL:
47 #if MACOSX
48  execlp(tool, tool, "--hide", "-a", VLC_PATH, url, "--args", "--play-and-exit", (char*)NULL);
49 #else
50  if (loop) execlp(tool, tool, "--quiet", "--repeat", url, (char*)NULL);
51  else execlp(tool, tool, "--quiet", url, "vlc:quit", (char*)NULL);
52 #endif
53  break;
54  }
55  error("%s %s", e_exec, tool);
56  signal(SIGCHLD, SIG_IGN);
57  exit(1);

60  execlp(tool, tool, "-a", QUICKTIME_PATH, url, (char*)NULL);

62 #endif
63  default:
64  break;
65  }
