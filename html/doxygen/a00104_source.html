105 /* compute kbps_sent & co */
106 void statAdjust()
107 {
108  static struct timeval then;
109  struct timeval now;
110 
111  gettimeofday(&now, NULL);
112  float d = Times::diffDates(then, now);
113  if (fabs(d) > 0.1) {
114  kbps_sent = bytes_sent/d/1000*8;
115  kbps_recvd = bytes_recvd/d/1000*8;
116  trace(DBG_NET, "kbps sent: %5.2f recvd: %5.2f",
117  bytes_sent/d/1000*8, bytes_recvd/d/1000*8);
118  then = now;
119  bytes_sent = bytes_recvd = 0;
120  }
121 }

173  }

175  fprintf(stderr, "bytes_rtp sent : %d\n", sum_bytes_rtp_sent);
