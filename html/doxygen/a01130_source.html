119 void BoneMesh::projectLight()
120 {
121  if (! triangleListCompiled) compileTriangleList();
122 
123  Vect3D lightDirection(0,0,1);
124  Vect3D lightDiffuse(1,1,1);
125  Vect3D lightSpecular(1,1,1);
126 
127  for (int i=0; i < triangles; i++) {
128  Vect3D mat(triangle[i]->colorRed, triangle[i]->colorGreen, triangle[i]->colorBlue);
129  float cosine = Vect3D::dotProduct(lightDirection, triangle[i]->initialNormal);
130  cosine = CROP(-1, cosine, 1);
131  if (cosine > 0) {
132  float cosPower = 64;
133  float cosinePower = pow(cosine, cosPower);
134  mat.x *= lightDiffuse.x * cosine + lightSpecular.x * cosinePower;
135  mat.y *= lightDiffuse.y * cosine + lightSpecular.y * cosinePower;
136  mat.z *= lightDiffuse.z * cosine + lightSpecular.z * cosinePower;
137  if (mat.x > 1) mat.x = 1;
138  if (mat.y > 1) mat.y = 1;
139  if (mat.z > 1) mat.z = 1;
140  }
141  else mat.reset();
142 
143  triangle[i]->finalRed = mat.x;
144  triangle[i]->finalGreen = mat.y;
145  triangle[i]->finalBlue = mat.z;
146  }
147 }
