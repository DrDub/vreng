99 void Stars::move()

113 bool Stars::pointed(int n)
114 {
115  float x0 = stars[n].x[0] * 2*pos.x / stars[n].v[0];
116  float y0 = stars[n].y[0] * 2*pos.y / stars[n].v[0];
117 
118  rotate(&x0, &y0, stars[n].rot);
119  x0 += 2*pos.x;
120  y0 += 2*pos.y;

192  else if (!stringcmp(l, "warp")) l = parse()->parseFloat(l, &warp, "warp");
