65 void Mirage::parser(char *l)
66 {
67  defaults();
68  l = tokenize(l);
69  begin_while_parse (l) {
70  l = parse()->parseAttributes(l, this);
71  if (!l) break;
72  if (!stringcmp(l, "aspeed")) {
73  l = parse()->parseFloat(l, &aspeed, "aspeed");
74  azspeed = ayspeed = axspeed = aspeed;
75  }
76  else if (!stringcmp(l, "azspeed")) l = parse()->parseFloat(l, &azspeed, "azspeed");
77  else if (!stringcmp(l, "ayspeed")) l = parse()->parseFloat(l, &ayspeed, "ayspeed");
78  else if (!stringcmp(l, "axspeed")) l = parse()->parseFloat(l, &axspeed, "axspeed");
79  else if (!stringcmp(l, "radius")) l = parse()->parseFloat(l, &radius, "radius");
80  else if (!stringcmp(l, "flare")) l = parse()->parseUInt8(l, &flares, "flare");
81  else if (!stringcmp(l, "scale")) l = parse()->parseFloat(l, &scale, "scale");
82  else if (!stringcmp(l, "color")) l = parse()->parseVector3f(l, color, "color");
83  else if (!stringcmp(l, "mode=")) {
84  char modestr[16];
85  l = parse()->parseString(l, modestr, "mode");
86  if (!stringcmp(modestr, "turn") || !stringcmp(modestr, "self")) turn = true;
87  if (!stringcmp(modestr, "roll")) roll = true;
88  if (!stringcmp(modestr, "tilt")) tilt = true;
89  else if (!stringcmp(modestr, "refresh") || !stringcmp(modestr, "volatile")) persist = false;
90  else if (!stringcmp(modestr, "orbit")) {
91  orbit = true;
92  x0x0y0y0 = pos.x*pos.x + pos.y*pos.y;
93  maxx = maxy = (float) sqrt(x0x0y0y0);
94  minx = miny = -maxx;
95  persist = false;
96  }
97  else if (!stringcmp(modestr, "circular")) {
98  circular = true;
99  centrex = pos.x;
100  centrey = pos.y;
101  initialaz = pos.az;
102  anglerot = pos.az;
103  persist = false;
104  }
105  }
106  }
107  end_while_parse(l);
108 }

199 /* system of equations handling permanent motion */
200 void Mirage::changePermanent(float lasting)
201 {
202  if (turn) {
203  pos.az -= lasting * azspeed;
204  pos.az -= M_2PI * (float) floor(pos.az / M_2PI);
205  }
206  if (roll) {
207  pos.ay -= lasting * ayspeed;
208  pos.ay -= M_2PI * (float) floor(pos.ay / M_2PI);
209  }
210  if (tilt) {
211  pos.ax -= lasting * axspeed;
212  pos.ax -= M_2PI * (float) floor(pos.ax / M_2PI);
213  }
214  if (orbit) {
215  float sx, sy, dx, dy;
216  if (pos.x >= maxx || pos.y <= miny) {
217  sx = +1; sy = +1;
218  }
219  else if (pos.y >= maxy || pos.x <= minx) {
220  sx = -1; sy = -1;
221  }
222  else if ((pos.y >= 0 && pos.x <= 0) || (pos.y <= 0 && pos.x <= 0)) {
223  sx = -1; sy = -1;
224  }
225  else {
226  sx = +1; sy = +1;
227  }
228  dy = sy / ::g.times.getRate();
229  pos.y += dy;
230  dx = (float) sqrt(fabs(x0x0y0y0 - pos.y*pos.y));
231  pos.x = sx * dx;
232  }
233  if (circular) {
234  anglerot -= (lasting * aspeed);
235  anglerot -= (M_2PI * (float) floor(anglerot / M_2PI));
236  pos.x = centrex + (radius * cos(anglerot));
237  pos.y = centrey + (radius * sin(anglerot));
238  pos.az = anglerot + initialaz;
239  }
240  updatePosition();
241 }
