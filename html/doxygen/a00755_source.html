57 void Text::parser(char *l)
58 {
59  defaults();
60  l = tokenize(l);
61  begin_while_parse(l) {
62  l = parse()->parseAttributes(l, this);
63  if (!l) break;
64  if (!stringcmp(l, "font")) l = parse()->parseString(l, names.url, "font");
65  else if (!stringcmp(l, "color")) l = parse()->parseVector3f(l, color, "color");
66  else if (!stringcmp(l, "verso")) l = parse()->parseUInt8(l, &verso, "verso");
67  else if (!stringcmp(l, "scale")) l = parse()->parseFloat(l, &scale, "scale");
68  else l = parse()->parseQuotedString(l, text);
69  }
70  end_while_parse(l);
71 
72  text[MAXLEN - 1] = '\0';
73 }

140 void Text::render()
141 {
142  if (! loaded) return;
143  if (state == INACTIVE) return;
144 
145  glPushAttrib(GL_ALL_ATTRIB_BITS);
146  glPushMatrix();
147  glEnable(GL_LIGHTING);
148  glEnable(GL_TEXTURE_2D);
149  if (verso) glDisable(GL_CULL_FACE); // both faces
150  else glEnable(GL_CULL_FACE); // front face
151 
152  txf->bindFontTexture();
153 
154  glMaterialfv(GL_FRONT, GL_DIFFUSE, color);
155  glTexEnvf(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, GL_MODULATE);
156 
157  glTranslatef(pos.x + shiftx, pos.y + shifty, pos.z + shiftz);
158  glRotatef(RAD2DEG(pos.az + shiftaz), 0, 0, 1);
159  glRotatef(RAD2DEG(pos.ax + shiftax), 1, 0, 0);
160  glScalef(SCALE*scale, SCALE*scale, SCALE*scale);
161 
162  // render the text
163  txf->renderString(text, (int) strlen(text));
164 
165  glEnable(GL_CULL_FACE);
166  glDisable(GL_TEXTURE_2D);
167  glDisable(GL_LIGHTING);
168  glPopMatrix();
169  glPopAttrib();
170 }
