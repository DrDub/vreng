48 void Music::parser(char *l)
49 {
50  defaults();
51  l = tokenize(l);
52  begin_while_parse(l) {
53  l = parse()->parseAttributes(l, this);
54  if (!l) break;
55  if (! stringcmp(l, "url")) l = parse()->parseUrl(l, names.url);
56  else if (! stringcmp(l, "state")) l = parse()->parseString(l, statestr, "state");
57  }
58  end_while_parse(l);
59 }

61 int Music::format()
62 {
63  char *pext = strrchr(names.url, '.');
64  pext++;
65 
66  if (!stringcmp(pext, "mp3") || !stringcmp(pext, "mp2"))
67  fmt = FMT_MP3;
68  else if (!stringcmp(pext, "wav"))
69  fmt = FMT_WAV;
70  else if (!stringcmp(pext, "au"))
71  fmt = FMT_AU;
72  else if (!stringcmp(pext, "mid") || !stringcmp(pext, "midi"))
73  fmt = FMT_MIDI;
74  else
75  fmt = FMT_NONE;
76 
77  return fmt;
78 }

107 void Music::play(Music *music, void *d, time_t s, time_t u)
108 {
109  music->repeat = (d == (void *) 1) ? true : false;
110 
111  // check if d is an url
112  if (music->repeat == false && (d != (void *)NULL)) {
113  strcpy(music->names.url, (char *) d);
114  music->fmt = music->format();
115  music->state = Music::INACTIVE;
116  }
117 
118  if (music->state == Music::INACTIVE) {
119  switch (music->fmt) {
120  case Music::FMT_MP3: Mp3::start(music->names.url); break;
121  case Music::FMT_WAV: Wav::start(music->names.url, music->repeat); break;
122  case Music::FMT_AU: Au::start(music->names.url); break;
123  case Music::FMT_MIDI: Midi::start(music->names.url); break;
124  default: music->state = Music::INACTIVE; return;
125  }
126  music->state = Music::PLAYING;
127  }
128 }
