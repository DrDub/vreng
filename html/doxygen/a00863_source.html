123 /* Initializes Object */
124 void WObject::initializeObject(uint8_t _mode)
125 {
126  mode = _mode;
127  type = typeId();
128  num = objectNum++;
129 
130  setWObjectId();
131  updateNames();
132 
133  //error("num=%d mode=%d type=%d", num, mode, type);
134  switch (_mode) {
135  case STILL:
136  addToStill();
137  break;
138  case MOBILE:
139  if (isBehavior(PERSISTENT)) {
140  getMySqlPosition(); // calls persistency MySql server
141  if (pos.x>1000 || pos.y>1000 || pos.z>1000 || pos.x<-1000 || pos.y<-1000 || pos.z<-1000) {
142  error("object %s discarded, bad position in MySql: %.2f,%.2f,%.2f", names.instance, pos.x, pos.y, pos.z);
143  enableBehavior(NO_BBABLE);
144  break;
145  }
146  }
147  addToMobile();
148  break;
149  case INVISIBLE:
150  enableBehavior(NO_BBABLE);
151  enableBehavior(UNSELECTABLE);
152  enableBehavior(NO_ELEMENTARY_MOVE);
153  addToInvisible();
154  break;
155  case EPHEMERAL:
156  addToMobile();
157  enableBehavior(NO_BBABLE);
158  enableBehavior(UNSELECTABLE);
159  enableBehavior(NO_ELEMENTARY_MOVE);
160  break;
161  case MOBILEINVISIBLE:
162  addToMobile();
163  enableBehavior(NO_BBABLE);
164  enableBehavior(UNSELECTABLE);
165  enableBehavior(NO_ELEMENTARY_MOVE);
166  addToInvisible();
167  break;
168  }
169  if (isBehavior(NO_BBABLE)) enableBehavior(COLLIDE_NEVER);
170  if (isBehavior(SPECIFIC_RENDER)) addToRender();
171 
172  update3D(pos);
173  if (bbBehavior()) {
174  updateBB();
175  insertIntoGrid();
176  }
177 }

199 void WObject::enableBehavior(uint32_t flag)
200 {
201  switch (flag) {
202  case UNVISIBLE:
203  setVisible(false);
204  break;
205  case COLLIDE_ONCE:
206  disableBehavior(COLLIDE_NEVER);
207  disableBehavior(COLLIDE_GHOST);
208  break;
209  case COLLIDE_NEVER:
210  disableBehavior(COLLIDE_ONCE);
211  disableBehavior(COLLIDE_GHOST);
212  break;
213  case COLLIDE_GHOST:
214  disableBehavior(COLLIDE_ONCE);
215  disableBehavior(COLLIDE_NEVER);
216  break;
217  }
218  behavior |= flag;
219 }
