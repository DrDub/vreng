136 void Face::facesHttpReader(void *_face, Http *http)
137 {
138  Face *face = (Face *) _face;
139  if (! http) { error("facesHttpReader: unable to open http connection"); return; }
140 
141  char line[URL_LEN];
142 
143  http->reset();
144  while (http->getLine(line)) {
145  char *faceurl = strdup(line);
146  trace(DBG_MAN, "facesHttpReader: add url=%s", faceurl);
147  face->urlList.addElement(faceurl);

152 void Face::change()
153 {
154  if (! index) return;
155 
156  currentUrl++;
157  trace(DBG_MAN, "change: currentUrl=%d count=%d", currentUrl, urlList.count());
158  currentUrl %= urlList.count();
159  char *urlface = urlList.getElemAt(currentUrl);
160  trace(DBG_MAN, "change: urlface=%s urlface=%p urlface[0]=%02x", urlface, urlface, urlface[0]);
161  if (! isascii(urlface[0])) {
162  error("change: BUG here! urlface=%s", urlface);
163  return;
164  }
165  load(urlface);
166 }

168 void Face::load(const char *url)
169 {
170  BoneMesh *newMesh = new BoneMesh();
171  BoneVertex *newRoot = new BoneVertex();
172 
173  if (Cache::setCacheName(url, cachefile) == 0) {
174  error("Face: file=%s url=%s", cachefile, url);
175  return;
176  }
177  Http::httpOpen(url, httpReader, (void *)url, 0);
178 
179  V3d::readV3Dfile(newMesh, newRoot, cachefile);
180 
181  animator.registerMesh(newMesh);
182  animator.registerSkeleton(newRoot);
183  animator.generateLinkList();
184 
185  if (mesh) delete mesh;
186  mesh = newMesh;
187  if (root) delete root;
188  root = newRoot;
189 }

257 void Face::animEyeLid(float angle, const char *root1, const char *lid, const char *left, const char *right)

273  bone->setCurrentRotation(sign * (1-scale) * 20., 1,0,0);
274  }
275  else
276  warning("%s not found", lid);
277  if ((bone = root->findChild(left)) != NULL) {
278  bone->resetCurrentPosition();
279  bone->scaleCurrentPosition(1, scale, 1);
280  }
281  if ((bone = root->findChild(right)) != NULL) {
282  bone->resetCurrentPosition();
283  bone->scaleCurrentPosition(1, scale, 1);
284  }
285  }
286  else
287  warning("%s not found", root1);

574  }
575  }
576  // == smile

578  if ((bone = root->findChild(lipsRoot)) != NULL) {
579  Vect3D smileDelta(0, cos(angle/10.0) / 4.0, 0);
