174 int Android::connectToBapServer(int _ipmode)
175 {
176  char setup_cmd[128];
177  struct hostent *hp;
178 
179  if ((sdtcp = Socket::openStream()) < 0) { error("socket vaps failed"); return 0; }
180  if ((hp = my_gethostbyname(vaps, AF_INET)) == NULL) { error("can't resolve name"); return 0; }
181  memset(&tcpsa, 0, sizeof(struct sockaddr_in));
182  tcpsa.sin_family = AF_INET;
183  tcpsa.sin_port = htons(VAPS_PORT);
184  memcpy(&tcpsa.sin_addr, hp->h_addr_list[0], hp->h_length);
185  my_free_hostent(hp);
186 
187  struct timeval timeout;
188  timeout.tv_sec = 10;
189  timeout.tv_usec = 0;
190 
191  if (setsockopt(sdtcp, SOL_SOCKET, SO_RCVTIMEO, (char *)&timeout, sizeof(timeout)) < 0)
192  error("setsockopt failed\n");
193  if (connect(sdtcp, (const struct sockaddr *) &tcpsa, sizeof(tcpsa)) < 0) {
194  error("Connection failed with the vaps server: %s (%s)", vaps, inet4_ntop(&tcpsa.sin_addr));
195  return 0;
196  }
197  //error("Connection established with vaps server: %s(%s)", vaps, inet4_ntop(&tcpsa.sin_addr));
198 #if 0 //dax
199  ipmode = _ipmode;
200  if (ipmode == MULTICAST) {
201  char group[GROUP_LEN];
202  Channel::getGroup(World::current()->getChan(), group);
203  sprintf(setup_cmd, "setup a=%s p=%d t=%d r=%.2f ",
204  group, vaps_port, Channel::getTtl(World::current()->getChan()),
205  ::g.times.getRate());
206  }
207  else
208 #endif
209  sprintf(setup_cmd, "setup p=%d r=%.2f ", vaps_port, ::g.times.getRate());
210  write(sdtcp, setup_cmd, strlen(setup_cmd)); // setup packet
211  return sdtcp;
212 }

214 void Android::disconnectFromBapServer()
215 {
216  if (sdtcp > 0) {
217  write(sdtcp, "stop", 4); Socket::closeStream(sdtcp);
218  if (ipmode == MULTICAST) Socket::dropMembership(sdudp, &mreq);
219  }

221  //error("Connection closed with the vaps server");
222  sdtcp = sdudp = -1;
223  state = INACTIVE;

227 int Android::readBapFrame()
228 {
229  fd_set set;
230  FD_ZERO(&set);
231  FD_SET(sdudp, &set);
232  struct timeval delay;
233  delay.tv_sec = delay.tv_usec = 0; // set select passing
234 
235  if (select(FD_SETSIZE, &set, NULL, NULL, &delay) == 0) {
236  //error("select=0 sdudp=%d", sdudp);
237  return 0; // nothing to read
238  }
239 
240  memset(bapline, 0, VAPS_BUFSIZ);
241  socklen_t slen = sizeof(struct sockaddr_in);
242  int len;
243  if ((len = recvfrom(sdudp, bapline, VAPS_BUFSIZ, 0,
244  (struct sockaddr *) &udpsa, &slen)) < 0) { // receive Bap line
245  error("error recvfrom");
246  return 0;
247  }
248  if (! len) {
249  disconnectFromBapServer(); // send stop
250 #if 0 //pdlastbap: memorize last-bapline for persistency
251  char last-bapline[BUFSIZ];
252  strcpy(last-bapline, bapline);
253 #endif
254  return 0; // eof
255  }
256  return 1; // bap is present
257 }

260 void Android::changePermanent(float lasting)
261 {
262  if (usercontrol) {
263  pos.x = localuser->pos.x;

269  updatePosition();
270  }
271 
272  if (state != PLAYING) return; // not connected
273

275  // bap is ready to be played
276  bap->readFrame(bapline); // get it
277 
278  switch (bap->getType()) {

284  if (bap->is(i) && bap->getFap(i) && body->face)
285  body->face->animate(i, (int) bap->getFap(i)); // play fap
286  break;
287  //TODO: predictive interpollation
288  }
289  }
290  else if (body->face) body->face->animate(); // local animation
291 }
