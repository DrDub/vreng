19 /*
20  * rfbproto.h - header file for the RFB protocol version 3.3
21  *
22  * Uses types CARD<n> for an n-bit unsigned integer, INT<n> for an n-bit signed
23  * integer (for n = 8, 16 and 32).
24  *
25  * All multiple byte integers are in big endian (network) order (most
26  * significant byte first). Unless noted otherwise there is no special
27  * alignment of protocol structures.
28  *
29  *
30  * Once the initial handshaking is done, all messages start with a type byte,
31  * (usually) followed by message-specific data. The order of definitions in
32  * this file is as follows:
33  *
34  * (1) Structures used in several types of message.
35  * (2) Structures used in the initial handshaking.
36  * (3) Message types.
37  * (4) Encoding types.
38  * (5) For each message type, the form of the data following the type byte.
39  * Sometimes this is defined by a single structure but the more complex
40  * messages have to be explained by comments.
41  */

157 /*-----------------------------------------------------------------------------
158  * Protocol Version
159  *
160  * The server always sends 12 bytes to start which identifies the latest RFB
161  * protocol version number which it supports. These bytes are interpreted
162  * as a string of 12 ASCII characters in the format "RFB xxx.yyy\n" where
163  * xxx and yyy are the major and minor version numbers (for version 3.3
164  * this is "RFB 003.003\n").
165  *
166  * The client then replies with a similar 12-byte message giving the version
167  * number of the protocol which should actually be used (which may be different
168  * to that quoted by the server).
169  *
170  * It is intended that both clients and servers may provide some level of
171  * backwards compatibility by this mechanism. Servers in particular should
172  * attempt to provide backwards compatibility, and even forwards compatibility
173  * to some extent. For example if a client demands version 3.1 of the
174  * protocol, a 3.0 server can probably assume that by ignoring requests for
175  * encoding types it doesn't understand, everything will still work OK. This
176  * will probably not be the case for changes in the major version number.
177  *
178  * The format string below can be used in sprintf or sscanf to generate or
179  * decode the version string respectively.
180  */
181 
182 #define rfbProtocolVersionFormat "RFB %03d.%03d\n"
183 #define rfbProtocolMajorVersion 3
184 #define rfbProtocolMinorVersion 3
185 
186 typedef char rfbProtocolVersionMsg[13]; /* allow extra byte for null */
187 
188 #define sz_rfbProtocolVersionMsg 12

191 /*-----------------------------------------------------------------------------
192  * Authentication
193  *
194  * Once the protocol version has been decided, the server then sends a 32-bit
195  * word indicating whether any authentication is needed on the connection.
196  * The value of this word determines the authentication scheme in use. For
197  * version 3.0 of the protocol this may have one of the following values:
198  */
199 
200 #define rfbConnFailed 0
201 #define rfbNoAuth 1
202 #define rfbVncAuth 2
203 
204 /*
205  * rfbConnFailed: For some reason the connection failed (e.g. the server
206  * cannot support the desired protocol version). This is
207  * followed by a string describing the reason (where a
208  * string is specified as a 32-bit length followed by that
209  * many ASCII characters).
210  *
211  * rfbNoAuth: No authentication is needed.
212  *
213  * rfbVncAuth: The VNC authentication scheme is to be used. A 16-byte
214  * challenge follows, which the client encrypts as
215  * appropriate using the password and sends the resulting
216  * 16-byte response. If the response is correct, the
217  * server sends the 32-bit word rfbVncAuthOK. If a simple
218  * failure happens, the server sends rfbVncAuthFailed and
219  * closes the connection. If the server decides that too
220  * many failures have occurred, it sends rfbVncAuthTooMany
221  * and closes the connection. In the latter case, the
222  * server should not allow an immediate reconnection by
223  * the client.
224  */
225 
226 #define rfbVncAuthOK 0
227 #define rfbVncAuthFailed 1
228 #define rfbVncAuthTooMany 2

415 /*- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
416  * Hextile Encoding. The rectangle is divided up into "tiles" of 16x16 pixels,
417  * starting at the top left going in left-to-right, top-to-bottom order. If
418  * the width of the rectangle is not an exact multiple of 16 then the width of
419  * the last tile in each row will be correspondingly smaller. Similarly if the
420  * height is not an exact multiple of 16 then the height of each tile in the
421  * final row will also be smaller. Each tile begins with a "subencoding" type
422  * byte, which is a mask made up of a number of bits. If the Raw bit is set
423  * then the other bits are irrelevant; w*h pixel values follow (where w and h
424  * are the width and height of the tile). Otherwise the tile is encoded in a
425  * similar way to RRE, except that the position and size of each subrectangle
426  * can be specified in just two bytes. The other bits in the mask are as
427  * follows:
428  *
429  * BackgroundSpecified - if set, a pixel value follows which specifies
430  * the background colour for this tile. The first non-raw tile in a
431  * rectangle must have this bit set. If this bit isn't set then the
432  * background is the same as the last tile.
433  *
434  * ForegroundSpecified - if set, a pixel value follows which specifies
435  * the foreground colour to be used for all subrectangles in this tile.
436  * If this bit is set then the SubrectsColoured bit must be zero.
437  *
438  * AnySubrects - if set, a single byte follows giving the number of
439  * subrectangles following. If not set, there are no subrectangles (i.e.
440  * the whole tile is just solid background colour).
441  *
442  * SubrectsColoured - if set then each subrectangle is preceded by a pixel
443  * value giving the colour of that subrectangle. If not set, all
444  * subrectangles are the same colour, the foreground colour; if the
445  * ForegroundSpecified bit wasn't set then the foreground is the same as
446  * the last tile.
447  *
448  * The position and size of each subrectangle is specified in two bytes. The
449  * Pack macros below can be used to generate the two bytes from x, y, w, h,
450  * and the Extract macros can be used to extract the x, y, w, h values from
451  * the two bytes.
452  */
453 
454 #define rfbHextileRaw (1 << 0)
455 #define rfbHextileBackgroundSpecified (1 << 1)
456 #define rfbHextileForegroundSpecified (1 << 2)
457 #define rfbHextileAnySubrects (1 << 3)
458 #define rfbHextileSubrectsColoured (1 << 4)
459 
460 #define rfbHextilePackXY(x,y) (((x) << 4) | (y))
461 #define rfbHextilePackWH(w,h) ((((w)-1) << 4) | ((h)-1))
462 #define rfbHextileExtractX(byte) ((byte) >> 4)
463 #define rfbHextileExtractY(byte) ((byte) & 0xf)
464 #define rfbHextileExtractW(byte) (((byte) >> 4) + 1)
465 #define rfbHextileExtractH(byte) (((byte) & 0xf) + 1)
