109 bool WObject::updateLasting(time_t sec, time_t usec, float *lasting)
110 {
111  *lasting = diffTime(sec, usec);
112  if (*lasting < move.ttl) {
113  move.ttl -= *lasting;
114  move.sec = sec;
115  move.usec = usec;
116  return true;
117  }
118  else {
119  if (move.next) {
120  move = *(move.next); // copy next motion into current
121  delete[] move.next; // delete next, now obsoleted
122  move.next = NULL;
123  move.sec = sec;
124  move.usec = usec;
125  move.perm_sec = sec;
126  move.perm_usec = usec;
127  if (move.without_col) {
128  trace(DBG_WO, "next ttl=%.2f", move.ttl);
129 #if HAVE_LIBPTHREAD
130  pthread_t tid;
131  pthread_create(&tid, NULL, endMovement, (void *) NULL);
132 #else
133  if (! fork()) {
134  endMovement(NULL);
135  exit(0);
136  }
137 #endif
138  move.without_col = false;
139  }
140  return true;
141  }
142  *lasting = 0;
143  stopImposedMovement();
144  return false;
145  }

148 /* modify user position in one direction */
149 void User::changePositionOneDir(int move_type, float lasting)
150 {
151  if (carrier && carrier->isTaking()) { // Manipulator
152  carrier->mouseEvent(move_type, lasting);
153  //carrier->keyEvent(move_type, lasting); //arrow keys
154  }
155  else { // Navigator
156  switch (move_type) {
157  case KEY_AV: // move forward left
158  pos.x += lasting * lspeed * Cos(pos.az);
159  pos.y += lasting * lspeed * Sin(pos.az);
160  break;
161  case KEY_AR: // move backward right
162  pos.x -= lasting * lspeed * Cos(pos.az);
163  pos.y -= lasting * lspeed * Sin(pos.az);
164  break;
165  case KEY_SD: // move forward right
166  pos.x += lasting * lspeed * sin(pos.az);
167  pos.y -= lasting * lspeed * Cos(pos.az);
168  break;
169  case KEY_SG: // move backward left
170  pos.x -= lasting * lspeed * Sin(pos.az);
171  pos.y += lasting * lspeed * Cos(pos.az);
172  break;
173  case KEY_DR: // turn right
174  pos.az -= lasting * aspeed;
175  pos.az -= M_2PI * (float) floor(pos.az / M_2PI);
176  break;
177  case KEY_GA: // turn left
178  pos.az += lasting * aspeed;
179  pos.az -= M_2PI * (float) floor(pos.az / M_2PI);
180  break;
181  case KEY_MT: // roll backward
182  pos.ay = MIN(pos.ay + lasting * aspeed, M_2PI_5);
183  break;
184  case KEY_DE: // roll forward
185  pos.ay = MAX(pos.ay - lasting * aspeed, -M_2PI_5);
186  break;
187  case KEY_HZ: // stand up
188  pos.ay = pos.ax = 0;
189  break;
190  case KEY_JU: // move up
191  pos.z += lasting * lspeed;
192  break;
193  case KEY_JD: // move down
194  pos.z -= lasting * lspeed;
195  break;
196  case KEY_TL: // tilt left
197  pos.ax = pos.ax + lasting * aspeed;
198  break;
199  case KEY_TR: // tilt right
200  pos.ax = pos.ax - lasting * aspeed;
201  break;
202  }
203  }
204 }

328 /* user general motion */
329 void User::userMovement(time_t sec, time_t usec)
330 {
331  Pos oldpos = pos;
332  copyPosAndBB(oldpos); // keep oldpos for network
333 
334  float keylastings[MAXKEYS];
335 
336  updateKeysTimes(sec, usec);
337  updateTime(keylastings);
338 
339  float lasting = -1.;
340  for (int k=0; k < MAXKEYS; k++)
341  if (keylastings[k] > lasting) lasting = keylastings[k];
342 
343  if (lasting > MIN_LASTING) { // user is moving
344  if (! isValid()) { error("usermove: type=%d", type); return; }
345  float maxlast = getLasting();
346  maxlast = maxlast ? maxlast : 1;
347  int moves = (int) (lasting / maxlast);
348  float tabdt[MAXKEYS];
349 
350  for (int m=0; m <= moves; m++) {
351  for (int k=0; k < MAXKEYS; k++) {
352  if (keylastings[k] > maxlast) {
353  tabdt[k] = maxlast;
354  keylastings[k] -= maxlast;
355  }
356  else {
357  tabdt[k] = keylastings[k]; // last movement
358  keylastings[k] = 0;
359  }
360  }
361  elementaryUserMovement(tabdt);
362  }
363  updateToNetwork(oldpos);
364  updatePositionAndGrid(oldpos);
365  updateCamera(pos);
366  }
367 }

385 void WObject::imposedMovement(time_t sec, time_t usec)
386 {
387  if (! isValid()) { error("imposedMovement: type=%d invalid", type); return; }
388  if (World::current()->isDead()) return;
389  if (! isMoving()) return; // no moving
390 
391  Pos oldpos = pos;
392  copyPosAndBB(oldpos); // keep oldpos for network
393 
394  float lasting = -1;
395  updateTime(sec, usec, &lasting); // handled by each object
396 
397  if (lasting > MIN_LASTING) {
398  float maxlast = getLasting();
399  maxlast = maxlast ? maxlast : 1;
400  // spliting movement in m elementary movements
401  float tabdt = 0.;

403 
404  for (int m=0; m <= moves; m++) {
405  if (lasting > maxlast) {
406  tabdt = maxlast;
407  lasting -= maxlast;
408  }
409  else {
410  tabdt = lasting;
411  lasting = 0;
412  }

414  if (isBehavior(NO_ELEMENTARY_MOVE)) return; // do movement only once
415  }
416  }
417

419  updatePositionAndGrid(oldpos);

479 /* Moves the user towards the object */
480 void WObject::moveUserToObject(float val, float _lttl, float attl)
481 {
482  // val could be whether a speed either a ttl, that depends on the context
483  if (! localuser) return;
484 
485  int sign = (val >= 0) ? 1 : -1;
486  float d = RATIO_GOTO * sign * MAX(pos.bbsize.v[0], pos.bbsize.v[1]);
487  float dx = pos.x - (d * Sin(pos.az)) - localuser->pos.x;
488  float dy = pos.y + (d * Cos(pos.az)) - localuser->pos.y;
489  float dz = MAX(pos.z - localuser->pos.z, 0.20);
490  float da = deltaAngle(pos.az, localuser->pos.az + 1.57);
491  float db = deltaAngle(pos.ay, localuser->pos.ay); //FIXME: atan
492 
493  float lttl;
494  if (_lttl) lttl = _lttl; // ttl given
495  else lttl = sqrt(dx*dx + dy*dy + dz*dz) / User::LSPEED; // compute ttl
496  //trace(DBG_FORCE, "lttl=%.2f d=%.2f %.2f,%.2f,%.2f", lttl,d,dx,dy,dz);
497 
498  localuser->move.ttl = attl; // needed for pthreaded endMovement
499  clearV3(localuser->move.lspeed);
500  clearV3(localuser->move.aspeed);
501  localuser->move.aspeed.v[0] = da / attl;
502  localuser->move.aspeed.v[1] = db / attl;
503  localuser->initImposedMovement(attl); // start orientation
504 
505  localuser->move.next = new Move[1]; // build 2nd movement eg. translation
506 
507  localuser->move.next->ttl = lttl; // needed for pthreaded endMovement
508  localuser->move.next->next = NULL;
509  localuser->move.next->without_col = true;
510  localuser->move.next->lspeed.v[0] = dx / lttl;
511  localuser->move.next->lspeed.v[1] = dy / lttl;
512  localuser->move.next->lspeed.v[2] = dz / lttl;
513  clearV3(localuser->move.aspeed);
514 }
