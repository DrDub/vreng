243 void Smoke::changePermanent(float dt)
244 {
245 #ifdef NEW_SMOKE
246  //error("smoke: change");
247  addParticle();
248  animParticle();
249 #else //NEW_SMOKE
250  struct sParticle *p = particles;
251  float da = pow(0.2, dt);
252 
253  time += dt;
254  //error("np=%d", np);
255  motionAnimate(dt);
256  for (int n=0; n < np; n++, p++) {
257  if ((p->t -= dt) <= 0)
258  *p = particles[--np]; // kill it
259  else { // animate it
260  p->v += fpart * dt;
261  p->p += p->v * dt;
262  motionWarp(p->p, dt);
263  p->a *= da;
264  }
265  }
266  lasttime += dt;
267  while (np < FIREMAX && (time - lasttime >= SMOKEDELTA)) {
268  Vector3 dir(0,0,.25);
269  lasttime += SMOKEDELTA;
270  p = particles + (np++);
271  p->p = src; // + DS1*rnd2(0.25) + DS2*rnd2(0.25);
272  p->v = dir;
273  motionWarp(p->p, (time - lasttime));
274  p->t = SMOKELIFE + lasttime - time;
275  float size = SMOKESIZE * (0.5+rnd2(0.5));
276  float alpha = SMOKEALPHA * (float)pow(size/SMOKESIZE, 0.5);
277  p->s.x = size;
278  p->s.y = size;
279  p->s.z = size;
280  p->a = alpha*pow(da, (time - lasttime));
281  p->s *= 0.5 + rnd2(0.5);
282  }
283  O.reset();
284  M.apply(src, &O);
285  O = src - O;
286  p = particles;
287  for (int n=0; n < np; n++, p++) {
288  float foc = 2.;
289  Vector3 v;
290  M.apply(p->p, &v);
291  v += O;
292  if (v.y > 1) {
293  p->ex = projx(v);
294  p->ey = projy(v);
295  projz(p->ez, v);
296  p->dx = foc*p->s.x/v.y;
297  p->dy = foc*p->s.z/v.y;
298  }
299  else p->dx = 0;
300  }
301 #endif //NEW_SMOKE
302 }

418 void Smoke::motionAnimate(float dt)
419 {
420  SMOKE_PARTICLE *p = &offset[0][0][0];
421  for (int z=0; z<NZ; z++) {
422  int zh = z+1; int zl = z-1;
423  for (int y=0; y<NY; y++) {
424  int yh = y+1; int yl = y-1;
425  for (int x=0; x<NX; x++,p++) {
426  int xh = x+1; int xl = x-1;
427  for (int zi=zl; zi<=zh; zi++) {
428  for (int yi=yl; yi<=yh; yi++) {
429  for (int xi=xl; xi<=xh; xi++) {
430  if ((xi != x) || (yi != y) || (zi!=z)) {
431  float norm = norms[abs(zi-z)][abs(yi-y)][abs(xi-x)];
432  SMOKE_PARTICLE *p2 = &offset[zi&(NZ-1)][yi&(NY-1)][xi&(NX-1)];
433  Vector3 spring = p2->p;
434  spring -= p->p;
435  spring.x += distvar[xi-x+1];
436  spring.y += distvar[yi-y+1];
437  spring.z += distvar[zi-z+1];
438  spring *= (norm-spring.length()) * (float)SPRINGF * dt;
439  p2->v += spring;
440  }
441  }
442  }
443  }
444  }
445  }
446  }
447  p = &offset[0][0][0];
448  float ft = (float) pow(DYNF, dt);
449  for (int z=0; z<NZ; z++) {
450  for (int y=0; y<NY; y++) {
451  for (int x=0; x<NX; x++, p++) {
452  p->p += p->v * dt;
453  if (p->p.length2() > (MAXWARPF*MAXWARPF))
454  p->v.reset();
455  else
456  p->v *= ft;
457  }
458  }
459  }
460 }

465 void Smoke::motionWarp(Vector3 &p, float dt)
466 {
467  static Vector3 v0,v1,v2,v3,v4,v5,v;
468 
469  int sx = (f2int((p.x)*(TILE*(1<<8)))+((NX/2)<<8)) & ((NX<<8)-1);
470  int sy = (f2int((p.y)*(TILE*(1<<8)))+((NY/2)<<8)) & ((NY<<8)-1);
471  int sz = (f2int((p.z)*(TILE*(1<<8)))) & ((NZ<<8)-1);
472  float fx = deplfact[sx&((1<<8)-1)];
473  float fy = deplfact[sy&((1<<8)-1)];
474  float fz = deplfact[sz&((1<<8)-1)];
475 
476  sx >>= 8; //notused int x2 = (sx+1)&(NX-1);
477  sy >>= 8; //notused int y2 = (sy+1)&(NY-1);
478  sz >>= 8; int z2 = (sz+1)&(NZ-1);
479 
480  v0.lerp(offset[sz][sy][sx].p, offset[sz][sy][sx+1].p, fx);
481  v1.lerp(offset[z2][sy][sx].p, offset[z2][sy][sx+1].p, fy);
482  v.lerp(v0, v1, fz);
483  v *= dt;
484  p += v;
485 }
