63 void Cloth::parser(char *l)
64 {
65  defaults();
66  l = tokenize(l);
67  begin_while_parse(l) {
68  l = parse()->parseAttributes(l, this);
69  if (!l) break;
70  if (!stringcmp(l, "article=")) {
71  char str[16];
72  l = parse()->parseString(l, str, "article");
73  if (!stringcmp(str, "halo")) { halo = true; article = HALO; }
74  else if (!stringcmp(str, "hat")) { hat = true; article = HAT; }
75  else if (!stringcmp(str, "dress")) { skirt = true; article = DRESS; }
76  else if (!stringcmp(str, "wings")) { wings = true; article = WINGS; }
77  }
78  else if (!stringcmp(l, "color=")) l = parse()->parseString(l, color, "color");
79  }
80  end_while_parse(l);
81 }

84 void Cloth::makeSolid()

89  sprintf(s,"solid shape=\"disk\" ri=\"%f\" re=\"%f\" dif=\"%s\" />", .09, .12, color);

93  parse()->parseSolid(s, SEP, this);
94  sprintf(s,"solid shape=\"disk\" ri=\"%f\" re=\"%f\" dif=\"%s\" />", .08, .15, color);
95  case DRESS:
96  sprintf(s,"solid shape=\"cone\" rb=\"%f\" rt=\"%f\" h=\"%f\" dif=\"%s\" />", .10, .30, -.50, color);
97  break;
98  case WINGS:
99  sprintf(s,"solid shape=\"bbox\" dim=\"%f %f %f\" />", 1.2, .02, 1.2);
100  default:
101  return;
102  }
103  parse()->parseSolid(s, SEP, this);
104 }

168 /* sets position near the avatar */
169 void Cloth::inits()
170 {
171  switch (article) {
172  case HALO:
173  case HAT:
174  if (localuser) dz += localuser->height / 2;
175  break;
176  }
177  // save original position
178  ox = pos.x; oy = pos.y; oz = pos.z;
179  oax = pos.ax; oay = pos.ay; oaz = pos.az;
180  // set new position
181  if (localuser) pos.x = localuser->pos.x + dx;
182  if (localuser) pos.y = localuser->pos.y + dy;
183  if (localuser) pos.z = localuser->pos.z + dz;
184  pos.ax = dax;
185  pos.ay = day;
186  if (localuser) pos.az = localuser->pos.az + daz;
187  updatePosition();
188 }
