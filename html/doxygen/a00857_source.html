88 /* Created from file */
89 void Wings::parser(char *l)
90 {
91  defaults();
92  l = tokenize(l);
93  begin_while_parse(l) {
94  l = parse()->parseAttributes(l, this); // <solid ... />
95  if (!l) break;
96  if (!stringcmp(l, "model=")) {
97  l = parse()->parseString(l, modelname, "model");
98  if (!stringcmp(modelname, "butterfly")) model = BUTTERFLY;
99  else if (!stringcmp(modelname, "libellule")) model = LIBELLULE;
100  else if (!stringcmp(modelname, "angel")) model = ANGEL;
101  else if (!stringcmp(modelname, "eagle")) model = EAGLE;
102  else if (!stringcmp(modelname, "helicopter")) model = HELICOPTER;
103  }
104  }
105  end_while_parse(l);
106 }

385 void Wings::changePermanent(float lasting)
386 {
387  if (! taken) return;
388 
389  pos.x = localuser->pos.x + dx;
390  pos.y = localuser->pos.y + dy;
391  pos.z = localuser->pos.z + dz;
392  pos.ax = localuser->pos.ax + DEG2RAD(dax);
393  pos.ay = localuser->pos.ay + DEG2RAD(day);
394  pos.az = localuser->pos.az + DEG2RAD(daz);
395  updatePosition();
396 
397  switch (model) {
398  case BUTTERFLY :
399  if (angle > 90) sign = -1;
400  if (angle < 0) sign = 1;
401  break;
402  case LIBELLULE :
403  if (angle > 15) sign = -1;
404  if (angle < 0) sign = 1;
405  break;
406  case ANGEL :
407  if (angle > 30) sign = -1;
408  if (angle < 0) sign = 1;
409  break;
410  case EAGLE :
411  sign = 0;
412  case HELICOPTER :
413  sign = 1;
414  angle += sign * 12;
415  break;
416  }
417  angle += sign * 6;
418 }

444  glPopMatrix();
445  glPushMatrix();

447  glCallList(dlist_right);
448  glPopMatrix();
449  break;
450  }
