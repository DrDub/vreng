80 bool VNCRFBproto::initialiseRFBConnection()
81 {
82  rfbProtocolVersionMsg pv;
83  rfbClientInitMsg ci;
84  int major, minor;
85  char *reason, *passwd = NULL;
86  uint32_t authScheme, reasonLen, authResult;
87  uint8_t challenge[CHALLENGESIZE];
88 
89  trace(DBG_VNC, "initialiseRFBConnection: initializing RFB connection");
90 
91  if (!VNC_sock.ReadFromRFBServer(pv, sz_rfbProtocolVersionMsg))
92  return false;
93 
94  pv[sz_rfbProtocolVersionMsg] = 0;
95 
96  if (sscanf(pv, rfbProtocolVersionFormat, &major, &minor) != 2) {
97  error("initialiseRFBConnection: not a valid VNC server");
98  return false;
99  }
100  trace(DBG_VNC, "initialiseRFBConnection: VNC server supports protocol version %d.%d (viewer %d.%d)",
101  major, minor, rfbProtocolMajorVersion, rfbProtocolMinorVersion);
102 
103  major = rfbProtocolMajorVersion;
104  minor = rfbProtocolMinorVersion;
105  sprintf(pv, rfbProtocolVersionFormat, major, minor);
106 
107  if (!VNC_sock.WriteExact(pv, sz_rfbProtocolVersionMsg))
108  return false;
109  if (!VNC_sock.ReadFromRFBServer((char *) &authScheme, 4))
110  return false;
111 
112  authScheme = (uint32_t) swap32IfLE(authScheme);
113  switch (authScheme) {
114 
115  case rfbConnFailed:
116  if (!VNC_sock.ReadFromRFBServer((char *) &reasonLen, 4))
117  return false;
118  reasonLen = swap32IfLE(reasonLen);
119  reason = new char[reasonLen];
120 
121  if (!VNC_sock.ReadFromRFBServer(reason, reasonLen))
122  return false;
123  error("VNC connection failed: %.*s", (int)reasonLen, reason);
124  delete[] reason;
125  return false;
126 
127  case rfbNoAuth:
128  trace(DBG_VNC, "No authentication needed");
129  break;
130 
131  case rfbVncAuth:
132  if (!VNC_sock.ReadFromRFBServer((char *) challenge, CHALLENGESIZE))
133  return false;
134 
135  //look for a password stored in passwordFile
136  passwd = vncDecryptPasswdFromFile(passwordFile);
137  if (!passwd) {
138  error("Cannot read valid password from file \"%s\"", passwordFile);
139  return false;
140  }
141  if (! passwd || (! strlen(passwd))) {
142  error("initialiseRFBConnection: reading password failed");
143  return false;
144  }
145  if (strlen(passwd) > 8)
146  passwd[8] = '\0';
147 
148  vncEncryptBytes(challenge, passwd);
149 
150  /* Lose the password from memory */
151  for (int i = strlen(passwd); i >= 0; i--)
152  passwd[i] = '\0';
153 
154  if (!VNC_sock.WriteExact((char *) challenge, CHALLENGESIZE))
155  return false;
156  if (!VNC_sock.ReadFromRFBServer((char *) &authResult, 4))
157  return false;
158 
159  authResult = swap32IfLE(authResult);
160  switch (authResult) {
161  case rfbVncAuthOK:
162  trace(DBG_VNC, "initialiseRFBConnection: VNC authentication succeeded");
163  break;
164  case rfbVncAuthFailed:
165  error("initialiseRFBConnection: VNC authentication failed");
166  return false;
167  case rfbVncAuthTooMany:
168  error("VNC authentication failed - too many tries");
169  return false;
170  default:
171  error("Unknown VNC authentication result: %d", (int)authResult);
172  return false;
173  }
174  break;
175 
176  default:
177  error("Unknown authentication scheme from VNC server: %d", (int)authScheme);
178  return false;
179  }
180 
181  ci.shared = (ShareDesktop ? 1 : 0);
182 
183  if (!VNC_sock.WriteExact((char *)&ci, sz_rfbClientInitMsg))
184  return false;
185  if (!VNC_sock.ReadFromRFBServer((char *)&si, sz_rfbServerInitMsg))
186  return false;
187 
188  si.framebufferWidth = swap16IfLE(si.framebufferWidth);
189  si.framebufferHeight = swap16IfLE(si.framebufferHeight);
190  si.format.redMax = swap16IfLE(si.format.redMax);
191  si.format.greenMax = swap16IfLE(si.format.greenMax);
192  si.format.blueMax = swap16IfLE(si.format.blueMax);
193  si.nameLength = swap32IfLE(si.nameLength);
194 
195  desktopName = new char[si.nameLength + 1];
196  if (!VNC_sock.ReadFromRFBServer(desktopName, si.nameLength)) {
197  delete[] desktopName;
198  desktopName = NULL;
199  return false;
200  }
201  desktopName[si.nameLength] = 0;
202  notice("Desktop: %s", desktopName);
203  delete[] desktopName;
204  desktopName = NULL;
205 
206  trace(DBG_VNC, "initialiseRFBConnection: connected to VNC server, using protocol version %d.%d",
207  rfbProtocolMajorVersion, rfbProtocolMinorVersion);
208  trace(DBG_VNC, "initialiseRFBConnection: VNC server default format:");
209  printPixelFormat(&si.format);
210 
211  return true;
212 }

214 bool VNCRFBproto::setFormatAndEncodings()
215 {
216  rfbSetPixelFormatMsg spf;
217  char buf[sz_rfbSetEncodingsMsg + MAX_ENCODINGS * 4];
218  rfbSetEncodingsMsg *se = (rfbSetEncodingsMsg *)buf;
219  uint32_t *encs = (uint32_t *)(&buf[sz_rfbSetEncodingsMsg]);
220  int len = 0;
221 
222  spf.type = rfbSetPixelFormat;
223  spf.format = pixFormat;
224  spf.format.redMax = swap16IfLE(spf.format.redMax);
225  spf.format.greenMax = swap16IfLE(spf.format.greenMax);
226  spf.format.blueMax = swap16IfLE(spf.format.blueMax);
227 
228  if (!VNC_sock.WriteExact((char *)&spf, sz_rfbSetPixelFormatMsg))
229  return false;
230 
231  se->type = rfbSetEncodings;
232  se->nEncodings = 0;
233 
234  if (VNC_sock.SameMachine()) {
235  trace(DBG_VNC, "Same machine: preferring raw encoding");
236  encs[se->nEncodings++] = swap32IfLE(rfbEncodingRaw);
237  }
238  encs[se->nEncodings++] = swap32IfLE(rfbEncodingCopyRect);
239  encs[se->nEncodings++] = swap32IfLE(rfbEncodingHextile);
240  encs[se->nEncodings++] = swap32IfLE(rfbEncodingCoRRE);
241  encs[se->nEncodings++] = swap32IfLE(rfbEncodingRRE);
242 
243  len = sz_rfbSetEncodingsMsg + se->nEncodings * 4;
244  se->nEncodings = swap16IfLE(se->nEncodings);
245 
246  if (!VNC_sock.WriteExact(buf, len))
247  return false;
248 
249  return true;
250 }

334 void VNCRFBproto::printPixelFormat(rfbPixelFormat *format)
335 {
336  if (format->bitsPerPixel == 1) {
337  trace(DBG_VNC, " Single bit per pixel");
338  trace(DBG_VNC, " %s significant bit in each byte is leftmost on the screen",

340  }
341  else {
342  trace(DBG_VNC, " %d bits per pixel", format->bitsPerPixel);
343  if (format->bitsPerPixel != 8) {
344  trace(DBG_VNC, " %s significant byte first in each pixel",
345  (format->bigEndian ? "Most" : "Least"));
346  }
347  if (format->trueColour) {
348  trace(DBG_VNC, " True color: max red %d green %d blue %d, shift red %d green %d blue %d",
349  format->redMax, format->greenMax, format->blueMax,
350  format->redShift, format->greenShift, format->blueShift);
351  }
352  else
353  trace(DBG_VNC, " Colour map (not true colour)");
354  }
