122  * Return codes:
123  * 0 = OK
124  * 1 = reached EOF
125  * 2 = not a AVI file
126  */
127 int32_t Avi::read_header()
128 {
129  int32_t tag, n, name;
130  int32_t lasttag = 0;
131  int32_t vids_strh_seen = 0;
132  int32_t vids_strf_seen = 0;
133  uint32_t args[64];
134 
135  /* Read first 12 bytes and check that this is an AVI file */
136  if (fread(&tag , 4, 1, file) != 1) return 1;
137  if (fread(&n , 4, 1, file) != 1) return 1;
138  if (fread(&name, 4, 1, file) != 1) return 1;
139  if (tag != RIFFtag || name != MAKEFOURCC('A','V','I',' ')) {
140  error("avi: not a AVI file");
141  return ERR_NO_AVI;
142  }
143 
144  /* Read all tags until we encounter a 'LIST xxx movi' tag */
145  while (1) {
146  if (fread(&tag, 4, 1, file) != 1) return 1;
147  if (fread(&n , 4, 1, file) != 1) return 1;
148 #if WORDS_BIGENDIAN
149  char tmp;
150  SWAPL(&n, tmp);
151 #endif
152  if (n&1) n++; /* Odd values are padded */
153  //error("avi: tag=%08x n=%d", tag, n);
154  /* Unless a LIST tag hasn't name movi, it is ignored */
155  if (tag == LISTtag) {
156  if (fread(&name, 4, 1, file) != 1) return 1;
157  //error("avi: name=%08x", name);
158  if (name == MAKEFOURCC('m','o','v','i')) {
159  //error("avi: movi");
160  return 0;
161  }
162  continue;
163  }
164 
165  /* Read the arguments of this tag, 256 bytes are sufficient, remainder is skipped */
166  if (n <= 256) {
167  if (fread(args, n, 1, file) != 1) return 1;
168  }
169  else {
170  if (fread(args, 256, 1, file) != 1) return 1;
171  if (fseek(file, n-256, SEEK_CUR)) return 1;
172  }
173 
174  /* Interpret the tag and its args */
175  switch (tag) {
176  case strhtag:
177  //error("avi: strhtag n=%d", n);
178  if (args[0] == MAKEFOURCC('v','i','d','s')) {
179  if (args[1] != MAKEFOURCC('M','J','P','G') &&
180  args[1] != MAKEFOURCC('m','j','p','g') &&
181  args[1] != MAKEFOURCC('M','S','V','C') &&
182  args[1] != MAKEFOURCC('m','s','v','c')) {
183  error("avi: args1 no mjpg %08x", args[1]);
184  return ERR_NO_MJPG;
185  }
186 #if WORDS_BIGENDIAN
187  SWAPL(&args[5], tmp);
188  SWAPL(&args[6], tmp);
189 #endif
190  if (args[5] != 0) fps = (double)args[6] / (double)args[5];
191  //error("avi: fps=%.2f", fps);
192  vids_strh_seen = 1;
193  lasttag = 1; /* vids */
194  }
195  else if (args[0] == MAKEFOURCC('a','u','d','s')) lasttag = 2; /* auds */
196  break;
197  case strftag:
198  //error("avi: strftag n=%d", n);
199  if (lasttag == 1) {
200 #if WORDS_BIGENDIAN
201  SWAPL(&args[1], tmp);
202  SWAPL(&args[2], tmp);
203 #endif
204  width = args[1];
205  height = args[2];
206  //error("avi: width=%d height=%d", width, height);
207  if (args[4] != MAKEFOURCC('M','J','P','G') &&
208  args[4] != MAKEFOURCC('m','j','p','g') &&
209  args[4] != MAKEFOURCC('C','R','A','M') &&
210  args[4] != MAKEFOURCC('c','r','a','m') &&
211  args[4] != MAKEFOURCC('M','S','V','C') &&
212  args[4] != MAKEFOURCC('m','s','v','c')) {
213  error("avi: args4 no mjpg %08x", args[4]);
214  return ERR_NO_MJPG;
215  }
216  vids_strf_seen = 1;
217  }
218  else if (lasttag == 2) {
219  /* Check audio format (must be PCM) */
220  if ((args[0]&0xffff) != 1) {
221  error("avi: no PCM tag=%08x", tag);
222  break;
223  //return ERR_NO_PCM;
224  }
225  a_chans = args[0]>>16;
226  a_rate = args[1];
227  a_bits = args[3]>>16;
228  }
229  break;
230  default:
231  //error("avi: default tag=%08x n=%d", tag, n);
232  lasttag = 0;
233  break;
234  }
235  }
236  if (!vids_strh_seen || !vids_strf_seen) return ERR_NO_VIDS;
237  return 0;
238 }

248 int Avi::read_data(uint8_t *vidbuf, uint32_t max_vid, int32_t *retlen)
249 {
250  uint32_t tag, n;
251 
252  while (file) {
253  /* Read tag and length */
254  if (fread(&tag, 4, 1, file) != 1 ||
255  fread(&n , 4, 1, file) != 1 ) {
256  //error("avi: read error");
257  file = NULL;
258  return 0;
259  }
260 #if WORDS_BIGENDIAN
261  char tmp;
262  SWAPL(&n, tmp);
263 #endif
264  if (n&1) n++; /* Odd values are padded */
265 
266  //error("avi: read tag=%08x n=%d", tag, n);
267  switch (tag) {
268  case MAKEFOURCC('0','0','d','b'):
269  case MAKEFOURCC('0','0','d','c'):
270  if (n > max_vid) {
271  error("avi: video buffer too small n=%d", n);
272  file = NULL;
273  return -1;
274  }
275  *retlen = n;
276  if (fread(vidbuf, n, 1, file) != 1) {
277  //error("avi: eof! n=%d", n);
278  file = NULL;
279  return 0;
280  }
281  return 1; // OK
282  break;
283  default:
284  if (fseek(file, n, SEEK_CUR)) {
285  error("avi: ignore tag=%08x n=%d", tag, n);
286  file = NULL;
287  return 0;
288  }
289  break;
290  }
291  }
292  return 0;
293 }

427 void Avi::write_header(int width, int height, int norm, int audio, int stereo, int size, int rate)
428 {
429  int32_t nbytes, pos, njunk, abytes, achans, sampsize;
430 
431  /* write index */
432  finish_list("movi");
433  fseek(file, 0, SEEK_END);
434  out4cc("idx1");
435  outlong(n_idx*4); /* # of bytes to follow */
436  nbytes = fwrite(idx, n_idx*4, 1, file);
437  if (nbytes != 1) { perror("write"); return; }
438 
439  /* get the file size */
440  fseek(file, 0, SEEK_END);
441  nbytes = ftell(file);
442  fseek(file, 0, SEEK_SET); /* Rewind */
443 
444  if (!audio && audio_samples)
445  error("AVI: Audio samples present, but no audio specified");
446 
447  /* The RIFF header */
448  out4cc("RIFF");
449  outlong(nbytes - 8); /* # of bytes to follow */
450  out4cc("AVI ");
451 
452  /* Start the header list */
453  start_list("hdrl");
454 
455  /* The main AVI header */
456  out4cc("avih");
457  outlong(56); /* # of bytes to follow */
458  if (norm == VIDEO_MODE_PAL)
459  outlong(40000); /* Microseconds per frame */
460  else
461  outlong(33780); /* Microseconds per frame, NTSC, might need adjustment */
462  outlong(10000000); /* MaxBytesPerSec, I hope this will never be used */
463  outlong(0); /* PaddingGranularity (whatever that might be) */
464  outlong(0x10010); /* Flags: AVIF_HASINDEX | AVIF_WASCAPTUREFILE */
465  //outlong(0x00000);
466  outlong(frames); /* TotalFrames */
467  outlong(0); /* InitialFrames */
468  if (audio_samples)
469  outlong(2); /* Streams */
470  else
471  outlong(1); /* Streams */
472  outlong(0); /* SuggestedBufferSize */
473  outlong(width); /* Width */
474  outlong(height); /* Height */
475  /* MS calls the following 'reserved': */
476  outlong(0); /* TimeScale: Unit used to measure time */
477  outlong(0); /* DataRate: Data rate of playback */
478  outlong(0); /* StartTime: Starting time of AVI data */
479  outlong(0); /* DataLength: Size of AVI data chunk */
480 
481  /* Start the video stream list */
482  start_list("strl");
483 
484  /* The video stream header */
485  out4cc("strh");
486  outlong(64); /* # of bytes to follow */
487  out4cc("vids");
488  outlong(0); // RGB 4cc
489  outlong(0); /* Flags */
490  outlong(0); /* Reserved, MS says: wPriority, wLanguage */
491  outlong(0); /* InitialFrames */
492  if (norm == VIDEO_MODE_PAL) outlong(40000); /* Scale */
493  else outlong(33780); /* Scale */
494  outlong(1000000); /* Rate: Rate/Scale == samples/second */
495  outlong(0); /* Start */
496  outlong(frames); /* Length */
497  outlong(0); /* SuggestedBufferSize */
498  outlong(-1); /* Quality */
499  outlong(0); /* SampleSize */
500  outlong(0); /* Frame */
501  outlong(0); /* Frame */
502  outlong(0); /* Frame */
503  outlong(0); /* Frame */
504 
505  /* The video stream format */
506  out4cc("strf");
507  outlong(68); /* # of bytes to follow */
508  outlong(68); /* Size */
509  outlong(width); /* Width */
510  outlong(height); /* Height */
511  outshrt(1); outshrt(24);/* Planes, Count */
512  //out4cc ("MJPG"); /* Compression */
513  outlong(0); // RGB 4cc
514  outlong(width*height); /* SizeImage ??? */
515  outlong(0); /* ??? */
516  outlong(0); /* ??? */
517  outlong(0); /* ??? */
518  outlong(0); /* ??? */
519  outlong(44); /* ??? */
520  outlong(24); /* ??? */
521  outlong(0); /* ??? */
522  outlong(2); /* ??? */
523  outlong(8); /* ??? */
524  outlong(2); /* ??? */
525  outlong(1); /* ??? */
526 
527  /* Finish stream list */
528  finish_list("strl");
529  if (audio_samples) {
530  abytes = (size==16) ? 2 : 1;
531  achans = stereo ? 2 : 1;
532  sampsize = abytes*achans;
533 
534  /* Start the audio stream list ---------------------------------- */
535  start_list("strl");
536 
537  /* The audio stream header */
538  out4cc("strh");
539  outlong(64); /* # of bytes to follow */
540  out4cc("auds");
541  out4cc("\0\0\0\0");
542  outlong(0); /* Flags */
543  outlong(0); /* Reserved, MS says: wPriority, wLanguage */
544  outlong(0); /* InitialFrames */
545  outlong(sampsize); /* Scale */
546  outlong(sampsize*rate); /* Rate: Rate/Scale == samples/second */
547  outlong(0); /* Start */
548  outlong(audio_samples/sampsize); /* Length */
549  outlong(0); /* SuggestedBufferSize */
550  outlong(-1); /* Quality */
551  outlong(sampsize); /* SampleSize */
552  outlong(0); /* Frame */
553  outlong(0); /* Frame */
554  outlong(0); /* Frame */
555  outlong(0); /* Frame */
556 
557  /* The audio stream format */
558  out4cc("strf");
559  outlong(16); /* # of bytes to follow */
560  outshrt(WAVE_FORMAT_PCM); /* Size */
561  outshrt(achans); /* Number of channels */
562  outlong(rate); /* SamplesPerSec */
563  outlong(sampsize*rate); /* AvgBytesPerSec */
564  outshrt(sampsize); /* BlockAlign */
565  outshrt(abytes*8); /* BitsPerSample */
566 
567  /* Finish stream list */
568  finish_list("strl");
569  }
570 
571  /* Finish header list */
572  finish_list("hdrl");
573 
574  /* Calculate the needed amount of junk bytes, output junk */
575  pos = ftell(file);
576  njunk = DATASTART - pos - 8 - 12;
577  out4cc("JUNK");
578  outlong(njunk);
579  File::closeFile(file);
580  file = NULL;
581 }
