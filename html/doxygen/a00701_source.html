197 void Smoke::changePermanent(float dt)
198 {
199  struct sParticle *p = particles;
200  float da = pow(0.2, dt);
201 
202  time += dt;
203  //error("np=%d", np);
204  motionAnimate(dt);
205  for (int n=0; n < np; n++, p++) {
206  if ((p->t -= dt) <= 0)
207  *p = particles[--np]; // kill it
208  else { // animate it
209  p->v += fpart * dt;
210  p->p += p->v * dt;
211  motionWarp(p->p, dt);
212  p->a *= da;
213  }
214  }
215  lasttime += dt;
216  while (np < FIREMAX && (time - lasttime >= SMOKEDELTA)) {
217  Vector3 dir(0,0,.25);
218  lasttime += SMOKEDELTA;
219  p = particles + (np++);
220  p->p = src; // + DS1*rnd2(0.25) + DS2*rnd2(0.25);
221  p->v = dir;
222  motionWarp(p->p, (time - lasttime));
223  p->t = SMOKELIFE + lasttime - time;
224  float size = SMOKESIZE * (0.5+rnd2(0.5));
225  float alpha = SMOKEALPHA * (float)pow(size/SMOKESIZE, 0.5);
226  p->s.x = size;
227  p->s.y = size;
228  p->s.z = size;
229  p->a = alpha*pow(da, (time - lasttime));
230  p->s *= 0.5 + rnd2(0.5);
231  }
232  O.reset();
233  M.apply(src, &O);
234  O = src - O;
235  p = particles;
236  for (int n=0; n < np; n++, p++) {
237  float foc = 2.;
238  Vector3 v;
239  M.apply(p->p, &v);
240  v += O;
241  if (v.y > 1) {
242  p->ex = projx(v);
243  p->ey = projy(v);
244  projz(p->ez, v);
245  p->dx = foc*p->s.x/v.y;
246  p->dy = foc*p->s.z/v.y;
247  }
248  else p->dx = 0;
249  }
250 }

252 // Handles the wobbling gel, which is used to give the WarpMap it's shape
253 void Smoke::motionAnimate(float dt)
254 {
255  SMOKE_PARTICLE *p = &offset[0][0][0];
256  for (int z=0; z<NTILEZ; z++) {
257  int zh = z+1; int zl = z-1;
258  for (int y=0; y<NTILEY; y++) {
259  int yh = y+1; int yl = y-1;
260  for (int x=0; x<NTILEX; x++,p++) {
261  int xh = x+1; int xl = x-1;
262  for (int zi=zl; zi<=zh; zi++) {
263  for (int yi=yl; yi<=yh; yi++) {
264  for (int xi=xl; xi<=xh; xi++) {
265  if ((xi != x) || (yi != y) || (zi!=z)) {
266  float norm = norms[abs(zi-z)][abs(yi-y)][abs(xi-x)];
267  SMOKE_PARTICLE *p2 = &offset[zi&(NTILEZ-1)][yi&(NTILEY-1)][xi&(NTILEX-1)];
268  Vector3 spring = p2->p;
269  spring -= p->p;
270  spring.x += distvar[xi-x+1];
271  spring.y += distvar[yi-y+1];
272  spring.z += distvar[zi-z+1];
273  spring *= (norm-spring.length()) * (float)SPRINGF * dt;
274  p2->v += spring;
275  }
276  }
277  }
278  }
279  }
280  }
281  }
282  p = &offset[0][0][0];
283  float ft = (float) pow(DYNF, dt);
284  for (int z=0; z<NTILEZ; z++) {
285  for (int y=0; y<NTILEY; y++) {
286  for (int x=0; x<NTILEX; x++, p++) {
287  p->p += p->v * dt;
288  if (p->p.length2() > (MAXLEN*MAXLEN))
289  p->v.reset();
290  else
291  p->v *= ft;
292  }
293  }
294  }
295 }

297 void Smoke::motionWarp(Vector3 &p, float dt)
298 {
299  static Vector3 v0,v1,v2,v3,v4,v5,v;
300 
301  int sx = (f2int((p.x)*(TILE*(1<<8)))+((NTILEX/2)<<8)) & ((NTILEX<<8)-1);
302  int sy = (f2int((p.y)*(TILE*(1<<8)))+((NTILEY/2)<<8)) & ((NTILEY<<8)-1);
303  int sz = (f2int((p.z)*(TILE*(1<<8)))) & ((NTILEZ<<8)-1);
304  float fx = deplfact[sx&((1<<8)-1)];
305  float fy = deplfact[sy&((1<<8)-1)];
306  float fz = deplfact[sz&((1<<8)-1)];
307  sx >>= 8; //notused int x2 = (sx+1)&(NTILEX-1);
308  sy >>= 8; //notused int y2 = (sy+1)&(NTILEY-1);
309  sz >>= 8; int z2 = (sz+1)&(NTILEZ-1);
310 
311  v0.lerp(offset[sz][sy][sx].p, offset[sz][sy][sx+1].p, fx);
312  v1.lerp(offset[z2][sy][sx].p, offset[z2][sy][sx+1].p, fy);
313  v.lerp(v0, v1, fz);
314  v *= dt;
315  p += v;
316 }
