110 Img * Img::loadDDS(void *tex, ImageReader read_func)
111 {
112  DDSurfaceDesc ddsd;
113  char magic[4];
114  long size, curr, end;
115 
116  const char *glexts = (const char *)glGetString (GL_EXTENSIONS);
117 
118  /* check for S3TC support */
119  if (!strstr (glexts, "GL_EXT_texture_compression_s3tc")) {
120  error("GL_EXT_texture_compression_s3tc is required for DDS textures!");
121  return NULL;
122  }
123  /* init extensions */
124 #ifdef _WIN32
125  glCompressedTexImage2D = (PFNGLCOMPRESSEDTEXIMAGE2DARBPROC)
126  wglGetProcAddress ("glCompressedTexImage2DARB");
127 #endif
128 
129  dds_t *dds = new dds_t[1];
130  if (! dds) return NULL;
131  memset(dds, 0, sizeof(dds_t));
132 
133  Texture *_tex = (Texture *) tex;
134  if ((dds->fp = Cache::openCache(_tex->url, _tex->http)) == NULL) return NULL;
135 
136  /* read magic number and check if valid .dds file */
137  fread(&magic, sizeof(char), 4, dds->fp);
138  if (strncmp (magic, "DDS ", 4) != 0) {
139  error("file \"%s\" is not a valid .dds file!", Cache::getFilePath(_tex->url));
140  File::closeFile(dds->fp);
141  return NULL;
142  }
143 
144  /* get the surface descriptor */
145  fread(&ddsd, sizeof(ddsd), 1, dds->fp);
146 
147  dds->width = ddsd.width;
148  dds->height = ddsd.height;
149  dds->numMipmaps = ddsd.mipMapLevels;
150 
151  trace(DBG_IMG, "dds: width=%d height=%d numMipmaps=%d",
152  dds->width, dds->height, dds->numMipmaps);
153 
154  switch (ddsd.channel.fourCC) {
155  case DXT1:
156  dds->channel = GL_COMPRESSED_RGBA_S3TC_DXT1_EXT;
157  dds->internal_channel = Img::RGB;
158  break;
159  case DXT3:
160  dds->channel = GL_COMPRESSED_RGBA_S3TC_DXT3_EXT;
161  dds->internal_channel = Img::RGBA;
162  break;
163  case DXT5:
164  dds->channel = GL_COMPRESSED_RGBA_S3TC_DXT5_EXT;
165  dds->internal_channel = Img::RGBA;
166  break;
167  default:
168  error("the file \"%s\" doesn't appear to be compressed using DXT1, DXT3, or DXT5! [%x]",
169  Cache::getFilePath(_tex->url), ddsd.channel.fourCC);
170  File::closeFile(dds->fp);
171  if (dds) delete[] dds;
172  return NULL;
173  }
174 
175  /* calculate pixel data size */
176  curr = ftell(dds->fp);
177  fseek(dds->fp, 0, SEEK_END);
178  end = ftell(dds->fp);
179  fseek(dds->fp, curr, SEEK_SET);
180  size = end - curr;
181 
182  /* alloc the data */
183  Img *img = new Img(dds->width, dds->height, Img::RGB, dds->numMipmaps, size);
184  dds->texels = img->pixmap;
185 
186  /* read pixel data with mipmaps */
187  fread(dds->texels, sizeof (GLubyte), size, dds->fp);
188 
189  File::closeFile(dds->fp);
190  if (dds) delete[] dds;
191 
192  return img;
193 }
