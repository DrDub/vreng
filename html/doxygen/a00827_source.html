67 void Water::parser(char *l)
68 {
69  defaults();
70  l = tokenize(l);
71  begin_while_parse(l) {
72  l = parse()->parseAttributes(l, this);
73  if (!l) break;
74  if (!stringcmp(l, "amplitude")) {
75  l = parse()->parseFloat(l, &amplitude, "amplitude");
76  amplitude *= DEF_AMPLITUDE; // coef
77  }
78  else if (!stringcmp(l, "freq")) {
79  l = parse()->parseFloat(l, &freq, "freq");
80  }
81  else if (!stringcmp(l, "phase")) {
82  l = parse()->parseFloat(l, &phase, "phase");
83  phase *= DEF_PHASE; // coef
84  }
85  else if (!stringcmp(l, "height")) {
86  l = parse()->parseFloat(l, &height, "height");
87  }
88  }
89  end_while_parse(l);
90 }

124 void Water::draw()
125 {
126  float d = 1./MESH;
127  for (int i=0; i < MESH; i++) {
128  glBegin(GL_TRIANGLE_STRIP);
129  for (int j=0; j < MESH; j++) {
130  float u, v, x, y, z;
131  u = j * d;
132  v = i * d;
133  x = -1. + 2. * u;
134  z = -1. + 2. * v;
135  y = amplitude * Sin(freq * M_2PI * v + off);
136  glTexCoord2f(u, v); glVertex3f(x, y, z);
137  u += d;
138  v += d;
139  x = -1. + 2. * u;
140  z = -1. + 2. * v;
141  y = amplitude * Sin(freq * M_2PI * v + off);
142  glTexCoord2f(u, v); glVertex3f(x, y, z);
143  if (play) {
144  off += phase;
145  if (off > MAX_OFF) off = 0.;
146  }
147  }
148  glEnd();
149  }
150 }

180 bool Water::whenIntersect(WObject *pcur, WObject *pold)
181 {
182  switch (pcur->type) {
183 
184  case BALL_TYPE:
185  pcur->pos.z += 3 * Ball::DELTAZ;
186  disableBehavior(COLLIDE_ONCE);
187  break;
188 
189  case USER_TYPE:
190  enableBehavior(COLLIDE_ONCE);
191  sound();
192  break;
193 
194  case CAULDRON_TYPE:
195  if (first) {
196  trace(DBG_WO, "WaterC: bbsize=%.2f,%.2f,%.2f bbcenter=%.2f,%.2f,%.2f",
197  pos.bbsize.v[0], pos.bbsize.v[1], pos.bbsize.v[2],
198  pos.bbcenter.v[0], pos.bbcenter.v[1], pos.bbcenter.v[2]);
199  first = false;
200  }
201  pcur->pos.z += 2 * Ball::DELTAZ;
202  pcur->pos.ay = DEG2RAD((float) floor(5*amplitude * freq * (off-MAX_OFF/2)));
203  break;
204  }
205  //dax pcur->updatePositionAndGrid(pcur->pos);
206  pcur->updatePosition();
207  return true;
208 }
