57 #if defined(WIN32) && !defined(CYGWIN32) // _spawn

61  if (toolid == WBD_TOOL) {
62  pid = _spawnlp(_P_NOWAIT, "wbd", "wbd", "-C", "VREng-wbd", "-t", ttl, "-l", "+r", newchan, NULL);
63  }
64  if (toolid == NTE_TOOL) {

66  }
67 #else

69  case -1:
70  error("%s whiteboard", e_fork);
71  break;
72  case 0:
73  if (toolid == WB_TOOL) {
74  execlp("wb", "wb", "-C", "VREng-wb", "-t", ttl, "-l", "+r", newchan, (char*)NULL);
75  error("%s wb", e_exec);
76  }
77  if (toolid == WBD_TOOL) {
78  execlp("wbd", "wbd", "-C", "VREng-wbd", "-t", ttl, "-l", "+r", newchan, (char*)NULL);
79  error("%s wbd", e_exec);
80  }
