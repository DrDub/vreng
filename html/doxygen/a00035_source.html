65 void HttpThread::begin()
66 {
67 #if defined(HAVE_LIBPTHREAD) && defined(WITH_PTHREAD)
68  if (modethr > 0) {
69  //trace(DBG_HTTP, "-> begin %s", url);
70  if (waitfifo) { // Wait authorization to begin
71  lockMutex(&nbsimcon_mutex);
72  //------------ lock
73  pthread_cond_wait(&waitfifo->cond, &nbsimcon_mutex);
74  nbsimcon++; // increments nbsimcon
75 
76  /* remove element from fifo */
77  fifofirst = waitfifo->next;
78  if (waitfifo) delete[] waitfifo;
79  waitfifo = NULL;
80  //------------ unlock
81  unlockMutex(&nbsimcon_mutex); // free fifo handling
82  }
83  }
84 #endif
85 }

87 void HttpThread::end()
88 {
89 #if defined(HAVE_LIBPTHREAD) && defined(WITH_PTHREAD)
90  if (modethr > 0) {
91  lockMutex(&nbsimcon_mutex); // lock access to global variable nbsimcon
92  //------------ lock
93  nbsimcon--; // decrements nbsimcon
94  if (nbsimcon < 0) nbsimcon = 0;
95  if (fifofirst) { // if something in fifo, awake it
96  trace(DBG_HTTP, "thread awake (%d) %s", nbsimcon, url);
97  pthread_cond_signal(&fifofirst->cond);
98  }
99  //------------ unlock
100  unlockMutex(&nbsimcon_mutex);
101  }
102 #endif

105 int HttpThread::putfifo()
106 {
107 #if defined(HAVE_LIBPTHREAD) && defined(WITH_PTHREAD)
108  lockMutex(&nbsimcon_mutex); // lock access to global variable nbsimcon
109  //------------ lock
110  if (nbsimcon >= ::g.pref.maxsimcon) { // test number of active connections
111  trace(DBG_HTTP, "too many threads=%d, waiting for %s", nbsimcon, url);
112 
113  tWaitFifo *wf = new tWaitFifo[1]; // new fifo
114 
115  pthread_cond_init(&(wf->cond), NULL); // put thread into fifo
116  wf->next = NULL;
117  if (!fifofirst) fifofirst = wf;
118  if (fifolast) fifolast->next = wf;
119  fifolast = wf;
120 
121  //------------ unlock
122  unlockMutex(&nbsimcon_mutex); // unlock the global variable
123  waitfifo = wf; // block the thread
124  }
125  else {
126  /* add a connection */
127  nbsimcon++;
128  trace(DBG_HTTP, "thread going now (%d) %s", nbsimcon, url);
129  //------------ unlock
130  unlockMutex(&nbsimcon_mutex);
131  waitfifo = NULL; // thread not blocked
132  }
133 
134  /* start new thread */
135  Vpthread_t tid;
136  return pthread_create(&tid, NULL, HttpThread::connection, (void *) this);
137 
138 #else
139  HttpThread::connection((void *) this);
140  return 0;
141 #endif

210 void HttpThread::checkProxy()
211 {
212  static bool done = false;
213  if (done) return;
214 
215  char *p;
216 
217  if (::g.pref.httpproxystr) p = ::g.pref.httpproxystr;
218  else p = getenv(HTTP_PROXY); // syntax: http://hostname:port/
219  if (p && *p) {
220  char envproxy[90];
221 
222  hostproxy = new char[strlen(p)];
223  if (p[strlen(p) - 1] == '/') p[strlen(p) - 1] = '\0';
224  strcpy(envproxy, p);
225  p = strrchr(envproxy, ':');
226  *p = '\0';
227  portproxy = atoi(++p);
228  if ((p = strrchr(envproxy, '/')) == NULL) sprintf(hostproxy, "http://%s", envproxy);
229  else strcpy(hostproxy, ++p);
230  proxy = 1;
231  trace(DBG_HTTP, "proxy=%s:%d", hostproxy, portproxy);
232  }
233  p = (::g.pref.noproxystr) ? ::g.pref.noproxystr : getenv(NO_PROXY);
234  if (p && *p) {
235  domnoproxy = new char[strlen(p)];
236  strcpy(domnoproxy, p);
237  noproxy = 1;
238  }
239  done = true;
240 }

248 void * HttpThread::connection(void *_ht)
249 {
250  HttpThread *ht = (HttpThread *) _ht;
251 
252  Http *http = ht->http;
253 
254  HttpThread::checkProxy();
255 
256  struct sockaddr_in sa;
257  bool hterr = false;
258  bool eoheader = false;
259  bool answerline = true; // position first line
260 
261  char host[MAXHOSTNAMELEN], scheme[8], path[URL_LEN], req[BUFSIZ];
262  memset(host, 0, sizeof(host));
263  memset(scheme, 0, sizeof(scheme));
264  memset(path, 0, sizeof(path));
265 
266  int urltype = Url::parser(ht->url, host, scheme, path);
267  trace(DBG_HTTP, "connection: url=%s", ht->url);
268  trace(DBG_HTTP, "url=%s, universe=%s scheme=%s host=%s path=%s type:%d", ::g.url,::g.universe,scheme,host,path,urltype);
269 
270  /* which kind of URL ? */
271  switch (urltype) {
272 
273  case Url::URLFILE: // file://
274  if ((http->fd = HttpThread::openFile(path)) < 0)
275  hterr = true;
276  else {
277  http->off = -1;
278  ht->httpReader(ht->thrhdl, http);
279  hterr = false;
280  }
281  break;
282 
283  case Url::URLHTTP: // http://
284  trace(DBG_HTTP, "HTTP: %s://%s%s", scheme, host, path);
285 again:
286  if (proxy && (!noproxy || strstr(host, domnoproxy) == 0)) { // proxy
287  struct hostent *hp;
288  if ((hp = my_gethostbyname(hostproxy, AF_INET)) == NULL) {
289  trace(DBG_HTTP, "my_gethostbyname hostproxy=%s", hostproxy);
290  proxy = 0;
291  noproxy = 0;
292  goto again;
293  }
294  memset(&sa, 0, sizeof(struct sockaddr_in));
295  sa.sin_family = AF_INET;
296  sa.sin_port = htons(portproxy);
297  memcpy(&sa.sin_addr, hp->h_addr_list[0], hp->h_length);
298  my_free_hostent(hp);
299  }
300  else {
301  int res;
302  if ((res = HttpThread::resolver(host, scheme, &sa)) != 0) {
303  if (! strncmp(host, "localhost", 9)) {
304  hterr = false;
305  }
306  else {
307  error("can't resolve %s err=%d", host, res);
308  hterr = true;
309  }
310  break;
311  }
312  }
313  if ((http->fd = HttpThread::connect(&sa)) < 0) {
314  error("can't connect %s", host);
315  hterr = true;
316  break;
317  }
318 
319  /*
320  * send the GET request to the http server with adding useful infos
321  */
322  if (proxy && (!noproxy || strstr(host, domnoproxy) == 0)) {
323  sprintf(req,
324  "GET %s?version=%s&target=%s-%s%s&user=%s HTTP/1.0\r\nHost: %s\r\n\r\n",
325  ht->url, PACKAGE_VERSION, ::g.env.machname(), ::g.env.sysname(),
326  ::g.env.relname(), ::g.user, host);
327  }
328  else {
329  sprintf(req,
330  "GET %s?version=%s&target=%s-%s%s&user=%s HTTP/1.0\r\nHost: %s\r\n\r\n",
331  path, PACKAGE_VERSION, ::g.env.machname(), ::g.env.sysname(),
332  ::g.env.relname(), ::g.user, host);
333 
334  }
335  if (HttpThread::send(http->fd, req, strlen(req)) < 0) {
336  error("can't send req=%s", req);
337  hterr = true;
338  break;
339  }
340 
341  /*
342  * parse HTTP/1.1 headers received from the server
343  */
344 
345  http->reset();
346 
347  do {
348  if ((http->len = HttpThread::answer(http->fd, http->buf, HTTP_BUFSIZ)) <= 0) {
349  eoheader = true;
350  hterr = true;
351  break;
352  }
353  //trace(DBG_FORCE, "http answer = %d %p", http->len, http->buf);
354 
355  for (http->off = 0; 1; ) {
356  int i = 0;
357  char httpheader[HTTP_BUFSIZ];
358 
359  while ((http->off < http->len) && (http->buf[http->off] != '\n'))
360  httpheader[i++] = http->buf[http->off++];
361 
362  if (http->off < http->len) {
363  if (httpheader[0] == '\r') { // test end of header
364  http->off++;
365  eoheader = true;
366  hterr = true;
367  break;
368  }
369  httpheader[i-1] = '\0'; // replace '\r' by '\0'
370  http->off++; // skip '\0'
371  trace(DBG_HTTP, "->%s", httpheader);
372 
373  if (answerline) {
374  /* first line => get error code */
375  int httperr, major, minor;
376 
377  sscanf(httpheader, "HTTP/%d.%d %d", &major, &minor, &httperr);
378  trace(DBG_HTTP, "HTTP-Code_err %d (%d.%d) - %s %s",
379  httperr, major, minor, httpheader+12, ht->url);
380 
381  switch (httperr) {
382  case HTTP_200:
383  case HTTP_202:
384  answerline = false; // answerline done
385  break;
386  case HTTP_301:
387  case HTTP_302:
388  case HTTP_307:
389  {
390  char *p, *q;
391  if ((p = strstr(httpheader, "Location:")) != 0) {
392  if ((q = strchr(p+17, '/')) !=0) {
393  *q = '\0';
394  strcpy(host, p+17); // redirect host
395  trace(DBG_HTTP, "redirect host = %s", host);
396  goto again;
397  }
398  }
399  }
400  break;
401  case 404: // not found
402  warning("HTTP-err: %d - %s %s on %s", httperr, httpheader, ht->url, host);
403  hterr = true;
404  break;
405  case HTTP_503:
406  warning("HTTP-err: %d - server %s unavailable", httperr, host);
407  hterr = true;
408  break;
409  default:
410  error("HTTP-err: %d - %s %s", httperr, httpheader+12, ht->url);
411  hterr = true;
412  break;
413  }
414  }
415  if (hterr)
416  break;
417 
418  if (! strncmp(httpheader, "Content-Type: ", 14)) {
419  char *p, *q;
420 
421  if ((p = strchr(httpheader, '/')) != NULL) {
422  p++;
423  if ((q = strchr(p, ';')) != NULL) *q = '\0';
424  else p[MIME_LEN] = 0;
425  trace(DBG_HTTP, "mime=%s %s", p, ht->url);
426  /* only for textures */
427  if (ht->thrhdl && strcmp(p, "plain")) {
428  Texture *tex = (Texture *) ht->thrhdl;
429  tex->setMime(p);
430  }
431  }
432  }
433  }
434  else
435  break;
436  }
437  } while (!eoheader);
438 
439  /*
440  * Call here the appropriated httpReader
441  */
442  ht->httpReader(ht->thrhdl, http);
443  hterr = false;
444  break;
445 
446  default: // scheme:// unknown
447  if (urltype) error("unknown scheme urltype=%d", urltype);
448  hterr = true;
449  break;
450  }
451 
452  if (hterr && ht)
453  ht->httpReader(ht->thrhdl, http);
454 
455  // free memory
456  if (ht) delete ht;
457  ht = NULL;
458  if (http) delete http;
459  http = NULL;
460 
461  return NULL;
462 }

527 int Http::httpRead(char *abuf, int maxl)
528 {
529  int l = 0, length;
530 
531  /* Modifie par Fabrice, l= longueur lue, maxl= longueur restante a recevoir */
532  length = (off < 0) ? 0 : len - off;
533 
534  if (length > 0) {
535  if (length > maxl) length = maxl;
536  memcpy(abuf, buf + off, length);
537  l += length;
538  maxl -= length;
539  off += length;
540  }
541 
542  while (maxl > 0) {
543  int r;
544  r = read(fd, abuf+l, maxl);
545  if (r < 0) {
546  error("%s (%d) l=%d maxl=%d off=%d len=%d", strerror(errno), errno, l, maxl, off, len);

550  l += r;
551  maxl -= r;
552  }
