210 void World::initGeneralFuncList()
211 {
212  for (int i=0; i <= OBJECTSNUMBER; i++) {
213 
214  // action functions
215  for (int j=0; j < ACTIONSNUMBER ; j++) {
216  generalActionList[i][j].name[0] = '\0';
217  generalActionList[i][j].method = NULL;
218  }
219 
220  // property functions
221  for (int j=0; j < PROPSNUMBER ; j++) {
222  getPropertyFuncList[i][j].method = NULL;
223  putPropertyFuncList[i][j].method = NULL;
224  }
225 
226  // property values
227  propertiesnumber[i] = 0;
228 
229  // maxlasting values
230  maxlastings[i] = MAXFLOAT;
231 
232  // init functions
233  if (generalInitList[i].initfunc) {
234  trace(DBG_WO, "initfunc: i=%d f=%p", i, generalInitList[i].initfunc);
235  generalInitList[i].initfunc();
236  }
237  }
238 }
