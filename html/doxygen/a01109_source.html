408 char * Body::getTok(char *l, int *tok)
409 {
410  char *t = l;
411  const struct sBodyToken *ptab;
412 
413  *tok = -1;
414  l = skipEqual(l);
415  if (l) *(l-1) = '\0'; // end of token '=', replaced by null terminated
416  for (ptab = btokens; *ptab->name != -1 ; ptab++) {
417  if (!strcmp(ptab->name, t)) {
418  *tok = ptab->num;
419  return l;
420  }
421  }
422  error("getTok: unknown \"%s\" in %s", t, l);
423  return l;
424 }

450 void Body::loadBodyParts(FILE *f)
451 {
452  char *l, jpline[256], tex[128];
453 
454  tex[0] = '\0';
455  // alloc bodyparts and joint-points arrays
456  bodyparts = new sBodyParts[MAX_PARTS];
457  jp.x = new float[MAX_PARTS];
458  jp.y = new float[MAX_PARTS];
459  jp.z = new float[MAX_PARTS];
460 
461  // init bodyparts
462  for (int i=0; i < MAX_PARTS; i++) {
463  bodyparts[i].loaded = false;
464  bodyparts[i].scale = 1;
465  for (int j=0; j<3 ; j++) bodyparts[i].scales[j] = 1;
466  for (int j=0; j<3 ; j++) bodyparts[i].color[j] = cloth[j];
467  bodyparts[i].texid = 0;
468  bodyparts[i].off = NULL;
469  bodyparts[i].obj = NULL;
470  bodyparts[i].url[0] = '\0';
471  bodyparts[i].texurl[0] = '\0';
472  }
473  // set default color
474  for (int j=0; j<3 ; j++) {
475  bodyparts[HEAD].color[j] = skin[j];
476  bodyparts[NECK].color[j] = skin[j];
477  bodyparts[L_FOREARM].color[j] = skin[j];
478  bodyparts[R_FOREARM].color[j] = skin[j];
479  bodyparts[L_HAND].color[j] = skin[j];
480  bodyparts[R_HAND].color[j] = skin[j];
481  }
482 
483  // parse first line of <body ...>
484  if (! fgets(jpline, sizeof(jpline), f)) return;
485  jpline[strlen(jpline) - 1] = '\0';
486  if (! stringcmp(jpline, "<body")) {
487  l = strtok(jpline, SEP); // tokenize jpline
488  l = strtok(NULL, SEP); // next token
489  while (l) {
490  if (! stringcmp(l, "scale=")) l = wobject->parse()->parseFloat(l, &bscale, "scale");
491  else if (! stringcmp(l, "tex=")) l = wobject->parse()->parseString(l, tex, "tex");
492  }
493  }
494 
495  if (*tex) {
496  for (int i=0; i < MAX_PARTS; i++)
497  strcpy(bodyparts[i].texurl, tex);
498  }
499 
500  // get body joint points and urls
501  for (int i=0; i < MAX_PARTS; i++) {
502  if (! fgets(jpline, sizeof(jpline), f)) break;
503  jpline[strlen(jpline) - 1] = '\0';
504  trace(DBG_MAN, "loadBodyParts: %s", jpline);
505 
506  int bpindex = 0;
507  char *p = jpline;
508  //if ((p = strstr(jpline, "</body>"))) continue;
509  if (jpline[0] == '#') continue;
510  else if ((p = strchr(jpline, '<'))) {
511  l = strtok(p, SEP);
512  l = getTok(l, &bpindex);
513  if (bpindex < 0) goto endparse; // eof
514  l = strtok(NULL, SEP);
515 
516  // parse others lines of <body ...> ... </body>
517  begin_while_parse(l) {
518  if (! stringcmp(l, "jp=")) {
519  l = skipEqual(l);
520  l++; // "
521  jp.x[bpindex] = (float)atof(l) / bscale; l = strtok(NULL, SEP);
522  jp.y[bpindex] = (float)atof(l) / bscale; l = strtok(NULL, SEP);
523  jp.z[bpindex] = (float)atof(l) / bscale; l = strtok(NULL, SEP);
524  }
525  else if (*l == '/') break; // eol
526  else if (! stringcmp(l, "url=")) {
527  l = wobject->parse()->parseString(l, bodyparts[i].url, "url");
528  drawparts++;
529  }
530  else if (! stringcmp(l, "scale=")) {
531  l = wobject->parse()->parseFloat(l, &bodyparts[i].scale, "scale");
532  bodyparts[i].scales[0] = bodyparts[i].scale;
533  bodyparts[i].scales[1] = bodyparts[i].scale;
534  bodyparts[i].scales[2] = bodyparts[i].scale;
535  }
536  else if (! stringcmp(l, "scalex="))
537  l = wobject->parse()->parseFloat(l, &bodyparts[i].scales[0],"scalex");
538  else if (! stringcmp(l, "scaley="))
539  l = wobject->parse()->parseFloat(l, &bodyparts[i].scales[1],"scaley");
540  else if (! stringcmp(l, "scalez="))
541  l = wobject->parse()->parseFloat(l, &bodyparts[i].scales[2],"scalez");
542  else if (! stringcmp(l, "color="))
543  l = wobject->parse()->parseVector3f(l, bodyparts[i].color, "color");
544  else if (! stringcmp(l, "tex="))
545  l = wobject->parse()->parseString(l, bodyparts[i].texurl, "tex");
546  } end_while_parse(l);
547  }
548  }
549 
550 endparse:
551  // now we can download all parts
552  for (int i=0; i < MAX_PARTS; i++) {
553  if (bodyparts[i].url[0]) { // if url exist
554  //error("body_url=%s", bodyparts[i].url);
555  bodyparts[i].model_t = Format::getModelByUrl(bodyparts[i].url);
556 
557  switch (bodyparts[i].model_t) {
558  case MODEL_OFF:
559  bodyparts[i].off = new Off(bodyparts[i].url);
560  //dax bodyparts[i].off = new Off(bscale * bodyparts[i].scale);
561  Http::httpOpen(bodyparts[i].url, Off::httpReader, bodyparts[i].off, 0);
562  break;
563  case MODEL_OBJ:
564  bodyparts[i].obj = new Obj(bodyparts[i].url, 1);
565  Http::httpOpen(bodyparts[i].url, Obj::httpReader, bodyparts[i].obj, 0);
566  bodyparts[i].obj->setScale(bscale * bodyparts[i].scale);
567  if (bodyparts[i].texurl[0]) // if url exist
568  bodyparts[i].texid = Texture::getFromCache(bodyparts[i].texurl);
569  break;
570  }
571  bodyparts[i].loaded = true;
572  }
573  }
574  model_t = bodyparts[0].model_t; // keep model used
575  if (f) File::closeFile(f);
576 }
