162 void Env::cleanCacheByExt(const char *ext)
163 {
164  chdir(vrengcache);
165 #if HAVE_READDIR
166  DIR *dirp = opendir(".");
167  if (dirp) {
168  struct dirent *dp;
169  struct stat bufstat;
170 
171  for (dp = readdir(dirp); dp; dp = readdir(dirp)) {
172  if (stat(dp->d_name, &bufstat) == 0) {
173  if (strstr(dp->d_name, ext)) {
174  char *p = strrchr(dp->d_name, '.');
175  if (p && strlen(p) == 4) {
176  unlink(dp->d_name);
177  }
178  }
179  }
180  }
181  closedir(dirp);
182  }
183 #else
184  char cmd[128];
185  sprintf(cmd, "IFS=' '; /bin/rm -f *.%s", ext);
186  system(cmd);
187 #endif
188  chdir(vrengcwd);
189 }

191 void Env::cleanCacheByTime(time_t cachetime)

203  if (stat(dp->d_name, &bufstat) == 0) {
204  if (S_ISREG(bufstat.st_mode)) {
205  if (now - bufstat.st_mtime > cachetime) {
