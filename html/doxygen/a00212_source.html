105 void Joystick::paintCanvasCB(UPaintEvent& e)
106 {
107  UGraph g(e);
108  g.setColor(::g.theme.joystickColor);
109 
110  // draw cercle
111  g.drawEllipse(0, 0, 2*radius, 2*radius);
112 
113  if (!is_drawing) {
114  // draw center point
115  g.fillEllipse(center_point.x-5, center_point.y-5, 10, 10);
116  }
117  else {
118  g.setColor(::g.theme.joystickArmColor);
119 
120  // draw center point
121  g.fillEllipse(center_point.x-5, center_point.y-5, 10, 10);
122 
123  float dx = current_point.x - center_point.x;
124  float dy = current_point.y - center_point.y;
125  float r = sqrt(dx * dx + dy * dy);
126  float a = dx == 0 ? 0 : atan(dy / dx);
127  float sign = dx > 0 ? +1 : -1;
128  if (r > radius) r = radius;
129  g.drawLine(center_point.x, center_point.y,
130  center_point.x + sign * r * cos(a),
131  center_point.y + sign * r * sin(a));
132  }
133 }
