90 
91  /* Read the header */
92  tMd2Header md2_hdr;

94 
95  File::read_buf(f, md2_hdr.ident, 4); o+=4;

103  md2_hdr.num_tris = File::read_long_le(f); o+=4;
104  md2_hdr.num_glcmds = File::read_long_le(f); o+=4;
105  md2_hdr.num_frames = File::read_long_le(f); o+=4;
106  md2_hdr.ofs_skins = File::read_long_le(f); o+=4;
107  md2_hdr.ofs_st = File::read_long_le(f); o+=4;
108  md2_hdr.ofs_tris = File::read_long_le(f); o+=4;
109  md2_hdr.ofs_frames = File::read_long_le(f); o+=4;
110  md2_hdr.ofs_glcmds = File::read_long_le(f); o+=4;
111  md2_hdr.ofs_end = File::read_long_le(f); o+=4;
112 
113  /* check if this is really a .md2 file */
114  if (strncmp(md2_hdr.ident, "IDP2", 4)) return false;
115  /* we do not need all the info from the header, just some of it */
116  numframes = md2_hdr.num_frames;
117  skinwidth = md2_hdr.skinwidth;
118  skinheight = md2_hdr.skinheight;
119 
120  /* check which info is first */
121  if (md2_hdr.ofs_frames > md2_hdr.ofs_glcmds) {
122  File::skip_byte(f, md2_hdr.ofs_glcmds - o); o = md2_hdr.ofs_glcmds;
123  o += getGLCmds(&md2_hdr, f);
124  File::skip_byte(f, md2_hdr.ofs_frames - o); o = md2_hdr.ofs_frames;
125  o += getFrames(&md2_hdr, f);
126  }
127  else {
128  File::skip_byte(f, md2_hdr.ofs_frames - o); o = md2_hdr.ofs_frames;
129  o += getFrames(&md2_hdr, f);
130  File::skip_byte(f, md2_hdr.ofs_glcmds - o); o = md2_hdr.ofs_glcmds;
131  o += getGLCmds(&md2_hdr, f);
132  }
133  loaded = true;
134  return loaded;
135 }

186 int32_t Md2::getFrames(tMd2Header *md2_hdr, FILE *f)
187 {
188  //trace(DBG_FORCE, "getFrames: numframes=%d", numframes);
189 
190  /* converts the FrameInfos to Frames */
191  frames = new tMd2Frame[numframes];
192 
193  // allocate display list
194  dlists = new GLint[numframes];
195 
196  int32_t o=0;
197  for (int fr=0; fr < numframes; fr++) {
198  tMd2Vec3 scale, origin;
199 
200  scale.x = File::read_float_le(f); o +=4;
201  scale.y = File::read_float_le(f); o +=4;
202  scale.z = File::read_float_le(f); o +=4;
203  origin.x = File::read_float_le(f); o +=4;
204  origin.y = File::read_float_le(f); o +=4;
205  origin.z = File::read_float_le(f); o +=4;
206  File::read_buf(f, frames[fr].name, 16); o +=16;
207 
208  frames[fr].vert_table = new tMd2Vertex[md2_hdr->num_xyz];
209 
210  /* loads the vertices */
211  for (int i=0; i < md2_hdr->num_xyz; i++) {
212  Trivertex cur_vert;
213  tMd2Vertex *p = (frames[fr].vert_table) + i;
214 
215  cur_vert.x = File::read_char(f); o++;
216  cur_vert.y = File::read_char(f); o++;
217  cur_vert.z = File::read_char(f); o++;
218  cur_vert.normal = File::read_char(f); o++;
219  p->x = ((cur_vert.x * scale.x) + origin.x);
220  p->y = ((cur_vert.y * scale.y) + origin.y);
221  p->z = ((cur_vert.z * scale.z) + origin.z);
222  }
223  }
224  return o;
225 }

273 int32_t Md2::getGLCmds(tMd2Header *md2_hdr, FILE *f)
274 {
275  int32_t num_vertices = ((md2_hdr->num_tris + 2 * md2_hdr->num_glcmds - 2) / 7);
276  //trace(DBG_FORCE, "getGLCmds: num_vertices=%d", num_vertices);
277 
278  texinfo = new tMd2TexInfo[num_vertices];
279  glcmds = new int32_t[md2_hdr->num_glcmds - 2 * num_vertices];
280  trace(DBG_VGL, "glcmds size = %d", sizeof(int32_t) * (md2_hdr->num_glcmds - 2 * num_vertices));
281 
282  /* now transform the GL commands */
283  int32_t *glcmds_copy = glcmds;
284  tMd2TexInfo *_texinfo = texinfo;
285 
286  int32_t glcmd, o;
287 
288  for (o = 0, glcmd = 0; (glcmd = File::read_long_le(f)) != 0; ) {
289  int32_t nb_verts;
290 
291  o += 4;
292 
293  /* determine the command to draw the triangles */
294  if (glcmd > 0) // triangle strip
295  nb_verts = glcmd;
296  else // triangle fan
297  nb_verts = -glcmd;
298  *(glcmds_copy++) = glcmd; // copy the command
299 
300  /* gets the texture information */
301  for (int i=0; i < nb_verts; i++) {
302  _texinfo->s = File::read_float_le(f); o+=4;
303  _texinfo->t = File::read_float_le(f); o+=4;
304  _texinfo++;
305 
306  /* we keep the vertex index */
307  glcmd = File::read_long_le(f); o+=4;
308  *(glcmds_copy++) = glcmd; // copy the vertex index
309  }
310  }
311  /* do not forget to copy the zero :-) */
312  *(glcmds_copy++) = glcmd;
313  return o+4;
314 }

386 void Md2::draw(int32_t frame, int32_t nextframe, float ratio, float scale)
387 {
388  /* keeps the frame value valid */
389  if (numframes == 0) {
390  error("drawMd2: numframes null");
391  return;
392  }
393 
394  frame %= numframes;
395  nextframe %= numframes;
396  //trace(DBG_FORCE, "draw: frame=%d", frame);
397 
398  /* get the frames information */
399  tMd2Frame *frame1 = frames + frame;
400  tMd2Frame *frame2 = frames + nextframe;
401 
402  /* do the gl commands */
403  int32_t *glcmd = glcmds;
404  int32_t nb_vert = 0;
405 
406  while (*glcmd != 0) {
407  int32_t num_verts;
408 
409  /* determine the command to draw the triangles */
410  if (*glcmd > 0) { // triangle strip
411  num_verts = *glcmd;
412  glBegin(GL_TRIANGLE_STRIP);
413  }
414  else { // triangle fan
415  num_verts = -(*glcmd);
416  glBegin(GL_TRIANGLE_FAN);
417  }
418  glcmd++;
419 
420  for (int i=0; i < num_verts; i++) {
421  tMd2Vec3 p; // Interpolated point
422  int32_t vert_index;
423 
424  /* Grab the vertex index */
425  vert_index = *glcmd;
426  glcmd++;
427 
428  /* Interpolate */
429  p.x = frame1->vert_table[vert_index].x +
430  (frame2->vert_table[vert_index].x -
431  frame1->vert_table[vert_index].x) * ratio;
432  p.y = frame1->vert_table[vert_index].y +
433  (frame2->vert_table[vert_index].y -
434  frame1->vert_table[vert_index].y) * ratio;
435  p.z = frame1->vert_table[vert_index].z +
436  (frame2->vert_table[vert_index].z -
437  frame1->vert_table[vert_index].z) * ratio;
438 
439  glTexCoord2f(texinfo[nb_vert].s, texinfo[nb_vert].t);
440 
441  /* apply scale */
442  Draw::vertex3f(p.x * scale, p.y * scale, p.z * scale);
443  updBbox(p.x * scale, p.y * scale, p.z * scale);
444  nb_vert++;
445  }
446  glEnd();
447  }
448 }
