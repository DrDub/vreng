167 void md5_loop(md5_ctxt *ctxt, uint8_t *input, uint32_t len)
168 {
169  uint32_t gap, i;
170 
171  ctxt->md5_n += len * 8; /* byte to bit */
172  gap = MD5_BUFLEN - ctxt->md5_i;
173 
174  if (len >= gap) {
175  memcpy((void *)(ctxt->md5_buf + ctxt->md5_i), (void *)input,
176  gap);
177  md5_calc(ctxt->md5_buf, ctxt);
178 
179  for (i = gap; i + MD5_BUFLEN <= len; i += MD5_BUFLEN) {
180  md5_calc((uint8_t *)(input + i), ctxt);
181  }
182 
183  ctxt->md5_i = len - i;
184  memcpy((void *)ctxt->md5_buf, (void *)(input + i), ctxt->md5_i);
185  } else {
186  memcpy((void *)(ctxt->md5_buf + ctxt->md5_i), (void *)input,
187  len);
188  ctxt->md5_i += len;
189  }
190 }

192 void md5_pad(md5_ctxt * ctxt)
193 {
194  uint32_t gap;
195 
196  /* Don't count up padding. Keep md5_n. */
197  gap = MD5_BUFLEN - ctxt->md5_i;
198  if (gap > 8) {
199  memcpy((void *)(ctxt->md5_buf + ctxt->md5_i),
200  (void *)md5_paddat,
201  gap - sizeof(ctxt->md5_n));
202  } else {
203  /* including gap == 8 */
204  memcpy((void *)(ctxt->md5_buf + ctxt->md5_i),(void *)md5_paddat,
205  gap);
206  md5_calc(ctxt->md5_buf, ctxt);
207  memcpy((void *)ctxt->md5_buf, (void *)(md5_paddat + gap),
208  MD5_BUFLEN - sizeof(ctxt->md5_n));
209  }
210 
211  /* 8 byte word */
212 #if MD5_LITTLE_ENDIAN
213  memcpy(&ctxt->md5_buf[56], &ctxt->md5_n8[0], 8);
214 #endif
215 #if MD5_BIG_ENDIAN
216  ctxt->md5_buf[56] = ctxt->md5_n8[7];
217  ctxt->md5_buf[57] = ctxt->md5_n8[6];
218  ctxt->md5_buf[58] = ctxt->md5_n8[5];
