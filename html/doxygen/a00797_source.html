105 void Vnc::buildScreen()
106 {
107  if (_solids.empty()) return; // no solid
108 
109  V3 dim;
110  getDim(dim);
111 
112  float left = MAX(dim.v[1], dim.v[0]);
113  float front = MIN(dim.v[0], dim.v[1]) + 0.02;
114  float top = dim.v[2];
115  float bot = -top;
116  //float depth = 0.9 * front;
117  float height;
118  float width;
119  float rsol = left / top;
120  float rtex = (u * tex_width) / (v * tex_height);
121 
122  if (rtex > rsol) {
123  height = bot + (2*0.95) * left/rtex;
124  width = 0.95 * left;
125  }
126  else {
127  height = (2*0.95-1) * top;
128  width = 0.95 * top * rtex;
129  }
130 
131  vertices[ 0] = front; vertices[ 1] = width; vertices[ 2] = bot;
132  vertices[ 3] = front; vertices[ 4] = width; vertices[ 5] = height;
133  vertices[ 6] = front; vertices[ 7] = -width; vertices[ 8] = height;
134  vertices[ 9] = front; vertices[10] = -width; vertices[11] = bot;
135 }

153 {

155  glBindTexture(GL_TEXTURE_2D, texture); // we use ours

174  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);

254 void Vnc::getVncCoords(int16_t &x, int16_t &y)
255 {
256  y = viewport[3] - y;
257 
258  // get intersection with z=0 plan
259  GLdouble ox, oy, oz;
260  gluUnProject((GLdouble) x, (GLdouble) y, (GLdouble) 0,
261  modelmat, projmat, viewport,
262  &ox, &oy, &oz);
263  // get intersection with z=1 plan
264  GLdouble ax, ay, az;
265  gluUnProject((GLdouble) x, (GLdouble) y, (GLdouble) 1,
266  modelmat, projmat, viewport,
267  &ax, &ay, &az);
268 
269  // calculate intersection coordinates in vnc screen plane
270  GLdouble dx, dy, dz;
271  dx = ax - ox;
272  if (dx >= 0) {
273  x = -1;
274  y = -1;
275  return;
276  }
277  dy = ay - oy;
278  dz = az - oz;
279 
280  GLdouble t = (vertices[0] - ox) / dx;
281  GLdouble ey = t*dy + oy;
282  GLdouble ez = t*dz + oz;
283 
284  ey = (ey-vertices[10]) / (vertices[4]-vertices[10]);
285  ez = (vertices[5]-ez) / (vertices[5]-vertices[11]);
286 
287  // build final coordinates
288  if ((ey > 0) && (ey < 1) && (ez > 0) && (ez < 1)) {
289  x = (int) (ey * vncClient->realScreenWidth);
290  y = (int) (ez * vncClient->realScreenHeight);
291  }
292  else {
293  x = -1;
294  y = -1;
295  }
296 }

298 /* Connect to a Vnc server */
299 void Vnc::connectServer()
300 {
301  if (! serverdefined) return;
302 
303  vncClient = new VNCClientTextured(servername, port, passwdfile);
304 
305  // client initialization
306  if (! vncClient->VNCInit()) {
307  serverdefined = false;
308  warning("VNC connect failed to %s:%d", servername, port);
309  delete vncClient;
310  vncClient = NULL;
311  return;
312  }
313  else {
314  connected = true;
315  notice("VNC connect successful on %s:%d", servername, port);
316 
317  vncClient->sendFramebufferUpdateRequest(0, 0,
318  vncClient->realScreenWidth,
319  vncClient->realScreenHeight,
320  false);
321  enablePermanentMovement();
322  }
323 
324  // texture initialization from the framebuffer
325  tex_pixmap = (GLubyte *) vncClient->framebuffer;
326  tex_width = vncClient->fbWidth;
327  tex_height = vncClient->fbHeight;
328 
329  setTexture(0); // without mipmaps
330 
331  u = (float) vncClient->realScreenWidth / tex_width;
332  v = (float) vncClient->realScreenHeight / tex_height;
333  trace(DBG_VNC, "real_width=%d tex_width=%d u=%.2f", vncClient->realScreenWidth, tex_width, u);
334  trace(DBG_VNC, "real_height=%d tex_height=%d v=%.2f", vncClient->realScreenHeight, tex_height, v);
335 }
