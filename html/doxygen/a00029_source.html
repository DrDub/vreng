179 /* fills ext */
180 bool Format::getExt(char *str, char *ext)
181 {
182  char *p, *pdot = NULL;
183 
184  if (str[strlen(str) - 1] == '/') {
185  strcpy(ext, EXT_HTML);
186  }
187  else if ((p = strrchr(str, '.')) == 0)
188  return false;
189  else {
190  p++;
191  if (! stringcmp(p, EXT_GZ)) {
192  pdot = --p;
193  *pdot = 0;
194  if ((p = strrchr(str, '.')) == 0)
195  return false;
196  else {
197  p++;
198  strncpy(ext, p, sizeof(ext) - 1);
199  *pdot = '.';
200  }
201  }
202  else {
203  strncpy(ext, p, sizeof(ext) - 1);
204  }
205  }
206  return true;
207 }
