57 void Spiral::build()
58 {
59  float sx = pos.bbsize.v[0]; // step width
60  float sy = pos.bbsize.v[1]; // step depth
61  float sz = pos.bbsize.v[2]; // step height
62 
63  nsteps = (int) ceil(height / sz);
64 
65  for (int n=0; n <= nsteps; n++) {
66  Pos newpos;
67  newpos.az = pos.az;
68  newpos.ax = newpos.ay = 0;
69  float deltaspiral = atan(sy / sx);
70  newpos.x = pos.x + (sx * (cos(deltaspiral * n) - 1));
71  newpos.y = pos.y + (sy * (sin(deltaspiral * n) - 1));
72  newpos.z = pos.z + (sz * n);
73  newpos.az = pos.az + (deltaspiral * n);
74 
75  nextstep = new Step(newpos, pos, geometry, false, height, 0, dir);
76  }
77 }
