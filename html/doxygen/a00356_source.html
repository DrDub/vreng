71 void Cart::addToCart(WObject *po)
72 {
73  // find object's world list and remove object from it
74  switch (po->mode) {
75  case MOBILE:
76  po->delFromMobile();
77  // render invisible the object
78  po->setVisible(false);
79 #if HAVE_MYSQL
80  psql = VRSql::getVRSql(); // first take the VRSql handle;
81  if (psql) {
82  psql->insertCol(CART_NAME, "owner", po->getInstance(), "");
83  psql->updateString(po, CART_NAME, "owner", po->getInstance(), "", ::g.pref.user);
84  trace(DBG_SQL, "cartRow: (%s,%s)", po->getInstance(), po->ownerName());
85  }
86 #endif
87  default:
88  break;
89  }
90 
91  // remove object from collision checking grid
92  po->deleteFromGrid();
93 
94  // informs the GUI
95  po->guip = ::g.gui.addCart(po);
96  if (! number) ::g.gui.showCartDialog(true); // popup cartDialog
97 
98  // net deletion declaration
99  if (! po->isPermanent() && po->noh) delete po->noh;
100  po->noh = NULL;
101 
102  cartList.push_back(po);
103  number++;
104 }

114 void Cart::leaveIntoWorld(WObject *po)
115 {
116  // remove object from the cartList
117  for (list<WObject*>::iterator it = cartList.begin(); it != cartList.end(); ++it)
118  if (*it == po) cartList.remove(*it);
119 
120  // set the object's new coordinates & state
121  float near = 0.5;
122  po->pos.x = localuser->pos.x + near * Cos(localuser->pos.az);
123  po->pos.y = localuser->pos.y + near * Sin(localuser->pos.az);
124  po->pos.z = localuser->pos.z + 0.5; // visible by eyes
125  po->move.ttl = 0;
126 
127  // restore object into mobileList
128  po->addToMobile();
129 
130  // render visible the object coming back into the world
131  po->setVisible(true);
132  po->enableBehavior(PERSISTENT);
133 
134  // show the object
135  po->updatePosition();
136 
137  // update the object's name with the type name as prefix
138  char tmpname[64];
139  sprintf(tmpname, "%s-%s", po->typeName(), po->named());
140  strcpy(po->names.named, tmpname);
141  po->updateNames();
142 
143  // owner is user
144  po->setOwner();
145 
146  // declare that the object has moved for MySql update
147  po->pos.moved = true;
148 
149 #if HAVE_MYSQL
150  psql = VRSql::getVRSql(); // first take the VRSql handle;
151  if (psql) {
152  po->psql = psql; // copy it into the object
153  psql->deleteRow(po, CART_NAME, po->getInstance(), "");
154  psql->insertRow(po);
155  psql->updatePos(po);
156  psql->updateOwner(po);
157  psql->updateGeom(po, po->geometry);
158  trace(DBG_SQL, "leaveFromCart: %s", po->getInstance());
159  }
160 #endif
161 
162  // declare the object creation to the network
163  if (! po->isPermanent() && po->noh)
164  po->noh->declareObjCreation();
165 
166  if (number) number--;
167  if (! number) ::g.gui.showCartDialog(false); // switch-off cartDialog
168  po->resetFlashy();
169 }
