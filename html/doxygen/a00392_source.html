62 /* Returns a constant describing intersection status of both Bounding Boxes */
63 int WObject::interAABB(V3 center1, V3 size1, V3 center2, V3 size2)
64 {
65  float dcx = ABSF(center2.v[0] - center1.v[0]);
66  float dcy = ABSF(center2.v[1] - center1.v[1]);
67  float dcz = ABSF(center2.v[2] - center1.v[2]);
68  float sx1 = ABSF(size1.v[0]);
69  float sx2 = ABSF(size2.v[0]);
70  float sy1 = ABSF(size1.v[1]);
71  float sy2 = ABSF(size2.v[1]);
72  float sz1 = ABSF(size1.v[2]);
73  float sz2 = ABSF(size2.v[2]);
74 
75  if ((dcz >= sz1+sz2 - 1e-3) ||
76  (dcx >= sx1+sx2 - 1e-3) ||
77  (dcy >= sy1+sy2 - 1e-3))
78  return NO_INTER; // doesn't intersect == 0
79  if (dcz < sz1+sz2 + 1e-3) return INTERSECT; // intersects == 3
80  if ((dcx+sx1 <= sx2) && (dcy+sy1 <= sy2) && (dcz+sz1 <= sz2))
81  return IN2; // obj1 inside obj2 == 1
82  if ((dcx+sx2 <= sx1) && (dcy+sy2 <= sy1) && (dcz+sz2 <= sz1))
83  return IN1; // obj2 inside obj1 == 2
84  if ((dcy < sy1+sy2) || (dcx < sx1+sx2)) return INTERSECT; // == 3
85  return NO_INTER; // doesn't intersect == 0
86 }

115 /* Checks intersection with Walls */
116 void WObject::ingoingWalls(WObject *pold)
117 {
118  V3 normal;
119  if (Walls::whenIntersect(pos.bbcenter, pos.bbsize, normal)) {
120  float nx = normal.v[0];
121  float ny = normal.v[1];
122  float cx = pos.x;
123  float cy = pos.y;
124  float ox = pold->pos.x;
125  float oy = pold->pos.y;
126 
127  if ((ox != cx) || (oy != cy)) {
128  float nn = nx*nx + ny*ny;
129  if (nn == 0) return;
130  //trace(DBG_WO, "(ox=%.2f oy=%.2f) (cx=%.2f cy=%.2f)", ox, oy, cx, cy);
131  pos.y = ((nx*ny) * (ox-cx) + ny*ny*oy + nx*nx*cy) / nn;
132  pos.x = ((nx*ny) * (oy-cy) + nx*nx*ox + ny*ny*cx) / nn;
133 
134  // user position has changed => need to update BBs
135  updatePositionAndGrid(pold);
136  }
137  pold->copyPositionAndBB(this);
138  }
139 }

141 /*
142  * Checks for ingoing intersection between current object and neighbor objects.
143  * There is collision if current object intersects a neighbor object
144  * and if its old position didn't intersect.
145  */
146 bool WObject::ingoingNeighbor(WObject *pold, WObject *neighbor)
147 {
148  if ((interAABB(pos.bbcenter, pos.bbsize,
149  neighbor->pos.bbcenter, neighbor->pos.bbsize) != NO_INTER) &&
150  (interAABB(pold->pos.bbcenter, pold->pos.bbsize,
151  neighbor->pos.bbcenter, neighbor->pos.bbsize) == NO_INTER)) {
152 
153  // check wether the neighbor is oblique
154  int az = (int) RAD2DEG(neighbor->pos.az) % 90;
155  int ax = (int) RAD2DEG(neighbor->pos.ax) % 90;
156  if (((az >= 2) && (az <= 88)) ||
157  ((ax >= 2) && (ax <= 88))) {
158  //trace(DBG_FORCE, "%s is oblique az=%d ax=%d", neighbor->getInstance(), az, ax);
159  // TODO: raffinate the collision
160  return true;
161  }
162  return true;
163  }
164  return false;
165 }

178 /*
179  * General function to handle intersections
180  *
181  * Warning: WObject *pold is an incomplete copy of *this
182  */
183 void WObject::generalIntersect(WObject *pold, OList *vicinity)
184 {
185  /* check walls first (maybe expensive) */
186  ingoingWalls(pold);
187 
188  /*
189  * check neighbors
190  */
191  int scans = 0;
192  int rescans = 0;
193  // held the first object
194  WObject *wofirst = (vicinity && vicinity->pobject) ? vicinity->pobject : NULL;
195 
196  // Scans neighbors for collision discovery
197  for (OList *vl = vicinity; vl ; scans++) {
198  if (! vl->pobject) { vl = vl->next; continue; } // discard non existant object
199 
200  WObject *neighbor = vl->pobject;
201 
202  // Hack-3! Assertion on valid neighbor
203  if (! neighbor || ! isValid()) {
204  vl = vl->next; continue;
205  }
206 
207  // Hack-1! Skip scanning if neighbor has already been seen
208  if ((neighbor == wofirst) && (scans > 1)) {
209  error("first=%s scans=%d", wofirst->getInstance(), scans);
210  vl = vl->next; continue;
211  }
212  if (neighbor->behavior & COLLIDE_NEVER) { vl = vl->next; continue; }
213 
214  if (ingoingNeighbor(pold, neighbor)) {
215  // current object intersects and its old instance didn't intersect
216  switch (neighbor->type) {
217  case AOI_TYPE:
218  if (this == (WObject *) localuser) {
219  if (currentAoi != neighbor) {
220  Aoi *aoi = (Aoi *) neighbor;
221  aoi->aoiEnter(); // avatars: change mcast address
222  }
223  }
224  else // other mobile objects: problem of property transfert
225  ;
226  break; // avoids a warning
227 
228  default:
229  if (! neighbor->whenIntersect(this, pold)) { // call the object itself
230  vl = vl->next; continue;
231  }
232 
233  // assertion
234  if (! isValid()) {
235  vl = vl->next; continue;
236  }
237 
238  switch (neighbor->collideBehavior()) {
239  case COLLIDE_ONCE: case COLLIDE_GHOST:
240  vl = vl->next; continue;
241  default:
242  if (isBehavior(COLLIDE_GHOST) || isBehavior(COLLIDE_ONCE)) {
243  vl = vl->next; continue;
244  }
245  else {
246  if (rescans++ > 999) {
247  error("collide loop between %s and %s %d", getInstance(), neighbor->getInstance(), rescans);
248  scans = rescans = 0;
249  vl = vl->next; continue;
250  }
251  scans = 0;
252  vl = vicinity; // not next COLLIDE_EVER ?
253  }
254  }
255  }
256  }
257  else if (outgoingNeighbor(pold, neighbor)) // current object leaves intersection
258  neighbor->whenIntersectOut(this, pold);
259  if (vl) vl = vl->next;
260  } //end neighbors
261 }

265 void WObject::computeNormal(WObject *mobil, V3 *normal)
266 {
267  float fxmin = pos.bbcenter.v[0] - pos.bbsize.v[0];
268  float fxmax = pos.bbcenter.v[0] + pos.bbsize.v[0];
269  float fymin = pos.bbcenter.v[1] - pos.bbsize.v[1];
270  float fymax = pos.bbcenter.v[1] + pos.bbsize.v[1];
271 
272  normal->v[2] = 1;
273  if (fxmax - fxmin < fymax - fymin) {
274  normal->v[0] = 1;
275  normal->v[1] = 0;
276  } else {
277  normal->v[0] = 0;
278  normal->v[1] = 1;
279  }
280  /* test which normal we take, computes dotprod */
281  float dp = (mobil->pos.bbcenter.v[0] - pos.bbcenter.v[0]) * normal->v[0]
282  + (mobil->pos.bbcenter.v[1] - pos.bbcenter.v[1]) * normal->v[1]
283  + (mobil->pos.bbcenter.v[2] - pos.bbcenter.v[2]) * normal->v[2];
284  if (dp < 0) {
285  normal->v[0] = -normal->v[0];
286  normal->v[1] = -normal->v[1];
287  normal->v[2] = -normal->v[2];
288  }
289 }

304  + (mobil.bbcenter.v[1] - still.bbcenter.v[1]) * normal->v[1]
305  + (mobil.bbcenter.v[2] - still.bbcenter.v[2]) * normal->v[2];

313 /* Projete le mouvement de l'objet mobile parallelement a l'objet fixe */
314 int WObject::projectPositionOnObstacle(Pos &mobil, Pos &mobilold, Pos &obstacle)
315 {
316  V3 normal;
317  float dx = mobil.x - mobilold.x;
318  float dy = mobil.y - mobilold.y;
319  float dz = 0; //pdcol mobil.z - mobilold.z;
320 
321  if ((ABSF(dx) + ABSF(dy) + ABSF(dz)) == 0) return 0; // nothing to do
322 
323  computeNormal(mobil, obstacle, &normal);
324  float nx = normal.v[0];
325  float ny = normal.v[1];
326  float nz = normal.v[2];
327  float dp = dx*nx + dy*ny + dz*nz; // dotprod
328 
329  if (dx) mobil.x = mobil.x - dp*nx;
330  if (dy) mobil.y = mobil.y - dp*ny;
331  if (dz) mobil.z = mobil.z - dp*nz;
332 
333  if (dp > 0) return 1;
334  if (dp < 0) return -1;
335  return 0;
336 }

350 void WObject::bounceTrajectory(WObject *pold, V3 *normal)
351 {
352  pold->copyPositionAndBB(this);
353 
354  float nx = normal->v[0];
355  float ny = normal->v[1];
356  float nz = normal->v[2];
357  float vx = move.lspeed.v[0];
358  float vy = move.lspeed.v[1];
359  float vz = move.lspeed.v[2];
360  float nn = sqrt(vx*vx + vy*vy + vz*vz);
361  if (nn == 0) return;
362 
363  float vxn = vx/nn;
364  float vyn = vy/nn;
365  float vzn = vz/nn;
366  float dp = vxn*nx + vyn*ny + vzn*nz;
367  if (dp >= 0) return;
368 
369  move.lspeed.v[0] = (vxn - 2*dp*nx) * nn;
370  move.lspeed.v[1] = (vyn - 2*dp*ny) * nn;
371  move.lspeed.v[2] = (vzn - 2*dp*nz) * nn;
372 }

402 {
403  igrid[0] = (int) ((bb[0] + dist.v[0] * (dim[0]/2)) / dist.v[0]);
