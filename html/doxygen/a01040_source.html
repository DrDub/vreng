100 int interpol(int v00, int v01, int v10, int v11, int xf, int yf)
101 {
102  int v0 = v00 + (((v01-v00) * xf) >> NORM8_BITS);

113 Img * Img::resize(uint16_t width_new, uint16_t height_new)
114 {
115  if (channel != RGB &&
116  channel != BW &&
117  channel != RGBA) {
118  error("resize invalid channel: f=%d w=%d h=%d", channel, width, height);
119  return NULL;
120  }
121  if (pixmap == NULL) {
122  warning("resize pixmap null");
123  return NULL;
124  }
125 
126  Img *img_new = new Img(width_new, height_new, channel);
127  GLubyte *pix_new = (GLubyte *) img_new->pixmap;
128 
129  uint16_t x1incr = ((width - 1) * NORM_8) / (width_new - 1);
130  uint16_t y1incr = ((height - 1) * NORM_8) / (height_new - 1);
131 
132  int x, y, xnew, ynew;
133 
134  for (ynew=0, y=0; y < height_new; y++) {
135  int xi, yi, xf, yf;
136  for (xnew=0, x=0; x < width_new; x++) {
137  xi = xnew >> (NORM16_BITS - NORM8_BITS);
138  yi = ynew >> (NORM16_BITS - NORM8_BITS);
139  xf = xnew & ((1 << (NORM16_BITS - NORM8_BITS))-1);
140  yf = ynew & ((1 << (NORM16_BITS - NORM8_BITS))-1);
141 
142  if (xi < (width-1) && yi < (height-1)) {
143  for (int k=0; k<channel; k++) {
144 #if 1 //DAX segfault
145  pix_new[k] = interpol(pixmap[(yi * width+xi) * channel+k],

149  xf, yf);

152 #endif
