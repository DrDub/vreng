85 int Rtp::updateSeq(sourceInfos *s, uint16_t seq)
86 {
87  uint16_t udelta = seq - s->max_seq;
88  const int MAX_DROPOUT = 3000;
89  const int MAX_MISORDER = 100;
90  const int MIN_SEQUENTIAL = 2;
91 
92  /*
93  * Source is not valid until MIN_SEQUENTIAL packets with
94  * sequential sequence numbers have been received.
95  */
96  if (s->probation) {
97  /* packet is in sequence */
98  if (seq == s->max_seq + 1) {
99  s->probation--;
100  s->max_seq = seq;
101  if (s->probation == 0) {
102  initSource(s, seq);
103  s->received++;
104  return 1;
105  }
106  } else {
107  s->probation = MIN_SEQUENTIAL - 1;
108  s->max_seq = seq;
109  }
110  return 0;
111  } else if (udelta < MAX_DROPOUT) {
112  /* in order, with permissible gap */
113  if (seq < s->max_seq) {
114  /*
115  * Sequence number wrapped - count another 64K cycle.
116  */
117  s->cycles += RTP_SEQ_MOD;
118  }
119  s->max_seq = seq;
120  } else if (udelta <= RTP_SEQ_MOD - MAX_MISORDER) {
121  /* the sequence number made a very large jump */
122  if (seq == s->bad_seq) {
123  /*
124  * Two sequential packets -- assume that the other side
125  * restarted without telling us so just re-sync
126  * (i.e., pretend this was the first packet).
127  */
128  initSource(s, seq);
129  }
130  else {
131  s->bad_seq = (seq + 1) & (RTP_SEQ_MOD-1);
132  return 0;
133  }
134  } else {
135  /* duplicate or reordered packet */
136  }
137  s->received++;
138  return 1;
139 }

307 int Rtp::recvRTCPPacket(struct sockaddr_in *from, uint8_t *pkt, int pkt_len)
308 {
309  rtcp_common_t rtcp_hdr;
310  rtcp_t *end;
311 #ifdef CHECK_RTCP_VALIDITY
312  rtcp_t *r;
313 #endif
314  uint16_t length = 0;
315  uint32_t ssrc = 0;
316  Channel *pchan = Channel::current(); // maybe indeterminated
317 
318  /* we are supposed to receive compounds RTCP packets */
319  for (int len = 0; len < pkt_len ; ) {
320  memcpy(&rtcp_hdr, pkt+len, sizeof(rtcp_common_t));
321  length = ntohs(rtcp_hdr.length);
322  //PD end = (rtcp_t *) ((uint32_t *) &rtcp_hdr + (length<<2) + sizeof(rtcp_common_t));
323  len += sizeof(rtcp_common_t);
324 
325 #ifdef CHECK_RTCP_VALIDITY
326  /* Check RTCP validity */
327  r = (rtcp_t *) &rtcp_hdr;
328  do
329  r = (rtcp_t *) ((uint32_t *) r + ntohs(r->common.length + sizeof(rtcp_common_t)));
330  while (r < end && r->common.version == RTP_VERSION);
331  if (r != end) {
332  error("RTCP wrong size: r=%x end=%x length=%d pkt_len=%d sizeof(rtcp_common_t)=%d", r, end, length<<2, pkt_len, sizeof(rtcp_common_t));
333  len += (length << 2);
334  continue;
335  }
336 #endif
337 
338  switch (rtcp_hdr.pt) {
339 
340  /* receives a SR */
341  case RTCP_SR: {
342  rtcp_sr_t sr;
343  Source *pso;
344 
345  trace(DBG_RTP, "Got SR: length=%d len=%d", length<<2, len);
346  memcpy(&sr, pkt+len, sizeof(rtcp_sr_t));
347  if ((pso = Source::getSource(ntohl(sr.ssrc))) != NULL)
348  pso->sr = sr;
349  len += (length << 2);
350  }
351  break;
352 
353  /* receives a SDES */
354  case RTCP_SDES: {
355  SdesItem *sitem = NULL;
356  Source *pso;
357  uint8_t *p = pkt + len;
358 
359  end = (rtcp_t *) ((uint32_t *) p + (length<<2) + sizeof(rtcp_common_t));
360  memcpy(&ssrc, p, sizeof(ssrc));
361  p += sizeof(ssrc);
362  trace(DBG_RTP, "Got SDES: ssrc=%x pkt_len=%d length=%d len=%d p=%x",
363  ntohl(ssrc), pkt_len, length<<2, len, p);
364 
365  if ((pso = Source::getSource(ntohl(ssrc))) != NULL) { // source found
366 
367  sitem = &pso->sdes; // first sitem which ever exists
368 
369  /* parses SDES chunks */
370  while (p < (uint8_t *) end) {
371  trace(DBG_RTP, "SDES: sitem=%p p=%p end=%p", sitem, p, end);
372  sitem->si_type = *p++;
373  sitem->si_len = *p++;
374  if (sitem->si_str == NULL) { // asumption
375  /* alloc string space, *FIX: memory leak no delete[] ! */
376  sitem->si_str = new uint8_t[sitem->si_len + 1];
377  }
378  memcpy(sitem->si_str, p, sitem->si_len);
379  sitem->si_str[sitem->si_len] = 0; // now string null terminated
380  trace(DBG_RTP, "SDES: %s [%d]", sitem->si_str, sitem->si_len);
381  p += sitem->si_len; // next item
382  if (*p == RTCP_SDES_END) {
383  trace(DBG_RTP, "end SDES: p=%p end=%p", p, end);
384  break; // end of SDES packet
385  }
386 
387  if (sitem->si_next == NULL) {
388  /* alloc next item */
389  trace(DBG_RTP, "alloc new item");
390  sitem->si_next = new SdesItem[1];
391  sitem->si_next->si_type = 0;
392  sitem->si_next->si_len = 0;
393  sitem->si_next->si_str = NULL;
394  sitem->si_next->si_next = NULL;
395  }
396  sitem = sitem->si_next;
397  trace(DBG_RTP, "SDES: sitem=%p", sitem);
398  }
399  }
400  len += (length << 2);
401  }
402  break;
403 
404  /* receives a BYE */
405  case RTCP_BYE: {
406  memcpy(&ssrc, pkt+len, sizeof(ssrc));
407  trace(DBG_RTP, "Got BYE: ssrc=%x", ntohl(ssrc));
408  if (pchan)
409  pchan->session->deleteSourceBySsrc(ntohl(ssrc));
410  len += (length << 2);
411  }
412  break;
413 
414  /* receives a APP */
415  case RTCP_APP:
416  trace(DBG_RTP, "Got APP");
417 
418  /* receives something unknown */
419  default:
420  len += (length << 2);
421  error("Got RTCP unknown type: pt=%u, pkt_len=%d len=%d",
422  rtcp_hdr.pt, pkt_len, len);
423  trace(DBG_FORCE, "%02x%02x%02x%02x %02x%02x%02x%02x %02x%02x%02x%02x",
424  pkt[0], pkt[1], pkt[2], pkt[3],
425  pkt[4], pkt[5], pkt[6], pkt[7],
426  pkt[8], pkt[9], pkt[10], pkt[11]);
427  return -3;
428  }
429  }
430  return 0;
