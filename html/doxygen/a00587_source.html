92 void Model::parser(char *l)
93 {
94  defaults();
95  l = tokenize(l);
96  begin_while_parse(l) {
97  l = parse()->parseAttributes(l, this);
98  if (!l) break;
99  if (!stringcmp(l, "url=")) l = parse()->parseUrl(l, names.url);
100  else if (!stringcmp(l, "scale=")) l = parse()->parseFloat(l, &scale, "scale");
101  else if (!stringcmp(l, "color=")) l = parse()->parseVector3f(l, color, "color");
102  else if (!stringcmp(l, "bvh=")) {
103  bvhurl = new char[URL_LEN];
104  l = parse()->parseString(l, bvhurl, "bvh");
105  }
106  else if (!stringcmp(l, "sound=")) {
107  sndurl = new char[URL_LEN];
108  l = parse()->parseString(l, sndurl, "sound");
109  }
110  else if (! stringcmp(l, "tex=")) {
111  texurl = new char[URL_LEN];
112  l = parse()->parseString(l, texurl, "tex");
113  texid = Texture::getFromCache(texurl);
114  }
115  }
116  end_while_parse(l);
117 }

246 void Model::preRender()
247 {
248  if (transform) return;
249 
250  //DAX glPushAttrib(GL_ALL_ATTRIB_BITS);
251  if (texid) {
252  glEnable(GL_LIGHTING);
253  glEnable(GL_TEXTURE_2D);
254  glBindTexture(GL_TEXTURE_2D, texid);
255  }
256  else glMaterialfv(GL_FRONT, GL_AMBIENT_AND_DIFFUSE, color);
257 
258  glPushMatrix();
259  glTranslatef(pos.x, pos.y, pos.z);
260  glRotatef(RAD2DEG(pos.az), 0,0,1);
261  glRotatef(RAD2DEG(pos.ax) + 90, 1,0,0);
262  //glRotatef(RAD2DEG(pos.ay), 0,1,0);
263 }
