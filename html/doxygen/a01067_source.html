52 void Texture::update()
53 {
54  lockMutex(tex_pmutex);
55  //-------------- lock
56 
57  for (list<Texture*>::iterator it = textureList.begin(); it != textureList.end(); ++it) {
58  if ((*it)->img) {
59  if (! (*it)->img->pixmap) {
60  error("updateTextures pixmap=null: n=%d w=%d h=%d u=%s", (*it)->id, (*it)->img->width, (*it)->img->height, (*it)->url);
61  continue;
62  }
63  glBindTexture(GL_TEXTURE_2D, (*it)->id);
64 
65  if (! (*it)->img->wellSized()) { /* image to resize */
66  Img *img1 = NULL;
67  if ((img1 = (*it)->img->resize(Img::SIDE_SIZE, Img::SIDE_SIZE)) == NULL) {
68  //error("updateTextures: n=%d u=%s", (*it)->id, (*it)->url);
69  continue;
70  }
71  glTexImage2D(GL_TEXTURE_2D, 0, 3, img1->width, img1->height, 0,
72  GL_RGB, GL_UNSIGNED_BYTE, img1->pixmap);
73  if (img1) delete img1;
74  }
75  else { /* image well sized */
76  if (! (*it)->img->nummipmaps) {
77  glTexImage2D(GL_TEXTURE_2D, 0, 3, (*it)->img->width, (*it)->img->height, 0,
78  GL_RGB, GL_UNSIGNED_BYTE, (*it)->img->pixmap);
79  }
80  else {
81  GLsizei mipWidth = (*it)->img->width;
82  GLsizei mipHeight = (*it)->img->height;
83  int blockSize = ((*it)->img->channel == Img::RGB) ? 8 : 16;
84  int offset = 0;
85 
86  /* setup some parameters for texture filters and mipmapping */
87  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
88  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
89 
90  /* upload mipmaps to video memory */
91  for (GLint mip = 0; mip < (*it)->img->nummipmaps; ++mip) {
92  GLsizei mipSize = ((mipWidth + 3) / 4) * ((mipHeight + 3) / 4) * blockSize;
93 
94  glCompressedTexImage2D(GL_TEXTURE_2D, mip, (*it)->img->channel,
95  mipWidth, mipHeight, 0, mipSize,
96  (*it)->img->pixmap + offset);
97  mipWidth = MAX(mipWidth >> 1, 1);
98  mipHeight = MAX(mipHeight >> 1, 1);
99  offset += mipSize;
100  }
101  }
102  }
103  // once the texture resized we can delete its container img
104  if ((*it)->loaded) {
105  if ((*it)->img) delete (*it)->img; // opengl has its own copy of pixels
106  (*it)->img = NULL;
107  }
108  }
109  }
110  //-------------- unlock
111  unlockMutex(tex_pmutex);
112 }

121 void Texture::httpReader(void *_tex, Http *_http)
122 {
123  if (! _http) return; // error while opening
124  Texture *tex = (Texture *) _tex;
125 
126  tex->http = _http;
127  trace(DBG_IMG, "texture: mime=%s url=%s", tex->mime, tex->url);
128 
129  Img *img = NULL;
130 
131  // get the loader type by mime type or file extension
132  uint8_t loader = Format::getLoaderByMime(tex->mime);
133  switch (loader) {
134  case IMG_NULL:
135  loader = Format::getLoaderByUrl(tex->url);
136  break;
137  }
138 
139  // call the appropriated loader
140  switch (loader) {
141  case IMG_GIF: img = Img::loadGIF(tex, readImg); break;
142  case IMG_PNG: img = Img::loadPNG(tex, readImg); break;
143  case IMG_JPG: img = Img::loadJPG(tex, readImg); break;
144  case IMG_PPM: img = Img::loadPPM(tex, readImg); break;
145  case IMG_PGM: img = Img::loadPGM(tex, readImg); break;
146  case IMG_BMP: img = Img::loadBMP(tex, readImg); break;
147  case IMG_TGA: img = Img::loadTGA(tex, readImg); break;
148  case IMG_SGI: img = Img::loadSGI(tex, readImg); break;
149  case IMG_PCX: img = Img::loadPCX(tex, readImg); break;
150  case IMG_DDS: img = Img::loadDDS(tex, readImg); break;
151  case IMG_TIF: img = Img::loadTIF(tex, readImg); break;
152  case IMG_XPM: img = Img::loadXPM(tex, readImg); break;
153  case IMG_XBM: img = Img::loadXBM(tex, readImg); break;
154  case IMG_PSD: img = Img::loadPSD(tex, readImg); break;
155  case IMG_NULL: return;
156  default:
157  warning("texture: unrecognized image format");
158  return;
159  }
160 
161  lockMutex(tex_pmutex);
162  tex->img = img;
163  tex->loaded = true;
164  unlockMutex(tex_pmutex);
165 }

167 Texture::Texture(const char *url)
168 {
169  if (! check(url)) return;
170 
171  id = 0;
172  http = NULL;
173  img = NULL;
174  loaded = false;
175  *mime = '\0';
176 
177  strcpy(this->url, url);
178  textureList.push_back(this);
179  id = getid();
180  last_texid = id;
181  //error("%d %s", id, url);
182  // load image
183  switch (Format::getLoaderByUrl((char*) url)) {
184  case IMG_GIF: Http::httpOpen(url, httpReader, this, 1); break; // multi-threaded
185  default: Http::httpOpen(url, httpReader, this, 0); break;
186  }
187  new_texture++;
188 }

237 GLuint Texture::getFromCache(const char *url)
238 {
239  if (! check(url)) return 0;
240 
241  GLuint texid = exist(url);
242  if (texid) return texid;
243 
244  Texture * texture = new Texture(); // new entry in cache
245  strcpy(texture->url, url);
246  texture->id = getid();
247  last_texid = texture->id;
248  //trace(DBG_IMG, "texture: id=%d %s", texture->id, url);
249 
250  /* we must download the texture now */
251  switch (Format::getLoaderByUrl((char*) url)) {
252  case IMG_GIF: Http::httpOpen(url, httpReader, texture, 1); break; // multi-threaded
253  default: Http::httpOpen(url, httpReader, texture, 0); break;
254  }
255  return texture->id;
256 }
