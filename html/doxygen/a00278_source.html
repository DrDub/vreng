46 /* parser */
47 void Animator::parser(char *l)
48 {
49  defaults();
50  l = tokenize(l);
51  begin_while_parse(l) {
52  l = parse()->parseAttributes(l, this);
53  if (!l) break;
54  if (!stringcmp(l, "state")) l = parse()->parseUInt16(l, (uint16_t*) &state, "state");
55  else if (!stringcmp(l, "anim")) l = parse()->parseUInt16(l, (uint16_t*) &state, "anim");
56  else if (!stringcmp(l, "begin")) l = parse()->parseUInt8(l, &frame, "begin");
57  else if (!stringcmp(l, "ttl")) l = parse()->parseFloat(l, &ttl, "ttl");
58  }
59  end_while_parse(l);
60 }
