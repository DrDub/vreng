59 void Step::parser(char *l)
60 {
61  defaults();
62  l = tokenize(l);
63  begin_while_parse(l) {
64  l = parse()->parseAttributes(l, this);
65  if (!l) break;
66  if (!stringcmp(l, "mode=")) {
67  char modestr[16];
68  l = parse()->parseString(l, modestr, "mode");
69  if (!stringcmp(modestr, "escalator")) { escalator = true; mobile = true; }
70  else if (!stringcmp(modestr, "travelator")) { travelator = true; mobile = true; }
71  else if (!stringcmp(modestr, "stair")) stair = true;
72  else if (!stringcmp(modestr, "spiral")) { stair = true; spiral = true; }
73  }
74  if (!stringcmp(l, "dir=")) {
75  char modestr[16];
76  l = parse()->parseString(l, modestr, "dir");
77  if (!stringcmp(modestr, "up")) dir = 1;
78  else if (!stringcmp(modestr, "down")) dir = -1;
79  }
80  else if (!stringcmp(l, "height")) l = parse()->parseFloat(l, &height, "height");
81  else if (!stringcmp(l, "length")) l = parse()->parseFloat(l, &length, "length");
82  else if (!stringcmp(l, "speed")) l = parse()->parseFloat(l, &speed, "speed");
83  }
84  end_while_parse(l);
85 
86  if (travelator) {
87  height = 0;
88  dir = 0;
89  }
90 }

102 void Step::build()
103 {
104  float size;
105  float sx = pos.bbsize.v[0]; // step width
106  float sy = pos.bbsize.v[1]; // step depth
107  float sz = pos.bbsize.v[2]; // step height
108 
109  char *geom = new char[strlen(geometry) + 1];
110  strcpy(geom, geometry); // step geometry to duplicate
111 
112  // steps are interlaced (sx sy sz)
113  if (travelator) {
114  nsteps = (int) ceil(length / sx);
115  size = length;
116  }
117  else { // escalator stair spiral
118  if (height) {
119  if (escalator) height += sz; // add the top step
120  }
121  else if (length && pos.ax) {
122  height = length * tan(pos.ax);
123  pos.ax = 0;
124  }
125  nsteps = (int) ceil(height / sz);
126  size = height;
127  }
128 
129  firstpos = pos; // save first step position for other steps
130 
131  for (int n=0; n <= nsteps; n++) {
132  Pos newpos;
133  newpos.az = pos.az;
134  newpos.ax = newpos.ay = 0;
135 
136  if (spiral) {
137  float deltaspiral = atan(sy / sx);
138  newpos.x = pos.x + (sx * (cos(deltaspiral * n) - 1));
139  newpos.y = pos.y + (sy * (sin(deltaspiral * n) - 1));
140  newpos.z = pos.z + (sz * n);
141  newpos.az = pos.az + (deltaspiral * n);
142  }
143  else {
144  if (travelator) {
145  //FIXME: wrong position
146  newpos.x = pos.x - (sin(pos.az) * sx * n);
147  newpos.y = pos.y - (cos(pos.az) * sy * n);
148  newpos.z = pos.z;
149  }
150  else { // escalator stair
151  newpos.x = pos.x + dir*(sin(pos.az) * sx * n);
152  newpos.y = pos.y + dir*(cos(pos.az) * sy * n);
153  newpos.z = pos.z + dir*(sz * n);
154  }
155  }
156  nextstep = new Step(newpos, firstpos, geom, mobile, size, speed, dir);
157  }
158  if (geom) delete[] geom;
159 
160  if (mobile) {
161  disableBehavior(PERSISTENT);
162  enablePermanentMovement(speed);
163  }
164 }

230 void Step::changePermanent(float lasting)
231 {
232  if (! mobile) return;
233  if (state == INACTIVE) return;
234 
235  float sx = pos.bbsize.v[0]; // step width
236  float sy = pos.bbsize.v[1]; // step depth
237  float sz = pos.bbsize.v[2]; // step height
238 
239  if (dir > 0) { // escalator upwards
240  pos.x += lasting * move.lspeed.v[0] * sin(pos.az);
241  pos.y += lasting * move.lspeed.v[1] * cos(pos.az);
242  pos.z += lasting * move.lspeed.v[2];
243  if (stuck) { // user follows this step
244  localuser->pos.x = pos.x;
245  localuser->pos.y = pos.y;
246  localuser->pos.z = pos.z + localuser->height/2;
247  if (pos.z >= (firstpos.z - sz + height)) { // user stops
248  localuser->pos.x += (sin(pos.az) * sx);
249  localuser->pos.y += (cos(pos.az) * sy);
250  localuser->pos.z += sz;
251  stuck = false;
252  }
253  localuser->updatePosition();
254  }
255  if (pos.z >= (firstpos.z - sz + height)) { // switch step
256  pos.x = firstpos.x - (sin(pos.az) * sx);
257  pos.y = firstpos.y - (cos(pos.az) * sy);
258  pos.z = firstpos.z - sz;
259  }
260  }
261  else if (dir < 0) { // escalator downwards
262  pos.x -= lasting * move.lspeed.v[0] * sin(pos.az);
263  pos.y -= lasting * move.lspeed.v[1] * cos(pos.az);
264  pos.z -= lasting * move.lspeed.v[2];
265  if (stuck) { // user follows this step
266  localuser->pos.x = pos.x;
267  localuser->pos.y = pos.y;
268  localuser->pos.z = pos.z + localuser->height/2;
269  if (pos.z <= (firstpos.z + sz - height)) { // user stops
270  localuser->pos.x -= (sin(pos.az) * sx);
271  localuser->pos.y -= (cos(pos.az) * sy);
272  stuck = false;
273  }
274  localuser->updatePosition();
275  }
276  if (pos.z <= (firstpos.z + sz - height)) {
277  pos.x = firstpos.x;
278  pos.y = firstpos.y;
279  pos.z = firstpos.z;
280  }
281  }
282  else { // travelator horizontal
283  pos.x -= lasting * move.lspeed.v[0] * sin(pos.az);
284  pos.y -= lasting * move.lspeed.v[1] * cos(pos.az);
285  if (stuck) { // user follows this step
286  localuser->pos.x = pos.x;
287  localuser->pos.y = pos.y;
288  if (pos.x >= (firstpos.x - sx + length)) { // user stops
289  localuser->pos.x -= (sin(pos.az) * sx);
290  localuser->pos.y -= (cos(pos.az) * sy);
291  stuck = false;
292  }
293  localuser->updatePosition();
294  }
295  if (pos.x >= (firstpos.x - sx + length)) {
296  pos.x = firstpos.x;
297  pos.y = firstpos.y;
298  }
299  }
300 
301  updatePositionAndGrid(pos);
302 }

309 /* object intersects: up */
310 bool Step::whenIntersect(WObject *pcur, WObject *pold)
311 {
312  switch (pcur->type) {
313 
314  case USER_TYPE:
315  if (mobile) { // escalator
316  if (! stuck) {
317  stuck = true;
318  return true;
319  }
320  return false;
321  }
322  else { // normal case
323  if (pcur->pos.x == pold->pos.x && pcur->pos.y == pold->pos.y) {
324  pold->copyPositionAndBB(pcur);
325  }
326  else {
327  if (pos.bbsize.v[2] < JUMP/8) pcur->pos.z += pos.bbsize.v[2];
328  else pcur->pos.z += JUMP; // up
329  pcur->updatePositionAndGrid(pold);
330  }
331  }
332  break;
333 
334  case STEP_TYPE:
335  case GUIDE_TYPE:
336  return false; // ignore this collision
337  break;
338 
339  case BALL_TYPE:
340  if (mobile) return false;
341  else pold->copyPositionAndBB(pcur);
342  break;
343 
344  default:
345  pold->copyPositionAndBB(pcur);
346  }
347  return true;
348 }
