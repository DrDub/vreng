106 int Movie::aviInit()
107 {
108  if (avi) return 0;
109 
110  avi = new Avi(names.url);
111 
112  int r = avi->read_header();
113  if (r) {
114  error("avi: error=%d", r);
115  delete avi;
116  avi = NULL;
117  return r;
118  }
119  fp = avi->getInfos(&width, &height, &fps);
120  //error("avi: w=%d h=%d f=%.3f", width, height, fps);
121  pixmap = new uint8_t[3 * width * height];
122  begin = true;
123  frame = 0;
124 
125  /* compute texsize */
126  int i, power = 0;
127  texsize = MAX(width, height);
128  while ((texsize = texsize/2) > 0) power++;
129  for (i=0, texsize=1; i <= power; i++) texsize *= 2;
130  //error("texsize=%d", texsize);
131 
132  /* get memory for tex pixmap */
133  pixtex = new GLubyte[3 * texsize * texsize];
134 
135  return r;
136 }

143 int Movie::mpegInit()
144 {
145  /* download Mpeg file */
146  cachempeg = new char[MAXHOSTNAMELEN];
147  if (Cache::download(names.url, cachempeg) == 0) {
148  error("can't download %s", cachempeg);
149  delete[] cachempeg;
150  return 0;
151  }
152  if ((fp = File::openFile(cachempeg, "r")) == NULL) {
153  error("can't open mpeg");
154  delete[] cachempeg;
155  return 0;
156  }
157 
158  /* init Mpeg */
159  imgmpeg = new ImageDesc[1];
160 
161  SetMPEGOption(MPEG_DITHER, FULL_COLOR_DITHER); //ORDERED_DITHER);
162  OpenMPEG(fp, imgmpeg);
163 
164  /* get Mpeg infos */
165  width = imgmpeg->Width;
166  height = imgmpeg->Height;
167  fps = imgmpeg->PictureRate;
168  pixmap = new uint8_t[imgmpeg->Size];
169 
170  /* compute texsize */
171  int i, power = 0;
172  texsize = MAX(width, height);
173  while ((texsize = texsize/2) > 0) power++;
174  for (i=0, texsize=1; i <= power; i++) texsize *= 2;
175  //error("mpg: size=%d w=%d h=%d texsize=%d f=%.1f", imgmpeg->Size,width,height,texsize,fps);
176 
177  /* get memory for tex pixmap */
178  pixtex = new GLubyte[3 * texsize * texsize];
179 
180  frame = 0;
181  begin = true;
182  signal(SIGABRT, mpegAbort);
183  return 1;
184 }

186 void Movie::changePermanent(float lasting)
187 {
188  if (state == INACTIVE || state == PAUSE) return;
189 
190  if (begin) {
191  gettimeofday(&start, NULL);
192  begin = false;
193  }
194 
195  uint16_t inter = frame;
196  struct timeval t;
197  gettimeofday(&t, NULL);
198  frame = (int) floor(((((float) (t.tv_sec - start.tv_sec) * 1000.) +
199  ((float) (t.tv_usec - start.tv_usec) / 1000.)
200  ) / 1000.) * (float) rate);
201 
202  for (int f=0; f < frame-inter; f++) {
203  /* get Mpeg frame or not */
204  if (f >= (int) fps) break;
205 
206  if (video == PLAYER_AVI) {
207  /* get Avi frame */
208  if (avi) {
209  int r, l;
210  static int n = 0;
211 
212  r = avi->read_data(pixmap, width * height * 3, &l);
213  //error("n=%d l=%d", n, l);
214  if (r == 0) {
215  //error("end of avi video");
216  File::closeFile(fp);
217  fp = NULL;
218  delete avi;
219  avi = NULL;
220  begin = true;
221  state = INACTIVE;
222  return;
223  }
224  n++;
225  int woff = (texsize - width) / 2;
226  int hoff = (texsize - height) / 2;
227  for (int j=0; j < height; j++) {
228  for (int i=0; i < width; i++) {
229  int u = 3 * (texsize * (j+hoff) + (i+woff));
230  int v = 3 * (width * j + i);
231  if (File::littleEndian()) {
232  pixtex[u+0] = pixmap[v+0];
233  pixtex[u+1] = pixmap[v+1];
234  pixtex[u+2] = pixmap[v+2];
235  }
236  else {
237  pixtex[u+0] = pixmap[v+2]; // RGB = ABG
238  pixtex[u+1] = pixmap[v+1];
239  pixtex[u+2] = pixmap[v+0];
240  }
241  }
242  }
243  }
244  }
245  else if (video == PLAYER_MPG) {
246  if (GetMPEGFrame((char *)pixmap) == false) {
247  /* The End */
248  if (state == LOOP) {
249  RewindMPEG(fp, imgmpeg);
250  begin = true;
251  return;
252  }
253  error("frames=%d", ++frame);
254  CloseMPEG();
255  state = INACTIVE;
256  return;
257  }
258  int woff = (texsize - width) / 2;
259  int hoff = (texsize - height) / 2;
260  trace(DBG_WO, "frame=%d f=%d texid=%d size=%d w=%d h=%d woff=%d hoff=%d", frame, f, texid, texsize, width, height, woff, hoff);
261 
262  ColormapEntry *colorMap = imgmpeg->Colormap;
263  if (colorMap) { // case of Colormap Index
264  for (int h=0; h < height; h++) {
265  for (int w=0; w < width; w++) {
266  int cmapindex = pixmap[h * width + w];
267  ColormapEntry *color = &colorMap[cmapindex];
268  int u = 3 * ((h+hoff) * texsize + (w+hoff));
269  pixtex[u+0] = color->red % 255;
270  pixtex[u+1] = color->green % 255;
271  pixtex[u+2] = color->blue % 255;
272  }
273  }
274  }
275  else {
276  for (int h=0; h < height; h++) {
277  for (int w=0; w < width; w++) {
278  int u = 3 * (texsize * (h+hoff) + (w+woff));
279  int v = 4 * (width * h + w);
280  if (File::littleEndian()) {
281  pixtex[u+0] = pixmap[v+0];
282  pixtex[u+1] = pixmap[v+1];
283  pixtex[u+2] = pixmap[v+2];
284  }
285  else {
286  pixtex[u+0] = pixmap[v+2]; // RGB = ABG
287  pixtex[u+1] = pixmap[v+1];
288  pixtex[u+2] = pixmap[v+0];
289  }
290  }
291  }
292  }
293  }
294  }
295 
296  // bind the frame texid
297  GLint param = (::g.pref.quality3D) ? GL_LINEAR : GL_NEAREST;
298  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, param);
299  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, param);
300  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
301  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
302  glEnable(GL_TEXTURE_2D);
303  glBindTexture(GL_TEXTURE_2D, texid);
304  glTexImage2D(GL_TEXTURE_2D, 0, 3, texsize, texsize, 0, GL_RGB, GL_UNSIGNED_BYTE, pixtex);
305 }
