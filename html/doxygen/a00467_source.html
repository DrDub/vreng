125 void Fractal::generate(float x, float z, float w, float h, float a, uint16_t l)
126 {
127  turtle_x = x;
128  turtle_z = z;
129  turtle_r = h;
130  step();
131  float x2 = turtle_x;
132  float z2 = turtle_z;
133  l--;
134  if (l < 3) {
135  glColor3f(0.0, 1.0, 0.0);
136  drawbranch(w, x, z, x2, z2);
137  }
138  else {
139  glColor3f(0.0, 0.0, 0.0);
140  drawbranch(w, x, z, x2, z2);
141  }
142  if (l > 0) {
143  turtle_theta = point(x, z, x2, z2);
144  turn(left_angle);
145  generate(turtle_x, turtle_z, left_w_factor*w, left_h_factor*h, left_angle, l);
146  turtle_theta = point(x, z, x2, z2);
147  turn(-right_angle);
148  generate(x2, z2, left_w_factor*w, left_h_factor*h, right_angle, l);
149  }
150 }

157 float Fractal::point(float x1, float z1, float x2, float z2)
158 {
159  float theta;
160 
161  if ((x2 - x1) == 0.0) {
162  if (z2 > z1) theta = 90.0;
163  else theta = 270.0;
164  }
165  else {
166  theta = atan((z2-z1) / (x2-x1)) * 57.295779;
167  }
168  if (x1 > x2) theta += 180.0;
169  return theta;
170 }
