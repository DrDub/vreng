186 void Teapot::normal(point n, point p, point q, point r)
187 {
188  n[0] = (q[1]-p[1])*(r[2]-p[2])-(q[2]-p[2])*(r[1]-p[1]);
189  n[1] = (q[2]-p[2])*(r[0]-p[0])-(q[0]-p[0])*(r[2]-p[2]);
190  n[2] = (q[0]-p[0])*(r[2]-p[2])-(q[2]-p[2])*(r[0]-p[0]);
191 }

193 /* division of convex hull of Bezier curve */
194 void Teapot::divide_curve(point *c, point *r, point *l)
195 {
196  for (int i=0; i<3 ; i++) {
197  point t;
198 
199  l[0][i]=c[0][i];
200  r[3][i]=c[3][i];
201  l[1][i]=(c[1][i]+c[0][i])/2;
202  r[2][i]=(c[2][i]+c[3][i])/2;
203  t[i]=(l[1][i]+r[2][i])/2;
204  l[2][i]=(t[i]+l[1][i])/2;
205  r[1][i]=(t[i]+r[2][i])/2;
206  l[3][i]=r[0][i]=(l[2][i]+r[1][i])/2;
207  }
208 }

210 /* transpose wastes time but makes program more readable */
211 void Teapot::transpose(point p[4][4])
212 {
213  for (int i=0; i<4 ; i++) for (int j=i; j<4 ; j++) for (int k=0; k<3 ; k++) {
214  GLfloat tt = p[i][j][k];
215  p[i][j][k] = p[j][i][k];
216  p[j][i][k] = tt;
217  }
218 }

220 /* subdivide curves in u direction, transpose results, divide
221  * in u direction again (equivalent to subdivision in v)
222  */
223 void Teapot::divide_patch(point p[4][4], uint8_t n)
224 {
225  point q[4][4], r[4][4], s[4][4], t[4][4];
226  point a[4][4], b[4][4];
227 
228  if (n==0) draw_patch(p); /* draw patch if recursion done */
229  else {
230  for (int k=0; k<4; k++)
231  divide_curve(p[k], a[k], b[k]);
232  transpose(a);
233  transpose(b);
234  for (int k=0; k<4; k++) {
235  divide_curve(a[k], q[k], r[k]);
236  divide_curve(b[k], s[k], t[k]);
237  }
238  /* recursive division of 4 resulting patches */
239  divide_patch(q, n-1);
240  divide_patch(r, n-1);
241  divide_patch(s, n-1);
242  divide_patch(t, n-1);
243  }
244 }
