77 void * Vac::connectThread(void *)
78 {
79  int sdvac;
80 
81  if ((sdvac = Socket::openStream()) < 0) {
82  error("socket vacs");
83  return NULL;
84  }
85  // resolve vacs address
86  struct hostent *hp;
87  if ((hp = my_gethostbyname(DEF_VACS_SERVER, AF_INET)) == NULL) {
88  error("can't resolve vacs");
89  }
90 
91  struct sockaddr_in savac;
92  struct timeval timeout;
93  timeout.tv_sec = 10;
94  timeout.tv_usec = 0;
95 
96  // set a timeout of 10 sec
97  if (setsockopt(sdvac, SOL_SOCKET, SO_RCVTIMEO, (char *)&timeout, sizeof(timeout)) < 0)
98  error("setsockopt failed\n");
99 
100  memset(&savac, 0, sizeof(struct sockaddr_in));
101  savac.sin_family = AF_INET;
102  savac.sin_port = htons(VACS_PORT);
103  memcpy(&savac.sin_addr, hp->h_addr_list[0], hp->h_length);
104  my_free_hostent(hp);
105 
106  if (connect(sdvac, (const struct sockaddr *) &savac, sizeof(savac)) < 0) {
107  perror("can't connect vacs");
108  return NULL;
109  }
110  //connected = true;
111  return NULL;
112 }

114 bool Vac::getList()

142 
143  vacList = this;
144  next = NULL;
145  Vac *prev = vacList;
146  int r;
147 
148  char *cache = new char[sizecache + 1];
149  memset(cache, 0, sizecache + 1);
150  while ((r = read(sdvac, cache, sizecache)) > 0) {
151  cache[r] = '\0';
152  //DEBUG printf("readvac: r=%d,%d l=%s\n", r, (int) strlen(cache), cache);
153  char *p = strtok(cache, " ");
154  while (p) {
155  if (strncmp(p, "http://", 7) != 0) break; // !!! ELC: pour eviter blocage
156  else{
157  Vac *vac = new Vac();

160  if (p) strcpy(vac->channel, p);
161  vac->next = NULL;
162  prev->next = vac;
163  prev = vac;
164  p = strtok(NULL, " ");
165  //DEBUG printf("u=%s c=%s\n", vac->url, vac->channel);
166  }
167  }
168  }
169  delete[] cache;
170  Socket::closeStream(sdvac);
171  trace(DBG_INIT, "Vacs initialized");
172  return true;
173 }

196  bool cached = false;
197  Vac *w = NULL, *wlast = NULL;
198  for (w = vacList->next; w ; w = w->next) {
199  if (! strcmp(url, w->url)) {
200  cached = true;
201  break;
202  }
203  wlast = w;
204  }
205  if (! cached) {
206 
207  Vac *vac = new Vac();
208 
209  strcpy(vac->url, url);
210  strcpy(vac->channel, line);
211  vac->next = NULL;
212  wlast->next = vac;
213  //trace(DBG_FORCE, "resolveWorld: channel=%s", vac->channel);
214  }
215  }
216  Socket::closeStream(sdvac);
217  return true;
218  }
219  else {
220  error("resolveWorld: channel NULL");
221  strcpy(chanstr, DEF_VRE_CHANNEL);
222  Socket::closeStream(sdvac);
