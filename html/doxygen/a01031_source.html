107 static int gifReadScreen(GifInfo *g)
108 {
109  uint8_t buf[7];
110 
111  if (gifread(g, buf, 7) != 7) return -4;
112  if (buf[6]) return -5;
113 
114  g->screenwidth = buf[0] + (buf[1] << 8);
115  g->screenheight = buf[2] + (buf[3] << 8);
116  g->global = buf[4] & 0x80;
117  g->colres = ((buf[4] & 0x70) >> 4) +1;
118  g->backclr = buf[5];
119  if (g->global) {
120  g->globalbits = (buf[4] & 0x07) + 1;
121  gifread(g, (uint8_t *) g->globalmap, 3*(1<<g->globalbits));
122  }
123  g->img = new Img(g->screenwidth, g->screenheight, Img::RGB);
124  return (g->img == NULL);
125 }

127 static int gifReadImage(GifInfo *g)
128 {
129  int nb_pass, pass_height, line_start, line_inc;
130  int localbits, ncolors;
131  uint8_t *pix, *pix1;
132  uint8_t buf[10];
133  uint8_t localmap[256][3];
134 
135  if ((gifread(g, buf, 10)) != 10) return -6;
136  int left = buf[0] + (buf[1] << 8);
137  int top = buf[2] + (buf[3] << 8);
138  int width = buf[4] + (buf[5] << 8);
139  int height = buf[6] + (buf[7] << 8);
140  int interleaved = buf[8] & 0x40;
141  int local = buf[8] & 0x80;
142  if (left+width>g->screenwidth || top+height>g->screenheight) return -7;
143 
144  if (local) {
145  localbits = (buf[8] & 0x7) + 1;
146  ncolors = 1 << localbits;
147  gifread(g, (uint8_t *) localmap, 3*ncolors);
148  }
149  else if (g->global) {
150  ncolors = (1 << g->globalbits);
151  memcpy(localmap, g->globalmap, 3*ncolors);
152  }
153  else return -8;
154 
155  int cbyte = buf[9];
156  if (cbyte<2 || cbyte>8) return -9;
157 
158  LZWDecodeInit(g, cbyte);
159 
160  if (interleaved) nb_pass=4;
161  else nb_pass=1;
162  uint8_t *line_buf = new uint8_t[width];
163 
164  for (int pass=0; pass < nb_pass; pass++) {
165  switch(pass) {
166  case 0:
167  line_start=0;
168  if (interleaved) {
169  pass_height=8;
170  line_inc=8;
171  } else {
172  pass_height=1;
173  line_inc=1;
174  }
175  break;
176  case 1:
177  pass_height=4;
178  line_start=4;
179  line_inc=8;
180  break;
181  case 2:
182  pass_height=2;
183  line_start=2;
184  line_inc=4;
185  break;
186  default:
187  pass_height=1;
188  line_start=1;
189  line_inc=2;
190  break;
191  }
192  for (int line=line_start; line < height; line += line_inc) {
193  LZWDecode(g, line_buf, width);
194  pix = (uint8_t *)g->img->pixmap + ((left + top * g->screenwidth) + (g->screenwidth * line)) * 3;
195  pix1 = pix;
196  for (int i=0; i < width; i++) {
197  int c = line_buf[i];
198  pix[0] = localmap[c][0];
199  pix[1] = localmap[c][1];
200  pix[2] = localmap[c][2];
201  pix += 3;
202  }
203  if (pass_height > 1) {
204  for (int j=1; j < pass_height; j++) {
205  memcpy(pix, pix1, g->screenwidth*3);
206  pix += g->screenwidth*3;
207  }
208  }
209  }
210  }
211  LZWDecodeEnd(g);
212  delete[] line_buf;
213  return 0;
214 }

318 /*
319  * Gif LZW Decoder
320  */
321 static void LZWDecode(GifInfo *g, uint8_t *out_buf, int out_len)
322 {
323  int sizbuf, bbits;
324  uint32_t bbuf;
325  uint8_t *pbuf, *obuf;
326  int olen;
327  int c, code, cursize, curmask, clear_code, end_code, newcodes;
328  int top_slot, slot, fc, oc;
329  uint8_t *stack, *suffix, *sp;
330  int16_t *prefix;
331 
332 /* cache rw */
333  cursize = g->cursize;
334  slot = g->slot;
335  oc = g->oc;
336  fc = g->fc;
337  sp = g->sp;
338 
339  sizbuf = g->sizbuf;
340  bbuf = g->bbuf;
341  bbits = g->bbits;
342  pbuf = g->pbuf;
343 
344 /* cache r */
345  clear_code = g->clear_code;
346  end_code = g->end_code;
347  newcodes = g->newcodes;
348  clear_code = g->clear_code;
349  stack = g->stack;
350  prefix = g->prefix;
351  suffix = g->suffix;
352 
353  top_slot = (1 << cursize);
354  curmask = top_slot - 1;
355  obuf = out_buf;
356  olen = out_len;
357 
358  if (olen == 0) goto _end;
359 
360  while (sp > stack) {
361  *obuf++ = *(--sp);
362  if ( (--olen) == 0 ) goto _end;
363  }
364 
365  while (1) {
366  GETCODE(c);
367  if (c == end_code) break;
368  if (c == clear_code) {
369  cursize = g->codesize+1;
370  top_slot = 1<<cursize;
371  curmask=top_slot - 1;
372  slot = newcodes;
373  do {
374  GETCODE(c);
375  } while (c == clear_code);
376  if (c == end_code) break;
377  fc = oc = c;
378  *obuf++ = c;
379  if ((--olen) == 0) break;
380  }
381  else {
382  code = c;
383  if (code >= slot) {
384  *sp++ = fc;
385  code = oc;
386  }
387  while (code >= newcodes) {
388  *sp++ = suffix[code]; //segfault snowleopard
389  code = prefix[code];
390  }
391  *sp++ = code;
392  if (slot < top_slot) {
393  suffix[slot] = fc = code;
394  prefix[slot++] = oc;
395  oc = c;
396  }
397  if (slot >= top_slot) {
398  if (cursize < MAXBITS) {
399  cursize++;
400  top_slot <<= 1;
401  curmask=top_slot - 1;
402  }
403  }
404  while (sp > stack) {
405  *obuf++ = *(--sp);
406  if ((--olen) == 0) goto _end;
407  }
408  }
409  }
410  _end:
411 
412 /* cache rw */
413  g->cursize = cursize;
414  g->slot = slot;
415  g->oc = oc;
416  g->fc = fc;
417  g->sp = sp;
418 
419  g->sizbuf = sizbuf;
420  g->bbuf = bbuf;
421  g->bbits = bbits;
422  g->pbuf = pbuf;
423 }
