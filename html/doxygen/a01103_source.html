151  }
152  else if (! stringcmp(l, HEAD_FAP_V20)) { // Fap2.0 Header
153  num_baps = NUM_FAPS;
154  resetMask(num_baps);

156  l = strtok(NULL, " \t"); l = strtok(NULL, " \t");

159  else if (! stringcmp(l, HEAD_FAP_V21)) { // Fap2.1 Header
160  num_baps = NUM_FAPS;

162  bfflag = TYPE_FAP_V21;

164  //debug error("readFrame fap2.1: num_baps=%d", num_baps);

166 
167  // Datas = Masks and Values
168  else {

170  for (int i=1; i <= num_baps; i++) {

188  }
189  }

191  for (int i=1; i <= num_baps; i++) {
192  if (l) {
193  bit[i] = atoi(l); //extract all the mask values
194  l = strtok(NULL, " \t");
195  }
196  }
197  trace(DBG_MAN, "readFrame fap: num_frame=%s", l);
198  for (int i=1; i <= num_baps; i++) {
199  if (bit[i] == 0) continue;
200  if ((l = strtok(NULL, " \t")) == NULL) break;
201  if (bfflag == TYPE_FAP_V20) fa[i] = atoi(l) / FAPV20_DIV; //unknown formula
202  else fa[i] = atoi(l) / FAPV21_DIV; //unknown formula
203  }
204  }
205  else { error("readFrame: unknown type=%d", bfflag); return 0; }
206  }
207  return 1; // frame ready to play
