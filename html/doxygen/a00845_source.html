89 void Widget::buildScreen()
90 {
91  if (_solids.empty()) return; // no solid
92 
93  V3 dim;
94  getDim(dim);
95 
96  float left = MAX(dim.v[1], dim.v[0]);
97  float front = MIN(dim.v[0], dim.v[1]) + 0.02;
98  float top = dim.v[2];
99  float bot = -top*0.9;
100  float rsol = left / top;
101  float rtex = (u * tex_width) / (v * tex_height);
102  //float depth = 0.9 * front;
103  float height;
104  float width;
105 
106  if (rtex > rsol) {
107  height = bot + (2*0.9) * left/rtex;
108  width = 0.9 * left;
109  }
110  else {
111  height = (2*0.9-1) * top;
112  width = 0.9 * top * rtex;
113  }
114 
115  vertices[ 0] = front; vertices[ 1] = width; vertices[ 2] = bot; // bot right
116  vertices[ 3] = front; vertices[ 4] = width; vertices[ 5] = height; // top right
117  vertices[ 6] = front; vertices[ 7] = -width; vertices[ 8] = height; // top left
118  vertices[ 9] = front; vertices[10] = -width; vertices[11] = bot; // bot left
119 }

136 /* Set texture parameters */
137 void Widget::setTexture(bool mipmap)
138 {
139  glEnable(GL_TEXTURE_2D); // we need to use a texture
140  glBindTexture(GL_TEXTURE_2D, texture); // we use ours
141 
142  // put it into the video memory
143  if (mipmap) { // with mipmap interpolation
144  gluBuild2DMipmaps(GL_TEXTURE_2D, 3, tex_width, tex_height,
145  GL_RGB, GL_UNSIGNED_BYTE, tex_pixmap);
146  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
147  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
148  }
149  else { // without mipmap interpolation
150  glTexImage2D(GL_TEXTURE_2D, 0, 3, tex_width, tex_height,
151  0, GL_RGB, GL_UNSIGNED_BYTE, tex_pixmap);
152  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
153  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
154  }
155 
156  //glTexEnvi(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, GL_DECAL); // bad fonts
157  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP);
158  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP);
159  glDisable(GL_TEXTURE_2D);
160 }

178 /* Renders at each loop */
179 void Widget::render()
180 {
181  updatePosition();
182 
183  // push
184  //DAX glPushAttrib(GL_ALL_ATTRIB_BITS);
185  glPushMatrix();
186  glEnable(GL_LIGHTING);
187  glEnable(GL_CULL_FACE);
188  glEnable(GL_TEXTURE_2D);
189  glBindTexture(GL_TEXTURE_2D, texture);
190 
191  // change system of coordonates
192  M4 posmat;
193  getPosition(posmat);
194  GLfloat gl_mat[16];
195  M4toV16(&posmat, gl_mat);
196  glMultMatrixf(gl_mat);
197 
198  // render
199  glRotatef(RAD2DEG(pos.az), 0, 0, 1);
200  drawScreen(); // draw screen
201 
202  // update parameters for mouse handling
203  GLint renderMode;
204  glGetIntegerv(GL_RENDER_MODE, &renderMode);
205  if (renderMode == GL_RENDER) {
206  glGetIntegerv(GL_VIEWPORT, viewport);
207  glGetDoublev(GL_MODELVIEW_MATRIX, modelmat);
208  glGetDoublev(GL_PROJECTION_MATRIX, projmat);
209  }
210 
211  // pop
212  glDisable(GL_CULL_FACE);
213  glDisable(GL_TEXTURE_2D);
214  glDisable(GL_LIGHTING);
215  glPopMatrix();
216  //DAX glPopAttrib();
217 }
