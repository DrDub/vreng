82 static
83 void getRow(SgiInfo *sgi, uint8_t *buf, int y, int z)
84 {
85  if ((sgi->type & 0xFF00) == 0x0100) { // storage = RLE (run length encoding)
86  fseek(sgi->fp, (uint32_t) sgi->rowStart[y + z*sgi->height], SEEK_SET);
87  fread(sgi->tmp, 1, (uint32_t) sgi->rowSize[y + z*sgi->height], sgi->fp);
88 
89  uint8_t *iPtr = sgi->tmp;
90  uint8_t *oPtr = buf;
91  for (;;) {
92  uint8_t pixel = *iPtr++;
93  int count = (int)(pixel & 0x7F);
94  if (!count)
95  return;
96  if (pixel & 0x80) {
97  while (count--) *oPtr++ = *iPtr++;
98  }
99  else {
100  pixel = *iPtr++;
101  while (count--) *oPtr++ = pixel;
102  }
103  }
104  }
105  else { // storage = VERBATIM
106  fseek(sgi->fp, 512 + (y*sgi->width) + (z*sgi->width*sgi->height), SEEK_SET);
107  fread(buf, 1, sgi->width, sgi->fp);
108  }
109 }

112 Img * Img::loadSGI(void *tex, ImageReader read_func)
113 {
114  SgiInfo *sgi = new SgiInfo[1];
115  if (! sgi) return NULL;
116 
117  Texture *_tex = (Texture *) tex;
118  if ((sgi->fp = Cache::openCache(_tex->url, _tex->http)) == NULL) return NULL;
119 
120  fread(sgi, 1, 12, sgi->fp); // header
121 
122  if (File::littleEndian())
123  File::convertShort(&sgi->imagic, 6);
124  sgi->tmp = new uint8_t[sgi->width*256];
125  if (!sgi->tmp) return NULL;
126 
127  if ((sgi->type & 0xFF00) == 0x0100) { // storage = RLE
128  int x = sgi->height * sgi->channel * sizeof(uint32_t);
129  sgi->rowStart = new uint32_t[x];
130  sgi->rowSize = new uint32_t[x];
131  if (!sgi->rowStart || !sgi->rowSize) return NULL;
132  sgi->rleEnd = 512 + (2 * x);
133  fseek(sgi->fp, 512, SEEK_SET);
134  fread(sgi->rowStart, 1, x, sgi->fp);
135  fread(sgi->rowSize, 1, x, sgi->fp);
136  if (File::littleEndian()) {
137  File::convertLong(sgi->rowStart, x/(int)sizeof(uint32_t));
138  File::convertLong((uint32_t *)sgi->rowSize, x/(int)sizeof(uint32_t));
139  }
140  }
141 
142  trace(DBG_IMG, "sgi: magic=%x type=%x width=%d height=%d channel=%d",
143  sgi->imagic, sgi->type, sgi->width, sgi->height, sgi->channel);
144 
145  /* alloc the data */
146  Img *img = new Img(sgi->width, sgi->height, Img::RGB);
147 
148  uint8_t *rbuf = new uint8_t[sgi->width];
149  uint8_t *gbuf = new uint8_t[sgi->width];
150  uint8_t *bbuf = new uint8_t[sgi->width];
151  uint8_t *abuf = new uint8_t[sgi->width];
152  if (!img->pixmap || !rbuf || !gbuf || !bbuf || !abuf) return NULL;
153 
154  /* read the data */
155  uint8_t *lptr = img->pixmap;
156  for (int y=0; y < sgi->height; y++) {
157  if (sgi->channel == 4) { // RGBA
158  getRow(sgi, rbuf, y, 0);
159  getRow(sgi, gbuf, y, 1);
160  getRow(sgi, bbuf, y, 2);
161  getRow(sgi, abuf, y, 3);
162  rgbatorgb(rbuf, gbuf, bbuf, abuf, lptr, sgi->width);
163  }
164  else if (sgi->channel == 3) { // RGB
165  getRow(sgi, rbuf, y, 0);
166  getRow(sgi, gbuf, y, 1);
167  getRow(sgi, bbuf, y, 2);
168  rgbtorgb(rbuf, gbuf, bbuf, lptr, sgi->width);
169  }
170  else if (sgi->channel == 2) { // LA
171  getRow(sgi, rbuf, y, 0);
172  getRow(sgi, abuf, y, 1);
173  latorgb(rbuf, abuf, lptr, sgi->width);
174  }
175  else { // BW
176  getRow(sgi, rbuf, y, 0);
177  bwtorgb(rbuf, lptr, sgi->width);
178  }
179  lptr += (sgi->width * Img::RGB);
180  }
181 
182  File::closeFile(sgi->fp);
183  delete[] sgi->tmp;
184  delete[] sgi;
185  delete[] rbuf;
186  delete[] gbuf;
187  delete[] bbuf;
188  delete[] abuf;
189 
190  return img;
191 }
